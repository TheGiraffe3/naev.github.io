94 void gl_screenshot( const char *filename )
95 {
96  GLubyte *screenbuf;
97  SDL_RWops *rw;
98  SDL_Surface *surface;
99  int i, w, h;
100 
101  /* Allocate data. */
102  w = gl_screen.rw;
103  h = gl_screen.rh;
104  screenbuf = malloc( sizeof(GLubyte) * 3 * w*h );
105  surface = SDL_CreateRGBSurface( 0, w, h, 24, RGBAMASK );
106 
107  /* Read pixels from buffer -- SLOW. */
108  glPixelStorei(GL_PACK_ALIGNMENT, 1); /* Force them to pack the bytes. */
109  glReadPixels( 0, 0, w, h, GL_RGB, GL_UNSIGNED_BYTE, screenbuf );
110 
111  /* Convert data. */
112  for (i=0; i<h; i++)
113  memcpy( (GLubyte*)surface->pixels + i * (3*w), &screenbuf[ (h - i - 1) * (3*w) ], 3*w );
114  free( screenbuf );
115 
116  /* Save PNG. */
117  if (!(rw = PHYSFSRWOPS_openWrite( filename )))
118  WARN( _("Aborting screenshot") );
119  else
120  IMG_SavePNG_RW( surface, rw, 1 );
121 
122  /* Check to see if an error occurred. */
123  gl_checkErr();
124 
125  /* Free memory. */
126  SDL_FreeSurface( surface );
127 }

164

167  errstr = _("GL invalid enum");
168  break;
169  case GL_INVALID_VALUE:
170  errstr = _("GL invalid value");
171  break;
172  case GL_INVALID_OPERATION:

266 static int gl_createWindow( unsigned int flags )
267 {
268  int ret, fallback;
269 
270  /* Create the window. */
271  gl_screen.window = SDL_CreateWindow( APPNAME,
272  SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,
273  conf.width, conf.height, flags | SDL_WINDOW_SHOWN | SDL_WINDOW_RESIZABLE
274  | SDL_WINDOW_ALLOW_HIGHDPI );
275  if (gl_screen.window == NULL)
276  ERR(_("Unable to create window! %s"), SDL_GetError());
277 
278  /* Set focus loss behaviour. */
279  SDL_SetHint( SDL_HINT_VIDEO_MINIMIZE_ON_FOCUS_LOSS,
280  conf.minimize ? "1" : "0" );
281 
282  /* Create the OpenGL context, note we don't need an actual renderer. */
283  for (fallback = 0; fallback <= 1; fallback++) {
284  gl_setupAttributes( fallback );
285  gl_screen.context = SDL_GL_CreateContext( gl_screen.window );
286  if (gl_screen.context != NULL)
287  break;
288  }
289  if (!gl_screen.context)
290  ERR(_("Unable to create OpenGL context! %s"), SDL_GetError());
291 
292  /* Set Vsync. */
293  if (conf.vsync) {
294  ret = SDL_GL_SetSwapInterval( 1 );
295  if (ret == 0)
296  gl_screen.flags |= OPENGL_VSYNC;
297  } else {
298  SDL_GL_SetSwapInterval( 0 );
299  }
300 
301  /* Finish getting attributes. */
302  gl_screen.current_fbo = 0; /* No FBO set. */
303  gl_screen.fbo[0] = GL_INVALID_VALUE;
304  gl_screen.fbo_tex[0] = GL_INVALID_VALUE;
305  gl_screen.fbo[1] = GL_INVALID_VALUE;
306  gl_screen.fbo_tex[1] = GL_INVALID_VALUE;
307  SDL_GL_GetAttribute( SDL_GL_DEPTH_SIZE, &gl_screen.depth );
308  gl_activated = 1; /* Opengl is now activated. */
309 
310  return 0;
311 }

385 static int gl_setupScaling (void)
386 {
387  double scalew, scaleh;
388 
389  /* Get the basic dimensions from SDL2. */
390  SDL_GetWindowSize(gl_screen.window, &gl_screen.w, &gl_screen.h);
391  SDL_GL_GetDrawableSize(gl_screen.window, &gl_screen.rw, &gl_screen.rh);
392  /* Calculate scale factor, if OS has native HiDPI scaling. */
393  gl_screen.dwscale = (double)gl_screen.w / (double)gl_screen.rw;
394  gl_screen.dhscale = (double)gl_screen.h / (double)gl_screen.rh;
395 
396  /* Combine scale factor from OS with the one in Naev's config */
397  gl_screen.scale = fmax(gl_screen.dwscale, gl_screen.dhscale) / conf.scalefactor;
398 
399  /* New window is real window scaled. */
400  gl_screen.nw = (double)gl_screen.rw * gl_screen.scale;
401  gl_screen.nh = (double)gl_screen.rh * gl_screen.scale;
402  /* Small windows get handled here. */
403  if ((gl_screen.nw < RESOLUTION_W_MIN)
404  || (gl_screen.nh < RESOLUTION_H_MIN)) {
405  if (gl_screen.scale != 1.)
406  DEBUG(_("Screen size too small, upscaling..."));
407  scalew = RESOLUTION_W_MIN / (double)gl_screen.nw;
408  scaleh = RESOLUTION_H_MIN / (double)gl_screen.nh;
409  gl_screen.scale *= MAX( scalew, scaleh );
410  /* Rescale. */
411  gl_screen.nw = (double)gl_screen.rw * gl_screen.scale;
412  gl_screen.nh = (double)gl_screen.rh * gl_screen.scale;
413  }
414  /* Viewport matches new window size. */
415  gl_screen.w = gl_screen.nw;
416  gl_screen.h = gl_screen.nh;
417  /* Set scale factors. */
418  gl_screen.wscale = (double)gl_screen.nw / (double)gl_screen.w;
419  gl_screen.hscale = (double)gl_screen.nh / (double)gl_screen.h;
420  gl_screen.mxscale = (double)gl_screen.w / (double)gl_screen.rw;
421  gl_screen.myscale = (double)gl_screen.h / (double)gl_screen.rh;
422 
423  return 0;
424 }

431 int gl_init (void)
432 {
433  unsigned int flags;
434  GLuint VaoId;
435 
436  /* Defaults. */
437  memset( &gl_screen, 0, sizeof(gl_screen) );
438 
439  flags = SDL_WINDOW_OPENGL | gl_getFullscreenMode();
440 
441  /* Initializes Video */
442  if (SDL_InitSubSystem(SDL_INIT_VIDEO) < 0) {
443  WARN(_("Unable to initialize SDL Video: %s"), SDL_GetError());
444  return -1;
445  }
446 
447  /* Create the window. */
448  gl_createWindow( flags );
449 
450  /* Apply the configured fullscreen display mode, if any. */
451  gl_setupFullscreen();
452 
453  /* Load extensions. */
454  if (!gladLoadGLLoader(SDL_GL_GetProcAddress))
455  ERR("Unable to load OpenGL using GLAD");
456 
457  /* We are interested in 3.1 because it drops all the deprecated stuff. */
458  if ( !GLAD_GL_VERSION_3_1 )
459  WARN( "Naev requires OpenGL 3.1, but got OpenGL %d.%d!", GLVersion.major, GLVersion.minor );
460 
461  /* Some OpenGL options. */
462  glClearColor( 0., 0., 0., 1. );
463 
464  /* Set default opengl state. */
465  gl_defState();
466 
467  /* Handles resetting the viewport and scaling, rw/rh are set in createWindow. */
468  gl_resize();
469 
470  /* Finishing touches. */
471  glClear( GL_COLOR_BUFFER_BIT ); /* must clear the buffer first */
472  gl_checkErr();
473 
474  /* Initialize subsystems.*/
475  gl_initMatrix();
476  gl_initTextures();
477  gl_initVBO();
478  gl_initRender();
479 
480  /* Get info about the OpenGL window */
481  gl_getGLInfo();
482 
483  /* Modern OpenGL requires at least one VAO */
484  glGenVertexArrays(1, &VaoId);
485  glBindVertexArray(VaoId);
486 
487  shaders_load();
488 
489  /* Set colorblind shader if necessary. */
490  gl_colorblind( conf.colorblind );
491 
492  /* Set colourspace. */
493  glEnable( GL_FRAMEBUFFER_SRGB );
494 
495  /* Cosmetic new line. */
496  DEBUG_BLANK();
497 
498  return 0;
499 }

528 void gl_viewport( int x, int y, int w, int h )
529 {
530  gl_Matrix4 proj;
531 
532  proj = gl_Matrix4_Ortho( 0., /* Left edge. */
533  gl_screen.nw, /* Right edge. */
534  0., /* Bottom edge. */
535  gl_screen.nh, /* Top edge. */
536  -1., /* near */
537  1. ); /* far */
538 
539  /* Take into account possible translation. */
540  gl_screen.x = x;
541  gl_screen.y = y;
542  proj = gl_Matrix4_Translate(proj, x, y, 0);
543 
544  /* Set screen size. */
545  gl_screen.w = w;
546  gl_screen.h = h;
547 
548  /* Take into account possible scaling. */
549  if (gl_screen.scale != 1.)
550  proj = gl_Matrix4_Scale(proj, gl_screen.wscale, gl_screen.hscale, 1);
551 
552  gl_view_matrix = proj;
553 }
