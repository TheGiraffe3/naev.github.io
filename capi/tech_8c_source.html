96 
97  /* Load the data. */

99  if (doc == NULL)
100  return -1;
101 
102  /* Load root element. */
103  parent = doc->xmlChildrenNode;
104  if (!xml_isNode(parent,XML_TECH_ID)) {
105  WARN(_("Malformed %s file: missing root element '%s'"), TECH_DATA_PATH, XML_TECH_ID);
106  return -1;
107  }
108 
109  /* Get first node. */
110  node = parent->xmlChildrenNode;
111  if (node == NULL) {
112  WARN(_("Malformed %s file: does not contain elements"), TECH_DATA_PATH);
113  return -1;
114  }
115 
116  /* Create the array. */
117  tech_groups = array_create( tech_group_t );
118 
119  /* First pass create the groups - needed to reference them later. */
120  ret = 0;
121  tech = NULL;
122  do {
123  xml_onlyNodes(node);
124  /* Must match tag. */
125  if (!xml_isNode(node, XML_TECH_TAG)) {
126  WARN(_("'%s' has unknown node '%s'."), XML_TECH_ID, node->name);
127  continue;
128  }
129  if (ret==0) /* Write over failures. */
130  tech = &array_grow( &tech_groups );
131  ret = tech_parseNode( tech, node );
132  } while (xml_nextNode(node));
133  array_shrink( &tech_groups );
134 
135  /* Now we load the data. */
136  node = parent->xmlChildrenNode;
137  s = array_size( tech_groups );
138  do {
139  /* Must match tag. */
140  if (!xml_isNode(node, XML_TECH_TAG))
141  continue;
142 
143  xmlr_attr_strd( node, "name", buf );
144  if (buf == NULL)
145  continue;
146 
147  /* Load next tech. */
148  for (i=0; i<s; i++) {
149  tech = &tech_groups[i];
150  if (strcmp(tech->name, buf)==0)
151  tech_parseNodeData( tech, node );
152  }
153 
154  /* Free memory. */
155  free(buf);
156  } while (xml_nextNode(node));
157 
158  /* Info. */
159  DEBUG( n_( "Loaded %d tech group", "Loaded %d tech groups", s ), s );
160 
161  /* Free memory. */
162  xmlFreeDoc(doc);
163 
164  return 0;
