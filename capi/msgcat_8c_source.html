101 const char* msgcat_ngettext( const msgcat_t* p, const char* msgid1, const char* msgid2, uint64_t n )
102 {
103  const char *trans = msgcat_mo_lookup(p->map, p->map_size, msgid1);
104  if (!trans) return NULL;
105 
106  /* Non-plural-processing gettext forms pass a null pointer as
107  * msgid2 to request that dcngettext suppress plural processing. */
108 
109  if (msgid2 && p->nplurals) {
110  uint64_t plural = msgcat_plural_eval(p->plural_rule, n);
111  if (plural > p->nplurals) return NULL;
112  while (plural--) {
113  size_t rem = p->map_size - (trans - (char *)p->map);
114  size_t l = strnlen(trans, rem);
115  if (l+1 >= rem)
116  return NULL;
117  trans += l+1;
118  }
119  }
120  return trans;
121 }

131 const char *msgcat_mo_lookup(const void *p, size_t size, const char *s)
132 {
133  const uint32_t *mo = p;
134  int sw = *mo - 0x950412de;
135  uint32_t b = 0, n = swapc(mo[2], sw);
136  uint32_t o = swapc(mo[3], sw);
137  uint32_t t = swapc(mo[4], sw);
138  if (n>=size/4 || o>=size-4*n || t>=size-4*n || ((o|t)%4))
139  return 0;
140  o/=4;
141  t/=4;
142  for (;;) {
143  uint32_t ol = swapc(mo[o+2*(b+n/2)], sw);
144  uint32_t os = swapc(mo[o+2*(b+n/2)+1], sw);
145  if (os >= size || ol >= size-os || ((char *)p)[os+ol])
146  return 0;
147  int sign = strcmp(s, (char *)p + os);
148  if (!sign) {
149  uint32_t tl = swapc(mo[t+2*(b+n/2)], sw);
150  uint32_t ts = swapc(mo[t+2*(b+n/2)+1], sw);
151  if (ts >= size || tl >= size-ts || ((char *)p)[ts+tl])
152  return 0;
153  return (char *)p + ts;
154  }
155  else if (n == 1) return 0;
156  else if (sign < 0)
157  n /= 2;
158  else {
159  b += n/2;
160  n -= n/2;
161  }
162  }
163  return 0;
164 }

168 /*
169 grammar:
170 
171 Start = Expr ';'
172 Expr = Or | Or '?' Expr ':' Expr
173 Or = And | Or '||' And
174 And = Eq | And '&&' Eq
175 Eq = Rel | Eq '==' Rel | Eq '!=' Rel
176 Rel = Add | Rel '<=' Add | Rel '>=' Add | Rel '<' Add | Rel '>' Add
177 Add = Mul | Add '+' Mul | Add '-' Mul
178 Mul = Prim | Mul '*' Prim | Mul '/' Prim | Mul '%' Prim
179 Prim = '(' Expr ')' | '!' Prim | decimal | 'n'
180 
181 internals:
182 
183 recursive descent expression evaluator with stack depth limit.
184 for binary operators an operator-precedence parser is used.
185 eval* functions store the result of the parsed subexpression
186 and return a pointer to the next non-space character.
187 */

203 static const char *evalprim(struct st *st, const char *s, int d)
204 {
205  char *e;
206  if (--d < 0) return "";
207  s = skipspace(s);
208  if (isdigit(*s)) {
209  st->r = strtoul(s, &e, 10);
210  if (e == s || st->r == UINT64_MAX) return "";
211  return skipspace(e);
212  }
213  if (*s == 'n') {
214  st->r = st->n;
215  return skipspace(s+1);
216  }
217  if (*s == '(') {
218  s = evalexpr(st, s+1, d);
219  if (*s != ')') return "";
220  return skipspace(s+1);
221  }
222  if (*s == '!') {
223  s = evalprim(st, s+1, d);
224  st->r = !st->r;
225  return s;
226  }
227  return "";
228 }

230 static int binop(struct st *st, int op, uint64_t left)
231 {
232  uint64_t a = left, b = st->r;
233  switch (op) {
234  case 0: st->r = a||b; return 0;
235  case 1: st->r = a&&b; return 0;
236  case 2: st->r = a==b; return 0;
237  case 3: st->r = a!=b; return 0;
238  case 4: st->r = a>=b; return 0;
239  case 5: st->r = a<=b; return 0;
240  case 6: st->r = a>b; return 0;
241  case 7: st->r = a<b; return 0;
242  case 8: st->r = a+b; return 0;
243  case 9: st->r = a-b; return 0;
244  case 10: st->r = a*b; return 0;
245  case 11: if (b) {st->r = a%b; return 0;} return 1;
246  case 12: if (b) {st->r = a/b; return 0;} return 1;
247  }
248  return 1;
249 }

251 static const char *parseop(struct st *st, const char *s)
252 {
253  static const char opch[11] = "|&=!><+-*%/";
254  static const char opch2[6] = "|&====";
255  int i;
256  for (i=0; i<11; i++)
257  if (*s == opch[i]) {
258  /* note: >,< are accepted with or without = */
259  if (i<6 && s[1] == opch2[i]) {
260  st->op = i;
261  return s+2;
262  }
263  if (i>=4) {
264  st->op = i+2;
265  return s+1;
266  }
267  break;
268  }
269  st->op = 13;
270  return s;
271 }

273 static const char *evalbinop(struct st *st, const char *s, int minprec, int d)
274 {
275  static const char prec[14] = {1,2,3,3,4,4,4,4,5,5,6,6,6,0};
276  uint64_t left;
277  int op;
278  d--;
279  s = evalprim(st, s, d);
280  s = parseop(st, s);
281  for (;;) {
282  /*
283  st->r (left hand side value) and st->op are now set,
284  get the right hand side or back out if op has low prec,
285  if op was missing then prec[op]==0
286  */
287  op = st->op;
288  if (prec[op] <= minprec)
289  return s;
290  left = st->r;
291  s = evalbinop(st, s, prec[op], d);
292  if (binop(st, op, left))
293  return "";
294  }
295 }

297 static const char *evalexpr(struct st *st, const char *s, int d)
298 {
299  uint64_t a, b;
300  if (--d < 0)
301  return "";
302  s = evalbinop(st, s, 0, d);
303  if (*s != '?')
304  return s;
305  a = st->r;
306  s = evalexpr(st, s+1, d);
307  if (*s != ':')
308  return "";
309  b = st->r;
310  s = evalexpr(st, s+1, d);
311  st->r = a ? b : st->r;
312  return s;
