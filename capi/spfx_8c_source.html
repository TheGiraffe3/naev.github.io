158 static int spfx_base_parse( SPFX_Base *temp, const char *filename )
159 {
160  xmlNodePtr node, cur, uniforms;
161  char *shadervert, *shaderfrag;
162  const char *name;
163  double x, y, z, w;
164  int ix, iy, iz, iw;
165  int isint;
166  GLint loc, dim;
167  xmlDocPtr doc;
168 
169  /* Load and read the data. */
170  doc = xml_parsePhysFS( filename );
171  if (doc == NULL)
172  return -1;
173 
174  /* Check to see if document exists. */
175  node = doc->xmlChildrenNode;
176  if (!xml_isNode(node,SPFX_XML_ID)) {
177  ERR( _("Malformed '%s' file: missing root element '%s'"), filename, SPFX_XML_ID);
178  return -1;
179  }
180 
181  /* Clear data. */
182  memset( temp, 0, sizeof(SPFX_Base) );
183  temp->shader = -1;
184  shadervert = NULL;
185  shaderfrag = NULL;
186  uniforms = NULL;
187 
188  xmlr_attr_strd( node, "name", temp->name );
189 
190  /* Extract the data. */
191  node = node->xmlChildrenNode;
192  do {
193  xml_onlyNodes(node);
194  xmlr_float(node, "anim", temp->anim);
195  xmlr_float(node, "ttl", temp->ttl);
196  if (xml_isNode(node,"gfx")) {
197  temp->gfx = xml_parseTexture( node,
198  SPFX_GFX_PATH"%s", 6, 5, 0 );
199  continue;
200  }
201 
202  if (xml_isNode(node,"shader")) {
203  cur = node->xmlChildrenNode;
204  do {
205  xml_onlyNodes(cur);
206  xmlr_strd(cur, "vert", shadervert);
207  xmlr_strd(cur, "frag", shaderfrag);
208  xmlr_float(cur, "size", temp->size);
209  if (xml_isNode(cur,"uniforms")) {
210  uniforms = cur;
211  continue;
212  }
213  } while (xml_nextNode(cur));
214  continue;
215  }
216  WARN(_("SPFX '%s' has unknown node '%s'."), temp->name, node->name);
217  } while (xml_nextNode(node));
218 
219  /* Convert from ms to s. */
220  temp->anim /= 1000.;
221  temp->ttl /= 1000.;
222  if (temp->ttl == 0.)
223  temp->ttl = temp->anim;
224 
225  /* Has shaders. */
226  if (shadervert != NULL && shaderfrag != NULL) {
227  temp->shader = gl_program_vert_frag( shadervert, shaderfrag );
228  temp->vertex = glGetAttribLocation( temp->shader, "vertex");
229  temp->projection = glGetUniformLocation( temp->shader, "projection");
230  temp->u_r = glGetUniformLocation( temp->shader, "u_r" );
231  temp->u_time = glGetUniformLocation( temp->shader, "u_time" );
232  temp->u_size = glGetUniformLocation( temp->shader, "u_size" );
233  if (uniforms != NULL) {
234  glUseProgram( temp->shader );
235  node = uniforms->xmlChildrenNode;
236  do {
237  xml_onlyNodes(node);
238  name = (char*)node->name;
239  loc = glGetUniformLocation( temp->shader, name );
240  if (loc < 0) {
241  WARN(_("SPFX '%s' is trying to set uniform '%s' not in shader!"), temp->name, name );
242  continue;
243  }
244  xmlr_attr_int_def(node,"int",isint,0);
245  /* Get dimension */
246  if (xmlHasProp(node,(xmlChar*)"w")) dim = 4;
247  else if (xmlHasProp(node,(xmlChar*)"z")) dim = 3;
248  else if (xmlHasProp(node,(xmlChar*)"y")) dim = 2;
249  else dim = 1;
250  /* Float values default to 0. */
251  if (isint) {
252  xmlr_attr_int(node, "x", ix );
253  xmlr_attr_int(node, "y", iy );
254  xmlr_attr_int(node, "z", iz );
255  xmlr_attr_int(node, "w", iw );
256  }
257  else {
258  xmlr_attr_float(node, "x", x );
259  xmlr_attr_float(node, "y", y );
260  xmlr_attr_float(node, "z", z );
261  xmlr_attr_float(node, "w", w );
262  }
263  switch (dim) {
264  case 1:
265  if (isint)
266  glUniform1i( loc, ix );
267  else
268  glUniform1f( loc, x );
269  break;
270  case 2:
271  if (isint)
272  glUniform2i( loc, ix, iy );
273  else
274  glUniform2f( loc, x, y );
275  break;
276  case 3:
277  if (isint)
278  glUniform3i( loc, ix, iy, iz );
279  else
280  glUniform3f( loc, x, y, z );
281  break;
282  case 4:
283  if (isint)
284  glUniform4i( loc, ix, iy, iz, iw );
285  else
286  glUniform4f( loc, x, y, z, w );
287  break;
288  default:
289  WARN(_("SPFX '%s' is trying to set uniform '%s' with '%d' dimensions!"), temp->name, name, dim );
290  continue;
291  }
292  } while (xml_nextNode(node));
293  glUseProgram( 0 );
294  }
295  gl_checkErr();
296  }
297 
298 #define MELEMENT(o,s) \
299  if (o) WARN( _("SPFX '%s' missing/invalid '%s' element"), temp->name, s)
300  MELEMENT(temp->anim==0.,"anim");
301  MELEMENT(temp->ttl==0.,"ttl");
302  MELEMENT(temp->gfx==NULL && (shadervert==NULL || shaderfrag==NULL),"gfx or shader");
303  MELEMENT(temp->shader>=0 && temp->size<=0., "shader/size");
304 #undef MELEMENT
305 
306  free(shadervert);
307  free(shaderfrag);
308 
309  /* Clean up. */
310  xmlFreeDoc(doc);
311 
312  return 0;
313 }

469  else if (layer == SPFX_LAYER_MIDDLE) /* middle layer */

484  anim = spfx_effects[effect].anim;

486  cur_spfx->timer = ttl + RNGF()*anim;
487  else
488  cur_spfx->timer = ttl;
489 
490  /* Shader magic. */
491  cur_spfx->unique = RNGF();

548 {

568 static void spfx_updateShake( double dt )
569 {
570  double mod, vmod, angle;
571  double force_x, force_y;
572  double dupdate;
573  int forced;
574 
575  /* Must still be on. */
576  if (shake_shader_pp_id == 0)
577  return;
578 
579  /* The shake decays over time */
580  forced = 0;
581  if (shake_force_mod > 0.) {
582  shake_force_mod -= SPFX_SHAKE_DECAY*dt;
583  if (shake_force_mod < 0.)
584  shake_force_mod = 0.;
585  else
586  forced = 1;
587  }
588  dupdate = dt*2.0;
589  shake_force_mean = dupdate*shake_force_mod + (1.0-dupdate)*shake_force_mean;
590 
591  /* See if it's settled down. */
592  mod = VMOD( shake_pos );
593  vmod = VMOD( shake_vel );
594  if (!forced && (mod < 0.01) && (vmod < 0.01)) {
595  render_postprocessRm( shake_shader_pp_id );
596  shake_shader_pp_id = 0;
597  if (fabs(shake_force_ang) > 1e3)
598  shake_force_ang = RNGF();
599  return;
600  }
601 
602  /* Calculate force. */
603  force_x = -SHAKE_K*shake_pos.x + -SHAKE_B*shake_vel.x;
604  force_y = -SHAKE_K*shake_pos.y + -SHAKE_B*shake_vel.y;
605 
606  /* Apply force if necessary. */
607  if (forced) {
608  shake_force_ang += dt;
609  angle = noise_simplex1( shake_noise, &shake_force_ang ) * 5.*M_PI;
610  force_x += shake_force_mod * cos(angle);
611  force_y += shake_force_mod * sin(angle);
612  }
613 
614  /* Update velocity. */
615  vect_cadd( &shake_vel, (1./SHAKE_MASS) * force_x * dt, (1./SHAKE_MASS) * force_y * dt );
616 
617  /* Update position. */
618  vect_cadd( &shake_pos, shake_vel.x * dt, shake_vel.y * dt );
619 
620  /* Set the uniform. */
621  glUseProgram( shaders.shake.program );
622  glUniform2f( shaders.shake.shake_pos, shake_pos.x / SCREEN_W, shake_pos.y / SCREEN_H );
623  glUniform2f( shaders.shake.shake_vel, shake_vel.x / SCREEN_W, shake_vel.y / SCREEN_H );
624  glUniform1f( shaders.shake.shake_force, shake_force_mean );
625  glUseProgram( 0 );
626 
627  gl_checkErr();
628 }

703 {

705  /* Remove outdated elements. */

726 void spfx_trail_sample( Trail_spfx* trail, double x, double y, TrailMode mode, int force )
727 {
728  TrailPoint p;
729 
730  if (!force && trail->spec->style[mode].col.a <= 0.)
731  return;
732 
733  p.x = x;
734  p.y = y;
735  p.t = 1.;
736  p.mode = mode;
737 
738  /* The "back" of the trail should always reflect our most recent state. */
739  trail_back( trail ) = p;
740 
741  /* We may need to insert a control point, but not if our last sample was recent enough. */
742  if (!force && trail_size(trail) > 1 && trail_at( trail, trail->iwrite-2 ).t >= 1.-TRAIL_UPDATE_DT)
743  return;
744 
745  /* If the last time we inserted a control point was recent enough, we don't need a new one. */
746  if (trail_size(trail) == trail->capacity) {
747  /* Full! Double capacity, and make the elements contiguous. (We've made space to grow rightward.) */
748  trail->point_ringbuf = realloc( trail->point_ringbuf, 2 * trail->capacity * sizeof(TrailPoint) );
749  trail->iread %= trail->capacity;
750  trail->iwrite = trail->iread + trail->capacity;
751  memmove( &trail->point_ringbuf[trail->capacity], trail->point_ringbuf, trail->iread * sizeof(TrailPoint) );
752  trail->capacity *= 2;
753  }
754  trail_at( trail, trail->iwrite++ ) = p;
755 }

930 static void spfx_hapticRumble( double mod )
931 {
932  SDL_HapticEffect *efx;
933  double len, mag;
934 
935  /* Not active. */

938

941  return;

944  SDL_HapticStopEffect( haptic, haptic_rumble );

956  WARN(_("Failed to update haptic effect: %s."), SDL_GetError());

958  }
959

961  SDL_HapticRunEffect( haptic, haptic_rumble, 1 );

1141 static int trailSpec_load (void)
1142 {

1144  xmlNodePtr first, node;
1145  xmlDocPtr doc;
1146  char *name, *inherits;
1147 
1148  /* Load the data. */
1149  doc = xml_parsePhysFS( TRAIL_DATA_PATH );
1150  if (doc == NULL)
1151  return -1;
1152 
1153  /* Get the root node. */
1154  node = doc->xmlChildrenNode;
1155  if (!xml_isNode(node,"Trail_types")) {
1156  WARN( _("Malformed '%s' file: missing root element 'Trail_types'"), TRAIL_DATA_PATH);
1157  return -1;
1158  }
1159 
1160  /* Get the first node. */
1161  first = node->xmlChildrenNode; /* first event node */
1162  if (node == NULL) {
1163  WARN( _("Malformed '%s' file: does not contain elements"), TRAIL_DATA_PATH);
1164  return -1;
1165  }
1166 
1167  trail_spec_stack = array_create( TrailSpec );
1168 
1169  node = first;
1170  do {
1171  xml_onlyNodes( node );
1172  if (xml_isNode(node,"trail")) {
1173  tc = &array_grow( &trail_spec_stack );
1174  memset( tc, 0, sizeof(TrailSpec) );
1175  for(int i=0; i<MODE_MAX; i++)
1176  tc->style[i].thick = 1.;
1177  xmlr_attr_strd( node, "name", tc->name );
1178 
1179  /* Do the first pass for non-inheriting trails. */
1180  xmlr_attr_strd( node, "inherits", inherits );
1181  if (inherits == NULL)
1182  trailSpec_parse( node, tc );
1183  else
1184  free( inherits );
1185  }
1186  } while (xml_nextNode(node));
1187 
1188  /* Second pass to complete inheritance. */
1189  node = first;
1190  do {
1191  xml_onlyNodes( node );
1192  if (xml_isNode(node,"trail")) {
1193  /* Only interested in inherits. */
1194  xmlr_attr_strd( node, "inherits", inherits );
1195  if (inherits == NULL)
1196  continue;
1197  parent = trailSpec_getRaw( inherits );
1198 
1199  /* Get the style itself. */
1200  xmlr_attr_strd( node, "name", name );
1201  tc = trailSpec_getRaw( name );
1202 
1203  /* Make sure we found stuff. */
1204  if ((tc==NULL) || (parent==NULL)) {
1205  WARN(_("Trail '%s' that inherits from '%s' has missing reference!"), name, inherits );
1206  continue;
1207  }
1208  free( inherits );
1209  free( name );
1210 
1211  /* Set properties. */
1212  name = tc->name;
1213  memcpy( tc, parent, sizeof(TrailSpec) );
1214  tc->name = name;
1215 
1216  /* Load remaining properties (overrides parent). */
1217  trailSpec_parse( node, tc );
1218  }
1219  } while (xml_nextNode(node));
1220 
1221  for (tc=array_begin(trail_spec_stack); tc!=array_end(trail_spec_stack); tc++) {
1222  for(int i=0; i<MODE_MAX; i++)
1223  tc->style[i].thick *= tc->def_thick;
1224  }
1225  array_shrink(&trail_spec_stack);
1226 
1227  /* Clean up. */
1228  xmlFreeDoc(doc);
1229 
1230  return 0;
1231 }
