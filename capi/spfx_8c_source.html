148 static int spfx_base_parse( SPFX_Base *temp, const xmlNodePtr parent )
149 {
150  xmlNodePtr node;
151 
152  /* Clear data. */
153  memset( temp, 0, sizeof(SPFX_Base) );
154 
155  xmlr_attr_strd( parent, "name", temp->name );
156 
157  /* Extract the data. */
158  node = parent->xmlChildrenNode;
159  do {
160  xml_onlyNodes(node);
161  xmlr_float(node, "anim", temp->anim);
162  xmlr_float(node, "ttl", temp->ttl);
163  if (xml_isNode(node,"gfx")) {
164  temp->gfx = xml_parseTexture( node,
165  SPFX_GFX_PATH"%s", 6, 5, 0 );
166  continue;
167  }
168  WARN(_("SPFX '%s' has unknown node '%s'."), temp->name, node->name);
169  } while (xml_nextNode(node));
170 
171  /* Convert from ms to s. */
172  temp->anim /= 1000.;
173  temp->ttl /= 1000.;
174  if (temp->ttl == 0.)
175  temp->ttl = temp->anim;
176 
177 #define MELEMENT(o,s) \
178  if (o) WARN( _("SPFX '%s' missing/invalid '%s' element"), temp->name, s)
179  MELEMENT(temp->anim==0.,"anim");
180  MELEMENT(temp->ttl==0.,"ttl");
181  MELEMENT(temp->gfx==NULL,"gfx");
182 #undef MELEMENT
183 
184  return 0;

447 static void spfx_updateShake( double dt )
448 {
449  double mod, vmod, angle;
450  double force_x, force_y;
451  int forced;
452 
453  /* Must still be on. */
454  if (shake_off)
455  return;
456 
457  /* The shake decays over time */
458  forced = 0;
459  if (shake_force_mod > 0.) {
460  shake_force_mod -= SHAKE_DECAY*dt;
461  if (shake_force_mod < 0.)
462  shake_force_mod = 0.;
463  else
464  forced = 1;
465  }
466 
467  /* See if it's settled down. */
468  mod = VMOD( shake_pos );
469  vmod = VMOD( shake_vel );
470  if (!forced && (mod < 0.01) && (vmod < 0.01)) {
471  shake_off = 1;
472  render_postprocessRm( shake_shader_pp_id );
473  shake_shader_pp_id = 0;
474  if (shake_force_ang > 1e3)
475  shake_force_ang = RNGF();
476  return;
477  }
478 
479  /* Calculate force. */
480  force_x = -SHAKE_K*shake_pos.x + -SHAKE_B*shake_vel.x;
481  force_y = -SHAKE_K*shake_pos.y + -SHAKE_B*shake_vel.y;
482 
483  /* Apply force if necessary. */
484  if (forced) {
485  shake_force_ang += dt;
486  angle = noise_simplex1( shake_noise, &shake_force_ang ) * 5.*M_PI;
487  force_x += shake_force_mod * cos(angle);
488  force_y += shake_force_mod * sin(angle);
489  }
490 
491  /* Update velocity. */
492  vect_cadd( &shake_vel, (1./SHAKE_MASS) * force_x * dt, (1./SHAKE_MASS) * force_y * dt );
493 
494  /* Update position. */
495  vect_cadd( &shake_pos, shake_vel.x * dt, shake_vel.y * dt );
496 
497  /* Set the uniform. */
498  glUseProgram( shaders.shake.program );
499  glUniform2f( shaders.shake.shake_pos, shake_pos.x / SCREEN_W, shake_pos.y / SCREEN_H );
500  glUseProgram( 0 );
501 }

586 {
587  TrailPoint p;
588 
589  if (trail->spec->style[mode].col.a <= 0.)
590  return;
591 
592  p.x = x;
593  p.y = y;
594  p.t = 1.;
595  p.mode = mode;
596 
597  /* The "back" of the trail should always reflect our most recent state. */
598  trail_back( trail ) = p;
599 
600  /* We may need to insert a control point, but not if our last sample was recent enough. */
601  if (trail_size(trail) > 1 && trail_at( trail, trail->iwrite-2 ).t >= 1.-TRAIL_UPDATE_DT)
602  return;
603

605  if (trail_size(trail) == trail->capacity) {

607  trail->point_ringbuf = realloc( trail->point_ringbuf, 2 * trail->capacity * sizeof(TrailPoint) );
608  trail->iread %= trail->capacity;
609  trail->iwrite = trail->iread + trail->capacity;
610  memmove( &trail->point_ringbuf[trail->capacity], trail->point_ringbuf, trail->iread * sizeof(TrailPoint) );
611  trail->capacity *= 2;
612  }
613  trail_at( trail, trail->iwrite++ ) = p;

716 void spfx_begin( const double dt, const double real_dt )
717 {
718  double ddt;
719 
720  /* Defaults. */
721  shake_set = 0;
722  if (shake_off)
723  return;
724 
725  /* Decrement the haptic timer. */
726  if (haptic_lastUpdate > 0.)
727  haptic_lastUpdate -= real_dt; /* Based on real delta-tick. */
728 
729  /* Micro basic simple control loop. */
730  if (dt > shake_fps_min) {
731  ddt = dt;
732  while (ddt > shake_fps_min) {
733  spfx_updateShake( shake_fps_min );
734  ddt -= shake_fps_min;
735  }
736  spfx_updateShake( ddt ); /* Leftover. */
737  }
738  else
739  spfx_updateShake( dt );
740 
741  /* set the new viewport */
742  shake_set = 1;
743 }

1011 static int trailSpec_load (void)
1012 {

1014  xmlNodePtr first, node;
1015  xmlDocPtr doc;
1016  char *name, *inherits;
1017 
1018  /* Load the data. */
1019  doc = xml_parsePhysFS( TRAIL_DATA_PATH );

1021  return -1;
1022 
1023  /* Get the root node. */
1024  node = doc->xmlChildrenNode;
1025  if (!xml_isNode(node,"Trail_types")) {
1026  WARN( _("Malformed '%s' file: missing root element 'Trail_types'"), TRAIL_DATA_PATH);
1027  return -1;
1028  }
1029 
1030  /* Get the first node. */
1031  first = node->xmlChildrenNode; /* first event node */
1032  if (node == NULL) {
1033  WARN( _("Malformed '%s' file: does not contain elements"), TRAIL_DATA_PATH);
1034  return -1;
1035  }
1036 
1037  trail_spec_stack = array_create( TrailSpec );
1038 
1039  node = first;
1040  do {
1041  xml_onlyNodes( node );
1042  if (xml_isNode(node,"trail")) {
1043  tc = &array_grow( &trail_spec_stack );
1044  memset( tc, 0, sizeof(TrailSpec) );
1045  for(int i=0; i<MODE_MAX; i++)
1046  tc->style[i].thick = 1.;
1047  xmlr_attr_strd( node, "name", tc->name );
1048 
1049  /* Do the first pass for non-inheriting trails. */
1050  xmlr_attr_strd( node, "inherits", inherits );
1051  if (inherits == NULL)
1052  trailSpec_parse( node, tc );
1053  else
1054  free( inherits );
1055  }
1056  } while (xml_nextNode(node));
1057 
1058  /* Second pass to complete inheritance. */
1059  node = first;
1060  do {
1061  xml_onlyNodes( node );
1062  if (xml_isNode(node,"trail")) {
1063  /* Only interested in inherits. */
1064  xmlr_attr_strd( node, "inherits", inherits );
1065  if (inherits == NULL)
1066  continue;
1067  parent = trailSpec_getRaw( inherits );
1068 
1069  /* Get the style itself. */
1070  xmlr_attr_strd( node, "name", name );
1071  tc = trailSpec_getRaw( name );
1072 
1073  /* Make sure we found stuff. */
1074  if ((tc==NULL) || (parent==NULL)) {
1075  WARN(_("Trail '%s' that inherits from '%s' has missing reference!"), name, inherits );
1076  continue;
1077  }
1078  free( inherits );
1079  free( name );
1080 
1081  /* Set properties. */
1082  name = tc->name;
1083  memcpy( tc, parent, sizeof(TrailSpec) );
1084  tc->name = name;
1085 
1086  /* Load remaining properties (overrides parent). */
1087  trailSpec_parse( node, tc );
1088  }
1089  } while (xml_nextNode(node));
1090 
1091  for (tc=array_begin(trail_spec_stack); tc!=array_end(trail_spec_stack); tc++) {
1092  for(int i=0; i<MODE_MAX; i++)
1093  tc->style[i].thick *= tc->def_thick;
1094  }
1095  array_shrink(&trail_spec_stack);
1096 
1097  /* Clean up. */
1098  xmlFreeDoc(doc);
1099 
1100  return 0;
1101 }
