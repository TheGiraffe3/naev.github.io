29 char *strnstr( const char *haystack, const char *needle, size_t size )
30 {
31  size_t needlesize;
32  const char *i, *j, *k, *end, *giveup;
33 
34  needlesize = strlen(needle);
35  /* We can give up if needle is empty, or haystack can never contain it */
36  if (needlesize == 0 || needlesize > size)
37  return NULL;
38  /* The pointer value that marks the end of haystack */
39  end = haystack + size;
40  /* The maximum value of i, because beyond this haystack cannot contain needle */
41  giveup = end - needlesize + 1;
42 
43  /* i is used to iterate over haystack */
44  for (i = haystack; i != giveup; i++) {
45  /* j is used to iterate over part of haystack during comparison */
46  /* k is used to iterate over needle during comparison */
47  for (j = i, k = needle; j != end && *k != '\0'; j++, k++) {
48  /* Bail on the first character that doesn't match */
49  if (*j != *k)
50  break;
51  }
52  /* If we've reached the end of needle, we've found a match */
53  /* i contains the start of our match */
54  if (*k == '\0')
55  return (char*) i;
56  }
57  /* Fell through the loops, nothing found */
58  return NULL;
59 }
