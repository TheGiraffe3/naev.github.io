121 static JumpPoint *luaL_validjumpSystem( lua_State *L, int ind, int *offset )
122 {
123  LuaJump *lj;
124  JumpPoint *jp;
125  StarSystem *a, *b;
126 
127  /* Defaults. */
128  jp = NULL;
129  a = NULL;
130  b = NULL;
131 
132  if (lua_isjump(L, ind)) {
133  lj = luaL_checkjump(L, ind);
134  a = system_getIndex( lj->srcid );
135  b = system_getIndex( lj->destid );
136  if (offset != NULL)
137  *offset = 1;
138  }
139  else if (lua_gettop(L) > 1) {
140  if (lua_isstring(L, ind))
141  a = system_get( lua_tostring( L, ind ));
142  else if (lua_issystem(L, ind))
143  a = system_getIndex( lua_tosystem(L, ind) );
144 
145  if (lua_isstring(L, ind+1))
146  b = system_get( lua_tostring( L, ind+1 ));
147  else if (lua_issystem(L, ind+1))
148  b = system_getIndex( lua_tosystem(L, ind+1) );
149 
150  if (offset != NULL)
151  *offset = 2;
152  }
153  else {
154  luaL_typerror(L, ind, JUMP_METATABLE);
155  // noreturn
156  }
157 
158  if (b != NULL && a != NULL)
159  jp = jump_getTarget( b, a );
160 
161  if (jp == NULL)
162  NLUA_ERROR(L, _("Jump is invalid"));
163 
164  return jp;
165 }
