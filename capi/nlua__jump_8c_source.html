133  b = system_getIndex( lj->destid );

233 static int jumpL_get( lua_State *L )
234 {
235  LuaJump lj;
236  StarSystem *a, *b;
237 
238  /* Defaults. */
239  a = NULL;
240  b = NULL;
241 
242  if (lua_gettop(L) > 1) {
243  if (lua_isstring(L, 1))
244  a = system_get( lua_tostring(L, 1));
245  else if (lua_issystem(L, 1))
246  a = system_getIndex( lua_tosystem(L, 1) );
247 
248  if (lua_isstring(L, 2))
249  b = system_get( lua_tostring(L, 2));
250  else if (lua_issystem(L, 2))
251  b = system_getIndex( lua_tosystem(L, 2) );
252 
253  if ((a == NULL) || (b == NULL)) {
254  NLUA_ERROR(L, _("No matching jump points found."));
255  return 0;
256  }
257 
258  if (jump_getTarget(b, a) != NULL) {
259  lj.srcid = a->id;
260  lj.destid = b->id;
261  lua_pushjump(L, lj);
262 
263  /* The inverse. If it doesn't exist, there are bigger problems. */
264  lj.srcid = b->id;
265  lj.destid = a->id;
266  lua_pushjump(L, lj);
267  return 2;
268  }
269  }
270  else
271  NLUA_INVALID_PARAMETER(L);
272 
273  return 0;
274 }
