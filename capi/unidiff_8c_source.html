640 static int diff_patch( xmlNodePtr parent )
641 {
642  int i, univ_update;
643  UniDiff_t *diff;
644  UniHunk_t *fail;
645  xmlNodePtr node;
646  char *target;
647  int nfailed;
648 
649  /* Prepare it. */
650  diff = diff_newDiff();
651  memset(diff, 0, sizeof(UniDiff_t));
652  xmlr_attr_strd(parent,"name",diff->name);
653 
654  /* Whether or not we need to update the universe. */
655  univ_update = 0;
656 
657  node = parent->xmlChildrenNode;
658  do {
659  xml_onlyNodes(node);
660  if (xml_isNode(node,"system")) {
661  univ_update = 1;
662  diff_patchSystem( diff, node );
663  }
664  else if (xml_isNode(node, "tech"))
665  diff_patchTech( diff, node );
666  else if (xml_isNode(node, "asset")) {
667  univ_update = 1;
668  diff_patchAsset( diff, node );
669  }
670  else if (xml_isNode(node, "faction")) {
671  univ_update = 1;
672  diff_patchFaction( diff, node );
673  }
674  else
675  WARN(_("Unidiff '%s' has unknown node '%s'."), diff->name, node->name);
676  } while (xml_nextNode(node));
677 
678  nfailed = array_size(diff->failed);
679  if (nfailed > 0) {
680  WARN(
681  n_( "Unidiff '%s' failed to apply %d hunk.", "Unidiff '%s' failed to apply %d hunks.", nfailed ),
682  diff->name, nfailed );
683  for (i=0; i<nfailed; i++) {
684  fail = &diff->failed[i];
685  target = fail->target.u.name;
686  switch (fail->type) {
687  case HUNK_TYPE_ASSET_ADD:

701  break;
702  case HUNK_TYPE_JUMP_REMOVE:
703  WARN(_(" [%s] jump remove: '%s'"), target, fail->u.name);
704  break;
705  case HUNK_TYPE_TECH_ADD:
706  WARN(_(" [%s] tech add: '%s'"), target,
707  fail->u.name );
708  break;
709  case HUNK_TYPE_TECH_REMOVE:
710  WARN(_(" [%s] tech remove: '%s'"), target,
711  fail->u.name );
712  break;
713  case HUNK_TYPE_ASSET_FACTION:
714  WARN(_(" [%s] asset faction: '%s'"), target,
715  fail->u.name );
716  break;
717  case HUNK_TYPE_ASSET_FACTION_REMOVE:

719  fail->u.name );
720  break;

783  /* Removing an asset. */
