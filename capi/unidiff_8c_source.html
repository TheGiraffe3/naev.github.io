243 static int diff_applyInternal( const char *name, int oneshot )
244 {
245  xmlNodePtr node;
246  xmlDocPtr doc;
247  char *filename;
248  int i;
249 
250  /* Check if already applied. */
251  if (diff_isApplied(name))
252  return 0;
253 
254  /* Reset change variable. */
255  if (oneshot)
256  diff_universe_changed = 0;
257 
258  filename = NULL;
259  for (i=0; i<array_size(diff_available); i++) {
260  if (strcmp(diff_available[i].name,name)==0) {
261  filename = diff_available[i].filename;
262  break;
263  }
264  }
265  if (filename == NULL) {
266  WARN(_("UniDiff '%s' not found in %s!"), name, UNIDIFF_DATA_PATH);
267  return -1;

271 
272  node = doc->xmlChildrenNode;
273  if (strcmp((char*)node->name,"unidiff")) {
274  ERR(_("Malformed unidiff file: missing root element 'unidiff'"));
275  return 0;
276  }
277 
278  /* Apply it. */
279  diff_patch( node );
280 
281  xmlFreeDoc(doc);
282 
283  /* Re-compute the economy. */
284  economy_execQueued();
285  economy_initialiseCommodityPrices();
286 
287  /* Update universe. */
288  if (oneshot)
289  diff_checkUpdateUniverse();

685  do {
686  xml_onlyNodes(node);
687  if (xml_isNode(node,"system")) {
