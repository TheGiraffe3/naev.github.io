205 static int shaderL_new( lua_State *L )
206 {
207  LuaShader_t shader;
208  const char *pixelcode, *vertexcode;
209  GLint i, ntex;
210  GLsizei length;
211  LuaUniform_t *u;
212  LuaTexture_t *t;
213 
214  /* Get arguments. */
215  pixelcode = luaL_checkstring(L,1);
216  vertexcode = luaL_checkstring(L,2);
217 
218  /* Initialize. */
219  memset( &shader, 0, sizeof(shader) );
220 
221  /* Do from string. */
222  shader.program = gl_program_vert_frag_string( vertexcode, strlen(vertexcode), pixelcode, strlen(pixelcode) );
223  if (shader.program == 0)
224  NLUA_ERROR(L,_("shader failed to compile!"));
225 
226  /* Set up defaults. */
227 #define ATTRIB(name) \
228  shader.name = glGetAttribLocation( shader.program, #name )
229 #define UNIFORM(name) \
230  shader.name = glGetUniformLocation( shader.program, #name )
231  UNIFORM( ViewSpaceFromLocal );
232  UNIFORM( ClipSpaceFromView );
233  UNIFORM( ClipSpaceFromLocal );
234  UNIFORM( ViewNormalFromLocal );
235  UNIFORM( MainTex );
236  UNIFORM( ConstantColor );
237  UNIFORM( love_ScreenSize );
238  ATTRIB( VertexPosition );
239  ATTRIB( VertexTexCoord );
240  ATTRIB( VertexColor );
241 
242  /* Do other uniforms. */
243  glGetProgramiv( shader.program, GL_ACTIVE_UNIFORMS, &shader.nuniforms );
244  shader.uniforms = calloc( shader.nuniforms, sizeof(LuaUniform_t) );
245  ntex = 0;
246  for (i=0; i<shader.nuniforms; i++) {
247  u = &shader.uniforms[i];
248  glGetActiveUniform( shader.program, (GLuint)i, SHADER_NAME_MAXLEN, &length, &u->size, &u->type, u->name );
249  u->id = glGetUniformLocation( shader.program, u->name );
250  u->tex = -1;
251 
252  /* Textures need special care. */
253  if ((u->type==GL_SAMPLER_2D) && (strcmp(u->name,"MainTex")!=0)) {
254  if (shader.tex == NULL)
255  shader.tex = array_create(LuaTexture_t);
256  t = &array_grow( &shader.tex );
257  ntex++;
258  t->active = GL_TEXTURE0+ntex;
259  t->texid = 0;
260  t->uniform = u->id;
261  t->value = ntex;
262  u->tex = ntex-1;
263  }
264  }
265  qsort( shader.uniforms, shader.nuniforms, sizeof(LuaUniform_t), shader_compareUniform );
266 
267  /* Check if there are textures. */
268 
269  gl_checkErr();
270 
271  lua_pushshader( L, shader );
272  return 1;
273 }

349 static int shaderL_sendHelper( lua_State *L, int ignore_missing )
350 {
351  LuaShader_t *ls;
352  LuaUniform_t *u;
353  const char *name;
354  int idx;
355  GLfloat values[4];
356  GLint ivalues[4];
357  glTexture *tex;
358 
359  ls = luaL_checkshader(L,1);
360  name = luaL_checkstring(L,2);
361 
362  u = shader_getUniform( ls, name );
363  if (u==NULL) {
364  if (ignore_missing)
365  return 0;
366  NLUA_ERROR(L,_("Shader does not have uniform '%s'!"), name);
367  }
368 
369  /* With OpenGL 4.1 or ARB_separate_shader_objects, there
370  * is no need to set the program first. */
371  glUseProgram( ls->program );
372  idx = 3;
373  switch (u->type) {
374  case GL_FLOAT:
375  shader_parseUniformArgsFloat( values, L, idx, 1 );
376  glUniform1f( u->id, values[0] );
377  break;
378  case GL_FLOAT_VEC2:
379  shader_parseUniformArgsFloat( values, L, idx, 2 );
380  glUniform2f( u->id, values[0], values[1] );
381  break;
382  case GL_FLOAT_VEC3:
383  shader_parseUniformArgsFloat( values, L, idx, 3 );
384  glUniform3f( u->id, values[0], values[1], values[2] );
385  break;
386  case GL_FLOAT_VEC4:
387  shader_parseUniformArgsFloat( values, L, idx, 4 );
388  glUniform4f( u->id, values[0], values[1], values[2], values[3] );
389  break;
390 
391  case GL_INT:
392  shader_parseUniformArgsInt( ivalues, L, idx, 1 );
393  glUniform1i( u->id, ivalues[0] );
394  break;
395  case GL_INT_VEC2:
396  shader_parseUniformArgsInt( ivalues, L, idx, 2 );
397  glUniform2i( u->id, ivalues[0], ivalues[1] );
398  break;
399  case GL_INT_VEC3:
400  shader_parseUniformArgsInt( ivalues, L, idx, 3 );
401  glUniform3i( u->id, ivalues[0], ivalues[1], ivalues[2] );
402  break;
403  case GL_INT_VEC4:
404  shader_parseUniformArgsInt( ivalues, L, idx, 4 );
405  glUniform4i( u->id, ivalues[0], ivalues[1], ivalues[2], ivalues[3] );
406  break;
407 
408  case GL_SAMPLER_2D:
409  tex = luaL_checktex(L,idx);
410  ls->tex[ u->tex ].texid = tex->texture;
411  break;
412 
413  default:
414  WARN(_("Unsupported shader uniform type '%d' for uniform '%s'. Ignoring."), u->type, u->name );
415  }
416  glUseProgram( 0 );
417 
418  gl_checkErr();
419 
420  return 0;
421 }
