200 
201  /* Calculate differential. */
202  dx = old_X;
203  dy = old_Y;
204 
205  /* Going to position. */
206  p = NULL;
207  if (camera_fly) {
208  if (camera_followpilot != 0) {
209  p = pilot_get( camera_followpilot );
210  if (p == NULL) {
211  camera_followpilot = 0;
212  camera_fly = 0;

214  else {
215  cam_updateFly( p->solid->pos.x, p->solid->pos.y, dt );
216  cam_updatePilotZoom( p, NULL, dt );

218  }

221  }
222  else {

231  }

239  dx = dt*(dx-camera_X);

253 static void cam_updateFly( double x, double y, double dt )
254 {
255  double k, dx,dy, max;
256 
257  max = camera_flyspeed*dt;
258  k = 25.*dt;
259  dx = (x - camera_X)*k;
260  dy = (y - camera_Y)*k;
261  if (pow2(dx)+pow2(dy) > pow2(max)) {
262  double a = atan2( dy, dx );
263  double r = max;
264  dx = r*cos(a);
265  dy = r*sin(a);
266  }
267  camera_X += dx;
268  camera_Y += dy;
269  background_moveStars( -dx, -dy );
270 
271  /* Stop within 100 pixels. */
272  if (FABS((pow2(camera_X)+pow2(camera_Y)) - (pow2(x)+pow2(y))) < 100*100) {
273  old_X = camera_X;
274  old_Y = camera_Y;
275  camera_fly = 0;
276  }
277 }

282 static void cam_updatePilot( Pilot *follow, double dt )
283 {
284  Pilot *target;
285  double dir, k;
286  double x,y, dx,dy, mx,my, targ_x,targ_y, bias_x,bias_y, vx,vy;
287 
288  /* Get target. */
289  if (!pilot_isFlag(follow, PILOT_HYPERSPACE) && (follow->target != follow->id))
290  target = pilot_getTarget( follow );
291  else
292  target = NULL;
293 
294  /* Real diagonal might be a bit too harsh since it can cut out the ship,
295  * we'll just use the largest of the two. */
296  /*diag2 = pow2(SCREEN_W) + pow2(SCREEN_H);*/
297  /*diag2 = pow2( MIN(SCREEN_W, SCREEN_H) );*/
298  const double diag2 = 100*100;
299  x = follow->solid->pos.x;
300  y = follow->solid->pos.y;
301 
302  /* Compensate player movement. */
303  mx = x - old_X;
304  my = y - old_Y;
305  camera_X += mx;
306  camera_Y += my;
307 
308  /* Set old position. */
309  old_X = x;
310  old_Y = y;
311 
312  /* No bias by default. */
313  bias_x = 0.;
314  bias_y = 0.;
315 
316  /* Bias towards target. */
317  if (target != NULL) {
318  bias_x += target->solid->pos.x - x;
319  bias_y += target->solid->pos.y - y;
320  }
321 
322  /* Bias towards velocity and facing. */
323  vx = follow->solid->vel.x*1.5;
324  vy = follow->solid->vel.y*1.5;
325  dir = angle_diff( atan2(vy,vx), follow->solid->dir);
326  dir = (M_PI - FABS(dir)) / M_PI; /* Normalize. */
327  vx *= dir;
328  vy *= dir;
329  bias_x += vx;
330  bias_y += vy;
331 
332  /* Limit bias. */
333  if (pow2(bias_x)+pow2(bias_y) > diag2/2.) {
334  double a = atan2( bias_y, bias_x );
335  double r = sqrt(diag2)/2.;
336  bias_x = r*cos(a);
337  bias_y = r*sin(a);
338  }
339 
340  /* Compose the target. */
341  targ_x = x + bias_x;
342  targ_y = y + bias_y;
343 
344  /* Head towards target. */
345  k = 0.5*dt/dt_mod;
346  dx = (targ_x-camera_X)*k;
347  dy = (targ_y-camera_Y)*k;
348  background_moveStars( -(mx+dx), -(my+dy) );
349 
350  /* Update camera. */
351  camera_X += dx;
352  camera_Y += dy;
353 
354  /* DEBUG. */
355 #if 0
356  glColor4d( 1., 1., 1., 1. );
357  glBegin(GL_LINES);
358  gl_gameToScreenCoords( &x, &y, x, y );
359  glVertex2d( x, y );
360  gl_gameToScreenCoords( &x, &y, camera_X, camera_Y );
361  glVertex2d( x, y );
362  glEnd(); /* GL_LINES */
363 #endif
364 
365  /* Update zoom. */
366  cam_updatePilotZoom( follow, target, dt );
367 }

390 static void cam_updatePilotZoom( const Pilot *follow, const Pilot *target, double dt )
391 {
392  double d, x,y, z,tz, dx, dy;
393  double zfar, znear;
394  double c;
395 
396  /* Must have auto zoom enabled. */
397  if (conf.zoom_manual || zoom_override)
398  return;
399 
400  /* Minimum depends on velocity normally.
401  *
402  * w*h = A, cte area constant
403  * w/h = K, cte proportion constant
404  * d^2 = A, cte geometric longitud
405  *
406  * A_v = A*(1+v/d) area of view is based on speed
407  * A_v / A = (1 + v/d)
408  *
409  * z = A / A_v = 1. / (1 + v/d)
410  */
411  d = sqrt(SCREEN_W*SCREEN_H);
412  znear = MIN( conf.zoom_near, 1. / (0.8 + VMOD(follow->solid->vel)/d) );
413 
414  /* Maximum is limited by nebulae. */
415  if (cur_system->nebu_density > 0.) {
416  c = MIN( SCREEN_W, SCREEN_H ) / 2;
417  zfar = CLAMP( conf.zoom_far, conf.zoom_near, c / nebu_getSightRadius() );
418  }
419  else
420  zfar = conf.zoom_far;
421  znear = MAX( znear, zfar );
422 
423  /*
424  * Set Zoom to pilot target.
425  */
426  z = cam_getZoom();
427  if (pilot_isFlag( follow, PILOT_STEALTH )) {
428  tz = zfar;
429  }
430  else {
431  if (target != NULL) {
432  /* Get current relative target position. */
433  gui_getOffset( &x, &y );
434  x += target->solid->pos.x - follow->solid->pos.x;
435  y += target->solid->pos.y - follow->solid->pos.y;
436 
437  /* Get distance ratio. */
438  dx = (SCREEN_W/2.) / (FABS(x) + 2.*target->ship->gfx_space->sw);
439  dy = (SCREEN_H/2.) / (FABS(y) + 2.*target->ship->gfx_space->sh);
440 
441  /* Get zoom. */
442  tz = MIN( dx, dy );
443  }
444  else
445  tz = znear; /* Aim at in. */
446  }
447 
448  /* Gradually zoom in/out. */
449  d = CLAMP(-conf.zoom_speed, conf.zoom_speed, tz - z);
450  d *= dt / dt_mod; /* Remove dt dependence. */
451  if (d < 0.)
452  d *= 2.;
453  camera_Z = CLAMP( zfar, znear, z + d);
454 }
