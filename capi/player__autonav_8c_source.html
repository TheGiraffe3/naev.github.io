308 static void player_autonav (void)
309 {
310  JumpPoint *jp;
311  Pilot *p;
312  int ret, map_npath;
313  double d, t, tint;
314  double vel;
315 
316  (void)map_getDestination( &map_npath );
317 
318  switch (player.autonav) {
319  case AUTONAV_JUMP_APPROACH:
320  /* Target jump. */
321  jp = &cur_system->jumps[ player.p->nav_hyperspace ];
322  ret = player_autonavApproach( &jp->pos, &d, 0 );
323  if (ret)
324  player.autonav = AUTONAV_JUMP_BRAKE;
325  else if (!tc_rampdown && (map_npath<=1)) {
326  vel = MIN( 1.5*player.p->speed, VMOD(player.p->solid->vel) );
327  t = d / vel * (1.2 - .1 * tc_base);
328  /* tint is the integral of the time in per time units.
329  *
330  * tc_mod
331  * ^
332  * |
333  * |\
334  * | \
335  * | \___
336  * |
337  * +------> time
338  * 0 3
339  *
340  * We decompose integral in a rectangle (3*1) and a triangle (3*(tc_mod-1.))/2.
341  * This is the "elapsed time" when linearly decreasing the tc_mod. Which we can
342  * use to calculate the actual "game time" that'll pass when decreasing the
343  * tc_mod to 1 during 3 seconds. This can be used then to compare when we want to
344  * start decrementing.
345  */
346  tint = 3. + 0.5*(3.*(tc_mod-tc_base));
347  if (t < tint) {
348  tc_rampdown = 1;
349  tc_down = (tc_mod-tc_base) / 3.;
350  }
351  }
352  break;
353 
354  case AUTONAV_JUMP_BRAKE:
355  /* Target jump. */
356  jp = &cur_system->jumps[ player.p->nav_hyperspace ];
357  if (player.p->stats.misc_instant_jump) {
358  ret = pilot_interceptPos( player.p, jp->pos.x, jp->pos.y );
359  if (!ret && space_canHyperspace(player.p))
360  ret = 1;
361  player_acc = player.p->solid->thrust / player.p->thrust;
362  }
363  else
364  ret = player_autonavBrake();
365 
366  /* Try to jump or see if braked. */
367  if (ret) {
368  if (space_canHyperspace(player.p))
369  player_jump();
370  player.autonav = AUTONAV_JUMP_APPROACH;
371  }
372 
373  /* See if should ramp down. */
374  if (!tc_rampdown && (map_npath<=1)) {
375  tc_rampdown = 1;
376  tc_down = (tc_mod-tc_base) / 3.;
377  }
378  break;
379 
380  case AUTONAV_POS_APPROACH:
381  ret = player_autonavApproach( &player.autonav_pos, &d, 1 );
382  if (ret) {
383  player_message( _("#oAutonav: arrived at position.") );
384  player_autonavEnd();
385  }
386  else if (!tc_rampdown)
387  player_autonavRampdown(d);
388  break;
389 
390  case AUTONAV_PNT_APPROACH:
391  ret = player_autonavApproach( &player.autonav_pos, &d, 1 );
392  if (ret) {
393  player_message( _("#oAutonav: arrived at #%c%s#0."),
394  player.autonavcol,
395  player.autonavmsg );
396  player_autonavEnd();
397  }
398  else if (!tc_rampdown)
399  player_autonavRampdown(d);
400  break;
401 
402  case AUTONAV_PNT_LAND_APPROACH:
403  ret = player_autonavApproach( &player.autonav_pos, &d, 1 );
404  if (ret)
405  player.autonav = AUTONAV_PNT_LAND_BRAKE;
406  else if (!tc_rampdown)
407  player_autonavRampdown(d);
408  break;
409 
410  case AUTONAV_PNT_LAND_BRAKE:
411  ret = player_autonavBrake();
412 
413  /* Try to land. */
414  if (ret) {
415  if (player_land(0) == PLAYER_LAND_DENIED)
416  player_autonavAbort(NULL);
417  else
418  player.autonav = AUTONAV_PNT_APPROACH;
419  }
420 
421  /* See if should ramp down. */
422  if (!tc_rampdown) {
423  tc_rampdown = 1;
424  tc_down = (tc_mod-tc_base) / 3.;
425  }
426  break;
427 
428  case AUTONAV_PLT_FOLLOW:
429  p = pilot_getTarget( player.p );
430  if (p == NULL)
431  p = pilot_get( PLAYER_ID );
432  if ((p->id == PLAYER_ID) || (!pilot_inRangePilot( player.p, p, NULL ))) {
433  /* TODO : handle the different reasons: pilot is too far, jumped, landed or died. */
434  player_message( _("#oAutonav: following target %s has been lost."),
435  player.autonavmsg );
436  player_accel( 0. );
437  player_autonavEnd();
438  }
439  else {
440  ret = (pilot_isDisabled(p) || pilot_isFlag(p,PILOT_BOARDABLE));
441  player_autonavFollow( &p->solid->pos, &p->solid->vel, !ret, &d );
442  if (ret && (!tc_rampdown))
443  player_autonavRampdown(d);
444  }
445  break;
446  }
447 }

458 static int player_autonavApproach( const Vector2d *pos, double *dist2, int count_target )
459 {
460  double d, t, vel, dist;
461 
462  /* Only accelerate if facing move dir. */
463  d = pilot_face( player.p, vect_angle( &player.p->solid->pos, pos ) );
464  if (FABS(d) < MIN_DIR_ERR) {
465  if (player_acc < 1.)
466  player_accel( 1. );
467  }
468  else if (player_acc > 0.)
469  player_accelOver();
470 
471  /* Get current time to reach target. */
472  t = MIN( 1.5*player.p->speed, VMOD(player.p->solid->vel) ) /
473  (player.p->thrust / player.p->solid->mass);
474 
475  /* Get velocity. */
476  vel = MIN( player.p->speed, VMOD(player.p->solid->vel) );
477 
478  /* Get distance. */
479  dist = vel*(t+1.1*M_PI/player.p->turn) -
480  0.5*(player.p->thrust/player.p->solid->mass)*t*t;
481 
482  /* Output distance^2 */
483  d = vect_dist( pos, &player.p->solid->pos );
484  dist = d - dist;
485  if (count_target)
486  *dist2 = dist;
487  else
488  *dist2 = d;
489 
490  /* See if should start braking. */
491  if (dist < 0.) {
492  player_accelOver();
493  return 1;
494  }
495  return 0;
496 }

507 static void player_autonavFollow( const Vector2d *pos, const Vector2d *vel, const int follow, double *dist2 )
508 {
509  double Kp, Kd, angle, radius, d;
510  Vector2d dir, point;
511 
512  /* Define the control coefficients. If needed, they could be adapted.
513  Maybe radius could be adjustable by the player. */
514  Kp = 10;
515  Kd = 20;
516  radius = 100;
517 
518  /* Find a point behind the target at a distance of radius unless stationary, or not following. */
519  if ( !follow || ( vel->x == 0 && vel->y == 0 ) )
520  radius = 0;
521  angle = M_PI + vel->angle;
522  vect_cset( &point, pos->x + radius * cos(angle),
523  pos->y + radius * sin(angle) );
524 
525  vect_cset( &dir, (point.x - player.p->solid->pos.x) * Kp +
526  (vel->x - player.p->solid->vel.x) *Kd,
527  (point.y - player.p->solid->pos.y) * Kp +
528  (vel->y - player.p->solid->vel.y) *Kd );
529 
530  d = pilot_face( player.p, VANGLE(dir) );
531 
532  if ((FABS(d) < MIN_DIR_ERR) && (VMOD(dir) > 300))
533  player_accel( 1. );
534  else
535  player_accel( 0. );
536 
537  /* If aiming exactly at the point, should say when approaching. */
538  if (!follow)
539  *dist2 = vect_dist( pos, &player.p->solid->pos );
540 }

615  will_reset = 1;
616  player.autonav_timer = MAX( player.autonav_timer, 2. );
617  }
618  }

621  last_armour = armour;
622

624  player_autonavResetSpeed();
625  return 1;
626  }

668 void player_updateAutonav( double dt )
669 {
670  const double dis_dead = 5.0;
671  const double dis_mod = 0.5;
672  const double dis_max = 4.0;
673  const double dis_ramp = 6.0;
674 
675  if (paused || (player.p==NULL) || pilot_isFlag(player.p, PILOT_DEAD))
676  return;
677 
678  /* We handle disabling here. */
679  if (pilot_isFlag(player.p, PILOT_DISABLED)) {
680  /* It is somewhat like:
681  * /------------\ 4x
682  * / \
683  * -----/ \----- 1x
684  *
685  * <---><-><----------><-><--->
686  * 5 6 X 6 5 Real time
687  * 5 15 X 15 5 Game time
688  *
689  * For triangles we have to add the rectangle and triangle areas.
690  */
691  /* 5 second deadtime. */
692  if (player.p->dtimer_accum < dis_dead)
693  tc_mod = tc_base;
694  else {
695  /* Ramp down. */
696  if (player.p->dtimer - player.p->dtimer_accum < dis_dead + (dis_max-tc_base)*dis_ramp/2 + tc_base*dis_ramp)
697  tc_mod = MAX( tc_base, tc_mod - dis_mod*dt );
698  /* Normal. */
699  else
700  tc_mod = MIN( dis_max, tc_mod + dis_mod*dt );
701  }
702  pause_setSpeed( tc_mod );
703  sound_setSpeed( tc_mod / player_dt_default() );
704  return;
705  }
706 
707  /* Must be autonaving. */
708  if (!player_isFlag(PLAYER_AUTONAV))
709  return;
710 
711  /* Ramping down. */
712  if (tc_rampdown) {
713  if (tc_mod != tc_base) {
714  tc_mod = MAX( tc_base, tc_mod-tc_down*dt );
715  pause_setSpeed( tc_mod );
716  sound_setSpeed( tc_mod / player_dt_default() );
717  }
718  return;
719  }
720 
721  /* We'll update the time compression here. */
722  if (tc_mod == player.tc_max)
723  return;
724  else
725  tc_mod += 0.2 * dt * (player.tc_max-tc_base);
726  /* Avoid going over. */
727  if (tc_mod > player.tc_max)
728  tc_mod = player.tc_max;
729  pause_setSpeed( tc_mod );
730  sound_setSpeed( tc_mod / player_dt_default() );
731 }
