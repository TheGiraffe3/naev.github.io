974 double pilot_aimAngle( Pilot *p, Pilot *target )
975 {
976  double x,y;
977  double t;
978  Vector2d tv, approach_vector, relative_location, orthoradial_vector;
979  double dist;
980  double speed;
981  double radial_speed;
982  double orthoradial_speed;
983 
984  /* Get the distance */
985  dist = vect_dist( &p->solid->pos, &target->solid->pos );
986 
987  /* Check if should recalculate weapon speed with secondary weapon. */
988  speed = pilot_weapSetSpeed( p, p->active_set, -1 );
989 
990  /* determine the radial, or approach speed */
991  /*
992  *approach_vector (denote Va) is the relative velocites of the pilot and target
993  *relative_location (denote Vr) is the vector that points from the target to the pilot
994  *
995  *Va dot Vr is the rate of approach between the target and the pilot.
996  *If this is greater than 0, the target is approaching the pilot, if less than 0, the target is fleeing.
997  *
998  *Va dot Vr + ShotSpeed is the net closing velocity for the shot, and is used to compute the time of flight for the shot.
999  */
1000  vect_cset(&approach_vector, VX(p->solid->vel) - VX(target->solid->vel), VY(p->solid->vel) - VY(target->solid->vel) );
1001  vect_cset(&relative_location, VX(target->solid->pos) - VX(p->solid->pos), VY(target->solid->pos) - VY(p->solid->pos) );
1002  vect_cset(&orthoradial_vector, VY(p->solid->pos) - VY(target->solid->pos), VX(target->solid->pos) - VX(p->solid->pos) );
1003 
1004  radial_speed = vect_dot(&approach_vector, &relative_location);
1005  radial_speed = radial_speed / VMOD(relative_location);
1006 
1007  orthoradial_speed = vect_dot(&approach_vector, &orthoradial_vector);
1008  orthoradial_speed = orthoradial_speed / VMOD(relative_location);
1009 
1010  /* Time for shots to reach that distance */
1011  /* t is the real positive solution of a 2nd order equation*/
1012  /* if the target is not hittable (i.e., fleeing faster than our shots can fly, determinant <= 0), just face the target */
1013  if ( ((speed*speed - VMOD(approach_vector)*VMOD(approach_vector)) != 0) && (speed*speed - orthoradial_speed*orthoradial_speed) > 0)
1014  t = dist * (sqrt( speed*speed - orthoradial_speed*orthoradial_speed ) - radial_speed) /
1015  (speed*speed - VMOD(approach_vector)*VMOD(approach_vector));
1016  else
1017  t = 0;
1018 
1019  /* if t < 0, try the other solution*/
1020  if (t < 0)
1021  t = - dist * (sqrt( speed*speed - orthoradial_speed*orthoradial_speed ) + radial_speed) /
1022  (speed*speed - VMOD(approach_vector)*VMOD(approach_vector));
1023 
1024  /* if t still < 0, no solution*/
1025  if (t < 0)
1026  t = 0;
1027 
1028  /* Position is calculated on where it should be */
1029  x = target->solid->pos.x + target->solid->vel.x*t
1030  - (p->solid->pos.x + p->solid->vel.x*t);
1031  y = target->solid->pos.y + target->solid->vel.y*t
1032  - (p->solid->pos.y + p->solid->vel.y*t);
1033  vect_cset( &tv, x, y );
1034 
1035  return VANGLE(tv);
1036 }

1344 double pilot_hit( Pilot* p, const Solid* w, const unsigned int shooter,
1345  const Damage *dmg, int reset )
1346 {
1347  int mod;
1348  double damage_shield, damage_armour, disable, knockback, dam_mod, ddmg, absorb, dmod, start;
1349  Pilot *pshooter;
1350 
1351  /* Invincible means no damage. */
1352  if (pilot_isFlag( p, PILOT_INVINCIBLE ) ||
1353  pilot_isFlag( p, PILOT_INVISIBLE ))
1354  return 0.;
1355 
1356  /* Defaults. */
1357  pshooter = NULL;
1358  dam_mod = 0.;
1359  ddmg = 0.;
1360 
1361  /* Calculate the damage. */
1362  absorb = 1. - CLAMP( 0., 1., p->dmg_absorb - dmg->penetration );
1363  disable = dmg->disable;
1364  dtype_calcDamage( &damage_shield, &damage_armour, absorb, &knockback, dmg, &p->stats );
1365 
1366  /*
1367  * Delay undisable if necessary. Amount varies with damage, as e.g. a
1368  * single Laser Cannon shot should not reset a Peacemaker's timer.
1369  */
1370  if (!pilot_isFlag(p, PILOT_DEAD) && (p->dtimer_accum > 0.))
1371  p->dtimer_accum -= MIN( pow(disable, .8), p->dtimer_accum );
1372 
1373  /* Ships that can not be disabled take raw armour damage instead of getting disabled. */
1374  if (pilot_isFlag( p, PILOT_NODISABLE )) {
1375  damage_armour += disable * absorb;
1376  disable = 0.;
1377  }
1378  else
1379  disable *= absorb;
1380 
1381  /*
1382  * Shields take entire blow.
1383  */
1384  if (p->shield - damage_shield > 0.) {
1385  start = p->shield;
1386  ddmg = damage_shield;
1387  p->shield -= damage_shield;
1388  dam_mod = damage_shield/p->shield_max;
1389 
1390  /*
1391  * Disabling damage leaks accordingly:
1392  * 50% + (50% - mean shield % / 2)
1393  *
1394  * 50% leaks at 100% shield, scales to 100% by 0% shield.
1395  *
1396  * The damage is adjusted based on the mean of the starting and ending
1397  * shield percentages. Using the starting percentage biases towards
1398  * low-damage, high-ROF weapons, while using the ending percentage
1399  * biases towards high-damage, low-ROF weapons.
1400  */
1401  p->stress += disable * (.5 + (.5 - ((start+p->shield) / p->shield_max) / 4.));
1402  }
1403  /*
1404  * Shields take part of the blow.
1405  */
1406  else if (p->shield > 0.) {
1407  start = p->shield;
1408  dmod = (1. - p->shield/damage_shield);
1409  ddmg = p->shield + dmod * damage_armour;
1410  p->shield = 0.;
1411 
1412  /* Leak some disabling damage through the remaining bit of shields. */
1413  p->stress += disable * (1. - dmod) * (.5 + (.5 - (start / p->shield_max / 4.)));
1414 
1415  /* Reduce stress as armour is eaten away. */
1416  p->stress *= (p->armour - dmod * damage_armour) / p->armour;
1417  p->armour -= dmod * damage_armour;
1418  p->stress += dmod * disable;
1419  dam_mod = (damage_shield + damage_armour) /
1420  ((p->shield_max + p->armour_max) / 2.);
1421 
1422  /* Increment shield timer or time before shield regeneration kicks in. */
1423  if (reset) {
1424  p->stimer = 3.;
1425  p->sbonus = 3.;
1426  }
1427  }
1428  /*
1429  * Armour takes the entire blow.
1430  */
1431  else if (p->armour > 0.) {
1432  ddmg = damage_armour;
1433  /* Reduce stress as armour is eaten away. */
1434  p->stress *= (p->armour - damage_armour) / p->armour;
1435  p->armour -= damage_armour;
1436  p->stress += disable;
1437 
1438  /* Increment shield timer or time before shield regeneration kicks in. */
1439  if (reset) {
1440  p->stimer = 3.;
1441  p->sbonus = 3.;
1442  }
1443  }
1444 
1445  /* Ensure stress never exceeds remaining armour. */
1446  if (p->stress > p->armour)
1447  p->stress = p->armour;
1448 
1449  /* Do not let pilot die. */
1450  if ((p->armour <= 0.) && pilot_isFlag( p, PILOT_NODEATH )) {
1451  p->armour = 1.;
1452  p->stress = 1.;
1453  }
1454 
1455  /* Disabled always run before dead to ensure combat rating boost. */
1456  pilot_updateDisable(p, shooter);
1457 
1458  /* Officially dead. */
1459  if (p->armour <= 0.) {
1460  p->armour = 0.;
1461  dam_mod = 0.;
1462 
1463  if (!pilot_isFlag(p, PILOT_DEAD)) {
1464  pilot_dead( p, shooter );
1465 
1466  /* adjust the combat rating based on pilot mass and ditto faction */
1467  if (pshooter == NULL)
1468  pshooter = pilot_get(shooter);
1469  if ((pshooter != NULL) && (pshooter->faction == FACTION_PLAYER)) {
1470 
1471  /* About 6 for a llama, 52 for hawking. */
1472  mod = 2 * (pow(p->base_mass, 0.4) - 1.);
1473 
1474  /* Modify faction for him and friends. */
1475  faction_modPlayer( p->faction, -mod, "kill" );
1476  }
1477  }
1478  }
1479 
1480  /* Some minor effects and stuff. */
1481  else if (p->shield <= 0.) {
1482  dam_mod = damage_armour/p->armour_max;
1483 
1484  if (p->id == PLAYER_ID) /* a bit of shaking */
1485  spfx_shake( SHAKE_MAX*dam_mod );

1487 
1488 
1489  if (w != NULL)
1490  /* knock back effect is dependent on both damage and mass of the weapon
1491  * should probably get turned into a partial conservative collision */
1492  vect_cadd( &p->solid->vel,
1493  knockback * (w->vel.x * (dam_mod/9. + w->mass/p->solid->mass/6.)),
