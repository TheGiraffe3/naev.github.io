1055 double pilot_aimAngle( Pilot *p, const Pilot *target )
1056 {
1057  double x,y;
1058  double t;
1059  Vector2d tv, approach_vector, relative_location, orthoradial_vector;
1060  double dist;
1061  double speed;
1062  double radial_speed;
1063  double orthoradial_speed;
1064 
1065  /* Get the distance */
1066  dist = vect_dist( &p->solid->pos, &target->solid->pos );
1067 
1068  /* Check if should recalculate weapon speed with secondary weapon. */
1069  speed = pilot_weapSetSpeed( p, p->active_set, -1 );
1070 
1071  /* determine the radial, or approach speed */
1072  /*
1073  *approach_vector (denote Va) is the relative velocites of the pilot and target
1074  *relative_location (denote Vr) is the vector that points from the target to the pilot
1075  *
1076  *Va dot Vr is the rate of approach between the target and the pilot.
1077  *If this is greater than 0, the target is approaching the pilot, if less than 0, the target is fleeing.
1078  *
1079  *Va dot Vr + ShotSpeed is the net closing velocity for the shot, and is used to compute the time of flight for the shot.
1080  */
1081  vect_cset(&approach_vector, VX(p->solid->vel) - VX(target->solid->vel), VY(p->solid->vel) - VY(target->solid->vel) );
1082  vect_cset(&relative_location, VX(target->solid->pos) - VX(p->solid->pos), VY(target->solid->pos) - VY(p->solid->pos) );
1083  vect_cset(&orthoradial_vector, VY(p->solid->pos) - VY(target->solid->pos), VX(target->solid->pos) - VX(p->solid->pos) );
1084 
1085  radial_speed = vect_dot(&approach_vector, &relative_location);
1086  radial_speed = radial_speed / VMOD(relative_location);
1087 
1088  orthoradial_speed = vect_dot(&approach_vector, &orthoradial_vector);
1089  orthoradial_speed = orthoradial_speed / VMOD(relative_location);
1090 
1091  /* Time for shots to reach that distance */
1092  /* t is the real positive solution of a 2nd order equation*/
1093  /* if the target is not hittable (i.e., fleeing faster than our shots can fly, determinant <= 0), just face the target */
1094  if ( ((speed*speed - VMOD(approach_vector)*VMOD(approach_vector)) != 0) && (speed*speed - orthoradial_speed*orthoradial_speed) > 0)
1095  t = dist * (sqrt( speed*speed - orthoradial_speed*orthoradial_speed ) - radial_speed) /
1096  (speed*speed - VMOD(approach_vector)*VMOD(approach_vector));
1097  else
1098  t = 0;
1099 
1100  /* if t < 0, try the other solution*/
1101  if (t < 0)
1102  t = - dist * (sqrt( speed*speed - orthoradial_speed*orthoradial_speed ) + radial_speed) /
1103  (speed*speed - VMOD(approach_vector)*VMOD(approach_vector));
1104 
1105  /* if t still < 0, no solution*/
1106  if (t < 0)
1107  t = 0;
1108 
1109  /* Position is calculated on where it should be */
1110  x = target->solid->pos.x + target->solid->vel.x*t
1111  - (p->solid->pos.x + p->solid->vel.x*t);
1112  y = target->solid->pos.y + target->solid->vel.y*t
1113  - (p->solid->pos.y + p->solid->vel.y*t);
1114  vect_cset( &tv, x, y );
1115 
1116  return VANGLE(tv);
1117 }

1433 double pilot_hit( Pilot* p, const Solid* w, const unsigned int shooter,
1434  const Damage *dmg, int reset )
1435 {

1437  double damage_shield, damage_armour, disable, knockback, dam_mod, ddmg, absorb, dmod, start;
1438  double tdshield, tdarmour;
1439  Pilot *pshooter;
1440 
1441  /* Invincible means no damage. */
1442  if (pilot_isFlag( p, PILOT_INVINCIBLE ) ||
1443  pilot_isFlag( p, PILOT_HIDE ))
1444  return 0.;
1445 
1446  /* Defaults. */
1447  pshooter = NULL;
1448  dam_mod = 0.;
1449  ddmg = 0.;
1450 
1451  /* Calculate the damage. */
1452  absorb = 1. - CLAMP( 0., 1., p->dmg_absorb - dmg->penetration );
1453  disable = dmg->disable;
1454  dtype_calcDamage( &damage_shield, &damage_armour, absorb, &knockback, dmg, &p->stats );
1455 
1456  /*
1457  * Delay undisable if necessary. Amount varies with damage, as e.g. a
1458  * single Laser Cannon shot should not reset a Peacemaker's timer.
1459  */
1460  if (!pilot_isFlag(p, PILOT_DEAD) && (p->dtimer_accum > 0.))
1461  p->dtimer_accum -= MIN( pow(disable, .8), p->dtimer_accum );
1462 
1463  /* Ships that can not be disabled take raw armour damage instead of getting disabled. */
1464  if (pilot_isFlag( p, PILOT_NODISABLE )) {
1465  damage_armour += disable * absorb;
1466  disable = 0.;
1467  }
1468  else
1469  disable *= absorb;
1470  tdshield = 0.;
1471  tdarmour = 0.;
1472 
1473 
1474  /*
1475  * Shields take entire blow.
1476  */
1477  if (p->shield - damage_shield > 0.) {
1478  start = p->shield;
1479  ddmg = damage_shield;
1480  p->shield -= damage_shield;
1481  dam_mod = damage_shield/p->shield_max;
1482 
1483  /*
1484  * Disabling damage leaks accordingly:
1485  * 50% + (50% - mean shield % / 2)
1486  *
1487  * 50% leaks at 100% shield, scales to 100% by 0% shield.
1488  *
1489  * The damage is adjusted based on the mean of the starting and ending
1490  * shield percentages. Using the starting percentage biases towards
1491  * low-damage, high-ROF weapons, while using the ending percentage
1492  * biases towards high-damage, low-ROF weapons.
1493  */

1498  }
1499  /*
1500  * Shields take part of the blow.
1501  */

1504  dmod = (1. - p->shield/damage_shield);
1505  ddmg = p->shield + dmod * damage_armour;
1506  tdshield = p->shield;
1507  p->shield = 0.;
1508

1510  p->stress += disable * (1. - dmod) * (.5 + (.5 - (start / p->shield_max / 4.)));
1511 
1512  /* Reduce stress as armour is eaten away. */
1513  p->stress *= (p->armour - dmod * damage_armour) / p->armour;
1514  tdarmour = dmod * damage_armour;
1515  p->armour -= tdarmour;
1516  p->stress += dmod * disable;

1518  ((p->shield_max + p->armour_max) / 2.);

1535 
1536  /* Increment shield timer or time before shield regeneration kicks in. */
1537  if (reset) {
