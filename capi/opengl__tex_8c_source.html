76 static int SDL_IsTrans( SDL_Surface* s, int x, int y )
77 {
78  int bpp;
79  Uint8 *p;
80  Uint32 pixelcolour;
81 
82  bpp = s->format->BytesPerPixel;
83  /* here p is the address to the pixel we want to retrieve */
84  p = (Uint8 *)s->pixels + y*s->pitch + x*bpp;
85 
86  pixelcolour = 0;
87  switch(bpp) {
88  case 1:
89  pixelcolour = *p;
90  break;
91 
92  case 2:
93  memcpy(&pixelcolour, p, sizeof(Uint16));
94  break;
95 
96  case 3:
97 #if SDL_BYTEORDER == SDL_BIG_ENDIAN
98  pixelcolour = p[0] << 16 | p[1] << 8 | p[2];
99 #else /* SDL_BYTEORDER == SDL_BIG_ENDIAN */
100  pixelcolour = p[0] | p[1] << 8 | p[2] << 16;
101 #endif /* SDL_BYTEORDER == SDL_BIG_ENDIAN */
102  break;
103 
104  case 4:
105  memcpy(&pixelcolour, p, sizeof(Uint32));
106  break;
107  }
108 
109  /* test whether pixels colour == colour of transparent pixels for that surface */
110  return ((pixelcolour & s->format->Amask) < (Uint32)(0.1*(double)s->format->Amask));
111 }

125 static uint8_t* SDL_MapTrans( SDL_Surface* s, int w, int h )
126 {
127  int i,j;
128  size_t size;
129  uint8_t *t;
130 
131  /* Get limit.s */
132  if (w < 0)
133  w = s->w;
134  if (h < 0)
135  h = s->h;
136 
137  /* alloc memory for just enough bits to hold all the data we need */
138  size = gl_transSize(w, h);
139  t = malloc(size);
140  if (t==NULL) {
141  WARN(_("Out of Memory"));
142  return NULL;
143  }
144  memset(t, 0, size); /* important, must be set to zero */
145 
146  /* Check each pixel individually. */
147  for (i=0; i<h; i++)
148  for (j=0; j<w; j++) /* sets each bit to be 1 if not transparent or 0 if is */
149  t[(i*w+j)/8] |= (SDL_IsTrans(s,j,i)) ? 0 : (1<<((i*w+j)%8));
150 
151  return t;
152 }

172 static GLuint gl_texParameters( unsigned int flags )
173 {
174  GLuint texture;
175 
176  /* opengl texture binding */
177  glGenTextures( 1, &texture ); /* Creates the texture */
178  glBindTexture( GL_TEXTURE_2D, texture ); /* Loads the texture */
179 
180  /* Filtering, LINEAR is better for scaling, nearest looks nicer, LINEAR
181  * also seems to create a bit of artifacts around the edges */
182  if ((gl_screen.scale != 1.) || (flags & OPENGL_TEX_MIPMAPS)) {
183  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
184  if (flags & OPENGL_TEX_MIPMAPS)
185  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST_MIPMAP_LINEAR);

191  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);

195  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
196  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
197 
198  /* Check errors. */
199  gl_checkErr();

390  free(data);

396  free(md5val);
397 
398  asprintf( &cachefile, "%scollisions/%s",
399  nfile_cachePath(), digest );
400

402  if (nfile_fileExists(cachefile)) {
403  trans = (uint8_t*)nfile_readFile( &filesize, cachefile );
404 
405  /* Consider cached data invalid if the length doesn't match. */
406  if (trans != NULL && cachesize != (unsigned int)filesize) {
407  free(trans);
408  trans = NULL;
409  }
410  /* Cached data matches, no need to overwrite. */
411  else {
412  free(cachefile);
413  cachefile = NULL;
414  }
415  }
416  }
417  else {
418  /* We could hash raw pixel data here, but that's slower than just
419  * generating the map from scratch.
420  */
421  WARN(_("Texture '%s' has no RWops"), name);
422  }
423 
424  if (trans == NULL) {
425  SDL_LockSurface(surface);
426  trans = SDL_MapTrans( surface, w, h );
427  SDL_UnlockSurface(surface);
428 
429  if (cachefile != NULL) {
430  /* Cache newly-generated transparency map. */
431  char dirpath[PATH_MAX];
432  snprintf( dirpath, sizeof(dirpath), "%s/%s", nfile_cachePath(), "collisions/" );
433  nfile_dirMakeExist( dirpath );
434  nfile_writeFile( (char*)trans, cachesize, cachefile );
435  free(cachefile);
436  }
437  }
438

440  texture->trans = trans;
441  return texture;
442 }

778 {
779  glTexList *cur, *last;
780 
781  if (texture == NULL)
782  return;
783 
784  /* see if we can find it in stack */
785  last = NULL;
786  for (cur=texture_list; cur!=NULL; cur=cur->next) {
787  if (cur->tex == texture) { /* found it */
788  cur->used--;
789  if (cur->used <= 0) { /* not used anymore */
790  /* free the texture */

792  free(texture->trans);
793  free(texture->name);
794  free(texture);
795

797  if (last == NULL) { /* case there's no texture before it */
798  if (cur->next != NULL)
799  texture_list = cur->next;
800  else /* case it's the last texture */
801  texture_list = NULL;
802  }
803  else
804  last->next = cur->next;
805  free(cur);
806  }
807  return; /* we already found it so we can exit */
808  }
809  last = cur;
810  }
811

813  if (texture->name != NULL) /* Surfaces will have NULL names */
814  WARN(_("Attempting to free texture '%s' not found in stack!"), texture->name);
815 
816  /* Free anyways */
817  glDeleteTextures( 1, &texture->texture );
818  free(texture->trans);
819  free(texture->name);
