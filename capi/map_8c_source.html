331 static void map_update_commod_av_price()
332 {
333  Commodity *c;
334  int i,j,k;
335  StarSystem *sys;
336  Planet *p;
337  if (cur_commod == -1 || map_selected == -1) {
338  commod_av_gal_price = 0;
339  return;
340  }
341  c=commod_known[cur_commod];
342  if ( cur_commod_mode !=0 ) {
343  double totPrice = 0;
344  int totPriceCnt = 0;
345  for (i=0; i<array_size(systems_stack); i++) {
346  sys = system_getIndex( i );
347 
348  /* if system is not known, reachable, or marked. and we are not in the editor */
349  if ((!sys_isKnown(sys) && !sys_isFlag(sys, SYSTEM_MARKED | SYSTEM_CMARKED)
350  && !space_sysReachable(sys)))
351  continue;
352  if ((sys_isKnown(sys)) && (system_hasPlanet(sys))) {
353  double sumPrice=0;
354  int sumCnt=0;
355  double thisPrice;
356  for ( j=0 ; j<array_size(sys->planets); j++) {
357  p=sys->planets[j];
358  for ( k=0; k<array_size(p->commodities); k++) {
359  if ( p->commodities[k] == c ) {
360  if ( p->commodityPrice[k].cnt > 0 ) {/*commodity is known about*/
361  thisPrice = p->commodityPrice[k].sum / p->commodityPrice[k].cnt;
362  sumPrice+=thisPrice;
363  sumCnt+=1;
364  break;
365  }
366  }
367  }
368  }
369  if ( sumCnt>0 ) {
370  totPrice += sumPrice / sumCnt;
371  totPriceCnt++;
372  }
373  }
374  }
375  if ( totPriceCnt > 0 )
376  totPrice /= totPriceCnt;
377  commod_av_gal_price = totPrice;
378 
379  } else {
380  commod_av_gal_price = 0;
381  }
382 }

738 static glTexture *gl_genFactionDisk( int radius )
739 {
740  int i, j;
741  uint8_t *pixels;
742  SDL_Surface *sur;
743  int dist;
744  double alpha;
745 
746  /* Calculate parameters. */
747  const int w = 2 * radius + 1;
748  const int h = 2 * radius + 1;
749 
750  /* Create the surface. */
751  sur = SDL_CreateRGBSurface( 0, w, h, 32, RGBAMASK );
752 
753  pixels = sur->pixels;
754  memset(pixels, 0xff, sizeof(uint8_t) * 4 * h * w);
755 
756  /* Generate the circle. */
757  SDL_LockSurface( sur );
758 
759  /* Draw the circle with filter. */
760  for (i=0; i<h; i++) {
761  for (j=0; j<w; j++) {
762  /* Calculate blur. */
763  dist = (i - radius) * (i - radius) + (j - radius) * (j - radius);
764  alpha = 0.;
765 
766  if (dist < radius * radius) {
767  /* Computes alpha with an empirically chosen formula.
768  * This formula accounts for the fact that the eyes
769  * has a logarithmic sensitivity to light */
770  alpha = 1. * dist / (radius * radius);
771  alpha = (exp(1 / (alpha + 1) - 0.5) - 1) * 0xFF;
772  }
773 
774  /* Sets the pixel alpha which is the forth byte
775  * in the pixel representation. */
776  pixels[i*sur->pitch + j*4 + 3] = (uint8_t)alpha;
777  }
778  }
779 
780  SDL_UnlockSurface( sur );
781 
782  /* Return texture. */
783  return gl_loadImage( sur, OPENGL_TEX_MIPMAPS );
