219 {
220  int *values;
221  int i, j, temp, max;
222  int new_table;
223 
224  NLUA_MIN_ARGS(1);
225  if (lua_isnumber(L,1)) {
226  max = lua_tointeger(L,1);
227  new_table = 1;
228  }
229  else if (lua_istable(L,1)) {
230  max = (int) lua_objlen(L,1);
231  new_table = 0;
232  }
233  else
234  NLUA_INVALID_PARAMETER(L);
235 
236  /* Create the list. */
237  values = malloc( sizeof(int)*max );
238  for (i=0; i<max; i++)
239  values[i]=i;
240 
241  /* Fisher-Yates shuffling algorithm */
242  for (i = max-1; i >= 0; --i){
243  /* Generate a random number in the range [0, max-1] */
244  j = randint() % (i+1);
245 
246  /* Swap the last element with an element at a random index. */
247  temp = values[i];
248  values[i] = values[j];
249  values[j] = temp;
250  }
251 
252  /* Now either return a new table or permute the given table. */
253  lua_newtable(L);
254  for (i=0; i<max; i++) {
255  lua_pushnumber( L, i+1 );
256  lua_pushnumber( L, values[i]+1 );
257  if (!new_table)

259  lua_settable( L, -3 );
260  }
