181 static int rnd_permutation( lua_State *L )
182 {
183  int *values;
184  int i, j, temp, max;
185  int new_table;
186 
187  NLUA_MIN_ARGS(1);
188  if (lua_isnumber(L,1)) {
189  max = lua_tointeger(L,1);
190  new_table = 1;
191  }
192  else if (lua_istable(L,1)) {
193  max = (int) lua_objlen(L,1);
194  new_table = 0;
195  }
196  else
197  NLUA_INVALID_PARAMETER(L);
198 
199  /* Create the list. */
200  values = malloc( sizeof(int)*max );
201  for (i=0; i<max; i++)
202  values[i]=i;
203 
204  /* Fisher-Yates shuffling algorithm */
205  for (i = max-1; i >= 0; --i){
206  /* Generate a random number in the range [0, max-1] */
207  j = randint() % (i+1);
208 
209  /* Swap the last element with an element at a random index. */
210  temp = values[i];
211  values[i] = values[j];
212  values[j] = temp;
213  }
214 
215  /* Now either return a new table or permute the given table. */
216  lua_newtable(L);
217  for (i=0; i<max; i++) {
218  lua_pushnumber( L, i+1 );
219  lua_pushnumber( L, values[i]+1 );
220  if (!new_table)
221  lua_gettable( L, 1 );
222  lua_settable( L, -3 );
223  }
224 
225  free( values );
226 
227  return 1;
