237  }
238  else

240 
241  /* Create the list. */
242  values = malloc( sizeof(int)*max );
243  for (int i=0; i<max; i++)
244  values[i]=i;
245 
246  /* Fisher-Yates shuffling algorithm */
247  for (int i = max-1; i >= 0; --i){
248  /* Generate a random number in the range [0, max-1] */
249  int j = randint() % (i+1);
250 
251  /* Swap the last element with an element at a random index. */
252  int temp = values[i];
253  values[i] = values[j];
254  values[j] = temp;
255  }
256

258  lua_newtable(L);

260  lua_pushnumber( L, values[i]+1 );
261  if (!new_table)

264  }
