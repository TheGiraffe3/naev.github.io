261 static int tkL_choice( lua_State *L )
262 {
263  int ret, opts;
264  const char *title, *str;
265  char *result;
266 
267  /* Handle parameters. */
268  opts = lua_gettop(L) - 2;
269  title = luaL_checkstring(L,1);
270  str = luaL_checkstring(L,2);
271 
272  /* Do an initial scan for invalid arguments. */
273  for (int i=0; i<opts; i++)
274  luaL_checkstring(L, i+3);
275 
276  /* Create dialogue. */
277  dialogue_makeChoice( title, str, opts );
278  for (int i=0; i<opts; i++)
279  dialogue_addChoice( title, str, luaL_checkstring(L,i+3) );
280  result = dialogue_runChoice();
281  if (result == NULL) /* Something went wrong, return nil. */
282  return 0;
283 
284  /* Handle results. */
285  ret = -1;
286  for (int i=0; i<opts && ret==-1; i++) {
287  if (strcmp(result, luaL_checkstring(L,i+3)) == 0)
288  ret = i+1; /* Lua uses 1 as first index. */
289  }
290 
291  /* Push parameters. */
292  lua_pushnumber(L,ret);
293  lua_pushstring(L,result);
294 
295  /* Clean up. */
296  free(result);
297 
298  return 2;
