223  int ret, opts, i;
224  const char *title, *str;
225  char *result;
226  NLUA_MIN_ARGS(3);
227 
228  /* Handle parameters. */
229  opts = lua_gettop(L) - 2;
230  title = luaL_checkstring(L,1);
231  str = luaL_checkstring(L,2);
232 
233  /* Do an initial scan for invalid arguments. */
234  for (i=0; i<opts; i++)
235  luaL_checkstring(L, i+3);
236 
237  /* Create dialogue. */
238  dialogue_makeChoice( title, str, opts );
239  for (i=0; i<opts; i++)
240  dialogue_addChoice( title, str, luaL_checkstring(L,i+3) );
241  result = dialogue_runChoice();
242  if (result == NULL) /* Something went wrong, return nil. */
243  return 0;
244 
245  /* Handle results. */
246  ret = -1;
247  for (i=0; i<opts && ret==-1; i++) {
248  if (strcmp(result, luaL_checkstring(L,i+3)) == 0)
249  ret = i+1; /* Lua uses 1 as first index. */
250  }
251 
252  /* Push parameters. */
253  lua_pushnumber(L,ret);
254  lua_pushstring(L,result);
255 
256  /* Clean up. */
257  free(result);
258 
259  return 2;
