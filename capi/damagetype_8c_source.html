66 static int DTYPE_parse( DTYPE *temp, const xmlNodePtr parent )
67 {
68  xmlNodePtr node;
69  char *stat;
70 
71  /* Clear data. */
72  memset( temp, 0, sizeof(DTYPE) );
73 
74  xmlr_attr_strd( parent, "name", temp->name );
75 
76  /* Extract the data. */
77  node = parent->xmlChildrenNode;
78  do {
79  xml_onlyNodes(node);
80 
81  if (xml_isNode(node, "shield")) {
82  temp->sdam = xml_getFloat(node);
83 
84  xmlr_attr_strd( node, "stat", stat );
85  if (stat != NULL) {
86  temp->soffset = ss_offsetFromType( ss_typeFromName(stat) );
87  free(stat);
88  }
89 
90  continue;
91  }
92  else if (xml_isNode(node, "armour")) {
93  temp->adam = xml_getFloat(node);
94 
95  xmlr_attr_strd( node, "stat", stat );
96  if (stat != NULL) {
97  temp->aoffset = ss_offsetFromType( ss_typeFromName(stat) );
98  free(stat);
99  }
100 
101  continue;
102  }
103  xmlr_float(node, "armour", temp->adam);
104  xmlr_float(node, "knockback", temp->knock);
105 
106  WARN(_("Unknown node of type '%s' in damage node '%s'."), node->name, temp->name);
107  } while (xml_nextNode(node));
108 
109 #define MELEMENT(o,s) \
110  if (o) WARN(_("DTYPE '%s' invalid '"s"' element"), temp->name)
111  MELEMENT(temp->sdam<0.,"shield");
112  MELEMENT(temp->adam<0.,"armour");
113  MELEMENT(temp->knock<0.,"knockback");
114 #undef MELEMENT
115 
116  return 0;
117 }

188

190  node = doc->xmlChildrenNode;

192  ERR("Malformed '"DTYPE_DATA_PATH"' file: missing root element '"DTYPE_XML_ID"'");
193  return -1;
194  }
195

197  node = node->xmlChildrenNode; /* first system node */
198  if (node == NULL) {
199  ERR("Malformed '"DTYPE_DATA_PATH"' file: does not contain elements");
200  return -1;
201  }
202

204  dtype_types = array_create(DTYPE);
205  do {

251 void dtype_calcDamage( double *dshield, double *darmour, double absorb, double *knockback, const Damage *dmg, ShipStats *s )
252 {
253  DTYPE *dtype;
254  char *ptr;
255  double multiplier;
256 
257  /* Must be valid. */
258  dtype = dtype_validType( dmg->type );
259  if (dtype == NULL)
260  return;
261 
262  /* Set if non-nil. */
263  if (dshield != NULL) {
264  if ((dtype->soffset <= 0) || (s == NULL))
265  *dshield = dtype->sdam * dmg->damage * absorb;
266  else {
267  /*
268  * If an offset has been specified, look for a double at that offset
269  * in the ShipStats struct, and used it as a multiplier.
270  *
271  * The 2. - n logic serves to undo the initialization done by
272  * ss_statsInit and turn the value into a multiplier.
273  */
274  ptr = (char*) s;
275  memcpy(&multiplier, &ptr[ dtype->soffset ], sizeof(double));
276  multiplier = MAX( 0., 2. - multiplier );
277  *dshield = dtype->sdam * dmg->damage * absorb * multiplier;
278  }
279  }
280  if (darmour != NULL) {
281  if ((dtype->aoffset) <= 0 || (s == NULL))
282  *darmour = dtype->adam * dmg->damage * absorb;
283  else {
284  ptr = (char*) s;
285  memcpy(&multiplier, &ptr[ dtype->aoffset ], sizeof(double));
286  multiplier = MAX( 0., 2. - multiplier );
287  *darmour = dtype->adam * dmg->damage * absorb * multiplier;
288  }
289  }
290 
291  if (knockback != NULL)
292  *knockback = dtype->knock;
293 }
