130 int event_runFunc( unsigned int eventid, const char *func, int nargs )
131 {
132  int ret;
133  const char* err;
134  int evt_delete;
135  Event_t *ev;
136 
137  ev = event_get( eventid );
138  if (ev == NULL)
139  return 0;
140 
141  ret = nlua_pcall(ev->env, nargs, 0);
142  if (ret != 0) { /* error has occurred */
143  err = (lua_isstring(naevL,-1)) ? lua_tostring(naevL,-1) : NULL;
144  if ((err==NULL) || (strcmp(err,NLUA_DONE)!=0)) {
145  WARN(_("Event '%s' -> '%s': %s"),
146  event_getData(ev->id), func, (err) ? err : _("unknown error"));
147  ret = -1;
148  }
149  else
150  ret = 1;
151  lua_pop(naevL, 1);
152  }
153 
154  /* Time to remove the event. */
155  ev = event_get( eventid ); /* The Lua call may have invalidated the pointer. */
156  nlua_getenv(ev->env, "__evt_delete");
157  evt_delete = lua_toboolean(naevL,-1);
158  lua_pop(naevL,1);
159  if (evt_delete) {
160  ret = 2;
161  event_remove( ev->id );
162  }
163 
164  return ret;
165 }

328  if (lua_istable(L,1)) {

334  else if (lua_isstring(L,-1))

336  lua_pop(L,1);
337  }
338  }
339  else if (lua_issystem(L, 1))
340  claim_addSys( claim, lua_tosystem( L, 1 ) );
341  else if (lua_isstring(L, 1))
342  claim_addStr( claim, lua_tostring( L, 1 ) );
343  else
344  NLUA_INVALID_PARAMETER(L);
345

347  if (onlytest) {
