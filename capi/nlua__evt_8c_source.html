142 int event_runFunc( unsigned int eventid, const char *func, int nargs )
143 {
144  int ret;
145  const char* err;
146  int evt_delete;
147  Event_t *ev;
148 
149  ev = event_get( eventid );
150  if (ev == NULL)
151  return 0;
152 
153 
154  ret = nlua_pcall(ev->env, nargs, 0);
155  if (ret != 0) { /* error has occurred */
156  err = (lua_isstring(naevL,-1)) ? lua_tostring(naevL,-1) : NULL;
157  if ((err==NULL) || (strcmp(err,NLUA_DONE)!=0)) {
158  WARN(_("Event '%s' -> '%s': %s"),
159  event_getData(ev->id), func, (err) ? err : _("unknown error"));
160  ret = -1;
161  }
162  else
163  ret = 1;
164  lua_pop(naevL, 1);
165  }
166 
167  /* Time to remove the event. */
168  ev = event_get( eventid ); /* The Lua call may have invalidated the pointer. */
169  nlua_getenv(ev->env, "__evt_delete");
170  evt_delete = lua_toboolean(naevL,-1);
171  lua_pop(naevL,1);
172  if (evt_delete) {
173  ret = 2;
174  event_remove( ev->id );
175  }
176 
177  return ret;
178 }

330 static int evtL_claim( lua_State *L )
331 {
332  Claim_t *claim;
333  Event_t *cur_event;
334  int onlytest;
335 
336  /* Get current event. */
337  cur_event = event_getFromLua(L);
338 
339  onlytest = lua_toboolean(L,2);
340 
341  /* Check to see if already claimed. */
342  if (!onlytest && (cur_event->claims != NULL)) {
343  NLUA_ERROR(L, _("Event trying to claim but already has."));
344  return 0;
345  }
346 
347  /* Create the claim. */
348  claim = claim_create();
349 
350  /* Handle parameters. */
351  if (lua_istable(L,1)) {
352  /* Iterate over table. */
353  lua_pushnil(L);
354  while (lua_next(L, 1) != 0) {
355  if (lua_issystem(L,-1))
356  claim_addSys( claim, lua_tosystem( L, -1 ) );
357  else if (lua_isstring(L,-1))
358  claim_addStr( claim, lua_tostring( L, -1 ) );
359  lua_pop(L,1);
360  }
361  }
362  else if (lua_issystem(L, 1))
363  claim_addSys( claim, lua_tosystem( L, 1 ) );
364  else if (lua_isstring(L, 1))
365  claim_addStr( claim, lua_tostring( L, 1 ) );
366  else
367  NLUA_INVALID_PARAMETER(L);
368 
369  /* Only test, but don't apply case. */
370  if (onlytest) {
371  lua_pushboolean( L, !claim_test( claim ) );
372  claim_destroy( claim );
373  return 1;
374  }
375 
376  /* Test claim. */
377  if (claim_test( claim )) {
378  claim_destroy( claim );
379  lua_pushboolean(L,0);
380  return 1;
381  }
382 
383  /* Set the claim. */
384  cur_event->claims = claim;
385  claim_activate( claim );
386  lua_pushboolean(L,1);
387  return 1;
