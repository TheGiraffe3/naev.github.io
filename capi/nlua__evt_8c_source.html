307 static int evt_claim( lua_State *L )
308 {
309  Claim_t *claim;
310  Event_t *cur_event;
311 
312  /* Get current event. */
313  cur_event = event_getFromLua(L);
314 
315  /* Check to see if already claimed. */
316  if (cur_event->claims != NULL) {
317  NLUA_ERROR(L, _("Event trying to claim but already has."));
318  return 0;
319  }
320 
321  /* Create the claim. */
322  claim = claim_create();
323 
324  /* Handle parameters. */
325  if (lua_istable(L,1)) {
326  /* Iterate over table. */
327  lua_pushnil(L);
328  while (lua_next(L, 1) != 0) {
329  if (lua_issystem(L,-1))
330  claim_addSys( claim, lua_tosystem( L, -1 ) );
331  else if (lua_isstring(L,-1))
332  claim_addStr( claim, lua_tostring( L, -1 ) );
333  lua_pop(L,1);
334  }
335  }
336  else if (lua_issystem(L, 1))
337  claim_addSys( claim, lua_tosystem( L, 1 ) );
338  else if (lua_isstring(L, 1))
339  claim_addStr( claim, lua_tostring( L, 1 ) );
340  else
341  NLUA_INVALID_PARAMETER(L);
342 
343  /* Test claim. */
344  if (claim_test( claim )) {
345  claim_destroy( claim );
346  lua_pushboolean(L,0);
347  return 1;
348  }
349 
350  /* Set the claim. */
351  cur_event->claims = claim;
352  claim_activate( claim );
353  lua_pushboolean(L,1);
