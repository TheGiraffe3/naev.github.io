255 static int audioL_new( lua_State *L )
256 {
257  LuaAudio_t la;
258  LuaFile_t *lf;
259  const char *name;
260  SDL_RWops *rw;
261  double master;
262 
263  name = NULL;
264  rw = NULL;
265 
266  if (lua_isstring(L,1))
267  name = lua_tostring(L,1);
268  else if (lua_isfile(L,1)) {
269  lf = lua_tofile(L,1);
270  name = lf->path;
271  }
272  else
273  NLUA_INVALID_PARAMETER(L);
274 
275  memset( &la, 0, sizeof(LuaAudio_t) );
276  if (!conf.nosound) {
277  rw = PHYSFSRWOPS_openRead( name );
278  if (rw==NULL)
279  NLUA_ERROR(L,"Unable to open '%s'", name );
280 
281  soundLock();
282  alGenSources( 1, &la.source );
283 
284  la.buf = malloc( sizeof(LuaBuffer_t) );
285  la.buf->refcount = 1;
286  alGenBuffers( 1, &la.buf->buffer );
287  sound_al_buffer( &la.buf->buffer, rw, name );
288 
289  /* Attach buffer. */
290  alSourcei( la.source, AL_BUFFER, la.buf->buffer );
291 
292  /* Defaults. */
293  master = sound_getVolumeLog();
294  alSourcef( la.source, AL_GAIN, master );
295  /* The behaviour of sources depends on whether or not they are mono or
296  * stereo. In the case they are stereo, no position stuff is actually
297  * done. However, if they are mono, they are played with absolute
298  * position and the sound heard depends on the listener. We can disable
299  * this by setting AL_SOURCE_RELATIVE which puts the listener always at
300  * the origin, and then setting the source at the same origin. It should
301  * be noted that depending on the sound model this can be bad if it is
302  * not bounded. */
303  alSourcei( la.source, AL_SOURCE_RELATIVE, AL_TRUE );
304  alSource3f( la.source, AL_POSITION, 0., 0., 0. );
305  al_checkErr();
306  soundUnlock();
307 
308  SDL_RWclose( rw );
309  }
310 
311  lua_pushaudio(L, la);
312  return 1;
313 }
