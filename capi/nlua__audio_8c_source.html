237 static int audioL_new( lua_State *L )
238 {
239  LuaAudio_t la;
240  LuaFile_t *lf;
241  const char *name;
242  SDL_RWops *rw;
243  double master;
244 
245  name = NULL;
246  rw = NULL;
247 
248  if (lua_isstring(L,1))
249  name = lua_tostring(L,1);
250  else if (lua_isfile(L,1)) {
251  lf = lua_tofile(L,1);
252  name = lf->path;
253  }
254  else
255  NLUA_INVALID_PARAMETER(L);
256 
257  memset( &la, 0, sizeof(LuaAudio_t) );
258  if (!conf.nosound) {
259  rw = PHYSFSRWOPS_openRead( name );
260  if (rw==NULL)
261  NLUA_ERROR(L,"Unable to open '%s'", name );
262 
263  soundLock();
264  alGenSources( 1, &la.source );
265  alGenBuffers( 1, &la.buffer );
266 
267  sound_al_buffer( &la.buffer, rw, name );
268 
269  /* Attach buffer. */
270  alSourcei( la.source, AL_BUFFER, la.buffer );
271 
272  /* Defaults. */
273  master = sound_getVolumeLog();
274  alSourcef( la.source, AL_GAIN, master );
275  alSourcei( la.source, AL_SOURCE_RELATIVE, AL_FALSE );
276  al_checkErr();
277  soundUnlock();
278 
279  SDL_RWclose( rw );
280  }
281 
282  lua_pushaudio(L, la);
283  return 1;
284 }

594 static int audioL_soundPlay( lua_State *L )
595 {
596  const char *name;
597  Vector2d *pos, *vel, vel0;
598  int dopos;
599 
600  /* Flag wether to use sound_playPos or sound_play. */
601  dopos = 0;
602 
603  /* Handle parameters. */
604  name = luaL_checkstring(L,1);
605  if (lua_gettop(L) > 1) {
606  dopos = 1;
607  pos = luaL_checkvector(L,2);
608  if (lua_gettop(L) > 2) {
609  vel = luaL_checkvector(L,3);
610  }
611  else {
612  vectnull( &vel0 );
613  vel = &vel0;
614  }
615  }
616 
617  if (dopos)
618  sound_playPos( sound_get(name), pos->x, pos->y, vel->x, vel->y );
619  else
620  sound_play( sound_get(name) );
621 
622  return 0;
623 }
