239 static int audioL_new( lua_State *L )
240 {
241  LuaAudio_t la;
242  LuaFile_t *lf;
243  const char *name;
244  SDL_RWops *rw;
245  double master;
246 
247  name = NULL;
248  rw = NULL;
249 
250  if (lua_isstring(L,1))
251  name = lua_tostring(L,1);
252  else if (lua_isfile(L,1)) {
253  lf = lua_tofile(L,1);
254  name = lf->path;
255  }
256  else
257  NLUA_INVALID_PARAMETER(L);
258 
259  memset( &la, 0, sizeof(LuaAudio_t) );
260  if (!conf.nosound) {
261  rw = PHYSFSRWOPS_openRead( name );
262  if (rw==NULL)
263  NLUA_ERROR(L,"Unable to open '%s'", name );
264 
265  soundLock();
266  alGenSources( 1, &la.source );
267  alGenBuffers( 1, &la.buffer );
268 
269  sound_al_buffer( &la.buffer, rw, name );
270 
271  /* Attach buffer. */
272  alSourcei( la.source, AL_BUFFER, la.buffer );
273 
274  /* Defaults. */
275  master = sound_getVolumeLog();
276  alSourcef( la.source, AL_GAIN, master );
277  /* The behaviour of sources depends on whether or not they are mono or
278  * stereo. In the case they are stereo, no position stuff is actually
279  * done. However, if they are mono, they are played with absolute
280  * position and the sound heard depends on the listener. We can disable
281  * this by setting AL_SOURCE_RELATIVE which puts the listener always at
282  * the origin, and then setting the source at the same origin. It should
283  * be noted that depending on the sound model this can be bad if it is
284  * not bounded. */
285  alSourcei( la.source, AL_SOURCE_RELATIVE, AL_TRUE );
286  alSource3f( la.source, AL_POSITION, 0., 0., 0. );
287  al_checkErr();
288  soundUnlock();
289 
290  SDL_RWclose( rw );
291  }
292 
293  lua_pushaudio(L, la);
294  return 1;
295 }

633 static int audioL_soundPlay( lua_State *L )
634 {
635  const char *name;
636  Vector2d *pos, *vel, vel0;
637  int dopos;
638 
639  /* Flag wether to use sound_playPos or sound_play. */
640  dopos = 0;
641 
642  /* Handle parameters. */
643  name = luaL_checkstring(L,1);
644  if (lua_gettop(L) > 1) {
645  dopos = 1;
646  pos = luaL_checkvector(L,2);
647  if (lua_gettop(L) > 2) {
648  vel = luaL_checkvector(L,3);
649  }
650  else {
651  vectnull( &vel0 );
652  vel = &vel0;
653  }
654  }
655 
656  if (dopos)
657  sound_playPos( sound_get(name), pos->x, pos->y, vel->x, vel->y );
658  else
659  sound_play( sound_get(name) );
660 
661  return 0;
662 }
