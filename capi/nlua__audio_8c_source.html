260 static int audioL_new( lua_State *L )
261 {
262  LuaAudio_t la;
263  LuaFile_t *lf;
264  const char *name;
265  SDL_RWops *rw;
266  double master;
267 
268  name = NULL;
269  rw = NULL;
270 
271  if (lua_isstring(L,1))
272  name = lua_tostring(L,1);
273  else if (lua_isfile(L,1)) {
274  lf = lua_tofile(L,1);
275  name = lf->path;
276  }
277  else
278  NLUA_INVALID_PARAMETER(L);
279 
280  memset( &la, 0, sizeof(LuaAudio_t) );
281  if (!sound_disabled) {
282  rw = PHYSFSRWOPS_openRead( name );
283  if (rw==NULL)
284  NLUA_ERROR(L,"Unable to open '%s'", name );
285 
286  soundLock();
287  alGenSources( 1, &la.source );
288 
289  la.buf = malloc( sizeof(LuaBuffer_t) );
290  la.buf->refcount = 1;
291  alGenBuffers( 1, &la.buf->buffer );
292  sound_al_buffer( &la.buf->buffer, rw, name );
293 
294  /* Attach buffer. */
295  alSourcei( la.source, AL_BUFFER, la.buf->buffer );
296 
297  /* Defaults. */
298  master = sound_getVolumeLog();
299  alSourcef( la.source, AL_GAIN, master );
300  /* The behaviour of sources depends on whether or not they are mono or
301  * stereo. In the case they are stereo, no position stuff is actually
302  * done. However, if they are mono, they are played with absolute
303  * position and the sound heard depends on the listener. We can disable
304  * this by setting AL_SOURCE_RELATIVE which puts the listener always at
305  * the origin, and then setting the source at the same origin. It should
306  * be noted that depending on the sound model this can be bad if it is
307  * not bounded. */
308  alSourcei( la.source, AL_SOURCE_RELATIVE, AL_TRUE );
309  alSource3f( la.source, AL_POSITION, 0., 0., 0. );
310  al_checkErr();
311  soundUnlock();
312 
313  SDL_RWclose( rw );
314  }
315 
316  lua_pushaudio(L, la);
317  return 1;
318 }
