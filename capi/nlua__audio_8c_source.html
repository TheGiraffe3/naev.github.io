237 static int audioL_new( lua_State *L )
238 {
239  LuaAudio_t la;
240  LuaFile_t *lf;
241  const char *name;
242  SDL_RWops *rw;
243  double master;
244 
245  name = NULL;
246  rw = NULL;
247 
248  if (lua_isstring(L,1))
249  name = lua_tostring(L,1);
250  else if (lua_isfile(L,1)) {
251  lf = lua_tofile(L,1);
252  name = lf->path;
253  }
254  else
255  NLUA_INVALID_PARAMETER(L);
256 
257  memset( &la, 0, sizeof(LuaAudio_t) );
258  if (!conf.nosound) {
259  rw = PHYSFSRWOPS_openRead( name );
260  if (rw==NULL)
261  NLUA_ERROR(L,"Unable to open '%s'", name );
262 
263  soundLock();
264  alGenSources( 1, &la.source );
265  alGenBuffers( 1, &la.buffer );
266 
267  sound_al_buffer( &la.buffer, rw, name );
268 
269  /* Attach buffer. */
270  alSourcei( la.source, AL_BUFFER, la.buffer );
271 
272  /* Defaults. */
273  master = sound_getVolumeLog();
274  alSourcef( la.source, AL_GAIN, master );
275  /* The behaviour of sources depends on whether or not they are mono or
276  * stereo. In the case they are stereo, no position stuff is actually
277  * done. However, if they are mono, they are played with absolute
278  * position and the sound heard depends on the listener. We can disable
279  * this by setting AL_SOURCE_RELATIVE which puts the listener always at
280  * the origin, and then setting the source at the same origin. It should
281  * be noted that depending on the sound model this can be bad if it is
282  * not bounded. */
283  alSourcei( la.source, AL_SOURCE_RELATIVE, AL_TRUE );
284  alSource3f( la.source, AL_POSITION, 0., 0., 0. );
285  al_checkErr();
286  soundUnlock();
287 
288  SDL_RWclose( rw );
289  }
290 
291  lua_pushaudio(L, la);
292  return 1;
293 }

603 static int audioL_soundPlay( lua_State *L )
604 {
605  const char *name;
606  Vector2d *pos, *vel, vel0;
607  int dopos;
608 
609  /* Flag wether to use sound_playPos or sound_play. */
610  dopos = 0;
611 
612  /* Handle parameters. */
613  name = luaL_checkstring(L,1);
614  if (lua_gettop(L) > 1) {
615  dopos = 1;
616  pos = luaL_checkvector(L,2);
617  if (lua_gettop(L) > 2) {
618  vel = luaL_checkvector(L,3);
619  }
620  else {
621  vectnull( &vel0 );
622  vel = &vel0;
623  }
624  }
625 
626  if (dopos)
627  sound_playPos( sound_get(name), pos->x, pos->y, vel->x, vel->y );
628  else
629  sound_play( sound_get(name) );
630 
631  return 0;
632 }
