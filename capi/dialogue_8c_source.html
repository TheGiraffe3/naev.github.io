176 static glFont* dialogue_getSize( const char* title,
177  const char* msg, int* width, int* height )
178 {
179  glFont* font;
180  double w, h, d;
181  int i, titlelen, msglen;
182 
183  /* Get title length. */
184  titlelen = gl_printWidthRaw( &gl_defFont, title );
185  msglen = gl_printWidthRaw( &gl_defFont, msg );
186 
187  /* Try widths from 300 to 800 in 50 px increments.
188  * Each subsequent width gets an additional line, following this table:
189  *
190  * 300 px: 2 lines, 540 px total
191  * 350 px: 3 lines, 930 px total
192  * ...
193  * 800 px: 12 lines, 9600 px total
194  */
195  for (i=0; i<11; i++)
196  if (msglen < (260 + i * 50) * (2 + i))
197  break;
198 
199  w = 300 + i * 50;
200  w = MAX(w, titlelen+40); /* Expand width if the title is long. */
201 
202  /* Now we look at proportion. */
203  font = &gl_defFont;
204  h = gl_printHeightRaw( font, w-40, msg );
205 
206  d = ((double)w/(double)h)*(3./4.); /* deformation factor. */
207  if (FABS(d) > 0.3) {
208  if (h > w)
209  w = h;
210  h = gl_printHeightRaw( font, w-40, msg );
211  }
212 
213  /* Set values. */
214  (*width) = w;
215  (*height) = h;
216 
217  return font;
218 }
