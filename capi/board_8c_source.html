200  if (!pilot_isDisabled(target))
201  return 0;
202  else if (vect_dist(&p->solid->pos, &target->solid->pos) >
203  target->ship->gfx_space->sw * PILOT_SIZE_APPROX )
204  return 0;
205  else if ((pow2(VX(p->solid->vel)-VX(target->solid->vel)) +
206  pow2(VY(p->solid->vel)-VY(target->solid->vel))) >
207  (double)pow2(MAX_HYPERSPACE_VEL))
208  return 0;
209  else if (pilot_isFlag(target,PILOT_BOARDED))
210  return 0;
211

213  pilot_setFlag(target, PILOT_BOARDED);
214  pilot_setFlag(p, PILOT_BOARDING);

217  p->ptimer = 3.;
