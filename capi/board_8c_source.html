320 static void board_stealAmmo( unsigned int wdw, char* str )
321 {
322  Pilot* p;
323  int nreloaded, i, nammo, x;
324  PilotOutfitSlot *target_outfit_slot, *player_outfit_slot;
325  Outfit *target_outfit, *ammo, *player_outfit;

328  p = pilot_get(player.p->target);
329  /* Target has no ammo */
330  if (pilot_countAmmo(p) <= 0) {
331  player_message(_("#rThe ship has no ammo."));
332  return;
333  }
334  /* Player is already at max ammo */
335  if (pilot_countAmmo(player.p) >= pilot_maxAmmo(player.p)) {
336  player_message(_("#rYou are already at max ammo."));
337  return;
338  }
339  if (board_fail(wdw))
340  return;
341  /* Steal the ammo */
342  for (i=0; i<array_size(p->outfits); i++) {
343  target_outfit_slot = p->outfits[i];
344  if (target_outfit_slot == NULL)
345  continue;
346  target_outfit = target_outfit_slot->outfit;
347  if (target_outfit == NULL)
348  continue;
349  /* outfit isn't a launcher */
350  if (!outfit_isLauncher(target_outfit)) {
351  continue;
352  }
353  nammo = target_outfit_slot->u.ammo.quantity;
354  ammo = target_outfit_slot->u.ammo.outfit;
355  /* launcher has no ammo */
356  if (ammo == NULL)
357  continue;
358  if (nammo <= 0) {
359  continue;
360  }
361  for (x=0; x<array_size(player.p->outfits); x++) {
362  int nadded = 0;
363  player_outfit_slot = player.p->outfits[x];
364  if (player_outfit_slot == NULL)
365  continue;
366  player_outfit = player_outfit_slot->outfit;
367  if (player_outfit == NULL)
368  continue;
369  if (!outfit_isLauncher(player_outfit)) {
370  continue;
371  }
372  if (strcmp(ammo->name, player_outfit_slot->u.ammo.outfit->name) != 0) {
373  continue;
374  }
375  /* outfit's ammo matches; try to add to player and remove from target */
376  nadded = pilot_addAmmo(player.p, player_outfit_slot, ammo, nammo);
377  nammo -= nadded;
378  pilot_rmAmmo(p, target_outfit_slot, nadded);
379  nreloaded += nadded;
380  if (nadded > 0)
381  player_message(_("#oYou looted: %d × %s"), nadded, _(ammo->name));
382  if (nammo <= 0) {
383  break;
384  }
385  }
386  if (nammo <= 0) {
387  continue;
388  }
389  }
390  if (nreloaded <= 0)
391  player_message(_("#rThere is no ammo compatible with your launchers on board."));
392  pilot_updateMass(player.p);
393  pilot_weaponSafe(player.p);
394  pilot_updateMass(p);
395  pilot_weaponSafe(p);
396  board_update(wdw);
397 }

539

541  if (!pilot_isDisabled(target))
542  return 0;

544  target->ship->gfx_space->sw * PILOT_SIZE_APPROX )
545  return 0;
