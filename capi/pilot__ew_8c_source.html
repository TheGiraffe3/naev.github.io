444  // continue;
445

448  /* TODO maybe not hardcode the close value. */
449  if ((close != NULL) && !pilot_isFlag(t,PILOT_STEALTH) &&
450  (dist < pow2( MAX( 0., p->ew_stealth * t->stats.ew_detect * 1.5 ))))
