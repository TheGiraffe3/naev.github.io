101 static void pilot_ewUpdate( Pilot *p )
102 {
103  p->ew_detection = p->ew_mass * p->ew_asteroid / p->stats.ew_hide;
104  p->ew_evasion = p->ew_detection * 0.75 * ew_interference / p->stats.ew_evade;
105  /* For stealth we apply the ew_asteroid and ew_interference bonus outside of the max, so that it can go below 1000 with in-system features. */
106  p->ew_stealth = MAX( 1000., p->ew_mass / p->stats.ew_hide * 0.25 / p->stats.ew_stealth ) * p->ew_asteroid * ew_interference;
107 }

356 int pilot_inRangeJump( const Pilot *p, int i )
357 {
358  double d;
359  JumpPoint *jp;
360  double sense;
361  double hide;
362 
363  /* pilot must exist */
364  if (p == NULL)
365  return 0;
366 
367  /* Get the jump point. */
368  jp = &cur_system->jumps[i];
369 
370  /* We don't want exit-only jumps. */
371  if (jp_isFlag(jp, JP_EXITONLY))
372  return 0;
373 
374  /* Jumps with 0. hide are considered to be highway points and always visible. */
375  hide = jp->hide;
376  if (hide==0.)
377  return 1;
378 
379  sense = EW_JUMPDETECT_DIST * p->stats.ew_jump_detect;
380  /* Handle hidden jumps separately, as they use a special range parameter. */
381  if (jp_isFlag(jp, JP_HIDDEN))
382  sense *= p->stats.misc_hidden_jump_detect;
383 
384  /* Get distance. */
385  d = vect_dist2( &p->solid->pos, &jp->pos );
386  if (d < pow2( MAX( 0., sense * hide )))
387  return 1;

408 static int pilot_ewStealthGetNearby( const Pilot *p, double *mod, int *close, int *isplayer )
409 {
410  Pilot *t;
411  Pilot *const* ps;
412  int i, n;
413  double dist;
414 
415  /* Check nearby non-allies. */
416  if (mod != NULL)
417  *mod = 0.;
418  if (close != NULL)
419  *close = 0;
420  if (isplayer != NULL)
421  *isplayer = 0;
422  n = 0;
423  ps = pilot_getAll();
424  for (i=0; i<array_size(ps); i++) {
425  t = ps[i];
426  if (areAllies( p->faction, t->faction ) ||
427  ((p->id == PLAYER_ID) && pilot_isFriendly(t)))
428  continue;
429  if (pilot_isDisabled(t))
430  continue;
431  if (!pilot_canTarget(t))
432  continue;
433 
434  /* Must not be landing nor taking off. */
435  if (pilot_isFlag(t, PILOT_LANDING) ||
436  pilot_isFlag(t, PILOT_TAKEOFF))
437  continue;
438 
439  /* Stealthed pilots don't reduce stealth. */
440  if (pilot_isFlag(t, PILOT_STEALTH))
441  continue;
442 
443  /* Compute distance. */
444  dist = vect_dist2( &p->solid->pos, &t->solid->pos );
445  /* TODO maybe not hardcode the close value. */
446  if ((close != NULL) && (dist < pow2( MAX( 0., p->ew_stealth * t->stats.ew_detect * 1.5 ))))
447  (*close)++;
448  if (dist > pow2( MAX( 0., p->ew_stealth * t->stats.ew_detect )))
449  continue;
450 
451  if (mod != NULL)
452  *mod += 1.0 - sqrt(dist) / (p->ew_stealth * t->stats.ew_detect);
453 
454  /* We found a pilot that is in range. */
455  n++;
456  if ((isplayer != NULL) && pilot_isPlayer(t))
457  *isplayer = 1;
458  }
459 
460  return n;
461 }

467 void pilot_ewUpdateStealth( Pilot *p, double dt )
468 {
469  int n, close, isplayer;
470  double mod;
471 
472  if (!pilot_isFlag( p, PILOT_STEALTH ))
473  return;
474 
475  /* Get nearby pilots. */
476  if (pilot_isPlayer(p))

478  else
479  n = pilot_ewStealthGetNearby( p, &mod, NULL, &isplayer );
480 
481  /* Stop autonav if pilots are nearby. */
482  if (pilot_isPlayer(p) && (close>0))
483  player_autonavResetSpeed();
484 
485  /* Increases if nobody nearby. */
486  if (n == 0) {
487  p->ew_stealth_timer += dt * 5000. / p->ew_stealth;
488  if (p->ew_stealth_timer > 1.)
489  p->ew_stealth_timer = 1.;
490  }
491  /* Otherwise decreases. */
492  else {
493  p->ew_stealth_timer -= dt * (p->ew_stealth / 10000. + mod);
494  if (p->ew_stealth_timer < 0.) {
495  pilot_destealth( p );
496  if (pilot_isPlayer(p))
497  player_message(_("You have been uncovered!"));
498  else if (isplayer)
499  player_message(_("You have uncovered '%s'!"), p->name);
500  }
501  }
