36 void pilot_ewUpdateStatic( Pilot *p )
37 {
38  /*
39  * Unlike the other values, heat isn't squared. The ew_hide formula is thus
40  * equivalent to: ew_base_hide * ew_mass * sqrt(ew_heat)
41  */
42  p->ew_mass = pow2( pilot_ewMass( p->solid->mass ) );
43  p->ew_heat = pilot_ewHeat( p->heat_T );
44  p->ew_asteroid = pilot_ewAsteroid( p );
45  p->ew_hide = p->ew_base_hide * p->ew_mass * p->ew_heat * p->ew_asteroid;
46  p->ew_evasion = p->ew_hide * EVASION_SCALE;
47 }

125 void pilot_updateSensorRange (void)
126 {
127  /* Adjust sensor range based on system interference. */
128  /* See: http://www.wolframalpha.com/input/?i=y+%3D+7500+%2F+%28%28x+%2B+200%29+%2F+200%29+from+x%3D0+to+1000 */
129  sensor_curRange = SENSOR_DEFAULT_RANGE / ((cur_system->interference + 200) / 200.);
130 
131  /* Speeds up calculations as we compare it against vectors later on
132  * and we want to avoid actually calculating the sqrt(). */
