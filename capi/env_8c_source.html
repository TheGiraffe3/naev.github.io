47 int nsetenv( const char *name, const char *value, int overwrite )
48 {
49 #if HAVE_DECL_SETENV

53 #if HAVE_DECL_GETENV_S
54  size_t envsize = 0;
55  int errcode = getenv_s( &envsize, NULL, 0, name );
56  if (errcode || envsize)
57  return errcode;
58 #else /* HAVE_DECL__PUTENV_S */
59  const char *envval = getenv( name );
60  if (envval != NULL)
61  return 0;
62 #endif /* HAVE_DECL__PUTENV_S */
63  }
64 #if HAVE_DECL__PUTENV_S
65  return _putenv_s(name, value);
66 #else /* HAVE_DECL__PUTENV_S */
67  char *buf;
68  asprintf( &buf, "%s=%s", name, value );
69  /* Per the standard, the string pointed to by putenv's argument becomes part of the environment
70  * ("so altering the string alters the environment" and "it is an error to call putenv() with an
71  * automatic variable as the argument".)
72  * If we're stuck using this wildly dangerous function, just leak the memory.
73  * */
74  return putenv( buf );
75 #endif /* HAVE_DECL__PUTENV_S */
76 #endif /* HAVE_DECL_SETENV */
77 }
