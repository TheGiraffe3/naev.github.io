374 static int shipL_getSlots( lua_State *L )
375 {
376  int i, j, k;
377  OutfitSlot *slot;
378  ShipOutfitSlot *sslot;
379  Ship *s = luaL_validship(L,1);
380  char *outfit_types[] = {"structure", "utility", "weapon"};
381  ShipOutfitSlot *outfit_arrays[] = {
382  s->outfit_structure,
383  s->outfit_utility,
384  s->outfit_weapon };
385 
386  lua_newtable(L);
387  k=1;
388  for (i=0; i<3 ; i++) {
389  for (j=0; j<array_size(outfit_arrays[i]) ; j++) {
390  slot = &outfit_arrays[i][j].slot;
391  sslot = &outfit_arrays[i][j];
392 
393  /* make the slot table and put it in */
394  lua_pushnumber(L, k++); /* slot table key */
395  lua_newtable(L);
396 
397  lua_pushstring(L, "type"); /* key */
398  lua_pushstring(L, outfit_types[i]); /* value */
399  lua_rawset(L, -3); /* table[key = value ]*/
400 
401  lua_pushstring(L, "size"); /* key */
402  lua_pushstring(L, slotSize( slot->size) );
403  lua_rawset(L, -3); /* table[key] = value */
404 
405  lua_pushstring(L, "property"); /* key */
406  lua_pushstring( L, sp_display(slot->spid)); /* value */
407  lua_rawset(L, -3); /* table[key] = value */
408 
409  lua_pushstring(L, "required"); /* key */
410  lua_pushboolean( L, sp_required(slot->spid)); /* value */
411  lua_rawset(L, -3); /* table[key] = value */
412 
413  lua_pushstring(L, "exclusive"); /* key */
414  lua_pushboolean( L, sp_exclusive(slot->spid)); /* value */
415  lua_rawset(L, -3); /* table[key] = value */
416 
417  if (sslot->data != NULL) {
418  lua_pushstring(L, "outfit"); /* key */
419  lua_pushoutfit(L, sslot->data); /* value*/
420  lua_rawset(L, -3); /* table[key] = value */
421  }
422 
423  lua_rawset(L, -3); /* put the slot table in */
424  }
425  }
426 
427  return 1;
428 }
