395 static int shipL_getSlots( lua_State *L )
396 {
397  int i, j, k;
398  OutfitSlot *slot;
399  ShipOutfitSlot *sslot;
400  Ship *s = luaL_validship(L,1);
401  char *outfit_types[] = {"Structure", "Utility", "Weapon"};
402  ShipOutfitSlot *outfit_arrays[] = {
403  s->outfit_structure,
404  s->outfit_utility,
405  s->outfit_weapon };
406 
407  lua_newtable(L);
408  k=1;
409  for (i=0; i<3 ; i++) {
410  for (j=0; j<array_size(outfit_arrays[i]) ; j++) {
411  slot = &outfit_arrays[i][j].slot;
412  sslot = &outfit_arrays[i][j];
413 
414  /* make the slot table and put it in */
415  lua_pushnumber(L, k++); /* slot table key */
416  lua_newtable(L);
417 
418  lua_pushstring(L, "type"); /* key */
419  lua_pushstring(L, outfit_types[i]); /* value */
420  lua_rawset(L, -3); /* table[key = value ]*/
421 
422  lua_pushstring(L, "size"); /* key */
423  lua_pushstring(L, slotSize( slot->size) );
424  lua_rawset(L, -3); /* table[key] = value */
425 
426  lua_pushstring(L, "property"); /* key */
427  lua_pushstring( L, sp_display(slot->spid)); /* value */
428  lua_rawset(L, -3); /* table[key] = value */
429 
430  lua_pushstring(L, "required"); /* key */
431  lua_pushboolean( L, sp_required(slot->spid)); /* value */
432  lua_rawset(L, -3); /* table[key] = value */
433 
434  lua_pushstring(L, "exclusive"); /* key */
435  lua_pushboolean( L, sp_exclusive(slot->spid)); /* value */
436  lua_rawset(L, -3); /* table[key] = value */
437 
438  if (sslot->data != NULL) {
439  lua_pushstring(L, "outfit"); /* key */
440  lua_pushoutfit(L, sslot->data); /* value*/
441  lua_rawset(L, -3); /* table[key] = value */
442  }
443 
444  lua_rawset(L, -3); /* put the slot table in */
445  }
446  }
447 
448  return 1;
449 }
