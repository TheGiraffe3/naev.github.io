364 static int shipL_getSlots( lua_State *L )
365 {
366  int i, j, k;
367  OutfitSlot *slot;
368  ShipOutfitSlot *sslot;
369  Ship *s = luaL_validship(L,1);
370  char *outfit_types[] = {"structure", "utility", "weapon"};
371  ShipOutfitSlot *outfit_arrays[] = {s->outfit_structure, s->outfit_utility, s->outfit_weapon};
372 
373  lua_newtable(L);
374  k=1;
375  for (i=0; i<3 ; i++) {
376  for (j=0; j<array_size(outfit_arrays[i]) ; j++) {
377  slot = &outfit_arrays[i][j].slot;
378  sslot = &outfit_arrays[i][j];
379 
380  /* make the slot table and put it in */
381  lua_pushnumber(L, k++); /* slot table key */
382  lua_newtable(L);
383 
384  lua_pushstring(L, "type"); /* key */
385  lua_pushstring(L, outfit_types[i]); /* value */
386  lua_rawset(L, -3); /* table[key = value ]*/
387 
388  lua_pushstring(L, "size"); /* key */
389  lua_pushstring(L, slotSize( slot->size) );
390  lua_rawset(L, -3); /* table[key] = value */
391 
392  lua_pushstring(L, "property"); /* key */
393  lua_pushstring( L, sp_display(slot->spid)); /* value */
394  lua_rawset(L, -3); /* table[key] = value */
395 
396  if (sslot->data != NULL) {
397  lua_pushstring(L, "outfit"); /* key */
398  lua_pushoutfit(L, sslot->data); /* value*/
399  lua_rawset(L, -3); /* table[key] = value */
400  }
401 
402  lua_rawset(L, -3); /* put the slot table in */
403  }
404  }
405 
406  return 1;
407 
408 }
