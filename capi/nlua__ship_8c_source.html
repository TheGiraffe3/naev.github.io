388 static int shipL_getSlots( lua_State *L )
389 {
390  int k;
391  const Ship *s = luaL_validship(L,1);
392  char *outfit_types[] = {"Structure", "Utility", "Weapon"};
393  const ShipOutfitSlot *outfit_arrays[] = {
394  s->outfit_structure,
395  s->outfit_utility,
396  s->outfit_weapon };
397 
398  lua_newtable(L);
399  k=1;
400  for (int i=0; i<3 ; i++) {
401  for (int j=0; j<array_size(outfit_arrays[i]) ; j++) {
402  const OutfitSlot *slot = &outfit_arrays[i][j].slot;
403  const ShipOutfitSlot *sslot = &outfit_arrays[i][j];
404 
405  /* make the slot table and put it in */
406  lua_newtable(L);
407 
408  lua_pushstring(L, "type"); /* key */
409  lua_pushstring(L, outfit_types[i]); /* value */
410  lua_rawset(L, -3); /* table[key = value ]*/
411 
412  lua_pushstring(L, "size"); /* key */
413  lua_pushstring(L, slotSize( slot->size) );
414  lua_rawset(L, -3); /* table[key] = value */
415 
416  lua_pushstring(L, "property"); /* key */
417  lua_pushstring( L, sp_display(slot->spid)); /* value */
418  lua_rawset(L, -3); /* table[key] = value */
419 
420  lua_pushstring(L, "required"); /* key */
421  lua_pushboolean( L, sp_required(slot->spid)); /* value */
422  lua_rawset(L, -3); /* table[key] = value */
423 
424  lua_pushstring(L, "exclusive"); /* key */
425  lua_pushboolean( L, sp_exclusive(slot->spid)); /* value */
426  lua_rawset(L, -3); /* table[key] = value */
427 
428  if (sslot->data != NULL) {
429  lua_pushstring(L, "outfit"); /* key */
430  lua_pushoutfit(L, sslot->data); /* value*/
431  lua_rawset(L, -3); /* table[key] = value */
432  }
433 
434  lua_rawseti(L, -2, k++); /* put the slot table in */
435  }
436  }
437 
438  return 1;
439 }
