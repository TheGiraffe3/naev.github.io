405 static int shipL_getSlots( lua_State *L )
406 {
407  int k;
408  const Ship *s = luaL_validship(L,1);
409  char *outfit_types[] = {"Structure", "Utility", "Weapon"};
410  const ShipOutfitSlot *outfit_arrays[] = {
411  s->outfit_structure,
412  s->outfit_utility,
413  s->outfit_weapon };
414 
415  lua_newtable(L);
416  k=1;
417  for (int i=0; i<3 ; i++) {
418  for (int j=0; j<array_size(outfit_arrays[i]) ; j++) {
419  const OutfitSlot *slot = &outfit_arrays[i][j].slot;
420  const ShipOutfitSlot *sslot = &outfit_arrays[i][j];
421 
422  /* make the slot table and put it in */
423  lua_newtable(L);
424 
425  lua_pushstring(L, "type"); /* key */
426  lua_pushstring(L, outfit_types[i]); /* value */
427  lua_rawset(L, -3); /* table[key = value ]*/
428 
429  lua_pushstring(L, "size"); /* key */
430  lua_pushstring(L, slotSize( slot->size) );
431  lua_rawset(L, -3); /* table[key] = value */
432 
433  lua_pushstring(L, "property"); /* key */
434  lua_pushstring( L, sp_display(slot->spid)); /* value */
435  lua_rawset(L, -3); /* table[key] = value */
436 
437  lua_pushstring(L, "required"); /* key */
438  lua_pushboolean( L, sp_required(slot->spid)); /* value */
439  lua_rawset(L, -3); /* table[key] = value */
440 
441  lua_pushstring(L, "exclusive"); /* key */
442  lua_pushboolean( L, sp_exclusive(slot->spid)); /* value */
443  lua_rawset(L, -3); /* table[key] = value */
444 
445  if (sslot->data != NULL) {
446  lua_pushstring(L, "outfit"); /* key */
447  lua_pushoutfit(L, sslot->data); /* value*/
448  lua_rawset(L, -3); /* table[key] = value */
449  }
450 
451  lua_rawseti(L, -2, k++); /* put the slot table in */
452  }
453  }
454 
455  return 1;
456 }
