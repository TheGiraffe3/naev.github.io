398 static int shipL_getSlots( lua_State *L )
399 {
400  int i, j, k;
401  const OutfitSlot *slot;
402  const ShipOutfitSlot *sslot;
403  const Ship *s = luaL_validship(L,1);
404  char *outfit_types[] = {"Structure", "Utility", "Weapon"};
405  const ShipOutfitSlot *outfit_arrays[] = {
406  s->outfit_structure,
407  s->outfit_utility,
408  s->outfit_weapon };
409 
410  lua_newtable(L);
411  k=1;
412  for (i=0; i<3 ; i++) {
413  for (j=0; j<array_size(outfit_arrays[i]) ; j++) {
414  slot = &outfit_arrays[i][j].slot;
415  sslot = &outfit_arrays[i][j];
416 
417  /* make the slot table and put it in */
418  lua_pushnumber(L, k++); /* slot table key */
419  lua_newtable(L);
420 
421  lua_pushstring(L, "type"); /* key */
422  lua_pushstring(L, outfit_types[i]); /* value */
423  lua_rawset(L, -3); /* table[key = value ]*/
424 
425  lua_pushstring(L, "size"); /* key */
426  lua_pushstring(L, slotSize( slot->size) );
427  lua_rawset(L, -3); /* table[key] = value */
428 
429  lua_pushstring(L, "property"); /* key */
430  lua_pushstring( L, sp_display(slot->spid)); /* value */
431  lua_rawset(L, -3); /* table[key] = value */
432 
433  lua_pushstring(L, "required"); /* key */
434  lua_pushboolean( L, sp_required(slot->spid)); /* value */
435  lua_rawset(L, -3); /* table[key] = value */
436 
437  lua_pushstring(L, "exclusive"); /* key */
438  lua_pushboolean( L, sp_exclusive(slot->spid)); /* value */
439  lua_rawset(L, -3); /* table[key] = value */
440 
441  if (sslot->data != NULL) {
442  lua_pushstring(L, "outfit"); /* key */
443  lua_pushoutfit(L, sslot->data); /* value*/
444  lua_rawset(L, -3); /* table[key] = value */
445  }
446 
447  lua_rawset(L, -3); /* put the slot table in */
448  }
449  }
450 
451  return 1;
452 }
