601 {

606  return;

650 static void hooks_updateDateExecute( ntime_t change )
651 {
652  int j;
653  Hook *h;
654 
655  /* Don't update without player. */
656  if ((player.p == NULL) || player_isFlag(PLAYER_CREATING))
657  return;
658 
659  /* Clear creation flags. */
660  for (h=hook_list; h!=NULL; h=h->next)
661  h->created = 0;
662 
663  /* On j=0 we increment all timers and try to run, then on j=1 we update the timers. */
664  hook_runningstack++; /* running hooks */
665  for (j=1; j>=0; j--) {
666  for (h=hook_list; h!=NULL; h=h->next) {
667  /* Not be deleting. */
668  if (h->delete)
669  continue;
670  /* Find valid date hooks. */
671  if (h->is_date == 0)
672  continue;
673  /* Don't update newly created hooks. */
674  if (h->created != 0)
675  continue;
676 
677  /* Decrement timer and check to see if should run. */
678  if (j==0)
679  h->acc += change;
680  if (h->acc < h->res)
681  continue;
682 
683  /* Run the timer hook. */
684  hook_run( h, NULL, j );
685  /* Date hooks are not deleted. */
686 
687  /* Time is modified at the end. */
688  if (j==1)
689  h->acc %= h->res; /* We'll skip all buggers. */
690  }
691  }
692  hook_runningstack--; /* not running hooks anymore */
693 
694  /* Second pass to delete. */
695  hooks_purgeList();
696 }

899 {
900  int j;
901  int run;
902  Hook *h;
903 
904  /* Don't update if player is dead. */
905  if ((player.p == NULL) || player_isFlag(PLAYER_DESTROYED))
906  return 0;
907 
908  /* Reset the current stack's ran and creation flags. */
909  for (h=hook_list; h!=NULL; h=h->next)
910  if (strcmp(stack, h->stack)==0) {
911  h->ran_once = 0;
912  h->created = 0;
913  }
914 
915  run = 0;
916  hook_runningstack++; /* running hooks */
917  for (j=1; j>=0; j--) {
918  for (h=hook_list; h!=NULL; h=h->next) {
919  /* Should be deleted. */
920  if (h->delete)
921  continue;
922  /* Don't run again. */
923  if (h->ran_once)
924  continue;
925  /* Don't update newly created hooks. */
926  if (h->created != 0)
927  continue;
928  /* Doesn't match stack. */
929  if (strcmp(stack, h->stack) != 0)
930  continue;
931 
932  /* Run hook. */
933  hook_run( h, param, j );
934  run++;
935  }
936  }
937  hook_runningstack--; /* not running hooks anymore */
938 
939  /* Check claims. */
940  if (run)
941  claim_activateAll();
942 
943  return run;
944 }
