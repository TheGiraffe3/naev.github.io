50 strcut (char *str, int begin, int len) {
51  size_t l;
52  l = strlen(str);
53 
54  if((int)l < 0 || (int)l > MAX_SAFE_INT) return -1;
55 
56  if (len < 0) len = l - begin + 1;
57  if (begin + len > (int)l) len = l - begin;
58  memmove(str + begin, str + begin + len, l - len + 1 - begin);
59

107 static char *
108 parse_slice (char *buf, char sep) {
109  char *pr, *part;
110  int plen;
111 
112  /* Find separator in buf */
113  pr = strchr(buf, sep);
114  if (pr == NULL) return NULL;
115  /* Length from separator to end of buf */
116  plen = strlen(pr);
117 
118  /* Copy from buf into new string */
119  part = (char*)calloc(plen + 1, sizeof(*part));
120  if (part == NULL) return NULL;
121  memcpy(part, pr + 1, plen);
122  /* Null terminate new string */
123  part[plen] = '\0';
124 
125  /* Terminate buf where separator was */
126  *pr = '\0';
127 
128  return part;

173 int
174 semver_parse_version (const char *str, semver_t *ver) {
175  size_t len;
176  int index, value;
177  char *slice, *next, *endptr;
178  slice = (char *) str;
179  index = 0;
180 
181  while (slice != NULL && index++ < 4) {
182  next = strchr(slice, DELIMITER[0]);
183  if (next == NULL)
184  len = strlen(slice);
185  else
186  len = next - slice;
187  if (len > SLICE_SIZE) return -1;
188 
189  /* Cast to integer and store */
190  value = strtol(slice, &endptr, 10);
191  if (endptr != next && *endptr != '\0') return -1;
192 
193  switch (index) {
194  case 1: ver->major = value; break;
195  case 2: ver->minor = value; break;
196  case 3: ver->patch = value; break;
197  }
198 
199  /* Continue with the next slice */
200  if (next == NULL)
201  slice = NULL;
202  else
203  slice = next + 1;
204  }
205 
206  return 0;
207 }

209 static int
210 compare_prerelease (char *x, char *y) {
211  char *lastx, *lasty, *xptr, *yptr, *endptr;
212  int xlen, ylen, xisnum, yisnum, xnum, ynum;
213  int xn, yn, min, res;
214  if (x == NULL && y == NULL) return 0;
215  if (y == NULL && x) return -1;
216  if (x == NULL && y) return 1;
217 
218  lastx = x;
219  lasty = y;
220  xlen = strlen(x);
221  ylen = strlen(y);
222 
223  while (1) {
224  if ((xptr = strchr(lastx, DELIMITER[0])) == NULL)
225  xptr = x + xlen;
226  if ((yptr = strchr(lasty, DELIMITER[0])) == NULL)
227  yptr = y + ylen;
228 
229  xnum = strtol(lastx, &endptr, 10);
230  xisnum = endptr == xptr ? 1 : 0;
231  ynum = strtol(lasty, &endptr, 10);
232  yisnum = endptr == yptr ? 1 : 0;
233 
234  if (xisnum && !yisnum) return -1;
235  if (!xisnum && yisnum) return 1;
236 
237  if (xisnum && yisnum) {
238  /* Numerical comparison */
239  if (xnum != ynum) return xnum < ynum ? -1 : 1;
240  } else {
241  /* String comparison */
242  xn = xptr - lastx;
243  yn = yptr - lasty;
244  min = xn < yn ? xn : yn;
245  if ((res = strncmp(lastx, lasty, min))) return res < 0 ? -1 : 1;
246  if (xn != yn) return xn < yn ? -1 : 1;
247  }
248 
249  lastx = xptr + 1;
250  lasty = yptr + 1;
251  if (lastx == x + xlen + 1 && lasty == y + ylen + 1) break;
252  if (lastx == x + xlen + 1) return -1;
253  if (lasty == y + ylen + 1) return 1;
254  }
255 
256  return 0;
257 }

444 int
445 semver_satisfies (semver_t x, semver_t y, const char *op) {
446  int first, second;
447  /* Extract the comparison operator */
448  first = op[0];
449  second = op[1];
450 
451  /* Caret operator */
452  if (first == SYMBOL_CF)
453  return semver_satisfies_caret(x, y);
454 
455  /* Tilde operator */
456  if (first == SYMBOL_TF)
457  return semver_satisfies_patch(x, y);
458 
459  /* Strict equality */
460  if (first == SYMBOL_EQ)
461  return semver_eq(x, y);
462 
463  /* Greater than or equal comparison */
464  if (first == SYMBOL_GT) {
465  if (second == SYMBOL_EQ) {
466  return semver_gte(x, y);
467  }
468  return semver_gt(x, y);
469  }
470 
471  /* Lower than or equal comparison */
472  if (first == SYMBOL_LT) {
473  if (second == SYMBOL_EQ) {
474  return semver_lte(x, y);
475  }
476  return semver_lt(x, y);
477  }
478 
479  return 0;
480 }
