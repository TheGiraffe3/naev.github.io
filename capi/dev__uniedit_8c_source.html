445 static int uniedit_mouse( unsigned int wid, SDL_Event* event, double mx, double my,
446  double w, double h, double rx, double ry, void *data )
447 {
448  (void) wid;
449  (void) data;
450  int i;
451  double x,y, t;
452  StarSystem *sys;
453  SDL_Keymod mod;
454 
455  t = 15.*15.; /* threshold */
456 
457  /* Handle modifiers. */
458  mod = SDL_GetModState();
459 
460  switch (event->type) {
461 
462  case SDL_MOUSEWHEEL:
463  /* Must be in bounds. */
464  if ((mx < 0.) || (mx > w) || (my < 0.) || (my > h))
465  return 0;
466 
467  if (event->wheel.y > 0)
468  uniedit_buttonZoom( 0, "btnZoomIn" );
469  else
470  uniedit_buttonZoom( 0, "btnZoomOut" );
471 
472  return 1;
473 
474  case SDL_MOUSEBUTTONDOWN:
475  /* Must be in bounds. */
476  if ((mx < 0.) || (mx > w) || (my < 0.) || (my > h))
477  return 0;
478 
479  /* selecting star system */
480  else {
481  mx -= w/2 - uniedit_xpos;
482  my -= h/2 - uniedit_ypos;
483 
484  if (uniedit_mode == UNIEDIT_NEWSYS) {
485  uniedit_newSys( mx, my );
486  uniedit_mode = UNIEDIT_DEFAULT;
487  return 1;
488  }
489 
490  for (i=0; i<array_size(systems_stack); i++) {
491  sys = system_getIndex( i );
492 
493  /* get position */
494  x = sys->pos.x * uniedit_zoom;
495  y = sys->pos.y * uniedit_zoom;
496 
497  if ((pow2(mx-x)+pow2(my-y)) < t) {
498 
499  /* Try to find in selected systems - begin drag move. */
500  for (i=0; i<uniedit_nsys; i++) {
501  /* Must match. */
502  if (uniedit_sys[i] != sys)
503  continue;
504 
505  /* Detect double click to open system. */
506  if ((SDL_GetTicks() - uniedit_dragTime < UNIEDIT_DRAG_THRESHOLD*2)
507  && (uniedit_moved < UNIEDIT_MOVE_THRESHOLD)) {
508  if (uniedit_nsys == 1) {
509  sysedit_open( uniedit_sys[0] );
510  return 1;
511  }
512  }
513 
514  /* Handle normal click. */
515  if (uniedit_mode == UNIEDIT_DEFAULT) {
516  uniedit_dragSys = 1;
517  uniedit_tsys = sys;
518 
519  /* Check modifier. */
520  if (mod & (KMOD_LCTRL | KMOD_RCTRL))
521  uniedit_tadd = 0;
522  else
523  uniedit_tadd = -1;
524  uniedit_dragTime = SDL_GetTicks();
525  uniedit_moved = 0;
526  }
527  return 1;
528  }
529 
530  if (uniedit_mode == UNIEDIT_DEFAULT) {
531  /* Add the system if not selected. */
532  if (mod & (KMOD_LCTRL | KMOD_RCTRL))
533  uniedit_selectAdd( sys );
534  else {
535  uniedit_deselect();
536  uniedit_selectAdd( sys );
537  }
538  uniedit_tsys = NULL;
539 
540  /* Start dragging anyway. */
541  uniedit_dragSys = 1;
542  uniedit_dragTime = SDL_GetTicks();
543  uniedit_moved = 0;
544  }
545  else if (uniedit_mode == UNIEDIT_JUMP) {
546  uniedit_toggleJump( sys );
547  uniedit_mode = UNIEDIT_DEFAULT;
548  }
549  return 1;
550  }
551  }
552 
553  /* Start dragging. */
554  if ((uniedit_mode == UNIEDIT_DEFAULT) && !(mod & (KMOD_LCTRL | KMOD_RCTRL))) {
555  uniedit_drag = 1;
556  uniedit_dragTime = SDL_GetTicks();
557  uniedit_moved = 0;
558  uniedit_tsys = NULL;
559  }
560  return 1;
561  }
562  break;
563 
564  case SDL_MOUSEBUTTONUP:
565  if (uniedit_drag) {
566  if ((SDL_GetTicks() - uniedit_dragTime < UNIEDIT_DRAG_THRESHOLD) && (uniedit_moved < UNIEDIT_MOVE_THRESHOLD)) {
567  if (uniedit_tsys == NULL)
568  uniedit_deselect();
569  else
570  uniedit_selectAdd( uniedit_tsys );
571  }
572  uniedit_drag = 0;
573  }
574  if (uniedit_dragSys) {
575  if ((SDL_GetTicks() - uniedit_dragTime < UNIEDIT_DRAG_THRESHOLD) &&
576  (uniedit_moved < UNIEDIT_MOVE_THRESHOLD) && (uniedit_tsys != NULL)) {
577  if (uniedit_tadd == 0)
578  uniedit_selectRm( uniedit_tsys );
579  else {
580  uniedit_deselect();
581  uniedit_selectAdd( uniedit_tsys );
582  }
583  }
584  uniedit_dragSys = 0;
585  if (conf.devautosave)
586  for (i=0; i<uniedit_nsys; i++)
587  dsys_saveSystem(uniedit_sys[i]);
588  }
589  break;
590 
591  case SDL_MOUSEMOTION:
592  /* Update mouse positions. */
593  uniedit_mx = mx;
594  uniedit_my = my;
595 
596  /* Handle dragging. */
597  if (uniedit_drag) {
598  /* axis is inverted */
599  uniedit_xpos -= rx;
600  uniedit_ypos += ry;
601 
602  /* Update mouse movement. */
603  uniedit_moved += ABS(rx) + ABS(ry);
604  }
605  else if (uniedit_dragSys && (uniedit_nsys > 0)) {
606  if ((uniedit_moved > UNIEDIT_MOVE_THRESHOLD) || (SDL_GetTicks() - uniedit_dragTime > UNIEDIT_DRAG_THRESHOLD)) {
607  for (i=0; i<uniedit_nsys; i++) {
608  uniedit_sys[i]->pos.x += rx / uniedit_zoom;
609  uniedit_sys[i]->pos.y -= ry / uniedit_zoom;
610  }
611  }
612 
613  /* Update mouse movement. */
614  uniedit_moved += ABS(rx) + ABS(ry);
615  }
616  break;
617  }
618 
619  return 0;
620 }
