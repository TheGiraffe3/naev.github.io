1235  o = s->outfits[j]->outfit;
1236  if (o==NULL)
1237  continue;

1259  eps += outfit_energy(o);
1260 
1261  continue;
1262  default:
1263  continue;

1269  dmg = outfit_damage(o->u.lau.ammo);
1270  else
1271  dmg = outfit_damage(o);
1272  dps += shots * mod_damage * dmg->damage;

1279  if (dps > 0.)
