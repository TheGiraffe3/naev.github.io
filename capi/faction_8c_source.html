654 static void faction_modPlayerLua( int f, double mod, const char *source, int secondary )
655 {
656  Faction *faction;
657  double old, delta;
658  HookParam hparam[3];
659 
660  faction = &faction_stack[f];
661 
662  /* Make sure it's not static. */
663  if (faction_isFlag(faction, FACTION_STATIC))
664  return;
665 
666  old = faction->player;
667 
668  if (faction->env == LUA_NOREF)
669  faction->player += mod;
670  else {
671 
672  /* Set up the function:
673  * faction_hit( current, amount, source, secondary ) */
674  nlua_getenv( faction->env, "faction_hit" );
675  lua_pushnumber( naevL, faction->player );
676  lua_pushnumber( naevL, mod );
677  lua_pushstring( naevL, source );
678  lua_pushboolean( naevL, secondary );
679 
680  /* Call function. */
681  if (nlua_pcall( faction->env, 4, 1 )) { /* An error occurred. */
682  WARN(_("Faction '%s': %s"), faction->name, lua_tostring(naevL,-1));
683  lua_pop( naevL, 1 );
684  return;
685  }
686 
687  /* Parse return. */
688  if (!lua_isnumber( naevL, -1 ))
689  WARN( _("Lua script for faction '%s' did not return a number from 'faction_hit(...)'."), faction->name );
690  else
691  faction->player = lua_tonumber( naevL, -1 );
692  lua_pop( naevL, 1 );
693  }
694 
695  /* Sanitize just in case. */
696  faction_sanitizePlayer( faction );
697 
698  /* Run hook if necessary. */
699  delta = faction->player - old;
700  if (FABS(delta) > 1e-10) {
701  hparam[0].type = HOOK_PARAM_FACTION;
702  hparam[0].u.lf = f;
703  hparam[1].type = HOOK_PARAM_NUMBER;
704  hparam[1].u.num = delta;
705  hparam[2].type = HOOK_PARAM_SENTINEL;
706  hooks_runParam( "standing", hparam );
707 
708  /* Tell space the faction changed. */
709  space_factionChange();
710  }
711 }
