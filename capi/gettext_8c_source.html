52 
53  setlocale( LC_ALL, "" );
54  /* If we don't disable LC_NUMERIC, lots of stuff blows up because 1,000 can be interpreted as
55  * 1.0 in certain languages. */
56  setlocale( LC_NUMERIC, "C" ); /* Disable numeric locale part. */
57 
58  for (i = 0; i < sizeof(env_vars)/sizeof(env_vars[0]); i++) {
59  language = getenv( env_vars[i] );
60  /* Extract languages codes, e.g. "en_US:de_DE" -> "en:de" */
61  j = 0;
62  while (language != NULL && *language != 0 && j < sizeof(gettext_systemLanguage)-1) {
63  if (isalpha(*language) || *language == ':')
64  gettext_systemLanguage[j++] = *language++;
65  else
66  language = strchr(language, ':');
67  gettext_systemLanguage[j] = 0;
68  }
69  if (j > 0)
70  return; /* The first env var with language settings wins. */
71  }
