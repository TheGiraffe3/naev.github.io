63 static void ndata_testVersion (void)
64 {
65  size_t i, size;
66  char *buf, cbuf[PATH_MAX];
67  int diff;
68 
69  if (!ndata_found())
70  ERR( _("Unable to find game data. You may need to install, specify a datapath, or run using naev.sh (if developing).") );
71 
72  /* Parse version. */
73  buf = ndata_read( "VERSION", &size );
74  for (i=0; i<MIN(size,PATH_MAX-1); i++)
75  cbuf[i] = buf[i];
76  cbuf[MIN(size-1,PATH_MAX-1)] = '\0';
77  diff = naev_versionCompare( cbuf );
78  if (diff != 0) {
79  WARN( _("ndata version inconsistency with this version of Naev!") );
80  WARN( _("Expected ndata version %s got %s."), VERSION, cbuf );
81  if (ABS(diff) > 2)
82  ERR( _("Please get a compatible ndata version!") );
83  if (ABS(diff) > 1)
84  WARN( _("Naev will probably crash now as the versions are probably not compatible.") );
85  }
86  free( buf );
87 }

157 void* ndata_read( const char* path, size_t *filesize )
158 {
159  char *buf;
160  PHYSFS_file *file;
161  PHYSFS_sint64 len, n;
162  size_t pos;
163  PHYSFS_Stat path_stat;
164 
165  if (!PHYSFS_stat( path, &path_stat )) {
166  WARN( _( "Error occurred while opening '%s': %s" ), path,
167  PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
168  *filesize = 0;
169  return NULL;
170  }
171  if (path_stat.filetype != PHYSFS_FILETYPE_REGULAR) {
172  WARN( _( "Error occurred while opening '%s': It is not a regular file" ), path );
173  *filesize = 0;
174  return NULL;
175  }
176 
177  /* Open file. */
178  file = PHYSFS_openRead( path );
179  if ( file == NULL ) {
180  WARN( _( "Error occurred while opening '%s': %s" ), path,
181  PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
182  *filesize = 0;
183  return NULL;
184  }
185 
186  /* Get file size. TODO: Don't assume this is always possible? */
187  len = PHYSFS_fileLength( file );
188  if ( len == -1 ) {
189  WARN( _( "Error occurred while seeking '%s': %s" ), path,
190  PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
191  PHYSFS_close( file );
192  *filesize = 0;
193  return NULL;
194  }
195 
196  /* Allocate buffer. */
197  buf = malloc( len+1 );
198  if (buf == NULL) {
199  WARN(_("Out of Memory"));
200  PHYSFS_close( file );
201  *filesize = 0;
202  return NULL;
203  }
204  buf[len] = '\0';
205 
206  /* Read the file. */
207  n = 0;
208  while ( n < len ) {
209  pos = PHYSFS_readBytes( file, &buf[ n ], len - n );
210  if ( pos <= 0 ) {
211  WARN( _( "Error occurred while reading '%s': %s" ), path,
212  PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
213  PHYSFS_close( file );
214  *filesize = 0;
215  free(buf);
216  return NULL;
217  }
218  n += pos;
219  }
220 
221  /* Close the file. */
222  PHYSFS_close(file);
223 
224  *filesize = len;
225  return buf;
226 }

236 char **ndata_listRecursive( const char *path )
237 {
238  char **files;
239  int i;
240 
241  files = array_create( char * );
242  PHYSFS_enumerate( path, ndata_enumerateCallback, &files );
243  /* Ensure unique. PhysicsFS can enumerate a path twice if it's in multiple components of a union. */
244  qsort( files, array_size(files), sizeof(char*), strsort );
245  for (i=0; i+1<array_size(files); i++)
246  if (strcmp(files[i], files[i+1]) == 0) {
247  free( files[i] );
248  array_erase( &files, &files[i], &files[i+1] );
249  i--; /* We're not done checking for dups of files[i]. */
250  }
251  return files;
252 }

257 static int ndata_enumerateCallback( void* data, const char* origdir, const char* fname )
258 {
259  char *path;
260  const char *fmt;
261  size_t dir_len;
262  PHYSFS_Stat stat;
263 
264  dir_len = strlen( origdir );
265  fmt = dir_len && origdir[dir_len-1]=='/' ? "%s%s" : "%s/%s";
266  asprintf( &path, fmt, origdir, fname );
267  if (!PHYSFS_stat( path, &stat )) {
268  WARN( _("PhysicsFS: Cannot stat %s: %s"), path,
269  PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
270  free( path );
271  }
272  else if (stat.filetype == PHYSFS_FILETYPE_REGULAR)
273  array_push_back( (char***)data, path );
274  else if (stat.filetype == PHYSFS_FILETYPE_DIRECTORY ) {
275  PHYSFS_enumerate( path, ndata_enumerateCallback, data );
276  free( path );
277  }
278  else
279  free( path );
280  return PHYSFS_ENUM_OK;
281 }

313 int ndata_copyIfExists( const char* file1, const char* file2 )
314 {
315  PHYSFS_File *f_in, *f_out;
316  char buf[ 8*1024 ];
317  PHYSFS_sint64 lr, lw;
318 
319  if (file1 == NULL)
320  return -1;
321 
322  /* Check if input file exists */
323  if (!PHYSFS_exists(file1))
324  return 0;
325 
326  /* Open files. */
327  f_in = PHYSFS_openRead( file1 );
328  f_out = PHYSFS_openWrite( file2 );
329  if ((f_in==NULL) || (f_out==NULL)) {
330  WARN( _("Failure to copy '%s' to '%s': %s"), file1, file2, PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
331  if (f_in!=NULL)
332  PHYSFS_close(f_in);
333  return -1;
334  }
335 
336  /* Copy data over. */
337  do {
338  lr = PHYSFS_readBytes( f_in, buf, sizeof(buf) );
339  if (lr == -1)
340  goto err;
341  else if (!lr) {
342  if (PHYSFS_eof( f_in ))
343  break;
344  goto err;
345  }
346 
347  lw = PHYSFS_writeBytes( f_out, buf, lr );
348  if (lr != lw)
349  goto err;
350  } while (lr > 0);
351 
352  /* Close files. */
353  PHYSFS_close( f_in );
354  PHYSFS_close( f_out );
355 
356  return 0;
357 
358 err:
359  WARN( _("Failure to copy '%s' to '%s': %s"), file1, file2, PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
360  PHYSFS_close( f_in );
361  PHYSFS_close( f_out );
362 
363  return -1;
364 }
