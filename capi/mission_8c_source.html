221 static int mission_meetReq( int mission, int faction,
222  const char* planet, const char* sysname )
223 {
224  MissionData* misn = mission_get( mission );
225  if (misn == NULL) /* In case it doesn't exist */
226  return 0;
227 
228  /* If planet, must match planet. */
229  if ((misn->avail.planet != NULL) && (strcmp(misn->avail.planet,planet)!=0))
230  return 0;
231 
232  /* If system, must match system. */
233  if ((misn->avail.system != NULL) && (strcmp(misn->avail.system,sysname)!=0))
234  return 0;
235 
236  /* Match faction. */
237  if ((faction >= 0) && !mission_matchFaction(misn,faction))
238  return 0;
239 
240  /* Must not be already done or running if unique. */
241  if (mis_isFlag(misn,MISSION_UNIQUE) &&
242  (player_missionAlreadyDone(mission) ||
243  mission_alreadyRunning(misn)))
244  return 0;
245 
246  /* Must meet Lua condition. */
247  if (misn->avail.cond != NULL) {
248  int c = cond_check(misn->avail.cond);
249  if (c < 0) {
250  WARN(_("Conditional for mission '%s' failed to run"), misn->name);
251  return 0;
252  }
253  else if (!c)
254  return 0;
255  }
256 
257  /* Must meet previous mission requirements. */
258  if ((misn->avail.done != NULL) &&
259  (player_missionAlreadyDone( mission_getID(misn->avail.done) ) == 0))
260  return 0;
261 
262  return 1;
263 }

755 static int mission_compare( const void* arg1, const void* arg2 )
756 {
757  Mission *m1, *m2;
758

760  m1 = (Mission*) arg1;
761  m2 = (Mission*) arg2;
762 
763  /* Check priority - lower is more important. */
764  if (m1->data->avail.priority < m2->data->avail.priority)
765  return -1;
766  else if (m1->data->avail.priority > m2->data->avail.priority)
767  return +1;

775  return strcmp( m1->title, m2->title );
776 
777  /* Tied. */
778  return strcmp(m1->data->name, m2->data->name);

792 Mission* missions_genList( int *n, int faction,
793  const char* planet, const char* sysname, int loc )
794 {
795  int m, alloced;
796  int rep;
797  Mission* tmp;
798 
799  /* Find available missions. */
800  tmp = NULL;
801  m = 0;
802  alloced = 0;
803  for (int i=0; i<array_size(mission_stack); i++) {
804  MissionData *misn = &mission_stack[i];
805  if (misn->avail.loc == loc) {
806  double chance;
807 
808  /* Must meet requirements. */
809  if (!mission_meetReq(i, faction, planet, sysname))
810  continue;
811 
812  /* Must hit chance. */
813  chance = (double)(misn->avail.chance % 100)/100.;
814  if (chance == 0.) /* We want to consider 100 -> 100% not 0% */
815  chance = 1.;
816  rep = MAX(1, misn->avail.chance / 100);
817 
818  for (int j=0; j<rep; j++) /* random chance of rep appearances */
819  if (RNGF() < chance) {
820  m++;
821  /* Extra allocation. */
822  if (m > alloced) {
823  if (alloced == 0)
824  alloced = 32;
825  else
826  alloced *= 2;
827  tmp = realloc( tmp, sizeof(Mission) * alloced );
828  }
829  /* Initialize the mission. */
830  if (mission_init( &tmp[m-1], misn, 1, 1, NULL ))
831  m--;
832  }
833  }
834  }
835 
836  /* Sort. */
837  if (tmp != NULL) {
838  qsort( tmp, m, sizeof(Mission), mission_compare );
839  (*n) = m;
840  }
841  else
842  (*n) = 0;
843 
844  return tmp;
845 }

1003 static int mission_parseFile( const char* file )
1004 {
1005  xmlDocPtr doc;
1006  xmlNodePtr node;
1007  size_t bufsize;
1008  char *filebuf;
1009  const char *pos, *start_pos;
1010  MissionData *temp;
1011 
1012  /* Load string. */
1013  filebuf = ndata_read( file, &bufsize );
1014  if (filebuf == NULL) {
1015  WARN(_("Unable to read data from '%s'"), file);
1016  return -1;
1017  }
1018 
1019  /* Skip if no XML. */
1020  pos = strnstr( filebuf, "</mission>", bufsize );
1021  if (pos==NULL) {
1022  pos = strnstr( filebuf, "function create", bufsize );
1023  if ((pos != NULL) && !strncmp(pos,"--common",bufsize))
1024  WARN(_("Mission '%s' has create function but no XML header!"), file);
1025  free(filebuf);
1026  return 0;
1027  }
1028 
1029  /* Separate XML header and Lua. */
1030  start_pos = strnstr( filebuf, "<?xml ", bufsize );
1031  pos = strnstr( filebuf, "--]]", bufsize );
1032  if (pos == NULL || start_pos == NULL) {
1033  WARN(_("Mission file '%s' has missing XML header!"), file);
1034  return -1;
1035  }
1036 
1037  /* Parse the header. */
1038  doc = xmlParseMemory( start_pos, pos-start_pos);
1039  if (doc == NULL) {
1040  WARN(_("Unable to parse document XML header for Mission '%s'"), file);
1041  return -1;
1042  }
1043 
1044  node = doc->xmlChildrenNode;
1045  if (!xml_isNode(node,XML_MISSION_TAG)) {
1046  ERR( _("Malformed XML header for '%s' mission: missing root element '%s'"), file, XML_MISSION_TAG );
1047  return -1;
1048  }
1049 
1050  temp = &array_grow(&mission_stack);
1051  mission_parseXML( temp, node );
1052  temp->lua = strdup(filebuf);
1053  temp->sourcefile = strdup(file);
1054 
1055 #ifdef DEBUGGING
1056  /* Check to see if syntax is valid. */
1057  int ret = luaL_loadbuffer(naevL, temp->lua, strlen(temp->lua), temp->name );
1058  if (ret == LUA_ERRSYNTAX) {
1059  WARN(_("Mission Lua '%s' syntax error: %s"),
1060  file, lua_tostring(naevL,-1) );
1061  } else {
1062  lua_pop(naevL, 1);
1063  }
1064 #endif /* DEBUGGING */
1065 
1066  /* Clean up. */
1067  xmlFreeDoc(doc);
1068  free(filebuf);
1069 
1070  return 0;
