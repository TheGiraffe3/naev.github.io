215 static int linoptL_setcol( lua_State *L )
216 {
217  LuaLinOpt_t *lp = luaL_checklinopt(L,1);
218  int idx = luaL_checkinteger(L,2);
219  const char *name = luaL_checkstring(L,3);
220  double coef = luaL_checknumber(L,4);
221  const char *skind = luaL_optstring(L,5,"real");
222  int haslb, hasub, type, kind;
223  double lb, ub;
224 
225  /* glpk stuff */
226  glp_set_col_name( lp->prob, idx, name );
227  glp_set_obj_coef( lp->prob, idx, coef );
228 
229  /* Determine bounds. */
230  haslb = !lua_isnoneornil(L,6);
231  hasub = !lua_isnoneornil(L,7);
232  lb = luaL_optnumber(L,6,0.0);
233  ub = luaL_optnumber(L,7,0.0);
234  if (haslb && hasub)
235  type = GLP_DB;
236  else if (haslb)
237  type = GLP_LO;
238  else if (hasub)
239  type = GLP_UP;
240  else
241  type = GLP_FR;
242  glp_set_col_bnds( lp->prob, idx, type, lb, ub );
243 
244  /* Get kind. */
245  if (strcmp(skind,"real")==0)
246  kind = GLP_CV;
247  else if (strcmp(skind,"integer")==0)
248  kind = GLP_IV;
249  else if (strcmp(skind,"binary")==0)
250  kind = GLP_BV;
251  else
252  NLUA_ERROR(L,_("Unknown column kind '%s'!"), skind);
253  glp_set_col_kind( lp->prob, idx, kind );
254 
255  return 0;
256 }

306 static int linoptL_loadmatrix( lua_State *L )
307 {
308  size_t i, n;
309  int *ia, *ja;
310  double *ar;
311  LuaLinOpt_t *lp = luaL_checklinopt(L,1);
312  luaL_checktype(L, 2, LUA_TTABLE);
313  luaL_checktype(L, 3, LUA_TTABLE);
314  luaL_checktype(L, 4, LUA_TTABLE);
315 
316  /* Make sure size is ok. */
317  n = lua_objlen(L,2);
318 #if DEBUGGING
319  if ((n != lua_objlen(L,3)) || (n != lua_objlen(L,4)))
320  NLUA_ERROR(L, _("Table lengths don't match!"));
321 #endif /* DEBUGGING */
322 
323  /* Load everything from tables, has to be 1-index based. */
324  ia = calloc( n+1, sizeof(int) );
325  ja = calloc( n+1, sizeof(int) );
326  ar = calloc( n+1, sizeof(double) );
327  for (i=1; i<=n; i++) {
328  lua_rawgeti(L, 2, i);
329  lua_rawgeti(L, 3, i);
330  lua_rawgeti(L, 4, i);
331 #if DEBUGGING
332  ia[i] = luaL_checkinteger(L,-3);
333  ja[i] = luaL_checkinteger(L,-2);
334  ar[i] = luaL_checknumber(L,-1);
335 #else /* DEBUGGING */
336  ia[i] = lua_tointeger(L,-3);
337  ja[i] = lua_tointeger(L,-2);
338  ar[i] = lua_tonumber(L,-1);
339 #endif /* DEBUGGING */
340  lua_pop(L,3);
341  }
342 
343  /* Set up the matrix. */
344  glp_load_matrix( lp->prob, n, ia, ja, ar );
345 
346  /* Clean up. */
347  free(ia);
348  free(ja);
349  free(ar);
350  return 0;
351 }

361 static int linoptL_solve( lua_State *L )
362 {
363  LuaLinOpt_t *lp = luaL_checklinopt(L,1);
364  double z;
365  //const char *name;
366  int i, ismip;
367 
368  /* Parameters. */
369  ismip = (glp_get_num_int( lp->prob ) > 0);
370 
371  /* Optimization. */
372  if (ismip) {
373  glp_iocp parm;
374  glp_init_iocp(&parm);
375  parm.presolve = GLP_ON;
376  //parm.msg_lev = GLP_MSG_ERR;
377  glp_intopt( lp->prob, &parm );
378  /* TODO handle errors. */
379  z = glp_mip_obj_val( lp->prob );
380  }
381  else {
382  glp_smcp parm;
383  glp_init_smcp(&parm);
384  //parm.msg_lev = GLP_MSG_ERR;
385  glp_simplex( lp->prob, &parm );
386  /* TODO handle errors. */
387  z = glp_get_obj_val( lp->prob );
388  }
389 
390  /* Output function value. */
391  lua_pushnumber(L,z);
392 
393  /* Go over variables and store them. */
394  lua_newtable(L); /* t */
395  for (i=1; i<=lp->ncols; i++) {
396  if (ismip)
397  z = glp_mip_col_val( lp->prob, i );
398  else
399  z = glp_get_col_prim( lp->prob, i );
400  lua_pushnumber( L, z ); /* t, z */
401  lua_rawseti( L, -2, i ); /* t */
402  }
403 
404  /* Go over constraints and store them. */
405  lua_newtable(L); /* t */
406  for (i=1; i<=lp->nrows; i++) {
407  if (ismip)
408  z = glp_mip_row_val( lp->prob, i );
409  else
410  z = glp_get_row_prim( lp->prob, i );
411  lua_pushnumber( L, z ); /* t, z */
412  lua_rawseti( L, -2, i ); /* t */
413  }
414 
415  return 3;
416 }
