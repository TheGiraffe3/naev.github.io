94 static void omsg_setMsg( omsg_t *omsg, const char *msg )
95 {
96  int l, n, s, m;
97  glFont *font;
98 
99  /* Clean up after old stuff. */
100  omsg_free( omsg );
101 
102  /* Create data. */
103  l = strlen( msg );
104  if (l==0)
105  return;
106  font = omsg_getFont( omsg->font );
107  /* First pass size. */
108  n = 0;
109  m = 0;
110  while (n < l) {
111  s = gl_printWidthForText( font, &msg[n], omsg_center_w, NULL );
112  n += s;
113  if ((msg[n] == '\n') || (msg[n] == ' '))
114  n++; /* Skip "empty char". */
115  m++;
116  }
117 
118  /* Avoid zero-length malloc. */
119  if (m == 0)
120  return;
121 
122  /* Second pass allocate. */
123  omsg->msg = malloc( m * sizeof(char*) );
124  omsg->nlines = m;
125  n = 0;
126  m = 0;
127  while (n < l) {
128  s = gl_printWidthForText( font, &msg[n], omsg_center_w, NULL );
129  omsg->msg[m] = strndup( &msg[n], s );
130  m++;
131  n += s;
132  if ((msg[n] == '\n') || (msg[n] == ' '))
133  n++; /* Skip "empty char". */
134  }
135 }

210 void omsg_render( double dt )
211 {
212  int i, j;
213  double x, y;
214  omsg_t *omsg;
215  glFont *font;
216  glColour col;
217 
218  /* Case nothing to do. */
219  if (omsg_array == NULL)
220  return;
221 
222  /* Center. */
223  x = omsg_center_x - omsg_center_w/2.;
224  y = omsg_center_y;
225 
226  /* Render. */
227  for (i=0; i<array_size(omsg_array); i++) {
228  omsg = &omsg_array[i];
229 
230  /* Render. */
231  font = omsg_getFont( omsg->font );
232  col = omsg->col;
233  if (omsg->duration < 1.)
234  col.a = omsg->duration;
235  gl_printRestoreClear();
236  for (j=0; j<omsg->nlines; j++) {
237  y -= font->h * 1.5;
238  gl_printRestoreLast();
239  gl_printMidRaw( font, omsg_center_w, x, y, &col, -1., omsg->msg[j] );
240  }
241 
242  /* Check if time to erase. */
243  omsg->duration -= dt;
244  if (omsg->duration < 0.) {
245  omsg_free( omsg );
246  array_erase( &omsg_array, &omsg[0], &omsg[1] );
247  i--;
248  continue;
249  }
250  }
