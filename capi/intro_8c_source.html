75 static int intro_load( const char *text )
76 {
77  size_t intro_size;
78  char *cur_line, *rest_of_file;
79 
80  has_side_gfx = 0;
81 
82  /* Load text. */
83  intro_buf = ndata_read( text, &intro_size );
84 
85  /* Create intro font. */
86  gl_fontInit( &intro_font, _(FONT_MONOSPACE_PATH), conf.font_size_intro, FONT_PATH_PREFIX, 0 );
87 
88  intro_cmds = array_create( intro_cmd_t );
89  rest_of_file = intro_buf;
90  while (rest_of_file) {
91  cur_line = rest_of_file;
92  rest_of_file = strchr(cur_line, '\n');
93  /* If there's a next line, split the string and point rest_of_file to it. */
94  if (rest_of_file != NULL) {
95  /* Check for CRLF endings -- if present, zero both parts. */
96  if (rest_of_file > cur_line && *(rest_of_file-1) == '\r')
97  *(rest_of_file-1) = '\0';
98  *rest_of_file++ = '\0';
99  }
100 
101  if (strncmp( cur_line, "[fadein ", 8 ) == 0) {
102  array_grow( &intro_cmds ).opcode = INTRO_FADEIN;
103  array_back( intro_cmds ).operand = &cur_line[8];
104  /* Zap the closing square-bracket. */
105  cur_line[strlen( cur_line ) - 1] = '\0';
106  /* Mark that there are graphics. */
107  has_side_gfx = 1;
108  }
109  else if (strncmp( cur_line, "[fadeout]", 9 ) == 0)
110  array_grow( &intro_cmds ).opcode = INTRO_FADEOUT;
111  else {
112  array_grow( &intro_cmds ).opcode = INTRO_TEXT;
113  /* Translate the line if it's not the empty string. */
114  array_back( intro_cmds ).operand = (cur_line[0] == '\0' ? cur_line : _(cur_line));
115  }
116  }
117 
118  return 0;
119 }

173 static void intro_fade_image_in( intro_img_t *side, intro_img_t *transition,
174  const char *img_file )
175 {
176  if (NULL == side->tex)
177  load_image( side, img_file ); /* Simple fade-in. */
178  else {
179  /*
180  * Transition or on-deck. The difference is whether one image is
181  * replacing the other (transition), or whether the first must fade out
182  * completely before the second comes in (on-deck).
183  *
184  * We can determine which is the case by seeing whether [fadeout] has been
185  * called. I.e., is side->fade_rate < 0?
186  */
187  if (NULL != transition->tex) {
188  /* Scrolling is happening faster than fading... */
189  WARN( _("Intro scrolling too fast!") );
190  gl_freeTexture( transition->tex );
191  }
192  load_image( transition, img_file );
193  if (side->fade_rate < 0.0)
194  transition->fade_rate = 0.0; /* put an image on deck. */
195  else {
196  /* transition. */
197  side->fade_rate = -0.1; /* begin fading out. */
198  side->c.a = 0.99;
199  }
200  }
201 }

209 static void intro_event_handler( int *stop, double *offset, double *vel )
210 {
211  SDL_Event event; /* user key-press, mouse-push, etc. */
212  while (SDL_PollEvent(&event)) {
213  if (event.type == SDL_WINDOWEVENT &&
214  event.window.event == SDL_WINDOWEVENT_RESIZED) {
215  naev_resize();
216  continue;
217  }
218 
219  if (event.type != SDL_KEYDOWN)
220  continue;
221 
222  /* Escape skips directly. */
223  if (event.key.keysym.sym == SDLK_ESCAPE) {
224  *stop = 1;
225  break;
226  }
227 
228  /* Slow down the text. */
229  else if (event.key.keysym.sym == SDLK_UP) {
230  *vel -= 8.0;
231  if (*vel < 0.0)
232  *vel = 0.0;
233  }
234 
235  /* Speed up the text. */
236  else if (event.key.keysym.sym == SDLK_DOWN) {
237  *vel += 8.0;
238  if (*vel > 100.0)
239  *vel = 100.0;
240  }
241 
242  /* Jump down. */
243  else if ((event.key.keysym.sym == SDLK_SPACE) ||
244  (event.key.keysym.sym == SDLK_RETURN))
245  *offset += 100;
246 
247  /* Jump up. */
248  else if (event.key.keysym.sym == SDLK_BACKSPACE)
249  *offset -= 100;
250 
251  /* User is clearly flailing on keyboard. */
252  else
253  *vel = 16.;
254  }
255 }

297 int intro_display( const char *text, const char *mus )
298 {
299  double offset; /* distance from bottom of the top line. */
300  double line_height; /* # pixels per line. */
301  int lines_per_screen; /* max appearing lines on the screen. */
302  char ** sb_arr; /* array of lines to render. */
303  int sb_index; /* Position in the line array. */
304  double vel = 16.; /* velocity: speed of text. */
305  int stop = 0; /* stop the intro. */
306  unsigned int tcur, tlast, tlag;/* timers. */
307  double delta; /* time diff from last render to this one. */
308  int cmd_index = 0; /* index into the big list of intro lines. */
309  intro_img_t side_image; /* image to go along with the text. */
310  intro_img_t transition; /* image for transitioning. */
311  glPrintLineIterator iter; /* the renderable lines coming from the current text directive. */
312 
313  /* Load the introduction. */
314  if (intro_load(text) < 0)
315  return -1;
316 
317  /* Change music to intro music. */
318  if (mus != NULL)
319  music_choose(mus);
320 
321  /* We need to clear key repeat to avoid infinite loops. */
322  toolkit_clearKey();
323 
324  /* Do a few calculations to figure out how many lines can be present on the
325  screen at any given time. */
326  line_height = (double)intro_font.h * 1.3;
327  lines_per_screen = (int)(SCREEN_H / line_height + 1.5); /* round up + 1 */
328 
329  /* Initialize the lines. */
330  gl_printLineIteratorInit( &iter, &intro_font, "", SCREEN_W - 2*SIDE_MARGIN - IMAGE_WIDTH );
331  (void) gl_printLineIteratorNext( &iter );
332  sb_arr = calloc( lines_per_screen, sizeof( char * ) );
333  sb_index = 0;
334 
335  /* Force the first line to be loaded immediately. */
336  offset = line_height;
337 
338  /* Unset the side image. */
339  initialize_image( &side_image );
340  initialize_image( &transition );
341 
342  tlast = SDL_GetTicks();
343  while (!stop) {
344  tcur = SDL_GetTicks();
345  delta = (double)(tcur - tlast) / 1000.;
346  tlast = tcur;
347 
348  /* Increment position. */
349  offset += vel * delta;
350  while (! (offset < line_height)) {
351  /* One line has scrolled off, and another one on. */
352  if (gl_printLineIteratorNext( &iter )) {
353  free( sb_arr[sb_index] );
354  sb_arr[sb_index] = strndup( &iter.text[iter.l_begin], iter.l_end - iter.l_begin );
355  offset -= line_height;
356  sb_index = ( sb_index + 1 ) % lines_per_screen;
357  }
358  else if (cmd_index < array_size(intro_cmds)) {
359  switch (intro_cmds[cmd_index].opcode) {
360  case INTRO_TEXT:
361  gl_printLineIteratorInit( &iter, &intro_font, intro_cmds[cmd_index].operand, iter.width );
362  break;
363  case INTRO_FADEIN:
364  intro_fade_image_in( &side_image, &transition, intro_cmds[cmd_index].operand );
365  break;
366  case INTRO_FADEOUT:
367  if (NULL == side_image.tex) {
368  WARN(_("Tried to fade out without an image.") );
369  break;
370  }
371  side_image.fade_rate = -0.1;
372  side_image.c.a = 0.99;
373  break;
374  }
375  ++cmd_index;
376  } else {
377  free( sb_arr[sb_index] );
378  sb_arr[sb_index] = NULL;
379  offset -= line_height;
380  sb_index = ( sb_index + 1 ) % lines_per_screen;
381  }
382  } /* while (offset > line_height) */
383 
384  /* Fade the side image. */
385  if (side_image.tex != NULL && side_image.c.a < 1.0) {
386  side_image.c.a += delta * vel * side_image.fade_rate;
387 
388  if (transition.tex != NULL && transition.fade_rate > 0.0)
389  transition.c.a += delta * vel * transition.fade_rate;
390 
391  if (side_image.c.a > 1.0) {
392  /* Faded in... */
393  side_image.c.a = 1.0;
394  side_image.fade_rate = 0.0;
395  }
396  else if (side_image.c.a < 0.0) {
397  /* Faded out... */
398  gl_freeTexture( side_image.tex );
399  if (transition.tex != NULL) {
400  side_image.tex = transition.tex;
401  side_image.c.a = transition.c.a;
402  side_image.w = transition.w;
403  side_image.h = transition.h;
404  side_image.y = transition.y;
405  side_image.fade_rate = 0.1;
406  transition.tex = NULL;
407  transition.c.a = 1.0;
408  }
409  else {
410  side_image.c.a = 1.0;
411  side_image.tex = NULL;
412  side_image.fade_rate = 0.0;
413  }
414  }
415  }
416 
417  /* Clear stuff. */
418  glClear(GL_COLOR_BUFFER_BIT);
419 
420  /* Only thing we actually care about updating is music. */
421  music_update( 0. );
422 
423  /* Draw text. */
424  stop = intro_draw_text( sb_arr, lines_per_screen, sb_index, offset, line_height );
425 
426  if (NULL != side_image.tex)
427  /* Draw the image next to the text. */
428  gl_renderScale( side_image.tex, side_image.x, side_image.y,
429  side_image.w, side_image.h, &side_image.c );
430 
431  if (NULL != transition.tex && transition.c.a > 0.0)
432  /* Draw the image in transition. */
433  gl_renderScale( transition.tex, transition.x, transition.y,
434  transition.w, transition.h, &transition.c );
435 
436  /* Display stuff. */
437  SDL_GL_SwapWindow( gl_screen.window );
438 
439  tlag = SDL_GetTicks() - tcur;
440  if (tlag < 25) /* No need to burn CPU. Note: swapping may involve a wait for vblank; 25 is 1.5 frames @60fps. */
441  SDL_Delay( 25 - tlag );
442 
443  /* Handle user events. */
444  intro_event_handler( &stop, &offset, &vel );
445 
446  } /* while (!stop) */
447 
448  /* free malloc'd memory. */
449  free( sb_arr );
450  gl_freeTexture( side_image.tex );
451  gl_freeTexture( transition.tex );
452 
453  /* Stop music, normal music will start shortly after. */
454  music_stop();
455 
456  /* Clean up after the introduction. */
457  intro_cleanup();
458 
459  return 0;
460 }
