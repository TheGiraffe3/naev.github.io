282 {
283  char *ptr;

286  int *i;
287  const ShipStatsLookup *sl = &ss_lookup[ list->type ];
288 
289  ptr = (char*) stats;

291  case SS_DATA_TYPE_DOUBLE:
292  case SS_DATA_TYPE_DOUBLE_ABSOLUTE:
293  fieldptr = &ptr[ sl->offset ];
294  memcpy(&dbl, &fieldptr, sizeof(double*));
295  *dbl += list->d.d;
296  if ((sl->data==SS_DATA_TYPE_DOUBLE) && (*dbl < 0.)) /* Don't let the values go negative. */
297  *dbl = 0.;
298 
299  /* We'll increment amount. */
300  if (amount != NULL) {
301  if ((sl->inverted && (list->d.d < 0.)) ||
302  (!sl->inverted && (list->d.d > 0.))) {
303  memcpy(&ptr, &amount, sizeof(char*));
304  fieldptr = &ptr[ sl->offset ];
305  memcpy(&dbl, &fieldptr, sizeof(double*));
306  (*dbl) += 1.0;
307  }
308  }
309  break;
310 
311  case SS_DATA_TYPE_INTEGER:
312  fieldptr = &ptr[ sl->offset ];
313  memcpy(&i, &fieldptr, sizeof(int*));
314  *i += list->d.i;
315  if (amount != NULL) {
316  if ((sl->inverted && (list->d.i < 0)) ||
317  (!sl->inverted && (list->d.i > 0))) {
318  memcpy(&ptr, &amount, sizeof(char*));
319  fieldptr = &ptr[ sl->offset ];
320  memcpy(&i, &fieldptr, sizeof(int*));
321  (*i) += 1;
322  }
323  }
324  break;
325 
326  case SS_DATA_TYPE_BOOLEAN:
327  fieldptr = &ptr[ sl->offset ];
328  memcpy(&i, &fieldptr, sizeof(int*));
329  *i = 1; /* Can only set to true. */
