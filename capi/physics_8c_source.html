253 static void solid_update_euler (Solid *obj, const double dt)
254 {
255  double px,py, vx,vy, ax,ay, th;
256  double cdir, sdir;
257 
258  /* make sure angle doesn't flip */
259  obj->dir += obj->dir_vel*dt;
260  if (obj->dir >= 2*M_PI)
261  obj->dir -= 2*M_PI;
262  if (obj->dir < 0.)
263  obj->dir += 2*M_PI;
264 
265  /* Initial positions. */
266  px = obj->pos.x;
267  py = obj->pos.y;
268  vx = obj->vel.x;
269  vy = obj->vel.y;
270  th = obj->thrust;
271 
272  /* Save direction. */
273  sdir = sin(obj->dir);
274  cdir = cos(obj->dir);
275 
276  /* Get acceleration. */
277  ax = th*cdir / obj->mass;
278  ay = th*sdir / obj->mass;
279 
280  /* p = v*dt + 0.5*a*dt^2 */
281  px += vx*dt + 0.5*ax * dt*dt;
282  py += vy*dt + 0.5*ay * dt*dt;
283 
284  /* v = a*dt */
285  vx += ax*dt;
286  vy += ay*dt;
287 
288  /* Update position and velocity. */
289  vect_cset( &obj->vel, vx, vy );
290  vect_cset( &obj->pos, px, py );
291 }

317 #define RK4_MIN_H 0.01
318 static void solid_update_rk4 (Solid *obj, const double dt)
319 {
320  int i, N; /* for iteration, and pass calculation */
321  double h, px,py, vx,vy; /* pass, and position/velocity values */
322  double ix,iy, tx,ty, ax,ay, th; /* initial and temporary cartesian vector values */
323  double vmod, vang;
324  int vint;
325  int limit; /* limit speed? */
326 
327  /* Initial positions and velocity. */
328  px = obj->pos.x;
329  py = obj->pos.y;
330  vx = obj->vel.x;
331  vy = obj->vel.y;
332  limit = (obj->speed_max >= 0.);
333 
334  /* Initial RK parameters. */
335  if (dt > RK4_MIN_H)
336  N = (int)(dt / RK4_MIN_H);
337  else
338  N = 1;
339  vmod = MOD( vx, vy );
340  vint = (int) vmod/100.;
341  if (N < vint)
342  N = vint;
343  h = dt / (double)N; /* step */
344 
345  /* Movement Quantity Theorem: m*a = \sum f */
346  th = obj->thrust / obj->mass;
347 
348  for (i=0; i < N; i++) { /* iterations */
349 
350  /* Calculate acceleration for the frame. */
351  ax = th*cos(obj->dir);
352  ay = th*sin(obj->dir);
353 
354  /* Limit the speed. */
355  if (limit) {
356  vmod = MOD( vx, vy );
357  if (vmod > obj->speed_max) {
358  /* We limit by applying a force against it. */
359  vang = ANGLE( vx, vy ) + M_PI;
360  vmod = 3. * (vmod - obj->speed_max);
361 
362  /* Update accel. */
363  ax += vmod * cos(vang);
364  ay += vmod * sin(vang);
365  }
366  }
367 
368  /* x component */
369  tx = ix = vx;
370  tx += 2.*ix + h*tx;
371  tx += 2.*ix + h*tx;
372  tx += ix + h*tx;
373  tx *= h/6.;
374 
375  px += tx;
376  vx += ax * h;
377 
378  /* y component */
379  ty = iy = vy;
380  ty += 2.*(iy + h/2.*ty);
381  ty += 2.*(iy + h/2.*ty);
382  ty += iy + h*ty;
383  ty *= h/6.;
384 
385  py += ty;
386  vy += ay * h;
387 
388  /* rotation. */
389  obj->dir += obj->dir_vel*h;
390  }
391  vect_cset( &obj->vel, vx, vy );
392  vect_cset( &obj->pos, px, py );
393 
394  /* Validity check. */
395  if (obj->dir >= 2.*M_PI)
396  obj->dir -= 2.*M_PI;
397  else if (obj->dir < 0.)
398  obj->dir += 2.*M_PI;
399 }

429

434  dest->dir = dir;

436  dest->dir = fmod(dest->dir, 2.*M_PI);
437

441  else
442  dest->vel = *vel;
443 
444  /* Set position. */
445  if (pos == NULL)

447  else
448  dest->pos = *pos;
449

452
