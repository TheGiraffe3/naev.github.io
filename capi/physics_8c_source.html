229 static void solid_update_euler (Solid *obj, const double dt)
230 {
231  double px,py, vx,vy, ax,ay, th;
232  double cdir, sdir;
233 
234  /* make sure angle doesn't flip */
235  obj->dir += obj->dir_vel*dt;
236  if (obj->dir >= 2*M_PI)
237  obj->dir -= 2*M_PI;
238  if (obj->dir < 0.)
239  obj->dir += 2*M_PI;
240 
241  /* Initial positions. */
242  px = obj->pos.x;
243  py = obj->pos.y;
244  vx = obj->vel.x;
245  vy = obj->vel.y;
246  th = obj->thrust;
247 
248  /* Save direction. */
249  sdir = sin(obj->dir);
250  cdir = cos(obj->dir);
251 
252  /* Get acceleration. */
253  ax = th*cdir / obj->mass;
254  ay = th*sdir / obj->mass;
255 
256  /* p = v*dt + 0.5*a*dt^2 */
257  px += vx*dt + 0.5*ax * dt*dt;
258  py += vy*dt + 0.5*ay * dt*dt;
259 
260  /* v = a*dt */
261  vx += ax*dt;
262  vy += ay*dt;
263 
264  /* Update position and velocity. */
265  vect_cset( &obj->vel, vx, vy );
266  vect_cset( &obj->pos, px, py );
267 }

293 static void solid_update_rk4 (Solid *obj, const double dt)
294 {
295  int N; /* for iteration, and pass calculation */
296  double h, px,py, vx,vy; /* pass, and position/velocity values */
297  double ix,iy, tx,ty, th; /* initial and temporary cartesian vector values */
298  double vmod, vang;
299  int vint;
300  int limit; /* limit speed? */
301 
302  /* Initial positions and velocity. */
303  px = obj->pos.x;
304  py = obj->pos.y;
305  vx = obj->vel.x;
306  vy = obj->vel.y;
307  limit = (obj->speed_max >= 0.);
308 
309  /* Initial RK parameters. */
310  if (dt > RK4_MIN_H)
311  N = (int)(dt / RK4_MIN_H);
312  else
313  N = 1;
314  vmod = MOD( vx, vy );
315  vint = (int) vmod/100.;
316  if (N < vint)
317  N = vint;
318  h = dt / (double)N; /* step */
319 
320  /* Movement Quantity Theorem: m*a = \sum f */
321  th = obj->thrust / obj->mass;
322 
323  for (int i=0; i < N; i++) { /* iterations */
324  /* Calculate acceleration for the frame. */
325  double ax = th*cos(obj->dir);
326  double ay = th*sin(obj->dir);
327 
328  /* Limit the speed. */
329  if (limit) {
330  vmod = MOD( vx, vy );
331  if (vmod > obj->speed_max) {
332  /* We limit by applying a force against it. */
333  vang = ANGLE( vx, vy ) + M_PI;
334  vmod = 3. * (vmod - obj->speed_max);
335 
336  /* Update accel. */
337  ax += vmod * cos(vang);
338  ay += vmod * sin(vang);
339  }
340  }
341 
342  /* x component */
343  tx = ix = vx;
344  tx += 2.*ix + h*tx;
345  tx += 2.*ix + h*tx;
346  tx += ix + h*tx;
347  tx *= h/6.;
348 
349  px += tx;
350  vx += ax * h;
351 
352  /* y component */
353  ty = iy = vy;
354  ty += 2.*(iy + h/2.*ty);
355  ty += 2.*(iy + h/2.*ty);
356  ty += iy + h*ty;
357  ty *= h/6.;
358 
359  py += ty;
360  vy += ay * h;
361 
362  /* rotation. */
363  obj->dir += obj->dir_vel*h;
364  }
365  vect_cset( &obj->vel, vx, vy );
366  vect_cset( &obj->pos, px, py );
367 
368  /* Validity check. */
369  if (obj->dir >= 2.*M_PI)
370  obj->dir -= 2.*M_PI;
371  else if (obj->dir < 0.)
372  obj->dir += 2.*M_PI;
373 }
