50 void pilot_lockUpdateSlot( Pilot *p, PilotOutfitSlot *o, Pilot *t, double *a, double dt )
51 {
52  double max, old;
53  double x,y, ang, arc;
54  int locked;
55 
56  /* No target. */
57  if (t == NULL)
58  return;
59 
60  /* Nota seeker. */
61  if (!outfit_isSeeker(o->outfit))
62  return;
63 
64  /* Check arc. */
65  arc = o->outfit->u.lau.arc;
66  if (arc > 0.) {
67 
68  /* We use an external variable to set and update the angle if necessary. */
69  if (*a < 0.) {
70  x = t->solid->pos.x - p->solid->pos.x;
71  y = t->solid->pos.y - p->solid->pos.y;
72  ang = ANGLE( x, y );
73  *a = FABS( angle_diff( ang, p->solid->dir ) );
74  }
75 
76  /* Decay if not in arc. */
77  if (*a > arc) {
78  /* Limit decay to the lockon time for this launcher. */
79  max = o->outfit->u.lau.lockon;
80 
81  /* When a lock is lost, immediately gain half the lock timer.
82  * This is meant as an incentive for the aggressor not to lose the lock,
83  * and for the target to try and break the lock. */
84  old = o->u.ammo.lockon_timer;
85  o->u.ammo.lockon_timer += dt;
86  if ((old <= 0.) && (o->u.ammo.lockon_timer > 0.))
87  o->u.ammo.lockon_timer += o->outfit->u.lau.lockon / 2.;
88 
89  /* Cap at max. */
90  if (o->u.ammo.lockon_timer > max)
91  o->u.ammo.lockon_timer = max;
92 
93  /* Out of arc. */
94  o->u.ammo.in_arc = 0;
95  return;
96  }
97  }
98 
99  /* In arc. */
100  o->u.ammo.in_arc = 1;
101  locked = (o->u.ammo.lockon_timer < 0.);
102 
103  /* Lower timer. When the timer reaches zero, the lock is established. */
104  max = -o->outfit->u.lau.lockon/3. * p->stats.launch_lockon;
105  if (o->u.ammo.lockon_timer > max) {
106  /* Compensate for enemy hide factor. */
107  o->u.ammo.lockon_timer -= dt * (o->outfit->u.lau.ew_target2 / t->ew_hide);
108 
109  /* Cap at -max/3. */
110  if (o->u.ammo.lockon_timer < max)
111  o->u.ammo.lockon_timer = max;
112 
113  /* Trigger lockon hook. */
114  if (!locked && (o->u.ammo.lockon_timer < 0.))
115  pilot_runHook( p, PILOT_HOOK_LOCKON );
116  }
117 }

156 int pilot_getMount( const Pilot *p, const PilotOutfitSlot *w, Vector2d *v )
157 {
158  double a, x, y;
159  double cm, sm;
160  const ShipMount *m;
161 
162  /* Calculate the sprite angle. */
163  a = (double)(p->tsy * p->ship->gfx_space->sx + p->tsx);
164  a *= p->ship->mangle;
165 
166  /* 2d rotation matrix
167  * [ x' ] [ cos sin ] [ x ]
168  * [ y' ] = [ -sin cos ] * [ y ]
169  *
170  * dir is inverted so that rotation is counter-clockwise.
171  */
172  m = &w->sslot->mount;
173  cm = cos(-a);
174  sm = sin(-a);
175  x = m->x * cm + m->y * sm;
176  y = m->x *-sm + m->y * cm;
177 
178  /* Correction for ortho perspective. */
179  y *= M_SQRT1_2;
180 
181  /* Don't forget to add height. */
182  y += m->h;
183 
184  /* Get the mount and add the player.p offset. */
185  vect_cset( v, x, y );
186 
187  return 0;
