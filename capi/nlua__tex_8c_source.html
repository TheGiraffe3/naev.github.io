187 static int texL_new( lua_State *L )
188 {
189  const char *path;
190  glTexture *tex;
191  LuaFile_t *lf;
192  LuaData_t *ld;
193  int sx, sy;
194  SDL_RWops *rw;
195  char *name;
196 
197  NLUA_CHECKRW(L);
198 
199  /* Defaults. */
200  lf = NULL;
201  ld = NULL;
202  path = NULL;
203 
204  /* Get path. */
205  if (lua_isdata(L,1)) {
206  int w, h;
207  ld = luaL_checkdata(L,1);
208  /* Since we don't know the size we need the width and height separately. */
209  w = luaL_checkinteger(L,2);
210  h = luaL_checkinteger(L,3);
211  if ((w < 0 ) || (h < 0))
212  NLUA_ERROR( L, _("Texture dimensions must be positive") );
213  sx = luaL_optinteger(L,4,1);
214  sy = luaL_optinteger(L,5,1);
215  if ((sx < 0 ) || (sy < 0))
216  NLUA_ERROR( L, _("Spritesheet dimensions must be positive") );
217  if (ld->type != LUADATA_NUMBER)
218  NLUA_ERROR( L, _("Data has invalid type for texture") );
219  if (w*h*ld->elem*4 != ld->size)
220  NLUA_ERROR( L, _("Texture dimensions don't match data size!") );
221  asprintf( &name, "nlua_texture_%03d", ++nlua_tex_counter );
222  tex = gl_loadImageData( (void*)ld->data, w, h, sx, sy, name );
223  free( name );
224  if (tex==NULL)
225  return 0;
226  lua_pushtex(L, tex);
227  return 1;
228  }
229  else if (lua_isfile(L,1))
230  lf = luaL_checkfile(L,1);
231  else
232  path = luaL_checkstring(L, 1);
233 
234  sx = luaL_optinteger(L,2,1);
235  sy = luaL_optinteger(L,3,1);
236  if ((sx < 0 ) || (sy < 0))
237  NLUA_ERROR( L, _("Spritesheet dimensions must be positive") );
238 
239  /* Push new texture. */
240  if (path != NULL)
241  tex = gl_newSprite( path, sx, sy, 0 );
242  else {
243  rw = PHYSFSRWOPS_openRead( lf->path );
244  if (rw==NULL)
245  NLUA_ERROR(L,"Unable to open '%s'", lf->path );
246  tex = gl_newSpriteRWops( lf->path, rw, sx, sy, 0 );
247  SDL_RWclose( rw );
248  }
249 
250  /* Failed to load. */
251  if (tex == NULL)
252  return 0;
253  /* Properly loaded. */
254  lua_pushtex( L, tex );
255  return 1;
256 }

259 static inline uint32_t get_pixel(SDL_Surface *surface, int x, int y)
260 {
261  int bpp = surface->format->BytesPerPixel;
262  /* Here p is the address to the pixel we want to retrieve */
263  uint8_t *p = (Uint8 *)surface->pixels + y * surface->pitch + x * bpp;
264 
265  switch(bpp) {
266  case 1:
267  return *p;
268  break;
269 
270  case 2:
271  return *(Uint16 *)p;
272  break;
273 
274  case 3:
275  if(SDL_BYTEORDER == SDL_BIG_ENDIAN)
276  return p[0] << 16 | p[1] << 8 | p[2];
277  else
278  return p[0] | p[1] << 8 | p[2] << 16;
279  break;
280 
281  case 4:
282  return *(Uint32 *)p;
283  break;
284 
285  default:
286  return 0; /* shouldn't happen, but avoids warnings */
287  }
288 }

300 static int texL_readData( lua_State *L )
301 {
302  LuaFile_t *lf;
303  LuaData_t ld;
304  SDL_Surface *surface;
305  SDL_RWops *rw;
306  const char *s;
307  size_t size;
308  uint8_t r, g, b, a;
309  uint32_t pix;
310  int i, j;
311  float *data;
312 
313  s = NULL;
314  if (lua_isfile(L,1)) {
315  lf = luaL_checkfile(L,1);
316  s = lf->path;
317  }
318  else
319  s = luaL_checkstring(L,1);
320  rw = PHYSFSRWOPS_openRead( s );
321  if (rw == NULL)
322  NLUA_ERROR(L, _("problem opening file '%s' for reading"), s );
323 
324  /* Try to read the image. */
325  surface = IMG_Load_RW( rw, 1 );
326  if (surface == NULL)
327  NLUA_ERROR(L, _("problem opening image for reading") );
328 
329  /* Convert surface to LuaData_t */
330  SDL_LockSurface( surface );
331  size = surface->w * surface->h;
332  ld.elem = sizeof(float);
333  ld.size = ld.elem * size * 4;
334  ld.data = calloc( ld.elem*4, size );
335  ld.type = LUADATA_NUMBER;
336  data = (float*)ld.data;
337  for (i=0; i<surface->h; i++) {
338  for (j=0; j<surface->w; j++) {
339  pix = get_pixel( surface, j, i );
340  SDL_GetRGBA( pix, surface->format, &r, &g, &b, &a );
341  size_t pos = 4*((surface->h-i-1)*surface->w+j);
342  data[ pos+0 ] = ((float)r)/255.;
343  data[ pos+1 ] = ((float)g)/255.;
344  data[ pos+2 ] = ((float)b)/255.;
345  data[ pos+3 ] = ((float)a)/255.;
346  }
347  }
348  SDL_UnlockSurface( surface );
349 
350  /* Return parameters. */
351  lua_pushdata(L, ld);
352  lua_pushinteger(L, surface->w);
353  lua_pushinteger(L, surface->h);
354 
355  /* Clean up. */
356  SDL_FreeSurface( surface );
357 
358  return 3;
359 }
