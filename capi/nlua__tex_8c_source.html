189 static int texL_new( lua_State *L )
190 {
191  const char *path;
192  glTexture *tex;
193  LuaFile_t *lf;
194  LuaData_t *ld;
195  int sx, sy;
196  SDL_RWops *rw;
197  char *name;
198 
199  NLUA_CHECKRW(L);
200 
201  /* Defaults. */
202  lf = NULL;
203  ld = NULL;
204  path = NULL;
205 
206  /* Get path. */
207  if (lua_isdata(L,1)) {
208  int w, h;
209  ld = luaL_checkdata(L,1);
210  /* Since we don't know the size we need the width and height separately. */
211  w = luaL_checkinteger(L,2);
212  h = luaL_checkinteger(L,3);
213  if ((w < 0 ) || (h < 0))
214  NLUA_ERROR( L, _("Texture dimensions must be positive") );
215  sx = luaL_optinteger(L,4,1);
216  sy = luaL_optinteger(L,5,1);
217  if ((sx < 0 ) || (sy < 0))
218  NLUA_ERROR( L, _("Spritesheet dimensions must be positive") );
219  if (ld->type != LUADATA_NUMBER)
220  NLUA_ERROR( L, _("Data has invalid type for texture") );
221  if (w*h*ld->elem*4 != ld->size)
222  NLUA_ERROR( L, _("Texture dimensions don't match data size!") );
223  asprintf( &name, "nlua_texture_%03d", ++nlua_tex_counter );
224  tex = gl_loadImageData( (void*)ld->data, w, h, sx, sy, name );
225  free( name );
226  if (tex==NULL)
227  return 0;
228  lua_pushtex(L, tex);
229  return 1;
230  }
231  else if (lua_isfile(L,1))
232  lf = luaL_checkfile(L,1);
233  else
234  path = luaL_checkstring(L, 1);
235 
236  sx = luaL_optinteger(L,2,1);
237  sy = luaL_optinteger(L,3,1);
238  if ((sx < 0 ) || (sy < 0))
239  NLUA_ERROR( L, _("Spritesheet dimensions must be positive") );
240 
241  /* Push new texture. */
242  if (path != NULL)
243  tex = gl_newSprite( path, sx, sy, 0 );
244  else {
245  rw = PHYSFSRWOPS_openRead( lf->path );
246  if (rw==NULL)
247  NLUA_ERROR(L,"Unable to open '%s'", lf->path );
248  tex = gl_newSpriteRWops( lf->path, rw, sx, sy, 0 );
249  SDL_RWclose( rw );
250  }
251 
252  /* Failed to load. */
253  if (tex == NULL)
254  return 0;
255  /* Properly loaded. */
256  lua_pushtex( L, tex );
257  return 1;
258 }

261 static inline uint32_t get_pixel(SDL_Surface *surface, int x, int y)
262 {
263  int bpp = surface->format->BytesPerPixel;
264  /* Here p is the address to the pixel we want to retrieve */
265  uint8_t *p = (Uint8 *)surface->pixels + y * surface->pitch + x * bpp;
266 
267  switch(bpp) {
268  case 1:
269  return *p;
270  break;
271 
272  case 2:
273  return *(Uint16 *)p;
274  break;
275 
276  case 3:
277  if(SDL_BYTEORDER == SDL_BIG_ENDIAN)
278  return p[0] << 16 | p[1] << 8 | p[2];
279  else
280  return p[0] | p[1] << 8 | p[2] << 16;
281  break;
282 
283  case 4:
284  return *(Uint32 *)p;
285  break;
286 
287  default:
288  return 0; /* shouldn't happen, but avoids warnings */
289  }
290 }

302 static int texL_readData( lua_State *L )
303 {
304  LuaFile_t *lf;
305  LuaData_t ld;
306  SDL_Surface *surface;
307  SDL_RWops *rw;
308  const char *s;
309  size_t size;
310  uint8_t r, g, b, a;
311  uint32_t pix;
312  int i, j;
313  float *data;
314 
315  s = NULL;
316  if (lua_isfile(L,1)) {
317  lf = luaL_checkfile(L,1);
318  s = lf->path;
319  }
320  else
321  s = luaL_checkstring(L,1);
322  rw = PHYSFSRWOPS_openRead( s );
323  if (rw == NULL)
324  NLUA_ERROR(L, _("problem opening file '%s' for reading"), s );
325 
326  /* Try to read the image. */
327  surface = IMG_Load_RW( rw, 1 );
328  if (surface == NULL)
329  NLUA_ERROR(L, _("problem opening image for reading") );
330 
331  /* Convert surface to LuaData_t */
332  SDL_LockSurface( surface );
333  size = surface->w * surface->h;
334  ld.elem = sizeof(float);
335  ld.size = ld.elem * size * 4;
336  ld.data = calloc( ld.elem*4, size );
337  ld.type = LUADATA_NUMBER;
338  data = (float*)ld.data;
339  for (i=0; i<surface->h; i++) {
340  for (j=0; j<surface->w; j++) {
341  pix = get_pixel( surface, j, i );
342  SDL_GetRGBA( pix, surface->format, &r, &g, &b, &a );
343  size_t pos = 4*((surface->h-i-1)*surface->w+j);
344  data[ pos+0 ] = ((float)r)/255.;
345  data[ pos+1 ] = ((float)g)/255.;
346  data[ pos+2 ] = ((float)b)/255.;
347  data[ pos+3 ] = ((float)a)/255.;
348  }
349  }
350  SDL_UnlockSurface( surface );
351 
352  /* Return parameters. */
353  lua_pushdata(L, ld);
354  lua_pushinteger(L, surface->w);
355  lua_pushinteger(L, surface->h);
356 
357  /* Clean up. */
358  SDL_FreeSurface( surface );
359 
360  return 3;
361 }
