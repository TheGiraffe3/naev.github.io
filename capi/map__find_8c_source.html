247 static int map_sortCompare( const void *p1, const void *p2 )
248 {
249  map_find_t *f1, *f2;
250 
251  /* Convert pointer. */
252  f1 = (map_find_t*) p1;
253  f2 = (map_find_t*) p2;
254 
255  /* Compare jumps. */
256  if (f1->jumps > f2->jumps)
257  return +1;
258  else if (f1->jumps < f2->jumps)
259  return -1;
260 
261  /* Compare distance. */
262  if (f1->distance > f2->distance)
263  return +1;
264  else if (f1->distance < f2->distance)
265  return -1;
266 
267  /* If they're the same it doesn't matter, so we'll sort by name. */
268  return strcasecmp( f1->sys->name, f2->sys->name );
269 }

284 static int map_findDistance( StarSystem *sys, Planet *pnt, int *jumps, double *distance )
285 {
286  int i, j;
287  StarSystem **slist, *ss;
288  double d;
289  Vector2d *vs, *ve;
290 
291  /* Defaults. */
292  ve = NULL;
293 
294  /* Special case it's the current system. */
295  if (sys == cur_system) {
296  *jumps = 0;
297  if (pnt != NULL)
298  *distance = vect_dist( &player.p->solid->pos, &pnt->pos );
299  else
300  *distance = 0.;
301 
302  return 0;
303  }
304 
305  /* Calculate jump path. */
306  slist = map_getJumpPath( cur_system->name, sys->name, 0, 1, NULL );
307  *jumps = array_size( slist );
308  if (slist==NULL)
309  /* Unknown. */
310  return -1;
311 
312  /* Distance to first jump point. */
313  vs = &player.p->solid->pos;
314  for (j=0; j < array_size(cur_system->jumps); j++) {
315  if (cur_system->jumps[j].target == slist[0]) {
316  ve = &cur_system->jumps[j].pos;
317  break;
318  }
319  }
320  if (ve == NULL) {
321  WARN(_("Jump to first system not found!"));
322  d = 0.;
323  }
324  else
325  d = vect_dist( vs, ve );
326 
327  /* Calculate distance. */
328  for (i=0; i<(*jumps-1); i++) {
329  ss = slist[i];
330 
331  /* Search jump points. */
332  for (j=0; j < array_size(ss->jumps); j++) {
333 
334  /* Get previous jump. */
335  if (i == 0) {
336  if (ss->jumps[j].target == cur_system)
337  vs = &ss->jumps[j].pos;
338  }
339  else {
340  if (ss->jumps[j].target == slist[i-1])
341  vs = &ss->jumps[j].pos;
342  }
343 
344  /* Get next jump. */
345  if (ss->jumps[j].target == slist[i+1]) {
346  ve = &ss->jumps[j].pos;
347  break;
348  }
349  }
350 
351  /* Use current position. */
352  if (i==0)
353  vs = &player.p->solid->pos;
354 
355 #ifdef DEBUGGING
356  if ((vs==NULL) || (ve==NULL)) {
357  WARN( _("Matching jumps not found, something is up...") );
358  continue;
359  }
360 #endif /* DEBUGGING */
361 
362  /* Calculate. */
363  d += vect_dist( vs, ve );
364  }
365 
366  /* Account final travel to planet for planet targets. */
367  if (pnt != NULL) {
368  if (i > 0) {
369  ss = slist[ i ];
370  for (j=0; j < array_size(ss->jumps); j++) {
371  if (ss->jumps[j].target == slist[i-1]) {
372  vs = &ss->jumps[j].pos;
373  break;
374  }
375  }
376  }
377 
378  ve = &pnt->pos;
379 
380  assert( vs != NULL );
381  assert( ve != NULL );
382 
383  d += vect_dist( vs, ve );
384  }
385 
386  /* Cleanup. */
387  array_free(slist);
388 
389  *distance = d;
390  return 0;
391 }

403 static void map_findAccumulateResult( map_find_t *found, int n, StarSystem *sys, Planet *pnt )
404 {
405  int ret;
406  char route_info[256];
407 
408  /* Set some values. */
409  found[n].pnt = pnt;
410  found[n].sys = sys;
411 
412  /* Set more values. */
413  ret = map_findDistance( sys, pnt, &found[n].jumps, &found[n].distance );
414  if (ret) {
415  found[n].jumps = 10000;
416  found[n].distance = 1e6;
417  snprintf( route_info, sizeof(route_info), "%s", _("unknown route") );
418  }
419  else
420  snprintf( route_info, sizeof(route_info),
421  n_( "%d jump, %.0fk distance", "%d jumps, %.0fk distance", found[n].jumps ),
422  found[n].jumps, found[n].distance/1000. );
423 
424  /* Set fancy name. */
425  if (pnt == NULL)
426  snprintf( found[n].display, sizeof(found[n].display),
427  _("%s (%s)"), _(sys->name), route_info );
428  else
429  snprintf( found[n].display, sizeof(found[n].display),
430  _("#%c%s%s (%s, %s)"), map_getPlanetColourChar(pnt),
431  map_getPlanetSymbol(pnt),
432  _(pnt->name), _(sys->name), route_info );
433 }

442 static int map_findSearchSystems( unsigned int parent, const char *name )
443 {
444  int i;
445  const char *sysname;
446  StarSystem *sys;
447  char **names;
448  int len, n;
449  map_find_t *found;
450 
451  /* Search for names. */
452  sysname = system_existsCase( name );
453  names = system_searchFuzzyCase( name, &len );
454  if (names == NULL)
455  return -1;
456 
457  /* Exact match. */
458  if ((sysname != NULL) && (len == 1)) {
459  /* Select and show. */
460  sys = system_get(sysname);
461  if (sys_isKnown(sys)) {
462  map_select( sys, 0 );
463  map_center( sysname );
464  free(names);
465  return 1;
466  }
467  }
468 
469  /* Construct found table. */
470  found = NULL;
471  n = 0;
472  for (i=0; i<len; i++) {
473 
474  /* System must be known. */
475  sys = system_get( names[i] );
476  if (!sys_isKnown(sys))
477  continue;
478 
479  if (found == NULL) /* Allocate results array on first match. */
480  found = malloc( sizeof(map_find_t) * len );
481 
482  map_findAccumulateResult( found, n, sys, NULL );
483  n++;
484  }
485  free(names);
486 
487  /* No visible match. */
488  if (n==0)
489  return -1;
490 
491  /* Display results. */
492  map_findDisplayResult( parent, found, n );
493  return 0;
494 }

749 static int map_findSearchOutfits( unsigned int parent, const char *name )
750 {
751  int i, j;
752  int len, n;
753  map_find_t *found;
754  Planet *pnt;
755  StarSystem *sys;
756  const char *oname, *sysname;
757  char **list;
758  const Outfit *o;
759  Outfit **olist;
760 
761  assert( "Outfit search is not reentrant!" && map_foundOutfitNames == NULL );
762 
763  /* Match planet first. */
764  o = NULL;
765  oname = outfit_existsCase( name );
766  map_foundOutfitNames = map_outfitsMatch( name );
767  len = array_size( map_foundOutfitNames );
768  if ((oname != NULL) && (len == 1))
769  o = outfit_get( oname );
770  /* Do fuzzy match. */
771  else if (len > 0) {
772  /* Ask which one player wants. */
773  list = malloc( len*sizeof(char*) );
774  for (i=0; i<len; i++)
775  list[i] = strdup( _(map_foundOutfitNames[i]) );
776  i = dialogue_listPanel( _("Search Results"), list, len, 452, 650,
777  map_addOutfitDetailFields, map_showOutfitDetail,

779  if (i < 0) {
780  array_free( map_foundOutfitNames );
781  map_foundOutfitNames = NULL;
782  return 0;
783  }
784  o = outfit_get( map_foundOutfitNames[i] );
785  }
786  array_free( map_foundOutfitNames );
787  map_foundOutfitNames = NULL;
788  if (o == NULL)
789  return -1;
790 
791  /* Construct found table. */
792  found = NULL;
793  n = 0;
794  len = array_size(map_known_techs);
795  for (i=0; i<len; i++) {
796  /* Try to find the outfit in the planet. */
797  olist = tech_getOutfit( map_known_techs[i] );
798  for (j=array_size(olist)-1; j>=0; j--)
799  if (olist[j] == o)
800  break;
801  array_free( olist );
802  olist = NULL;
803  if (j < 0)
804  continue;
805  pnt = map_known_planets[i];

955 {

959  /* Get the name. */
960  name = window_getInput( wid, "inpSearch" );
961  if ( name[0] == '\0' )
962  return;
963 
964  /* Prevent reentrancy, e.g. the toolkit spontaneously deciding a future mouseup event was the
965  * user releasing the clicked "Find" button and should reactivate it, never mind that they were
966  * actually clicking on the dialogue_listPanel we opened to present the results.
967  * FIXME: That behavior doesn't seem right, but I'm not sure if it's an actual bug or not. */
968  window_disableButton( wid, "btnSearch" );
969 
970  /* Clean up if necessary. */
971  free( map_found_cur );
972  map_found_cur = NULL;
973 
974  /* Handle different search cases. */
975  if (map_find_systems) {
976  ret = map_findSearchSystems( wid, name );
977  searchname = _("System");
978  }
979  else if (map_find_planets) {
980  ret = map_findSearchPlanets( wid, name );
981  searchname = _("Planet");
982  }
983  else if (map_find_outfits) {
984  ret = map_findSearchOutfits( wid, name );
985  searchname = _("Outfit");
986  }
987  else if (map_find_ships) {
988  ret = map_findSearchShips( wid, name );
989  searchname = _("Ship");
990  }
991  else
992  ret = 1;
993 
994  if (ret < 0)
995  dialogue_alert( _("%s matching '%s' not found!"), searchname, name );
996 
997  /* Safe at last. */
998  window_enableButton( wid, "btnSearch" );
999 
1000  if (ret > 0)
1001  map_findClose( wid, str );
