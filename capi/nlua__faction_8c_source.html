144 {
145  int id;

193 
194  ret = 0;

570 static int factionL_dynAdd( lua_State *L )
571 {
572  LuaFaction fac, newfac;
573  const char *name, *display, *ai;
574  int clear_allies, clear_enemies;
575 
576  NLUA_CHECKRW(L);
577 
578  if (!lua_isnoneornil(L, 1))
579  fac = luaL_validfaction(L,1);
580  else
581  fac = -1;
582  name = luaL_checkstring(L,2);
583  display = luaL_optstring(L,3,name);
584 
585  /* Parse parameters. */
586  if (lua_istable(L,4)) {
587  lua_getfield(L,4,"ai");
588  ai = luaL_optstring(L,-1,NULL);
589  lua_pop(L,1);
590 
591  lua_getfield(L,4,"clear_allies");
592  clear_allies = lua_toboolean(L,-1);
593  lua_pop(L,1);
594 
595  lua_getfield(L,4,"clear_enemies");
596  clear_enemies = lua_toboolean(L,-1);
597  lua_pop(L,1);
598  }
599  else {
600  ai = NULL;
601  clear_allies = 0;
602  clear_enemies = 0;
603  }
604 
605  /* Check if exists. */
606  if (faction_exists(name))
607  NLUA_ERROR(L,_("Faction '%s' already exists!"), name);
608 
609  /* Create new faction. */
610  newfac = faction_dynAdd( fac, name, display, ai );
611 
612  /* Clear if necessary. */
613  if (clear_allies)
614  faction_clearAlly( newfac );
615  if (clear_enemies)
616  faction_clearEnemy( newfac );
617 
618  lua_pushfaction( L, newfac );
619  return 1;
620 }
