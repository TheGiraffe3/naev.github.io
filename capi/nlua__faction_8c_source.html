574 static int factionL_dynAdd( lua_State *L )
575 {
576  LuaFaction fac, newfac;
577  const char *name, *display, *ai;
578  int clear_allies, clear_enemies;
579 
580  NLUA_CHECKRW(L);
581 
582  if (!lua_isnoneornil(L, 1))
583  fac = luaL_validfaction(L,1);
584  else
585  fac = -1;
586  name = luaL_checkstring(L,2);
587  display = luaL_optstring(L,3,name);
588 
589  /* Parse parameters. */
590  if (lua_istable(L,4)) {
591  lua_getfield(L,4,"ai");
592  ai = luaL_optstring(L,-1,NULL);
593  lua_pop(L,1);
594 
595  lua_getfield(L,4,"clear_allies");
596  clear_allies = lua_toboolean(L,-1);
597  lua_pop(L,1);
598 
599  lua_getfield(L,4,"clear_enemies");
600  clear_enemies = lua_toboolean(L,-1);
601  lua_pop(L,1);
602  }
603  else {
604  ai = NULL;
605  clear_allies = 0;
606  clear_enemies = 0;
607  }
608 
609  /* Check if exists. */
610  if (faction_exists(name))
611  NLUA_ERROR(L,_("Faction '%s' already exists!"), name);
612 
613  /* Create new faction. */
614  newfac = faction_dynAdd( fac, name, display, ai );
615 
616  /* Clear if necessary. */
617  if (clear_allies)
618  faction_clearAlly( newfac );
619  if (clear_enemies)
620  faction_clearEnemy( newfac );
621 
622  lua_pushfaction( L, newfac );
623  return 1;
624 }
