152 {

194 int lua_isfaction( lua_State *L, int ind )
195 {
196  int ret;
197 
198  if (lua_getmetatable(L,ind)==0)
199  return 0;
200  lua_getfield(L, LUA_REGISTRYINDEX, FACTION_METATABLE);
201 
202  ret = 0;
203  if (lua_rawequal(L, -1, -2)) /* does it have the correct mt? */
204  ret = 1;
205 
206  lua_pop(L, 2); /* remove both metatables */
207  return ret;
208 }
