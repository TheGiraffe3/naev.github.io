121  * Note, taking off and landing takes about 1e7 ntime, which is 1 period.
122  * Time does not advance when on a planet.
123  * Journey with a single jump takes approx 3e7, so about 3 periods.
124  */

135  /* Check if found. */
136  if (i >= array_size(econ_comm)) {
137  WARN(_("Price for commodity '%s' not known."), com->name);
138  return 0;
139  }
140 
141  /* and get the index on this planet */
142  for ( i=0; i<array_size(p->commodities); i++) {
143  if ( ( strcmp(p->commodities[i]->name, com->name) == 0 ) )
144  break;
145  }
146  if (i >= array_size(p->commodities)) {
147  WARN(_("Price for commodity '%s' not known on this planet."), com->name);
148  return 0;
149  }
150  commPrice = &p->commodityPrice[i];
151  /* Calculate price. */
152  /* price = (double) com->price; */
153  /* price *= sys->prices[i]; */
154  price = (commPrice->price + commPrice->sysVariation
155  * sin(2 * M_PI * t / commPrice->sysPeriod)
156  + commPrice->planetVariation

315 {
316  int i;
317  double I;
318  double prodfactor, p;
319  double ddt;
320  Planet *planet;
321 
322  ddt = (double)(dt / NTIME_UNIT_LENGTH);
323 
324  /* Calculate production level. */
325  p = 0.;
326  for (i=0; i<sys->nplanets; i++) {
327  planet = sys->planets[i];
328  if (planet_hasService(planet, PLANET_SERVICE_INHABITED)) {
329  /*
330  * Calculate production.
331  */
332  /* We base off the current production. */
333  prodfactor = planet->cur_prodfactor;
334  /* Add a variability factor based on the Gaussian distribution. */
335  prodfactor += ECON_PROD_VAR * RNG_2SIGMA() * ddt;
336  /* Add a tendency to return to the planet's base production. */
337  prodfactor -= ECON_PROD_VAR *
338  (planet->cur_prodfactor - prodfactor)*ddt;
339  /* Save for next iteration. */
340  planet->cur_prodfactor = prodfactor;

342  p += prodfactor * sqrt(planet->population);

344  }

599 static int economy_calcPrice( Planet *planet, Commodity *commodity, CommodityPrice *commodityPrice ) {
600 
601  CommodityModifier *cm;
602  double base, scale, factor;
603  const char *factionname;
604 
605  /* Check the faction is not NULL.*/
606  if ( planet->faction == -1 ) {
607  WARN(_("Planet '%s' appears to have commodity '%s' defined, but no faction."), planet->name, commodity->name);
608  return 1;
609  }
610 
611  /* Reset price to the base commodity price. */
612  commodityPrice->price = commodity->price;
613 
614  /* Get the cost modifier suitable for planet type/class. */
615  cm = commodity->planet_modifier;
616  scale = 1.;
617  while ( cm != NULL ) {
618  if ( ( strcmp( planet->class, cm->name ) == 0 ) ) {
619  scale = cm->value;
620  break;
621  }
622  cm = cm->next;
623  }
624  commodityPrice->price *= scale;
625  commodityPrice->planetVariation = 0.5;
626  commodityPrice->sysVariation = 0.;
627  /*commodityPrice->sum = 0.;
628  commodityPrice->sum2 = 0.;
629  commodityPrice->cnt = 0;
630  commodityPrice->updateTime = 0;*/
631  /* Use filename to specify a variation period. */
632  base = 100;
633  commodity->period = 32 * (planet->gfx_spaceName[strlen(PLANET_GFX_SPACE_PATH)] % 32) + planet->gfx_spaceName[strlen(PLANET_GFX_SPACE_PATH) + 1] % 32;
634  commodityPrice->planetPeriod = commodity->period + base;
635 
636  /* Use filename of exterior graphic to modify the variation period.
637  No rhyme or reason, just gives some variability. */
638  scale = 1 + (strlen(planet->gfx_exterior) - strlen(PLANET_GFX_EXTERIOR_PATH) - 19) / 100.;
639  commodityPrice->planetPeriod *= scale;
640 
641  /* Use population to modify price and variability. The tanh function scales from -1 (small population)
642  to +1 (large population), done on a log scale. Price is then modified by this factor, scaled by a
643  value defined in the xml, as is variation. So for some commodities, prices increase with population,
644  while for others, prices decrease. */
645  factor = -1;
646  if ( planet->population > 0 )
647  factor = tanh( ( log((double)planet->population) - log(1e8) ) / 2 );
648  base = commodity->population_modifier;
649  commodityPrice->price *= 1 + factor * base;
650  commodityPrice->planetVariation *= 0.5 - factor * 0.25;
651  commodityPrice->planetPeriod *= 1 + factor * 0.5;
652 
653  /* Modify price based on faction (as defined in the xml).
654  Some factions place a higher value on certain goods.
655  Some factions are more stable than others.*/
656  scale = 1.;
657  cm = commodity->planet_modifier;
658 
659  factionname = faction_name(planet->faction);
660  while ( cm != NULL ) {
661  if ( strcmp( factionname, cm->name ) == 0 ) {
662  scale = cm->value;
663  break;
664  }
665  cm = cm->next;
666  }
667  commodityPrice->price *= scale;
668 
669  /* Range seems to go from 0-5, with median being 2. Increased range
670  * will increase safety and so lower prices and improve stability */
671  commodityPrice->price *= (1 - planet->presenceRange/30.);
672  commodityPrice->planetPeriod *= 1 / (1 - planet->presenceRange/30.);
673 
674  /* Make sure the price is always positive and non-zero */
675  commodityPrice->price = MAX( commodityPrice->price, 1 );
676
