485 void loadscreen_load (void)
486 {
487  char file_path[PATH_MAX];
488  char **loadpaths, **loadscreens, *load;
489  size_t nload, nreal, nbuf;
490  const char *loading_prefix = "webp";
491 
492  /* Count the loading screens */
493  loadpaths = PHYSFS_enumerateFiles( GFX_PATH"loading/" );
494 
495  for (nload=0; loadpaths[nload]!=NULL; nload++) {}
496  loadscreens = calloc( nload, sizeof(char*) );
497  nreal = 0;
498  for (nload=0; loadpaths[nload]!=NULL; nload++) {
499  if (!ndata_matchExt( loadpaths[nload], loading_prefix ))
500  continue;
501  loadscreens[nreal++] = loadpaths[nload];
502  }
503 
504  /* Must have loading screens */
505  if (nreal==0) {
506  WARN( _("No loading screens found!") );
507  PHYSFS_freeList( loadpaths );
508  free( loadscreens );
509  loading = NULL;
510  return;
511  }
512 
513  /* Load the loading font. */
514  gl_fontInit( &loading_font, _(FONT_DEFAULT_PATH), 24, FONT_PATH_PREFIX, 0 ); /* initializes default font to size */
515 
516  /* Set the zoom. */
517  cam_setZoom( conf.zoom_far );
518 
519  /* Choose the screen. */
520  load = loadscreens[ RNG_BASE(0,nreal-1) ];
521 
522  /* Load the texture */
523  snprintf( file_path, sizeof(file_path), GFX_PATH"loading/%s", load );
524  loading = gl_newImage( file_path, 0 );
525 
526  /* Load the metadata. */
527  snprintf( file_path, sizeof(file_path), GFX_PATH"loading/%s.txt", load );
528  free( loading_txt );
529  loading_txt = ndata_read( file_path, &nbuf );
530 
531  /* Create the stars. */
532  background_initStars( 1000 );
533 
534  /* Clean up. */
535  PHYSFS_freeList( loadpaths );
536  free( loadscreens );
537 }

546 void loadscreen_render( double done, const char *msg )
547 {
548  const double SHIP_IMAGE_WIDTH = 512.;
549  const double SHIP_IMAGE_HEIGHT = 512.;
550  glColour col;
551  double bx;
552  double by;
553  double x;
554  double y;
555  double w;
556  double h;
557  double rh;
558  SDL_Event event;
559 
560  /* Clear background. */
561  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
562 
563  /* Draw stars. */
564  background_renderStars( 0. );
565 
566  /*
567  * Dimensions.
568  */
569  /* Image. */
570  bx = (SCREEN_W-SHIP_IMAGE_WIDTH)/2.;
571  by = (SCREEN_H-SHIP_IMAGE_HEIGHT)/2.;
572  /* Loading bar. */
573  w = SCREEN_W * 0.4;
574  h = SCREEN_H * 0.02;
575  rh = h + gl_defFont.h + 4.;
576  x = (SCREEN_W-w)/2.;
577  y = (SCREEN_H-SHIP_IMAGE_HEIGHT)/2. - rh - 5.;
578 
579  /* Draw loading screen image. */
580  if (loading != NULL)
581  gl_blitScale( loading, bx, by, SHIP_IMAGE_WIDTH, SHIP_IMAGE_HEIGHT, NULL );
582  if (loading_txt != NULL) {
583  int tw = gl_printWidthRaw( &loading_font, loading_txt );
584  gl_printRaw( &loading_font, bx+SHIP_IMAGE_WIDTH-tw, by+20, &cFontWhite, 1, loading_txt );
585  }
586 
587  /* Draw progress bar. */
588  /* BG. */
589  col.r = cBlack.r;
590  col.g = cBlack.g;
591  col.b = cBlack.b;
592  col.a = 0.7;
593  gl_renderRect( x-2., y-2., w+4., rh+4., &col );
594  /* FG. */
595  col.r = cGreen.r;
596  col.g = cGreen.g;
597  col.b = cGreen.b;
598  col.a = 0.2;
599  gl_renderRect( x+done*w, y, (1.-done)*w, h, &col );
600  col.r = cPrimeGreen.r;
601  col.g = cPrimeGreen.g;
602  col.b = cPrimeGreen.b;
603  col.a = 0.7;
604  gl_renderRect( x, y, done*w, h, &col );
605 
606  /* Draw text. */
607  gl_printRaw( &gl_defFont, x, y + h + 3., &cFontGreen, -1., msg );
608 
609  /* Get rid of events again. */
610  while (SDL_PollEvent(&event));
611 
612  /* Flip buffers. HACK: Also try to catch a late-breaking resize from the WM (...or a crazy user?). */
613  SDL_GL_SwapWindow( gl_screen.window );
614  naev_resize();
615 }

825 static void fps_init (void)
826 {
827 #if HAS_POSIX && defined(CLOCK_MONOTONIC)
828  use_posix_time = 1;
829  /* We must use clock_gettime here instead of gettimeofday mainly because this
830  * way we are not influenced by changes to the time source like say ntp which
831  * could skew up the dt calculations. */
832  if (clock_gettime(CLOCK_MONOTONIC, &global_time)==0)
833  return;
834  WARN( _("clock_gettime failed, disabling POSIX time.") );
835  use_posix_time = 0;
836 #endif /* HAS_POSIX && defined(CLOCK_MONOTONIC) */
837  time_ms = SDL_GetTicks();
838 }
844 static double fps_elapsed (void)
845 {
846  double dt;
847  unsigned int t;
848 
849 #if HAS_POSIX && defined(CLOCK_MONOTONIC)
850  struct timespec ts;
851 
852  if (use_posix_time) {
853  if (clock_gettime(CLOCK_MONOTONIC, &ts)==0) {
854  dt = ts.tv_sec - global_time.tv_sec;
855  dt += (ts.tv_nsec - global_time.tv_nsec) / 1000000000.0;
856  global_time = ts;
857  return dt;
858  }
859  WARN( _("clock_gettime failed!") );
860  }
861 #endif /* HAS_POSIX && defined(CLOCK_MONOTONIC) */
862 
863  t = SDL_GetTicks();
864  dt = (double)(t - time_ms); /* Get the elapsed ms. */
865  dt /= 1000.; /* Convert to seconds. */
866  time_ms = t;
867 
868  return dt;
869 }

875 static void fps_control (void)
876 {
877  double delay;
878  double fps_max;
879 #if HAS_POSIX
880  struct timespec ts;
881 #endif /* HAS_POSIX */
882 
883  /* dt in s */
884  real_dt = fps_elapsed();
885  game_dt = real_dt * dt_mod; /* Apply the modifier. */
886 
887  /* if fps is limited */
888  if (!conf.vsync && conf.fps_max != 0) {
889  fps_max = 1./(double)conf.fps_max;
890  if (real_dt < fps_max) {
891  delay = fps_max - real_dt;
892 #if HAS_POSIX
893  ts.tv_sec = floor( delay );
894  ts.tv_nsec = fmod( delay, 1. ) * 1e9;
895  nanosleep( &ts, NULL );
896 #else /* HAS_POSIX */
897  SDL_Delay( (unsigned int)(delay * 1000) );
898 #endif /* HAS_POSIX */
899  fps_dt += delay; /* makes sure it displays the proper fps */
900  }
901  }

960 static void update_all (void)
961 {
962  int i, n;
963  double nf, microdt, accumdt;
964 
965  if ((real_dt > 0.25) && (fps_skipped==0)) { /* slow timers down and rerun calculations */
966  fps_skipped = 1;
967  return;
968  }
969  else if (game_dt > fps_min) { /* we'll force a minimum FPS for physics to work alright. */
970 
971  /* Number of frames. */
972  nf = ceil( game_dt / fps_min );
973  microdt = game_dt / nf;
974  n = (int) nf;
975 
976  /* Update as much as needed, evenly. */
977  accumdt = 0.;
978  for (i=0; i<n; i++) {
979  update_routine( microdt, 0 );
980  /* OK, so we need a bit of hackish logic here in case we are chopping up a
981  * very large dt and it turns out time compression changes so we're now
982  * updating in "normal time compression" zone. This amounts to many updates
983  * being run when time compression has changed and thus can cause, say, the
984  * player to exceed their target position or get mauled by an enemy ship.
985  */
986  accumdt += microdt;
987  if (accumdt > dt_mod*real_dt)
988  break;
989  }
990 
991  /* Note we don't touch game_dt so that fps_display works well */
992  }
993  else /* Standard, just update with the last dt */
994  update_routine( game_dt, 0 );
995 
996  fps_skipped = 0;
997 }
