26 /*
27  * SOUND OVERVIEW
28  *
29  *
30  * We use a priority virtual voice system with pre-allocated buffers.
31  *
32  * Naming:
33  * buffer - sound sample
34  * source - openal object that plays sound
35  * voice - virtual object that wants to play sound
36  *
37  *
38  * 1) First we allocate all the buffers based on what we find inside the
39  * datafile.
40  * 2) Then we allocate all the possible sources (giving the music system
41  * what it needs).
42  * 3) Now we allow the user to dynamically create voices, these voices will
43  * always try to grab a source from the source pool. If they can't they
44  * will pretend to play the buffer.
45  * 4) Every so often we'll check to see if the important voices are being
46  * played and take away the sources from the lesser ones.
47  *
48  *
49  * EFX
50  *
51  * We use multiple effects, namely:
52  *
53  * - Air absorption factor
54  * - Reverb
55  */

200 int sound_al_init (void)
201 {
202  int ret;
203  ALuint s;
204  ALint freq;
205  ALint attribs[4] = { 0, 0, 0, 0 };
206 
207  /* Default values. */
208  ret = 0;
209 
210  /* we'll need a mutex */
211  sound_lock = SDL_CreateMutex();
212  soundLock();
213 
214  /* opening the default device */
215  al_device = alcOpenDevice(NULL);
216  if (al_device == NULL) {
217  WARN(_("Unable to open default sound device"));
218  ret = -1;
219  goto snderr_dev;
220  }
221 
222  /* Query EFX extension. */
223  if (conf.al_efx) {
224  al_info.efx = alcIsExtensionPresent( al_device, "ALC_EXT_EFX" );
225  if (al_info.efx == AL_TRUE) {
226  attribs[0] = ALC_MAX_AUXILIARY_SENDS;
227  attribs[1] = 4;
228  }
229  }
230  else
231  al_info.efx = AL_FALSE;
232 
233  /* Create the OpenAL context */
234  al_context = alcCreateContext( al_device, attribs );
235  if (al_context == NULL) {
236  WARN(_("Unable to create OpenAL context"));
237  ret = -2;
238  goto snderr_ctx;
239  }
240 
241  /* Clear the errors */
242  alGetError();
243 
244  /* Set active context */
245  if (alcMakeContextCurrent( al_context )==AL_FALSE) {
246  WARN(_("Failure to set default context"));
247  ret = -4;
248  goto snderr_act;
249  }
250 
251  /* Get context information. */
252  alcGetIntegerv( al_device, ALC_FREQUENCY, sizeof(freq), &freq );
253 
254  /* Try to enable EFX. */
255  if (al_info.efx == AL_TRUE) {
256  al_enableEFX();
257  }
258  else {
259  al_info.efx_reverb = AL_FALSE;
260  al_info.efx_echo = AL_FALSE;
261  }
262 
263  /* Check for errors. */
264  al_checkErr();
265 
266  /* Start allocating the sources - music has already taken theirs */
267  source_nstack = 0;
268  source_mstack = conf.snd_voices;
269  source_stack = malloc( sizeof( ALuint ) * source_mstack );
270  while (source_nstack < conf.snd_voices) {
271  alGenSources( 1, &s );
272  source_stack[source_nstack] = s;
273 
274  /* How OpenAL distance model works:
275  *
276  * Clamped:
277  * gain = distance_function( CLAMP( AL_REFERENCE_DISTANCE, AL_MAX_DISTANCE, distance ) );
278  *
279  * Distance functions:
280  * AL_REFERENCE_DISTANCE
281  * * Inverse = ------------------------------------------------------------------------------
282  * AL_REFERENCE_DISTANCE + AL_ROLLOFF_FACTOR ( distance - AL_REFERENCE_DISTANCE )
283  *
284  * 1 - AL_ROLLOFF_FACTOR ( distance - AL_REFERENCE_DISTANCE )
285  * * Linear = ----------------------------------------------------------
286  * AL_MAX_DISTANCE - AL_REFERENCE_DISTANCE
287  *
288  * / distance \ -AL_ROLLOFF_FACTOR
289  * * Exponential = | --------------------- |
290  * \ AL_REFERENCE_DISTANCE /
291  *
292  *
293  * Some values:
294  *
295  * model falloff reference 100 1000 5000 10000
296  * linear 1 500 1.000 0.947 0.526 0.000
297  * inverse 1 500 1.000 0.500 0.100 0.050
298  * exponent 1 500 1.000 0.500 0.100 0.050
299  * inverse 0.5 500 1.000 0.667 0.182 0.095
300  * exponent 0.5 500 1.000 0.707 0.316 0.223
301  * inverse 2 500 1.000 0.333 0.052 0.026
302  * exponent 2 500 1.000 0.250 0.010 0.003
303  */
304  alSourcef( s, AL_REFERENCE_DISTANCE, 500. ); /* Close distance to clamp at (doesn't get louder). */
305  alSourcef( s, AL_MAX_DISTANCE, 25000. ); /* Max distance to clamp at (doesn't get quieter). */
306  alSourcef( s, AL_ROLLOFF_FACTOR, 1. ); /* Determines how it drops off. */
307 
308  /* Set the filter. */
309  if (al_info.efx == AL_TRUE)
310  alSource3i( s, AL_AUXILIARY_SEND_FILTER, efx_directSlot, 0, AL_FILTER_NULL );
311 
312  /* Check for error. */
313  if (alGetError() == AL_NO_ERROR)
314  source_nstack++;
315  else
316  break;
317  }
318 
319  if ( source_nstack == 0 ) {
320  WARN( _( "OpenAL failed to initialize sources" ) );
321  source_mstack = 0;
322  free( source_stack );
323  source_stack = NULL;
324  }
325  else {
326  /* Reduce ram usage. */
327  source_mstack = source_nstack;
328  source_stack = realloc( source_stack, sizeof( ALuint ) * source_mstack );
329  /* Copy allocated sources to total stack. */
330  source_ntotal = source_mstack;
331  source_total = malloc( sizeof( ALuint ) * source_mstack );
332  memcpy( source_total, source_stack, sizeof( ALuint ) * source_mstack );
333 
334  /* Copy allocated sources to all stack. */
335  source_nall = source_mstack;
336  source_all = malloc( sizeof( ALuint ) * source_mstack );
337  memcpy( source_all, source_stack, sizeof( ALuint ) * source_mstack );
338  }
339 
340  /* Set up how sound works. */
341  alDistanceModel( AL_INVERSE_DISTANCE_CLAMPED ); /* Clamping is fundamental so it doesn't sound like crap. */
342  alDopplerFactor( 1. );
343  sound_al_env( SOUND_ENV_NORMAL, 0. );
344 
345  /* Check for errors. */
346  al_checkErr();
347 
348  /* we can unlock now */
349  soundUnlock();
350 
351  /* debug magic */
352  DEBUG(_("OpenAL started: %d Hz"), freq);
353  DEBUG(_("Renderer: %s"), alGetString(AL_RENDERER));
354  if (al_info.efx)
355  DEBUG(_("Version: %s with EFX %d.%d"), alGetString(AL_VERSION),
356  al_info.efx_major, al_info.efx_minor);
357  else
358  DEBUG(_("Version: %s without EFX"), alGetString(AL_VERSION));
359  DEBUG_BLANK();
360 
361  return ret;
362 
363  /*
364  * error handling
365  */
366 snderr_act:
367  alcDestroyContext( al_context );
368 snderr_ctx:
369  al_context = NULL;
370  alcCloseDevice( al_device );
371 snderr_dev:
372  al_device = NULL;
373  soundUnlock();
374  SDL_DestroyMutex( sound_lock );
375  sound_lock = NULL;
376  return ret;
377 }

397  alcGetIntegerv( al_device, ALC_EFX_MAJOR_VERSION, 1, &al_info.efx_major );
398  alcGetIntegerv( al_device, ALC_EFX_MINOR_VERSION, 1, &al_info.efx_minor );
399 
400  /* Get function pointers. */
401  nalGenAuxiliaryEffectSlots = alGetProcAddress( "alGenAuxiliaryEffectSlots" );
402  nalDeleteAuxiliaryEffectSlots = alGetProcAddress( "alDeleteAuxiliaryEffectSlots" );
403  nalIsAuxiliaryEffectSlot = alGetProcAddress( "alIsAuxiliaryEffectSlot" );
404  nalAuxiliaryEffectSloti = alGetProcAddress( "alAuxiliaryEffectSloti" );
405  nalAuxiliaryEffectSlotiv = alGetProcAddress( "alAuxiliaryEffectSlotiv" );
406  nalAuxiliaryEffectSlotf = alGetProcAddress( "alAuxiliaryEffectSlotf" );
407  nalAuxiliaryEffectSlotfv = alGetProcAddress( "alAuxiliaryEffectSlotfv" );
408  nalGetAuxiliaryEffectSloti = alGetProcAddress( "alGetAuxiliaryEffectSloti" );
409  nalGetAuxiliaryEffectSlotiv = alGetProcAddress( "alGetAuxiliaryEffectSlotiv" );
410  nalGetAuxiliaryEffectSlotf = alGetProcAddress( "alGetAuxiliaryEffectSlotf" );
411  nalGetAuxiliaryEffectSlotfv = alGetProcAddress( "alGetAuxiliaryEffectSlotfv" );
412  nalGenFilters = alGetProcAddress( "alGenFilters" );
413  nalDeleteFilters = alGetProcAddress( "alDeleteFilters" );
414  nalFilteri = alGetProcAddress( "alFilteri" );

422  nalEffectf = alGetProcAddress( "alEffectf" );

426  !nalAuxiliaryEffectSloti || !nalAuxiliaryEffectSlotiv ||
427  !nalAuxiliaryEffectSlotf || !nalAuxiliaryEffectSlotfv ||

429  !nalGetAuxiliaryEffectSlotf || !nalGetAuxiliaryEffectSlotfv ||

432  !nalGenEffects || !nalDeleteEffects ||
433  !nalEffecti || !nalEffectiv || !nalEffectf || !nalEffectfv) {
434  DEBUG(_("OpenAL EFX functions not found, disabling EFX."));
435  al_info.efx = AL_FALSE;
436  return -1;
437  }
438 
439  /* Create auxiliary slot. */
440  nalGenAuxiliaryEffectSlots( 1, &efx_directSlot );
441 
442  /* Create reverb effect. */
443  nalGenEffects( 1, &efx_reverb );
444  nalEffecti( efx_reverb, AL_EFFECT_TYPE, AL_EFFECT_REVERB );
445  if (alGetError() != AL_NO_ERROR) {
446  DEBUG(_("OpenAL Reverb not found, disabling."));
447  al_info.efx_reverb = AL_FALSE;
448  nalDeleteEffects( 1, &efx_reverb );
449  }
450  else {
451  al_info.efx_reverb = AL_TRUE;
452 
453  /* Set Reverb parameters. */
454  /*nalEffectf( efx_reverb, AL_REVERB_DECAY_TIME, 15. );*/
455  }
456 
457  /* Create echo effect. */
458  nalGenEffects( 1, &efx_echo );

462  al_info.efx_echo = AL_FALSE;
463  nalDeleteEffects( 1, &efx_echo );
464  }
465  else {
466  al_info.efx_echo = AL_TRUE;
467 
468  /* Set Echo parameters. */
469  nalEffectf( efx_echo, AL_ECHO_DELAY, 0.207 );
470  }
471 
472  /* Set up the listener. */
473  alListenerf( AL_METERS_PER_UNIT, 5. );
474 
475  /* Check for errors. */
476  al_checkErr();
477 
478  return 0;

485 void sound_al_exit (void)
486 {
487  int i;
488 
489  soundLock();
490 
491  /* Free groups. */
492  for (i=0; i<al_ngroups; i++) {
493  free(al_groups[i].sources);
494  al_groups[i].sources = NULL;
495  al_groups[i].nsources = 0;
496  }
497  free(al_groups);
498  al_groups = NULL;
499  al_ngroups = 0;
500 
501  /* Free stacks. */
502  if (source_all != NULL) {
503  alSourceStopv( source_nall, source_all );
504  alDeleteSources( source_nall, source_all );
505  free(source_all);
506  }
507  source_all = NULL;
508  source_nall = 0;
509  free(source_total);
510  source_total = NULL;
511  source_ntotal = 0;
512  free(source_stack);
513  source_stack = NULL;
514  source_nstack = 0;
515  source_mstack = 0;
516 
517  /* Clean up EFX stuff. */
518  if (al_info.efx == AL_TRUE) {
519  nalDeleteAuxiliaryEffectSlots( 1, &efx_directSlot );
520  if (al_info.efx_reverb == AL_TRUE)
521  nalDeleteEffects( 1, &efx_reverb );
522  if (al_info.efx_echo == AL_TRUE)
523  nalDeleteEffects( 1, &efx_echo );
524  }
525 
526  /* Clean up global stuff. */
527  if (al_context) {
528  alcMakeContextCurrent(NULL);
529  alcDestroyContext( al_context );
530  }
531  if (al_device)
532  alcCloseDevice( al_device );
533 
534  soundUnlock();
535  SDL_DestroyMutex( sound_lock );
536 }

547 static int sound_al_loadWav( ALuint *buf, SDL_RWops *rw )
548 {
549  SDL_AudioSpec wav_spec;
550  Uint32 wav_length;
551  Uint8 *wav_buffer;
552  ALenum format;
553 
554  SDL_RWseek( rw, 0, SEEK_SET );
555 
556  /* Load WAV. */
557  if (SDL_LoadWAV_RW( rw, 0, &wav_spec, &wav_buffer, &wav_length) == NULL) {
558  WARN(_("SDL_LoadWav_RW failed: %s"), SDL_GetError());
559  return -1;
560  }
561 
562  /* Handle format. */
563  switch (wav_spec.format) {
564  case AUDIO_U8:
565  case AUDIO_S8:
566  format = (wav_spec.channels==1) ? AL_FORMAT_MONO8 : AL_FORMAT_STEREO8;
567  break;
568  case AUDIO_U16LSB:
569  case AUDIO_S16LSB:
570  format = (wav_spec.channels==1) ? AL_FORMAT_MONO16 : AL_FORMAT_STEREO16;
571  break;
572  case AUDIO_U16MSB:
573  case AUDIO_S16MSB:
574  WARN( _("Big endian WAVs unsupported!") );
575  return -1;
576  default:
577  WARN( _("Invalid WAV format!") );
578  return -1;
579  }
580 
581  /* Load into openal. */
582  soundLock();
583  /* Create new buffer. */
584  alGenBuffers( 1, buf );
585  /* Put into the buffer. */
586  alBufferData( *buf, format, wav_buffer, wav_length, wav_spec.freq );
587  soundUnlock();
588 
589  /* Clean up. */
590  free( wav_buffer );
591  al_checkErr();
592  return 0;
593 }

623 static int sound_al_loadOgg( ALuint *buf, OggVorbis_File *vf )
624 {
625  int ret;
626  long i;
627  int section;
628  vorbis_info *info;
629  ALenum format;
630  ogg_int64_t len;
631  char *data;
632 
633  /* Finish opening the file. */
634  ret = ov_test_open(vf);
635  if (ret) {
636  WARN(_("Failed to finish loading Ogg file: %s"), vorbis_getErr(ret) );
637  return -1;
638  }
639 
640  /* Get file information. */
641  info = ov_info( vf, -1 );
642  format = (info->channels == 1) ? AL_FORMAT_MONO16 : AL_FORMAT_STEREO16;
643  len = ov_pcm_total( vf, -1 ) * info->channels * 2;
644 
645  /* Allocate memory. */
646  data = malloc( len );
647 
648  /* Fill buffer. */
649  i = 0;
650  while (i < len) {
651  /* Fill buffer with data in the 16 bit signed samples format. */
652  i += ov_read( vf, &data[i], len-i, (SDL_BYTEORDER == SDL_BIG_ENDIAN), 2, 1, &section );
653  }
654 
655  soundLock();
656  /* Create new buffer. */
657  alGenBuffers( 1, buf );
658  /* Put into buffer. */
659  alBufferData( *buf, format, data, len, info->rate );
660  al_checkErr();
661  soundUnlock();
662 
663  /* Clean up. */
664  free(data);
665  ov_clear(vf);
666 
667  return 0;
668 }

700  }

703

705  al_checkErr();

720 int sound_al_load( alSound *snd, SDL_RWops *rw, const char *name )
721 {
722  int ret;
723  ALint freq, bits, channels, size;
724 
725  ret = sound_al_buffer( &snd->buf, rw, name );
726  if (ret != 0) {
727  WARN(_("Failed to load sound file '%s'."), name);
728  return ret;
729  }
730 
731  soundLock();
732 
733  /* Get the length of the sound. */
734  alGetBufferi( snd->buf, AL_FREQUENCY, &freq );
735  alGetBufferi( snd->buf, AL_BITS, &bits );
