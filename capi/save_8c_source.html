99  char file[PATH_MAX];
100  xmlDocPtr doc;
101  xmlTextWriterPtr writer;
102 
103  /* Do not save if saving is off. */
104  if (player_isFlag(PLAYER_NOSAVE))
105  return 0;
106 
107  /* Create the writer. */
108  writer = xmlNewTextWriterDoc(&doc, conf.save_compress);
109  if (writer == NULL) {
110  ERR(_("testXmlwriterDoc: Error creating the xml writer"));
111  return -1;
112  }
113 
114  /* Set the writer parameters. */
115  xmlw_setParams( writer );
116 
117  /* Start element. */
118  xmlw_start(writer);
119  xmlw_startElem(writer,"naev_save");
120 
121  /* Save the version and such. */
122  xmlw_startElem(writer,"version");
123  xmlw_elem( writer, "naev", "%s", VERSION );
124  xmlw_elem( writer, "data", "%s", start_name() );
125  xmlw_endElem(writer); /* "version" */
126 
127  /* Save last played. */
128  xmlw_saveTime( writer, "last_played", time(NULL) );
129 
130  /* Save the data. */
131  if (save_data(writer) < 0) {
132  ERR(_("Trying to save game data"));
133  goto err_writer;
134  }
135 
136  /* Finish element. */
137  xmlw_endElem(writer); /* "naev_save" */
138  xmlw_done(writer);
139 
140  /* Write to file. */
141  if (PHYSFS_mkdir("saves") == 0) {
142  snprintf(file, sizeof(file), "%s/saves", PHYSFS_getWriteDir());
143  WARN(_( "Dir '%s' does not exist and unable to create: %s" ), file, PHYSFS_getErrorByCode( PHYSFS_getLastErrorCode() ) );
144  goto err_writer;
145  }
146  snprintf(file, sizeof(file), "saves/%s.ns", player.name);
147 
148  /* Back up old saved game. */
149  if (!save_loaded) {
150  if (ndata_backupIfExists(file) < 0) {
151  WARN(_("Aborting save..."));
152  goto err_writer;
153  }
154  }
155  save_loaded = 0;
156 
157  /* Critical section, if crashes here player's game gets corrupted.
158  * Luckily we have a copy just in case... */
159  xmlFreeTextWriter(writer);
160  snprintf(file, sizeof(file), "%s/saves/%s.ns", PHYSFS_getWriteDir(), player.name); /* TODO: write via physfs */
161  if (xmlSaveFileEnc(file, doc, "UTF-8") < 0) {
162  WARN(_("Failed to write saved game! You'll most likely have to restore it by copying your backup saved game over your current saved game."));
163  goto err;
164  }
165  xmlFreeDoc(doc);
166 
167  return 0;
