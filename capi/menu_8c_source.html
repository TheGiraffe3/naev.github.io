107  * Background system for the menu.

109 static int menu_main_bkg_system (void)
110 {
111  const nsave_t *ns;
112  const char *sys;
113  Planet *pnt;
114  double cx, cy;
115 
116  /* Clean pilots. */
117  pilots_cleanAll();
118  sys = NULL;
119 
120  /* Refresh saves. */
121  load_refresh();
122 
123  /* Load saves. */
124  ns = load_getList();
125 
126  /* Try to apply unidiff. */
127  if (array_size( ns ) > 0) {
128  load_gameDiff( ns[0].path );
129 
130  /* Get start position. */
131  if (planet_exists( ns[0].planet )) {
132  pnt = planet_get( ns[0].planet );
133  if (pnt != NULL) {
134  sys = planet_getSystem( ns[0].planet );
135  if (sys != NULL) {
136  cx = pnt->pos.x;
137  cy = pnt->pos.y;
138  }
139  }
140  }
141  }
142 
143  /* Fallback if necessary. */
144  if (sys == NULL) {
145  sys = start_system();
146  start_position( &cx, &cy );
147  }
148 
149  /* Have to normalize values by zoom. */
150  cx += SCREEN_W/4. / conf.zoom_far;
151  cy += SCREEN_H/8. / conf.zoom_far;
152 
153  /* Initialize. */
154  space_init( sys );
155  cam_setTargetPos( cx, cy, 0 );
156  cam_setZoom( conf.zoom_far );
157  pause_setSpeed( 1. );
158  sound_setSpeed( 1. );
159 
160  return 0;
161 }

167 void menu_main (void)
168 {
169  int offset_logo, offset_wdw, freespace;

171  glTexture *tex;
172  int h, y;
173 
174  if (menu_isOpen(MENU_MAIN)) {
175  WARN( _("Menu main is already open.") );

177  }
178 
179  /* Close all open windows. */
180  toolkit_closeAll();
181 
182  /* Clean up GUI - must be done before using SCREEN_W or SCREEN_H. */
183  gui_cleanup();
184  player_soundStop(); /* Stop sound. */
185  player_resetSpeed();
186 
187  /* Play load music. */
188  music_choose("load");
189 
190  /* Load background and friends. */
191  tex = gl_newImage( GFX_PATH"Naev.webp", 0 );
192  main_naevLogo = tex;
193  menu_main_bkg_system();
194 
195  /* Set dimensions */
196  y = 20 + (BUTTON_HEIGHT+20)*4;
197  h = y + 80;
198  if (conf.devmode) {
199  h += BUTTON_HEIGHT + 20;
200  y += BUTTON_HEIGHT + 20;
201  }
202 
203  /* Calculate Logo and window offset. */
204  freespace = SCREEN_H - tex->sh - h;
205  if (freespace < 0) { /* Not enough freespace, this can get ugly. */
206  offset_logo = SCREEN_W - tex->sh;
207  offset_wdw = 0;
208  }
209  /* Otherwise space evenly. */
210  else {
211  offset_logo = -freespace/4;
212  offset_wdw = freespace/2;
213  }
214 
215  /* create background image window */
216  bwid = window_create( "wdwBG", "", -1, -1, -1, -1 );
217  window_onClose( bwid, menu_main_cleanBG );
218  window_setBorder( bwid, 0 );
219  window_addImage( bwid, (SCREEN_W-tex->sw)/2., offset_logo, 0, 0, "imgLogo", tex, 0 );
220  window_addText( bwid, 0, 10, SCREEN_W, 30., 1, "txtBG", NULL,
221  &cWhite, naev_version(1) );
222 
223  /* create menu window */
224  wid = window_create( "wdwMainMenu", _("Main Menu"), -1, offset_wdw, MAIN_WIDTH, h );
225  window_setCancel( wid, main_menu_promptClose );
226 
227  /* Buttons. */
228  window_addButtonKey( wid, 20, y, BUTTON_WIDTH, BUTTON_HEIGHT,
229  "btnLoad", _("Load Game"), menu_main_load, SDLK_l );
230  y -= BUTTON_HEIGHT+20;
231  window_addButtonKey( wid, 20, y, BUTTON_WIDTH, BUTTON_HEIGHT,
232  "btnNew", _("New Game"), menu_main_new, SDLK_n );
233  y -= BUTTON_HEIGHT+20;
234  if (conf.devmode) {
235  window_addButtonKey( wid, 20, y, BUTTON_WIDTH, BUTTON_HEIGHT,
236  "btnEditor", _("Editors"), menu_editors_open, SDLK_e );
237  y -= BUTTON_HEIGHT+20;
238  }
239  window_addButtonKey( wid, 20, y, BUTTON_WIDTH, BUTTON_HEIGHT,
240  "btnOptions", _("Options"), menu_options_button, SDLK_o );
241  y -= BUTTON_HEIGHT+20;

274  int menu_id, bg_id;

285 
286  freespace = SCREEN_H - main_naevLogo->sh - h;

295 
296  window_moveWidget( bg_id, "imgLogo",
297  (bgw - main_naevLogo->sw)/2., offset_logo );
298 
299  window_dimWidget( bg_id, "txtBG", &tw, &th );
300 
301  if (tw > SCREEN_W) {
302  /* RIP abstractions. X must be set manually because window_moveWidget
303  * transforms negative coordinates. */
304  wgt = window_getwgt( bg_id, "txtBG" );
305  if (wgt)
306  wgt->x = (SCREEN_W - tw) / 2;
307  }
308  else
309  window_moveWidget( bg_id, "txtBG", (SCREEN_W - tw)/2, 10. );
310 
311  window_move( menu_id, -1, offset_wdw );
312 }
