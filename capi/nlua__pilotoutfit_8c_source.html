170 static int poL_state( lua_State *L )
171 {
172  PilotOutfitSlot *po = luaL_validpilotoutfit(L,1);
173  const char *state = luaL_optstring(L,2,NULL);
174  PilotOutfitState pos = po->state;
175 
176  if (state==NULL || strcmp(state,"off")==0)
177  po->state = PILOT_OUTFIT_OFF;
178  else if (strcmp(state,"warmup")==0)
179  po->state = PILOT_OUTFIT_WARMUP;
180  else if (strcmp(state,"on")==0)
181  po->state = PILOT_OUTFIT_ON;
182  else if (strcmp(state,"cooldown")==0)
183  po->state = PILOT_OUTFIT_COOLDOWN;
184  else
185  NLUA_ERROR( L, _("Unknown PilotOutfit state '%s'!"), state );
186 
187  /* Mark as modified if state changed. */
188  if (pos != po->state)
189  pilotoutfit_modified = 1;
190 
191  return 0;
192 }
