174 static int poL_state( lua_State *L )
175 {
176  PilotOutfitSlot *po = luaL_validpilotoutfit(L,1);
177  const char *state = luaL_optstring(L,2,NULL);
178  PilotOutfitState pos = po->state;
179 
180  if (state==NULL || strcmp(state,"off")==0)
181  po->state = PILOT_OUTFIT_OFF;
182  else if (strcmp(state,"warmup")==0)
183  po->state = PILOT_OUTFIT_WARMUP;
184  else if (strcmp(state,"on")==0)
185  po->state = PILOT_OUTFIT_ON;
186  else if (strcmp(state,"cooldown")==0)
187  po->state = PILOT_OUTFIT_COOLDOWN;
188  else
189  NLUA_ERROR( L, _("Unknown PilotOutfit state '%s'!"), state );
190 
191  /* Mark as modified if state changed. */
192  if (pos != po->state)
193  pilotoutfit_modified = 1;
194 
195  return 0;
196 }
