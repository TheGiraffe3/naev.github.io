907  fct = faction_getGroup(1);
908  else if (strcmp(cmd, "hostile") == 0)
909  fct = faction_getGroup(3);

911  fct = faction_getGroup(2);
912  else /* Invalid command string. */
913  used = 0;
914  }

917  /* A faction id was given. */
918  f = luaL_validfaction(L, 2);
