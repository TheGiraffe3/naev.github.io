869  used = 1;

872  if (strcmp(cmd, "all") == 0)
873  fct = faction_getGroup(0);
874  else if (strcmp(cmd, "friendly") == 0)
875  fct = faction_getGroup(1);
876  else if (strcmp(cmd, "hostile") == 0)
877  fct = faction_getGroup(3);
878  else if (strcmp(cmd, "neutral") == 0)
879  fct = faction_getGroup(2);
880  else /* Invalid command string. */
881  used = 0;
882  }
883

885  /* A faction id was given. */
886  int f = luaL_validfaction(L, 2);
887  fct = array_create(int);
888  array_push_back(&fct, f);
889  }

892  presence = 0;

894  /* Only count positive presences. */

896  if (v > 0)
897  presence += v;
898  }
899

902
