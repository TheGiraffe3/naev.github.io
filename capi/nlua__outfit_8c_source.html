504 static int outfitL_weapStats( lua_State *L )
505 {
506  double eps, dps, shots;
507  double mod_energy, mod_damage, mod_shots;
508  const Damage *dmg;
509  Outfit *o = luaL_validoutfit( L, 1 );
510  Pilot *p = (lua_ispilot(L,2)) ? luaL_validpilot(L,2) : NULL;
511 
512  /* Just return 0 for non-wapons. */
513  if (o->slot.type != OUTFIT_SLOT_WEAPON) {
514  lua_pushnumber( L, 0. );
515  lua_pushnumber( L, 0. );
516  return 2;
517  }
518 
519  /* Special case beam weapons .*/
520  if (outfit_isBeam(o)) {
521  if (p) {
522  /* Special case due to continuous fire. */
523  if (o->type == OUTFIT_TYPE_BEAM) {
524  mod_energy = p->stats.fwd_energy;
525  mod_damage = p->stats.fwd_damage;
526  mod_shots = 1. / p->stats.fwd_firerate;
527  }
528  else {
529  mod_energy = p->stats.tur_energy;
530  mod_damage = p->stats.tur_damage;
531  mod_shots = 1. / p->stats.tur_firerate;
532  }
533  }
534  else {
535  mod_energy = 1.;
536  mod_damage = 1.;
537  mod_shots = 1.;
538  }
539  shots = outfit_duration(o);
540  mod_shots = shots / (shots + mod_shots * outfit_delay(o));
541  dps = mod_shots * mod_damage * outfit_damage(o)->damage;
542  eps = mod_shots * mod_energy * outfit_energy(o);
543  lua_pushnumber( L, dps );
544  lua_pushnumber( L, eps );
545  return 2;
546  }
547 
548  if (p) {
549  switch (o->type) {
550  case OUTFIT_TYPE_BOLT:
551  mod_energy = p->stats.fwd_energy;
552  mod_damage = p->stats.fwd_damage;
553  mod_shots = 1. / p->stats.fwd_firerate;
554  break;
555  case OUTFIT_TYPE_TURRET_BOLT:
556  mod_energy = p->stats.tur_energy;
557  mod_damage = p->stats.tur_damage;
558  mod_shots = 1. / p->stats.tur_firerate;
559  break;
560  case OUTFIT_TYPE_LAUNCHER:
561  case OUTFIT_TYPE_TURRET_LAUNCHER:
562  mod_energy = 1.;
563  mod_damage = p->stats.launch_damage;
564  mod_shots = 1. / p->stats.launch_rate;
565  break;
566  case OUTFIT_TYPE_BEAM:
567  case OUTFIT_TYPE_TURRET_BEAM:
568  default:
569  return 0;
570  }
571  }
572  else {
573  mod_energy = 1.;
574  mod_damage = 1.;
575  mod_shots = 1.;
576  }
577 
578  shots = 1. / (mod_shots * outfit_delay(o));
579  /* Special case: Ammo-based weapons. */
580  if (outfit_isLauncher(o))
581  dmg = outfit_damage(o->u.lau.ammo);
582  else
583  dmg = outfit_damage(o);
584  dps = shots * mod_damage * dmg->damage;
585  eps = shots * mod_energy * MAX( outfit_energy(o), 0. );
586 
587  lua_pushnumber( L, dps );
588  lua_pushnumber( L, eps );
589  return 2;
590 }
