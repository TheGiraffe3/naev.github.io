533 static int outfitL_weapStats( lua_State *L )
534 {
535  double eps, dps, disable, shots;
536  double mod_energy, mod_damage, mod_shots;
537  double sdmg, admg;
538  const Damage *dmg;
539  Outfit *o = luaL_validoutfit( L, 1 );
540  Pilot *p = (lua_ispilot(L,2)) ? luaL_validpilot(L,2) : NULL;
541 
542  /* Just return 0 for non-wapons. */
543  if (o->slot.type != OUTFIT_SLOT_WEAPON)
544  return 0;
545 
546  /* Special case beam weapons .*/
547  if (outfit_isBeam(o)) {
548  if (p) {
549  /* Special case due to continuous fire. */
550  if (o->type == OUTFIT_TYPE_BEAM) {
551  mod_energy = p->stats.fwd_energy;
552  mod_damage = p->stats.fwd_damage;
553  mod_shots = 1. / p->stats.fwd_firerate;
554  }
555  else {
556  mod_energy = p->stats.tur_energy;
557  mod_damage = p->stats.tur_damage;
558  mod_shots = 1. / p->stats.tur_firerate;
559  }
560  }
561  else {
562  mod_energy = 1.;
563  mod_damage = 1.;
564  mod_shots = 1.;
565  }
566  shots = outfit_duration(o);
567  mod_shots = shots / (shots + mod_shots * outfit_delay(o));
568  dmg = outfit_damage(o);
569  /* Modulate the damage by average of damage types. */
570  dtype_raw( dmg->type, &sdmg, &admg, NULL );
571  mod_damage *= 0.5*(sdmg+admg);
572  /* Calculate good damage estimates. */
573  dps = mod_shots * mod_damage * dmg->damage;
574  disable = mod_shots * mod_damage * dmg->disable;
575  eps = mod_shots * mod_energy * outfit_energy(o);
576  lua_pushnumber( L, dps );
577  lua_pushnumber( L, disable );
578  lua_pushnumber( L, eps );
579  lua_pushnumber( L, outfit_range(o) );
580  return 4;
581  }
582 
583  if (p) {
584  switch (o->type) {
585  case OUTFIT_TYPE_BOLT:
586  mod_energy = p->stats.fwd_energy;
587  mod_damage = p->stats.fwd_damage;
588  mod_shots = 1. / p->stats.fwd_firerate;
589  break;
590  case OUTFIT_TYPE_TURRET_BOLT:
591  mod_energy = p->stats.tur_energy;
592  mod_damage = p->stats.tur_damage;
593  mod_shots = 1. / p->stats.tur_firerate;
594  break;
595  case OUTFIT_TYPE_LAUNCHER:
596  case OUTFIT_TYPE_TURRET_LAUNCHER:
597  mod_energy = 1.;
598  mod_damage = p->stats.launch_damage;
599  mod_shots = 1. / p->stats.launch_rate;
600  break;
601  case OUTFIT_TYPE_BEAM:
602  case OUTFIT_TYPE_TURRET_BEAM:
603  default:
604  return 0;
605  }
606  }
607  else {
608  mod_energy = 1.;
609  mod_damage = 1.;
610  mod_shots = 1.;
611  }
612 
613  shots = 1. / (mod_shots * outfit_delay(o));
614  /* Special case: Ammo-based weapons. */
615  if (outfit_isLauncher(o))
616  dmg = outfit_damage(o->u.lau.ammo);
617  else
618  dmg = outfit_damage(o);
619  if (dmg==NULL)
620  return 0;
621  /* Modulate the damage by average of damage types. */
622  dtype_raw( dmg->type, &sdmg, &admg, NULL );
623  mod_damage *= 0.5*(sdmg+admg);
624  /* Calculate good damage estimates. */
625  dps = shots * mod_damage * dmg->damage;
626  disable = shots * mod_damage * dmg->disable;
627  eps = shots * mod_energy * MAX( outfit_energy(o), 0. );
628 
629  lua_pushnumber( L, dps );
630  lua_pushnumber( L, disable );
631  lua_pushnumber( L, eps );
632  lua_pushnumber( L, outfit_range(o) );
633  lua_pushnumber( L, outfit_trackmin(o) );
634  lua_pushnumber( L, outfit_trackmax(o) );
635  if (outfit_isLauncher(o)) {
636  lua_pushnumber( L, o->u.lau.lockon );
637  return 7;
638  }
639  return 6;
640 }
