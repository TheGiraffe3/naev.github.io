510 static int outfitL_weapStats( lua_State *L )
511 {
512  double eps, dps, disable, shots;
513  double mod_energy, mod_damage, mod_shots;
514  double sdmg, admg;
515  const Damage *dmg;
516  const Outfit *o = luaL_validoutfit( L, 1 );
517  Pilot *p = (lua_ispilot(L,2)) ? luaL_validpilot(L,2) : NULL;
518 
519  /* Just return 0 for non-wapons. */
520  if (o->slot.type != OUTFIT_SLOT_WEAPON)
521  return 0;
522 
523  /* Special case beam weapons .*/
524  if (outfit_isBeam(o)) {
525  if (p) {
526  /* Special case due to continuous fire. */
527  if (o->type == OUTFIT_TYPE_BEAM) {
528  mod_energy = p->stats.fwd_energy;
529  mod_damage = p->stats.fwd_damage;
530  mod_shots = 1. / p->stats.fwd_firerate;
531  }
532  else {
533  mod_energy = p->stats.tur_energy;
534  mod_damage = p->stats.tur_damage;
535  mod_shots = 1. / p->stats.tur_firerate;
536  }
537  }
538  else {
539  mod_energy = 1.;
540  mod_damage = 1.;
541  mod_shots = 1.;
542  }
543  shots = outfit_duration(o);
544  mod_shots = shots / (shots + mod_shots * outfit_delay(o));
545  dmg = outfit_damage(o);
546  /* Modulate the damage by average of damage types. */
547  if( dtype_raw( dmg->type, &sdmg, &admg, NULL ) != 0)
548  NLUA_ERROR(L, _("Outfit has invalid damage type."));
549  mod_damage *= 0.5*(sdmg+admg);
550  /* Calculate good damage estimates. */
551  dps = mod_shots * mod_damage * dmg->damage;
552  disable = mod_shots * mod_damage * dmg->disable;
553  eps = mod_shots * mod_energy * outfit_energy(o);
554  lua_pushnumber( L, dps );
555  lua_pushnumber( L, disable );
556  lua_pushnumber( L, eps );
557  lua_pushnumber( L, outfit_range(o) );
558  return 4;
559  }
560 
561  if (p) {
562  switch (o->type) {
563  case OUTFIT_TYPE_BOLT:
564  mod_energy = p->stats.fwd_energy;
565  mod_damage = p->stats.fwd_damage;
566  mod_shots = 1. / p->stats.fwd_firerate;
567  break;
568  case OUTFIT_TYPE_TURRET_BOLT:
569  mod_energy = p->stats.tur_energy;
570  mod_damage = p->stats.tur_damage;
571  mod_shots = 1. / p->stats.tur_firerate;
572  break;
573  case OUTFIT_TYPE_LAUNCHER:
574  case OUTFIT_TYPE_TURRET_LAUNCHER:
575  mod_energy = 1.;
576  mod_damage = p->stats.launch_damage;
577  mod_shots = 1. / p->stats.launch_rate;
578  break;
579  case OUTFIT_TYPE_BEAM:
580  case OUTFIT_TYPE_TURRET_BEAM:
581  default:
582  return 0;
583  }
584  }
585  else {
586  mod_energy = 1.;
587  mod_damage = 1.;
588  mod_shots = 1.;
589  }
590 
591  shots = 1. / (mod_shots * outfit_delay(o));
592  /* Special case: Ammo-based weapons. */
593  if (outfit_isLauncher(o))
594  dmg = outfit_damage(o->u.lau.ammo);
595  else
596  dmg = outfit_damage(o);
597  if (dmg==NULL)
598  return 0;
599  /* Modulate the damage by average of damage types. */
600  dtype_raw( dmg->type, &sdmg, &admg, NULL );
601  mod_damage *= 0.5*(sdmg+admg);
602  /* Calculate good damage estimates. */
603  dps = shots * mod_damage * dmg->damage;
604  disable = shots * mod_damage * dmg->disable;
605  eps = shots * mod_energy * MAX( outfit_energy(o), 0. );
606 
607  lua_pushnumber( L, dps );
608  lua_pushnumber( L, disable );
609  lua_pushnumber( L, eps );
610  lua_pushnumber( L, outfit_range(o) );
611  lua_pushnumber( L, outfit_trackmin(o) );
612  lua_pushnumber( L, outfit_trackmax(o) );
613  if (outfit_isLauncher(o)) {
614  lua_pushnumber( L, o->u.lau.lockon );
615  return 7;
616  }
617  return 6;
618 }
