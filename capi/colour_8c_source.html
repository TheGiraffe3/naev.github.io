136 void col_hsv2rgb( glColour *c, float h, float s, float v )
137 {
138  float var_h, var_i, var_1, var_2, var_3;
139 
140  if (v > 1)
141  v = 1;
142 
143  if (s == 0) {
144  c->r = v;
145  c->g = v;
146  c->b = v;
147  }
148  else {
149  var_h = h * 6 / 360.;
150  var_i = floor(var_h);
151  var_1 = v * (1 - s);
152  var_2 = v * (1 - s * (var_h - var_i));
153  var_3 = v * (1 - s * (1 - (var_h - var_i)));
154 
155  if (var_i == 0) { c->r = v ; c->g = var_3 ; c->b = var_1; }
156  else if (var_i == 1) { c->r = var_2 ; c->g = v ; c->b = var_1; }
157  else if (var_i == 2) { c->r = var_1 ; c->g = v ; c->b = var_3; }
158  else if (var_i == 3) { c->r = var_1 ; c->g = var_2 ; c->b = v; }
159  else if (var_i == 4) { c->r = var_3 ; c->g = var_1 ; c->b = v; }
160  else { c->r = v ; c->g = var_1 ; c->b = var_2; }
161  }
162 }

179 void col_rgb2hsv( float *H, float *S, float *V, float R, float G, float B )
180 {
181  float H1, S1, V1;
182 #ifdef HSV_TRAVIS
183  float R1, G1, B1;
184 #endif /* HSV_TRAVIS */
185  float max, min, diff;
186 
187  max = max3( R, G, B );
188  min = min3( R, G, B );
189  diff = max - min;
190 
191  if (max == 0)
192  H1 = S1 = V1 = 0;
193  else {
194  V1 = max;
195  S1 = diff/max;
196  if (S1 == 0)
197  /* H1 is undefined, but give it a value anyway */
198  H1 = 0;
199  else {
200 #ifdef HSV_TRAVIS
201  R1 = (max - R)/diff;
202  G1 = (max - G)/diff;
203  B1 = (max - B)/diff;
204 
205  if ((R == max) && (G == min))
206  H1 = 5 + B1;
207  else {
208  if ((R == max) && (G != min))
209  H1 = 1 - G1;
210  else {
211  if ((G == max) && (B == min))
212  H1 = 1 + R1;
213  else {
214  if ((G == max) && (B != min))
215  H1 = 3 - B1;
216  else {
217  if (R == max)
218  H1 = 3 + G1;
219  else
220  H1 = 5 - R1;
221  }
222  }
223  }
224  }
225 
226  H1 *= 60; /* convert to range [0, 360] degrees */
227 #else /* HSV_TRAVIS */
228  H1 = 0.; /* Shuts up Clang. */
229  /* assume Foley & VanDam HSV */
230  if (R == max)
231  H1 = (G - B)/diff;
232  if (G == max)
233  H1 = 2 + (B - R)/diff;
234  if (B == max)
235  H1 = 4 + (R - G)/diff;
236 
237  H1 *= 60; /* convert to range [0, 360] degrees */
238  if (H1 < 0)
239  H1 += 360;
240 #endif /* HSV_TRAVIS */
241  }
242  }
243  *H = H1;
244  *S = S1;
245  *V = V1;
246 }
