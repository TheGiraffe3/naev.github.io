177 void col_hsv2rgb( glColour *c, float h, float s, float v )
178 {
179  float var_h, var_i, var_1, var_2, var_3;
180 
181  if (v > 1)
182  v = 1;
183 
184  if (s == 0) {
185  c->r = v;
186  c->g = v;
187  c->b = v;
188  }
189  else {
190  var_h = h * 6 / 360.;
191  var_i = floor(var_h);
192  var_1 = v * (1 - s);
193  var_2 = v * (1 - s * (var_h - var_i));
194  var_3 = v * (1 - s * (1 - (var_h - var_i)));
195 
196  if (var_i == 0) { c->r = v ; c->g = var_3 ; c->b = var_1; }
197  else if (var_i == 1) { c->r = var_2 ; c->g = v ; c->b = var_1; }
198  else if (var_i == 2) { c->r = var_1 ; c->g = v ; c->b = var_3; }
199  else if (var_i == 3) { c->r = var_1 ; c->g = var_2 ; c->b = v; }
200  else if (var_i == 4) { c->r = var_3 ; c->g = var_1 ; c->b = v; }
201  else { c->r = v ; c->g = var_1 ; c->b = var_2; }
202  }
203 
204  c->r = gammaToLinear( c->r );
205  c->g = gammaToLinear( c->g );
206  c->b = gammaToLinear( c->b );
207 }

224 void col_rgb2hsv( float *H, float *S, float *V, float R, float G, float B )
225 {
226  float H1, S1, V1;
227 #ifdef HSV_TRAVIS
228  float R1, G1, B1;
229 #endif /* HSV_TRAVIS */
230  float max, min, diff;
231 
232  R = gammaToLinear( R );
233  G = gammaToLinear( G );
234  B = gammaToLinear( B );
235 
236  max = max3( R, G, B );
237  min = min3( R, G, B );
238  diff = max - min;
239 
240  if (max == 0)
241  H1 = S1 = V1 = 0;
242  else {
243  V1 = max;
244  S1 = diff/max;
245  if (S1 == 0)
246  /* H1 is undefined, but give it a value anyway */
247  H1 = 0;
248  else {
249 #ifdef HSV_TRAVIS
250  R1 = (max - R)/diff;
251  G1 = (max - G)/diff;
252  B1 = (max - B)/diff;
253 
254  if ((R == max) && (G == min))
255  H1 = 5 + B1;
256  else {
257  if ((R == max) && (G != min))
258  H1 = 1 - G1;
259  else {
260  if ((G == max) && (B == min))
261  H1 = 1 + R1;
262  else {
263  if ((G == max) && (B != min))
264  H1 = 3 - B1;
265  else {
266  if (R == max)
267  H1 = 3 + G1;
268  else
269  H1 = 5 - R1;
270  }
271  }
272  }
273  }
274 
275  H1 *= 60; /* convert to range [0, 360] degrees */
276 #else /* HSV_TRAVIS */
277  H1 = 0.; /* Shuts up Clang. */
278  /* assume Foley & VanDam HSV */
279  if (R == max)
280  H1 = (G - B)/diff;
281  if (G == max)
282  H1 = 2 + (B - R)/diff;
283  if (B == max)
284  H1 = 4 + (R - G)/diff;
285 
286  H1 *= 60; /* convert to range [0, 360] degrees */
287  if (H1 < 0)
288  H1 += 360;
289 #endif /* HSV_TRAVIS */
290  }
291  }
292  *H = H1;
293  *S = S1;
294  *V = V1;
295 }
