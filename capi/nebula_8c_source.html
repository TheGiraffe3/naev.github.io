102 int nebu_resize (void)
103 {
104  double scale;
105  GLfloat fbo_w, fbo_h;
106 
107  scale = conf.nebu_scale * gl_screen.scale;
108  fbo_w = round(gl_screen.nw/scale);
109  fbo_h = round(gl_screen.nh/scale);
110  if (scale == nebu_scale && fbo_w == nebu_render_w && fbo_h == nebu_render_h)
111  return 0;
112 
113  nebu_scale = scale;
114  nebu_render_w = fbo_w;
115  nebu_render_h = fbo_h;
116  nebu_dofbo = (nebu_scale != 1.);
117  glDeleteTextures( 1, &nebu_tex );
118  glDeleteFramebuffers( 1, &nebu_fbo );
119 
120  if (nebu_dofbo)
121  gl_fboCreate( &nebu_fbo, &nebu_tex, nebu_render_w, nebu_render_h );
122 
123  /* Set up the matrices. */
124  nebu_render_P = gl_Matrix4_Identity();
125  nebu_render_P = gl_Matrix4_Translate(nebu_render_P, -nebu_render_w/2., -nebu_render_h/2., 0. );
126  nebu_render_P = gl_Matrix4_Scale(nebu_render_P, nebu_render_w, nebu_render_h, 1);
127  glUseProgram(shaders.nebula_background.program);
128  gl_Matrix4_Uniform(shaders.nebula_background.projection, nebu_render_P);
129  glUseProgram(shaders.nebula.program);
130  gl_Matrix4_Uniform(shaders.nebula.projection, nebu_render_P);
131  glUseProgram(0);
132 
133  return 0;

184 {

188  if (nebu_dofbo) {

321 static void nebu_renderPuffs( int below_player )
322 {
323  int i;
324  NebulaPuff *puff;
325  glColour col;
326 
327  /* Main menu shouldn't have puffs */
328  if (menu_isOpen(MENU_MAIN))
329  return;
330 
331  for (i=0; i<nebu_npuffs; i++) {
332  puff = &nebu_puffs[i];
333 
334  /* Separate by layers */
335  if ((below_player && (puff->height < 1.)) ||
336  (!below_player && (puff->height > 1.))) {
337 
338  /* calculate new position */
339  puff->x += puff_x * puff->height;
340  puff->y += puff_y * puff->height;
341 
342  /* Check boundaries */
343  if (puff->x > SCREEN_W + NEBULA_PUFF_BUFFER)
344  puff->x -= SCREEN_W + 2*NEBULA_PUFF_BUFFER;
345  else if (puff->y > SCREEN_H + NEBULA_PUFF_BUFFER)
346  puff->y -= SCREEN_H + 2*NEBULA_PUFF_BUFFER;
347  else if (puff->x < -NEBULA_PUFF_BUFFER)
348  puff->x += SCREEN_W + 2*NEBULA_PUFF_BUFFER;
349  else if (puff->y < -NEBULA_PUFF_BUFFER)
350  puff->y += SCREEN_H + 2*NEBULA_PUFF_BUFFER;
351 
352  /* Render */
353  col_blend( &col, &puff->col, &cBlack, conf.nebu_brightness );
354  gl_blitStatic( nebu_pufftexs[puff->tex],
355  puff->x, puff->y, &col );
356  }
357  }

379 {
380  (void)volatility;
381  int i;

384

395  glUseProgram(shaders.trail.program);

397  glUseProgram(0);
398 
399  /* Set density parameters. */
400  nebu_density = density;
401  nebu_update( 0. );
402  nebu_dt = (2.*density + 200.) / 10000.; /* Faster at higher density */
403  nebu_dx = 15000. / pow(density, 1./3.); /* Closer at higher density */
404  nebu_time = 0.;
405 
406  nebu_npuffs = density/2.;
407  nebu_puffs = realloc(nebu_puffs, sizeof(NebulaPuff)*nebu_npuffs);
408  for (i=0; i<nebu_npuffs; i++) {
409  /* Position */
410  nebu_puffs[i].x = (double)RNG(-NEBULA_PUFF_BUFFER,
411  SCREEN_W + NEBULA_PUFF_BUFFER);
412  nebu_puffs[i].y = (double)RNG(-NEBULA_PUFF_BUFFER,
413  SCREEN_H + NEBULA_PUFF_BUFFER);
414 
415  /* Maybe make size related? */
416  nebu_puffs[i].tex = RNG(0,NEBULA_PUFFS-1);
417  nebu_puffs[i].height = RNGF() + 0.2;
418 
419  /* Set the colour, with less saturation. */
420  puffhue = nebu_hue * 360.0 + 0.1*(RNGF()*2.-1.);
421  col_hsv2rgb( &nebu_puffs[i].col, puffhue, 0.6, 1.0 );
422  nebu_puffs[i].col.a = 1.0;
423  }
