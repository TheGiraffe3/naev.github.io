47 const char* debug_sigCodeToStr( int sig, int sig_code )
48 {
49  if (sig == SIGFPE)
50  switch (sig_code) {
51  case SI_USER: return _("SIGFPE (raised by program)");
52  case FPE_INTDIV: return _("SIGFPE (integer divide by zero)");
53  case FPE_INTOVF: return _("SIGFPE (integer overflow)");
54  case FPE_FLTDIV: return _("SIGFPE (floating-point divide by zero)");
55  case FPE_FLTOVF: return _("SIGFPE (floating-point overflow)");
56  case FPE_FLTUND: return _("SIGFPE (floating-point underflow)");
57  case FPE_FLTRES: return _("SIGFPE (floating-point inexact result)");
58  case FPE_FLTINV: return _("SIGFPE (floating-point invalid operation)");
59  case FPE_FLTSUB: return _("SIGFPE (subscript out of range)");
60  default: return _("SIGFPE");
61  }
62  else if (sig == SIGSEGV)
63  switch (sig_code) {
64  case SI_USER: return _("SIGSEGV (raised by program)");
65  case SEGV_MAPERR: return _("SIGSEGV (address not mapped to object)");
66  case SEGV_ACCERR: return _("SIGSEGV (invalid permissions for mapped object)");
67  default: return _("SIGSEGV");
68  }
69  else if (sig == SIGABRT)
70  switch (sig_code) {
71  case SI_USER: return _("SIGABRT (raised by program)");
72  default: return _("SIGABRT");
73  }
74 
75  /* No suitable code found. */
76  return strsignal(sig);
77 }

179  abfd = bfd_openr("/proc/self/exe", NULL);
180  if (abfd != NULL) {
181  bfd_check_format_matches(abfd, bfd_object, &matching);
182 
183  /* Read symbols */
184  if (bfd_get_file_flags(abfd) & HAS_SYMS) {
185 
186  /* static */
187  symcount = bfd_read_minisymbols (abfd, FALSE, (void **)&syms, &size);
188  if ( symcount == 0 && abfd != NULL ) /* dynamic */
189  symcount = bfd_read_minisymbols (abfd, TRUE, (void **)&syms, &size);
190  assert(symcount >= 0);
191  }
192  }
193 
194  /* Set up handler. */
195  sa.sa_handler = NULL;
196  sa.sa_sigaction = debug_sigHandler;
197  sigemptyset(&sa.sa_mask);
198  sa.sa_flags = SA_SIGINFO;
199 
200  /* Attach signals. */
201  sigaction(SIGSEGV, &sa, &so);
202  if (so.sa_handler == SIG_IGN)
203  DEBUG( _("Unable to set up SIGSEGV signal handler.") );
204  sigaction(SIGFPE, &sa, &so);
205  if (so.sa_handler == SIG_IGN)
206  DEBUG( _("Unable to set up SIGFPE signal handler.") );
207  sigaction(SIGABRT, &sa, &so);
208  if (so.sa_handler == SIG_IGN)
209  DEBUG( _("Unable to set up SIGABRT signal handler.") );
210  DEBUG( _("BFD backtrace catching enabled.") );
211 #endif /* LINUX && HAS_BFD && DEBUGGING */
212 }
