1141 static void system_scheduler( double dt, int init )
1142 {
1143  int i, n;
1144  nlua_env env;
1145  SystemPresence *p;
1146  Pilot *pilot;
1147 
1148  /* Go through all the factions and reduce the timer. */
1149  for (i=0; i < array_size(cur_system->presence); i++) {
1150  p = &cur_system->presence[i];
1151  env = faction_getScheduler( p->faction );
1152 
1153  /* Must have a valid scheduler. */
1154  if (env==LUA_NOREF)
1155  continue;
1156 
1157  /* Spawning is disabled for this faction. */
1158  if (p->disabled)
1159  continue;
1160 
1161  /* Run the appropriate function. */
1162  if (init) {
1163  nlua_getenv( env, "create" ); /* f */
1164  if (lua_isnil(naevL,-1)) {
1165  WARN(_("Lua Spawn script for faction '%s' missing obligatory entry point 'create'."),
1166  faction_name( p->faction ) );
1167  lua_pop(naevL,1);
1168  continue;
1169  }
1170  n = 0;
1171  }
1172  else {
1173  /* Decrement dt, only continue */
1174  p->timer -= dt;
1175  if (p->timer >= 0.)
1176  continue;
1177 
1178  nlua_getenv( env, "spawn" ); /* f */
1179  if (lua_isnil(naevL,-1)) {
1180  WARN(_("Lua Spawn script for faction '%s' missing obligatory entry point 'spawn'."),
1181  faction_name( p->faction ) );
1182  lua_pop(naevL,1);
1183  continue;
1184  }
1185  lua_pushnumber( naevL, p->curUsed ); /* f, presence */
1186  n = 1;
1187  }
1188  lua_pushnumber( naevL, p->value ); /* f, [arg,], max */
1189 
1190  /* Actually run the function. */
1191  if (nlua_pcall(env, n+1, 2)) { /* error has occurred */
1192  WARN(_("Lua Spawn script for faction '%s' : %s"),
1193  faction_name( p->faction ), lua_tostring(naevL,-1));
1194  lua_pop(naevL,1);
1195  continue;
1196  }
1197 
1198  /* Output is handled the same way. */
1199  if (!lua_isnumber(naevL,-2)) {
1200  WARN(_("Lua spawn script for faction '%s' failed to return timer value."),
1201  faction_name( p->faction ) );
1202  lua_pop(naevL,2);
1203  continue;
1204  }
1205  p->timer += lua_tonumber(naevL,-2);
1206  /* Handle table if it exists. */
1207  if (lua_istable(naevL,-1)) {
1208  lua_pushnil(naevL); /* tk, k */
1209  while (lua_next(naevL,-2) != 0) { /* tk, k, v */
1210  /* Must be table. */
1211  if (!lua_istable(naevL,-1)) {
1212  WARN(_("Lua spawn script for faction '%s' returns invalid data (not a table)."),
1213  faction_name( p->faction ) );
1214  lua_pop(naevL,2); /* tk, k */
1215  continue;
1216  }
1217 
1218  lua_getfield( naevL, -1, "pilot" ); /* tk, k, v, p */
1219  if (!lua_ispilot(naevL,-1)) {
1220  WARN(_("Lua spawn script for faction '%s' returns invalid data (not a pilot)."),
1221  faction_name( p->faction ) );
1222  lua_pop(naevL,2); /* tk, k */
1223  continue;
1224  }
1225  pilot = pilot_get( lua_topilot(naevL,-1) );
1226  if (pilot == NULL) {
1227  lua_pop(naevL,2); /* tk, k */
1228  continue;
1229  }
1230  lua_pop(naevL,1); /* tk, k, v */
1231  lua_getfield( naevL, -1, "presence" ); /* tk, k, v, p */
1232  if (!lua_isnumber(naevL,-1)) {
1233  WARN(_("Lua spawn script for faction '%s' returns invalid data (not a number)."),
1234  faction_name( p->faction ) );
1235  lua_pop(naevL,2); /* tk, k */
1236  continue;
1237  }
1238  pilot->presence = lua_tonumber(naevL,-1);
1239  if (pilot->faction != p->faction) {
1240  WARN( _("Lua spawn script for faction '%s' actually spawned a '%s' pilot."),
1241  faction_name( p->faction ),
1242  faction_name( pilot->faction ) );
1243  n = getPresenceIndex( cur_system, pilot->faction );
1244  p = &cur_system->presence[n];
1245  }
1246  p->curUsed += pilot->presence;
1247  lua_pop(naevL,2); /* tk, k */
1248  }
1249  }
1250  lua_pop(naevL,2);
1251  }
