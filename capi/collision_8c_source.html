339 int pointInPolygon( const CollPoly* at, const Vector2d* ap,
340  float x, float y )
341 {
342  int i;
343  float vprod, sprod, angle;
344  float dxi, dxip, dyi, dyip;
345 
346  /* See if the pixel is inside the polygon:
347  We increment the angle when doing a loop along all the points
348  If the final angle is 0, we are outside the polygon
349  Otherwise, the angle should be +/- 2pi*/
350  angle = 0.0;
351  for (i=0; i<=at->npt-2; i++) {
352  dxi = at->x[i] +VX(*ap)-x;
353  dxip = at->x[i+1]+VX(*ap)-x;
354  dyi = at->y[i] +VY(*ap)-y;
355  dyip = at->y[i+1]+VY(*ap)-y;
356  sprod = dxi * dxip + dyi * dyip;
357  vprod = dxi * dyip - dyi * dxip;
358  angle += atan2(vprod, sprod);
359  }
360  dxi = at->x[at->npt-1] + VX(*ap) - x;
361  dxip = at->x[0] + VX(*ap) - x;
362  dyi = at->y[at->npt-1] + VY(*ap)- y;
363  dyip = at->y[0] + VY(*ap) - y;
364  sprod = dxi * dxip + dyi * dyip;
365  vprod = dxi * dyip - dyi * dxip;
366  angle += atan2(vprod, sprod);
367 
368  if (FABS(angle) < 1e-5)
369  return 0;
370 
371  return 1;
372 }

429 int CollideLineLine( double s1x, double s1y, double e1x, double e1y,
430  double s2x, double s2y, double e2x, double e2y, Vector2d* crash )
431 {
432  double ua_t, ub_t, u_b;
433  double ua, ub;
434 
435  ua_t = (e2x - s2x) * (s1y - s2y) - (e2y - s2y) * (s1x - s2x);
436  ub_t = (e1x - s1x) * (s1y - s2y) - (e1y - s1y) * (s1x - s2x);
437  u_b = (e2y - s2y) * (e1x - s1x) - (e2x - s2x) * (e1y - s1y);
438 
439  if (u_b != 0.) {
440  ua = ua_t / u_b;
441  ub = ub_t / u_b;
442 
443  /* Intersection at a point. */
444  if ((0. <= ua) && (ua <= 1.) && (0. <= ub) && (ub <= 1.)) {
445  crash->x = s1x + ua * (e1x - s1x);
446  crash->y = s1y + ua * (e1y - s1y);
447  return 1;
448  }
449  /* No intersection. */
450  else
451  return 0;
452  }
453  else {
454  /* Coincident. */
455  if ((ua_t == 0.) || (ub_t == 0.))
456  return 3;
457  /* Parallel. */
458  else
459  return 2;
460  }
461 }

491 
492  /* Make sure texture has transparency map. */
493  if (bt->trans == NULL) {
494  WARN(_("Texture '%s' has no transparency map"), bt->name);
495  return 0;
496  }
497 
498  /* Set up end point of line. */
499  ep[0] = ap->x + al*cos(ad);
500  ep[1] = ap->y + al*sin(ad);
501 
502  /* Set up top right corner of the rectangle. */
503  tr[0] = bp->x + bt->sw/2.;
504  tr[1] = bp->y + bt->sh/2.;
505  /* Set up bottom left corner of the rectangle. */
506  bl[0] = bp->x - bt->sw/2.;
507  bl[1] = bp->y - bt->sh/2.;
508 
509  /*
510  * Start check for rectangular collisions.
511  */
512  hits = 0;
513  /* Left border. */
514  if (CollideLineLine(ap->x, ap->y, ep[0], ep[1],
515  bl[0], bl[1], bl[0], tr[1], &tmp_crash) == 1) {
516  border[hits].x = tmp_crash.x;
517  border[hits].y = tmp_crash.y;
518  hits++;
519  }
520  /* Top border. */
521  if (CollideLineLine(ap->x, ap->y, ep[0], ep[1],
522  bl[0], tr[1], tr[0], tr[1], &tmp_crash) == 1) {
523  border[hits].x = tmp_crash.x;
524  border[hits].y = tmp_crash.y;
525  hits++;
526  }
527  /* Now we have to make sure hits isn't 2. */
528  /* Right border. */
529  if ((hits < 2) && CollideLineLine(ap->x, ap->y, ep[0], ep[1],
530  tr[0], tr[1], tr[0], bl[1], &tmp_crash) == 1) {
531  border[hits].x = tmp_crash.x;
532  border[hits].y = tmp_crash.y;
533  hits++;
534  }
535  /* Bottom border. */
536  if ((hits < 2) && CollideLineLine(ap->x, ap->y, ep[0], ep[1],
537  tr[0], bl[1], bl[0], bl[1], &tmp_crash) == 1) {
538  border[hits].x = tmp_crash.x;
539  border[hits].y = tmp_crash.y;
540  hits++;
541  }
542 
543  /* No hits - missed. */
544  if (hits == 0)
545  return 0;
546 
547  /* Beam must die in the rectangle. */
548  if (hits == 1) {
549  border[1].x = ep[0];
550  border[1].y = ep[1];
551  }
552

554  * Now we do a pixel perfect approach.
555  */
556  real_hits = 0;

559  v[1] = border[1].y - border[0].y;

562  v[0] /= mod;

569  bby = rbsy*(int)(bt->sh);
570

572  x = border[0].x - bl[0] + v[0];
573  y = border[0].y - bl[1] + v[1];
574  while ((x > 0.) && (x < bt->sw) && (y > 0.) && (y < bt->sh)) {
575  /* Is non-transparent. */
576  if (!gl_isTrans(bt, bbx+(int)x, bby+(int)y)) {
577  crash[real_hits].x = x + bl[0];
578  crash[real_hits].y = y + bl[1];
579  real_hits++;
580  break;
581  }
582  x += v[0];
583  y += v[1];
584  }
585 
586  /* Now we check the second border. */
587  x = border[1].x - bl[0] - v[0];
588  y = border[1].y - bl[1] - v[1];
589  while ((x > 0.) && (x < bt->sw) && (y > 0.) && (y < bt->sh)) {
590  /* Is non-transparent. */
591  if (!gl_isTrans(bt, bbx+(int)x, bby+(int)y)) {
592  crash[real_hits].x = x + bl[0];
593  crash[real_hits].y = y + bl[1];
594  real_hits++;
595  break;
596  }
597  x -= v[0];
598  y -= v[1];
599  }
600 
601  /* Actually missed. */
602  if (real_hits == 0)
603  return 0;
604 
605  /* Strange situation, should never happen but just in case we duplicate
606  * the hit. */
607  if (real_hits == 1) {
608  crash[1].x = crash[0].x;
609  crash[1].y = crash[0].y;
610  }
611 
612  /* We hit. */
613  return 1;
614 }
