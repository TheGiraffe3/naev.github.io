331 int pointInPolygon( const CollPoly* at, const Vector2d* ap,
332  float x, float y )
333 {
334  int i;
335  float vprod, sprod, angle;
336  float dxi, dxip, dyi, dyip;
337 
338  /* See if the pixel is inside the polygon:
339  We increment the angle when doing a loop along all the points
340  If the final angle is 0, we are outside the polygon
341  Otherwise, the angle should be +/- 2pi*/
342  angle = 0.0;
343  for (i=0; i<=at->npt-2; i++) {
344  dxi = at->x[i] +VX(*ap)-x;
345  dxip = at->x[i+1]+VX(*ap)-x;
346  dyi = at->y[i] +VY(*ap)-y;
347  dyip = at->y[i+1]+VY(*ap)-y;
348  sprod = dxi * dxip + dyi * dyip;
349  vprod = dxi * dyip - dyi * dxip;
350  angle += atan2(vprod, sprod);
351  }
352  dxi = at->x[at->npt-1] + VX(*ap) - x;
353  dxip = at->x[0] + VX(*ap) - x;
354  dyi = at->y[at->npt-1] + VY(*ap)- y;
355  dyip = at->y[0] + VY(*ap) - y;
356  sprod = dxi * dxip + dyi * dyip;
357  vprod = dxi * dyip - dyi * dxip;
358  angle += atan2(vprod, sprod);
359 
360  if (FABS(angle) < 1e-5)
361  return 0;
362 
363  return 1;
364 }

418 int CollideLineLine( double s1x, double s1y, double e1x, double e1y,
419  double s2x, double s2y, double e2x, double e2y, Vector2d* crash )
420 {
421  double ua_t, ub_t, u_b;
422  double ua, ub;
423 
424  ua_t = (e2x - s2x) * (s1y - s2y) - (e2y - s2y) * (s1x - s2x);
425  ub_t = (e1x - s1x) * (s1y - s2y) - (e1y - s1y) * (s1x - s2x);
426  u_b = (e2y - s2y) * (e1x - s1x) - (e2x - s2x) * (e1y - s1y);
427 
428  if (u_b != 0.) {
429  ua = ua_t / u_b;
430  ub = ub_t / u_b;
431 
432  /* Intersection at a point. */
433  if ((0. <= ua) && (ua <= 1.) && (0. <= ub) && (ub <= 1.)) {
434  crash->x = s1x + ua * (e1x - s1x);
435  crash->y = s1y + ua * (e1y - s1y);
436  return 1;
437  }
438  /* No intersection. */
439  else
440  return 0;
441  }
442  else {
443  /* Coincident. */
444  if ((ua_t == 0.) || (ub_t == 0.))
445  return 3;
446  /* Parallel. */
447  else
448  return 2;
449  }
450 }

756 int CollideLineCircle( const Vector2d* p1, const Vector2d* p2,
757  const Vector2d *cc, double cr, Vector2d crash[2] )
758 {
759  double x0 = cc->x;
760  double y0 = cc->y;
761  double x1 = p1->x;
762  double y1 = p1->y;
763  double x2 = p2->x;
764  double y2 = p2->y;
765 
766  double A = y2 - y1;
767  double B = x1 - x2;
768  double C = x2 * y1 - x1 * y2;
769 
770  double a = pow2(A) + pow2(B);
771  double b, c, d;
772 
773  int bnz, cnt;
774 
775  double x, y, d1;
776 
777  /* Non-vertical case. */
778  if (fabs(B) >= 1e-8) {
779  b = 2. * (A * C + A * B * y0 - pow2(B) * x0);
780  c = pow2(C) + 2. * B * C * y0 - pow2(B) * (pow2(cr) - pow2(x0) - pow2(y0));
781  bnz = 1;
782  }
783  /* Have to have special care when line is vertical. */
784  else {
785  b = 2. * (B * C + A * B * x0 - pow2(A) * y0);
786  c = pow2(C) + 2. * A * C * x0 - pow2(A) * (pow2(cr) - pow2(x0) - pow2(y0));
787  bnz = 0;
788  }
789  d = pow2(b) - 4. * a * c; /* Discriminant. */
790  if (d < 0.)
791  return 0;
792 
793  cnt = 0;
794  d1 = hypot( x2-x1, y2-y1 );
795  /* Line is tangent, so only one intersection. */
796  if (d == 0.) {
797  if (bnz) {
798  x = -b / (2. * a);
799  y = FX(A, B, C, x);
800  if (linePointOnSegment( d1, x1, y1, x2, y2, x, y )) {
801  crash[cnt].x = x;
802  crash[cnt].y = y;
803  cnt++;
804  }
805  } else {
806  y = -b / (2. * a);
807  x = FY(A, B, C, y);
808  if (linePointOnSegment( d1, x1, y1, x2, y2, x, y )) {
809  crash[cnt].x = x;
810  crash[cnt].y = y;
811  cnt++;
812  }
813  }
814  }
815  /* Two intersections. */
816  else {
817  d = sqrt(d);
818  if (bnz) {
819  x = (-b + d) / (2. * a);
820  y = FX(A, B, C, x);
821  if (linePointOnSegment( d1, x1, y1, x2, y2, x, y )) {
822  crash[cnt].x = x;
823  crash[cnt].y = y;
824  cnt++;
825  }
826  x = (-b - d) / (2. * a);
827  y = FX(A, B, C, x);
828  if (linePointOnSegment( d1, x1, y1, x2, y2, x, y )) {
829  crash[cnt].x = x;
830  crash[cnt].y = y;
831  cnt++;
832  }
833  } else {
834  y = (-b + d) / (2. * a);
835  x = FY(A, B, C, y);
836  if (linePointOnSegment( d1, x1, y1, x2, y2, x, y )) {
837  crash[cnt].x = x;
838  crash[cnt].y = y;
839  cnt++;
840  }
841  y = (-b - d) / (2. * a);
842  x = FY(A, B, C, y);
843  if (linePointOnSegment( d1, x1, y1, x2, y2, x, y )) {
844  crash[cnt].x = x;
845  crash[cnt].y = y;
846  cnt++;
847  }
848  }
849  }
850  return cnt;
851 }
