111

113  memset( temp, 0, sizeof(Fleet) );
114  temp->faction = -1;
115

117  if (temp->name == NULL)
118  WARN( _("Fleet in %s has invalid or no name"), FLEET_DATA_PATH );
119

121  xml_onlyNodes(node);
122 
123  /* Set faction. */

133

136  cur = node->children;
137  do {
138  xml_onlyNodes(cur);
139  WARN(_("Fleet '%s' has unknown flag node '%s'."), temp->name, cur->name);
140  } while (xml_nextNode(cur));
141  continue;
142  }
143 
144  /* Load pilots. */
145  else if (xml_isNode(node,"pilots")) {

207 static int fleet_loadFleets (void)
208 {
209  int mem;
210  xmlNodePtr node;
211  xmlDocPtr doc;
212 
213  /* Load the data. */
214  doc = xml_parsePhysFS( FLEET_DATA_PATH );
215  if (doc == NULL)
216  return -1;
217 
218  node = doc->xmlChildrenNode; /* fleets node */
219  if (strcmp((char*)node->name,"Fleets")) {
220  ERR( _("Malformed %s file: missing root element 'Fleets'."), FLEET_DATA_PATH);
221  return -1;
222  }
223 
224  node = node->xmlChildrenNode; /* first fleet node */
225  if (node == NULL) {
226  ERR( _("Malformed %s file: does not contain elements."), FLEET_DATA_PATH );
227  return -1;
228  }
229 
230  mem = 0;
231  do {
232  if (xml_isNode(node,"fleet")) {
233  /* See if memory must grow. */
234  nfleets++;
235  if (nfleets > mem) {
236  mem += CHUNK_SIZE;
237  fleet_stack = realloc(fleet_stack, sizeof(Fleet) * mem);
238  }
239 
240  /* Load the fleet. */
241  fleet_parse( &fleet_stack[nfleets-1], node );
242  }
243  } while (xml_nextNode(node));
244  /* Shrink to minimum. */
245  fleet_stack = realloc(fleet_stack, sizeof(Fleet) * nfleets);
246 
247  xmlFreeDoc(doc);
248 
249  return 0;
250 }
