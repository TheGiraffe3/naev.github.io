75 static int pilot_weapSetFire( Pilot *p, PilotWeaponSet *ws, int level )
76 {
77  int i, j, ret, s;
78  Pilot *pt;
79  AsteroidAnchor *field;
80  Asteroid *ast;
81  double time;
82  Outfit *o;
83 
84  ret = 0;
85  for (i=0; i<array_size(ws->slots); i++) {
86  o = ws->slots[i].slot->outfit;
87 
88  /* Ignore NULL outfits. */
89  if (o == NULL)
90  continue;
91 
92  /* Only "active" outfits. */
93  if ((level != -1) && (ws->slots[i].level != level))
94  continue;
95 
96  /* Only run once for each weapon type in the group. */
97  s = 0;
98  for (j=0; j<i; j++) {
99  /* Only active outfits. */
100  if ((level != -1) && (ws->slots[j].level != level))
101  continue;
102  /* Found a match. */
103  if (ws->slots[j].slot->outfit == o) {
104  s = 1;
105  break;
106  }
107  }
108  if (s!=0)
109  continue;
110 
111  /* Only "locked on" outfits. */
112  if (outfit_isSeeker(o) &&
113  (ws->slots[i].slot->u.ammo.lockon_timer > 0.))
114  continue;
115 
116  /* If inrange is set we only fire at targets in range. */
117  time = INFINITY; /* With no target we just set time to infinity. */
118 
119  /* Calculate time to target if it is there. */
120  if (p->target != p->id) {
121  pt = pilot_get( p->target );
122  if (pt != NULL)
123  time = pilot_weapFlyTime( o, p, &pt->solid->pos, &pt->solid->vel);
124  }
125  /* Looking for a closer targeted asteroid */
126  if (p->nav_asteroid != -1) {
127  field = &cur_system->asteroids[p->nav_anchor];
128  ast = &field->asteroids[p->nav_asteroid];
129  time = MIN( time, pilot_weapFlyTime( o, p, &ast->pos, &ast->vel) );
130  }
131 
132  /* Only "inrange" outfits. */
133  if (ws->inrange && outfit_duration(o) < time)
134  continue;
135 
136  /* Shoot the weapon of the weaponset. */
137  ret += pilot_shootWeaponSetOutfit( p, ws, o, level, time );
138  }
139 
140  return ret;
141 }

165 void pilot_weapSetPress( Pilot* p, int id, int type )
166 {
167  int i, l, on, n;
168  PilotWeaponSet *ws;
169  ws = pilot_weapSet(p,id);
170  /* Case no outfits. */
171  if (ws->slots == NULL)
172  return;
173 
174  /* Handle fire groups. */
175  switch (ws->type) {
176  case WEAPSET_TYPE_CHANGE:
177  /* On press just change active weapon set to whatever is available. */
178  if (type > 0) {
179  if (id != p->active_set)
180  pilot_weapSetUpdateOutfits( p, ws );
181  p->active_set = id;
182  }
183  break;
184 
185  case WEAPSET_TYPE_WEAPON:
186  /* Activation philosophy here is to turn on while pressed and off
187  * when it's not held anymore. */
188  if (type > 0)
189  ws->active = 1;
190  else if (type < 0)
191  ws->active = 0;
192  break;
193 
194  case WEAPSET_TYPE_ACTIVE:
195  /* The behaviour here is more complex. What we do is consider a group
196  * to be entirely off if not all outfits are either on or cooling down.
197  * In the case it's deemed to be off, all outfits that are off get turned
198  * on, otherwise all outfits that are on are turrned to cooling down. */
199  /* Only care about presses. */
200  if (type < 0)
201  break;
202 
203  /* Must not be disabled or cooling down. */
204  if ((pilot_isDisabled(p)) || (pilot_isFlag(p, PILOT_COOLDOWN)))
205  return;
206 
207  /* Decide what to do. */
208  on = 1;
209  l = array_size(ws->slots);
210  for (i=0; i<l; i++) {
211  if (ws->slots[i].slot->state == PILOT_OUTFIT_OFF) {
212  on = 0;
213  break;
214  }
215  }
216 
217  /* Turn them off. */
218  n = 0;
219  if (on) {
220  for (i=0; i<l; i++) {
221  if (ws->slots[i].slot->state != PILOT_OUTFIT_ON)
222  continue;
223 
224  n += pilot_outfitOff( p, ws->slots[i].slot );
225  }
226  }
227  /* Turn them on. */
228  else {
229  for (i=0; i<l; i++) {
230  if (ws->slots[i].slot->state != PILOT_OUTFIT_OFF)
231  continue;
232  if (outfit_isAfterburner(ws->slots[i].slot->outfit))
233  pilot_afterburn( p );
234  else {
235  ws->slots[i].slot->state = PILOT_OUTFIT_ON;
236  ws->slots[i].slot->stimer = outfit_duration( ws->slots[i].slot->outfit );
237  }
238  n++;
239  }
240  }
241  /* Must recalculate stats. */
242  if (n > 0)
243  pilot_calcStats( p );
244 
245  break;
246  }
247 }

544 static void pilot_weapSetUpdateRange( PilotWeaponSet *ws )
545 {
546  int i, lev;
547  double range, speed;
548  double range_accum[PILOT_WEAPSET_MAX_LEVELS];
549  int range_num[PILOT_WEAPSET_MAX_LEVELS];
550  double speed_accum[PILOT_WEAPSET_MAX_LEVELS];
551  int speed_num[PILOT_WEAPSET_MAX_LEVELS];
552 
553  /* Calculate ranges. */
554  for (i=0; i<PILOT_WEAPSET_MAX_LEVELS; i++) {
555  range_accum[i] = 0.;
556  range_num[i] = 0;
557  speed_accum[i] = 0.;
558  speed_num[i] = 0;
559  }
560  for (i=0; i<array_size(ws->slots); i++) {
561  if (ws->slots[i].slot->outfit == NULL)
562  continue;
563 
564  /* Get level. */
565  lev = ws->slots[i].level;
566  if (lev >= PILOT_WEAPSET_MAX_LEVELS)
567  continue;
568 
569  /* Empty Launchers aren't valid */
570  if (outfit_isLauncher(ws->slots[i].slot->outfit) && (ws->slots[i].slot->u.ammo.quantity <= 0))
571  continue;
572 
573  /* Get range. */
574  range = outfit_range(ws->slots[i].slot->outfit);
575  if (range >= 0.) {
576  /* Calculate. */
577  range_accum[ lev ] += range;
578  range_num[ lev ]++;
579  }
580 
581  /* Get speed. */
582  speed = outfit_speed(ws->slots[i].slot->outfit);
583  if (speed >= 0.) {
584  /* Calculate. */
585  speed_accum[ lev ] += speed;
586  speed_num[ lev ]++;
587  }
588  }
589 
590  /* Postprocess. */
591  for (i=0; i<PILOT_WEAPSET_MAX_LEVELS; i++) {
592  /* Postprocess range. */
593  if (range_num[i] == 0)
594  ws->range[i] = 0;
595  else
596  ws->range[i] = range_accum[i] / (double) range_num[i];
597 
598  /* Postprocess speed. */
599  if (speed_num[i] == 0)
600  ws->speed[i] = 0;
601  else
602  ws->speed[i] = speed_accum[i] / (double) speed_num[i];
603  }
604 }

836 double pilot_weapFlyTime( Outfit *o, Pilot *parent, Vector2d *pos, Vector2d *vel)
837 {
838  Vector2d approach_vector, relative_location, orthoradial_vector;
839  double speed, radial_speed, orthoradial_speed, dist, t;
840 
841  dist = vect_dist( &parent->solid->pos, pos );
842 
843  /* Beam weapons */
844  if (outfit_isBeam(o)) {
845  if (dist > o->u.bem.range)
846  return INFINITY;
847  return 0.;
848  }
849 
850  /* A bay doesn't have range issues */
851  if (outfit_isFighterBay(o))
852  return 0.;
853 
854  /* Missiles use absolute velocity while bolts and unguided rockets use relative vel */
855  if (outfit_isLauncher(o) && o->u.lau.ammo->u.amm.ai != AMMO_AI_UNGUIDED)
856  vect_cset( &approach_vector, - vel->x, - vel->y );
857  else
858  vect_cset( &approach_vector, VX(parent->solid->vel) - vel->x,
859  VY(parent->solid->vel) - vel->y );
860 
861  speed = outfit_speed(o);
862 
863  /* Get the vector : shooter -> target */
864  vect_cset( &relative_location, pos->x - VX(parent->solid->pos),
865  pos->y - VY(parent->solid->pos) );
866 
867  /* Get the orthogonal vector */
868  vect_cset(&orthoradial_vector, VY(parent->solid->pos) - pos->y,
869  pos->x - VX(parent->solid->pos) );
870 
871  radial_speed = vect_dot( &approach_vector, &relative_location );
872  radial_speed = radial_speed / VMOD(relative_location);
873 
874  orthoradial_speed = vect_dot(&approach_vector, &orthoradial_vector);
875  orthoradial_speed = orthoradial_speed / VMOD(relative_location);
876 
877  if ( ((speed*speed - VMOD(approach_vector)*VMOD(approach_vector)) != 0) && (speed*speed - orthoradial_speed*orthoradial_speed) > 0)
878  t = dist * (sqrt( speed*speed - orthoradial_speed*orthoradial_speed ) - radial_speed) /
879  (speed*speed - VMOD(approach_vector)*VMOD(approach_vector));
880  else
881  return INFINITY;
882 
883  /* if t < 0, try the other solution */
884  if (t < 0)
885  t = - dist * (sqrt( speed*speed - orthoradial_speed*orthoradial_speed ) + radial_speed) /
886  (speed*speed - VMOD(approach_vector)*VMOD(approach_vector));
887 
888  /* if t still < 0, no solution */
889  if (t < 0)
890  return INFINITY;
891 
892  return t;
893 }
