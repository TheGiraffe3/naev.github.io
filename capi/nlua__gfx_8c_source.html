243  tx = luaL_checknumber( L, 8 );
244  ty = luaL_checknumber( L, 9 );
245  tw = luaL_checknumber( L, 10 );
246  th = luaL_checknumber( L, 11 );
247  col = luaL_optcolour(L, 12, &cWhite );
248  angle = luaL_optnumber(L,13,0.);
249 
250  /* Some safety checking. */
251 #if DEBUGGING
252  if (sx >= t->sx)

254  t->name, sx+1, t->sx );

257  t->name, sy+1, t->sy );
258 #endif /* DEBUGGING */

260  /* Translate as needed. */
261  tx = (tx * t->sw + t->sw * (double)(sx)) / t->w;
262  tw = tw * t->srw;

266  th = th * t->srh;

268  ty -= th;
269

271  return 0;

554 static int gfxL_printfWrap( lua_State *L )
555 {
556  const char *s;
557  int width, outw, maxw;
558  glFont *font;
559  int p, l, slen;
560  int linenum;
561  char *tmp;
562 
563  /* Parse parameters. */
564  font = luaL_checkfont(L,1);
565  s = luaL_checkstring(L,2);
566  width = luaL_checkinteger(L,3);
567  if (width < 0)
568  NLUA_ERROR(L,_("width has to be a positive value."));
569 
570  /* Process output into table. */
571  lua_newtable(L);
572  tmp = strdup(s);
573  slen = strlen(s);
574  p = 0;
575  linenum = 1;
576  maxw = 0;
577  do {
578  int lp = p;
579  if ((tmp[p] == ' ') || (tmp[p] == '\n'))
580  p++;
581  /* Don't handle tab for now. */
582  if (tmp[p]=='\t')
583  tmp[p] = ' ';
584  l = gl_printWidthForText(font, &tmp[p], width, &outw );
585  if (outw > maxw)
586  maxw = outw;
587 
588  /* Create entry of form { string, width } in the table. */
589  lua_pushnumber(L, linenum++); /* t, n */
590  lua_newtable(L); /* t, n, t */
591  lua_pushinteger(L, 1); /* t, n, t, 1 */
592  lua_pushlstring(L, &tmp[p], l); /* t, n, t, 1, s */
593  lua_rawset(L, -3); /* t, n, t */
594  lua_pushinteger(L, 2); /* t, n, t, 2 */
595  lua_pushinteger(L, outw); /* t, n, t, 2, n */
596  lua_rawset(L, -3); /* t, n, t */
597  lua_rawset(L, -3); /* t */
598 
599  p += l;
600  if (lp==p)
601  break;
602  } while (p < slen);
603 
604  /* Push max width. */
605  lua_pushinteger(L, maxw);
606 
607  free( tmp );
608  return 2;
609 }
