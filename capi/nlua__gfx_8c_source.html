547 static int gfxL_printfWrap( lua_State *L )
548 {
549  const char *s;
550  int width, outw, maxw;
551  glFont *font;
552  int p, l, slen;
553  int linenum;
554  char *tmp;
555 
556  /* Parse parameters. */
557  font = luaL_checkfont(L,1);
558  s = luaL_checkstring(L,2);
559  width = luaL_checkinteger(L,3);
560  if (width < 0)
561  NLUA_ERROR(L,_("width has to be a positive value."));
562 
563  /* Process output into table. */
564  lua_newtable(L);
565  tmp = strdup(s);
566  slen = strlen(s);
567  p = 0;
568  linenum = 1;
569  maxw = 0;
570  do {
571  int lp = p;
572  if ((tmp[p] == ' ') || (tmp[p] == '\n'))
573  p++;
574  /* Don't handle tab for now. */
575  if (tmp[p]=='\t')
576  tmp[p] = ' ';
577  l = gl_printWidthForText(font, &tmp[p], width, &outw );
578  if (outw > maxw)
579  maxw = outw;
580 
581  /* Create entry of form { string, width } in the table. */
582  lua_pushnumber(L, linenum++); /* t, n */
583  lua_newtable(L); /* t, n, t */
584  lua_pushinteger(L, 1); /* t, n, t, 1 */
585  lua_pushlstring(L, &tmp[p], l); /* t, n, t, 1, s */
586  lua_rawset(L, -3); /* t, n, t */
587  lua_pushinteger(L, 2); /* t, n, t, 2 */
588  lua_pushinteger(L, outw); /* t, n, t, 2, n */
589  lua_rawset(L, -3); /* t, n, t */
590  lua_rawset(L, -3); /* t */
591 
592  p += l;
593  if (lp==p)
594  break;
595  } while (p < slen);
596 
597  /* Push max width. */
598  lua_pushinteger(L, maxw);
599 
600  free( tmp );
601  return 2;
602 }

815

817  srcRGB = GL_SRC_ALPHA;

827  else if (!strcmp( mode, "multiply" ))
828  srcRGB = srcA = GL_DST_COLOR;
829  else if (!strcmp( mode, "subtract" ))
830  func = GL_FUNC_REVERSE_SUBTRACT;
831  else if (!strcmp( mode, "add" ) ) {
832  func = GL_FUNC_REVERSE_SUBTRACT;
833  srcA = GL_ZERO;

835  }
836  else if (!strcmp( mode, "lighten" ))
837  func = GL_MAX;
838  else if (!strcmp( mode, "darken" ))
839  func = GL_MIN;
840  else if (!strcmp( mode, "screen" ))
841  dstRGB = dstA = GL_ONE_MINUS_SRC_COLOR;

843  NLUA_INVALID_PARAMETER(L);
