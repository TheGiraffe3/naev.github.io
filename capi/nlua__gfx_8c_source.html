288 static int gfxL_renderTexRaw( lua_State *L )
289 {
290  glTexture *t;
291  const glColour *col;
292  double px,py, pw,ph, tx,ty, tw,th;
293  double angle;
294  int sx, sy;
295 
296  NLUA_CHECKRW(L);
297 
298  /* Parameters. */
299  t = luaL_checktex( L, 1 );
300  px = luaL_checknumber( L, 2 );
301  py = luaL_checknumber( L, 3 );
302  pw = luaL_checknumber( L, 4 );
303  ph = luaL_checknumber( L, 5 );
304  sx = luaL_checkinteger( L, 6 ) - 1;
305  sy = luaL_checkinteger( L, 7 ) - 1;
306  tx = luaL_checknumber( L, 8 );
307  ty = luaL_checknumber( L, 9 );
308  tw = luaL_checknumber( L, 10 );
309  th = luaL_checknumber( L, 11 );
310  col = luaL_optcolour(L, 12, &cWhite );
311  angle = luaL_optnumber(L,13,0.);
312 
313  /* Some safety checking. */
314 #if DEBUGGING
315  if (sx >= t->sx)
316  NLUA_ERROR( L, _("Texture '%s' trying to render out of bounds (X position) sprite: %d > %d."),
317  t->name, sx+1, t->sx );
318  if (sx >= t->sx)
319  NLUA_ERROR( L, _("Texture '%s' trying to render out of bounds (Y position) sprite: %d > %d."),
320  t->name, sy+1, t->sy );
321 #endif /* DEBUGGING */
322 
323  /* Translate as needed. */
324  tx = (tx * t->sw + t->sw * (double)(sx)) / t->w;
325  tw = tw * t->srw;
326  if (tw < 0)
327  tx -= tw;
328  ty = (ty * t->sh + t->sh * (t->sy - (double)sy-1)) / t->h;
329  th = th * t->srh;
330  if (th < 0)
331  ty -= th;
332 
333  gl_renderTexture( t, px, py, pw, ph, tx, ty, tw, th, col, angle );
334  return 0;
335 }

348 {

357

359  t = luaL_checktex( L,1 );
360  shader = luaL_checkshader( L,2 );
361  H = luaL_checktransform( L,3 );
362  col = luaL_optcolour(L,4,&cWhite);

364 
365  glUseProgram( shader->program );
366 
367  /* Set the vertex. */
368  glEnableVertexAttribArray( shader->VertexPosition );
369  gl_vboActivateAttribOffset( gl_squareVBO, shader->VertexPosition,
370  0, 2, GL_FLOAT, 0 );
371 
372  /* Set up texture vertices if necessary. */
373  if (shader->VertexTexCoord >= 0) {
374  gl_Matrix4_Uniform( shader->ViewSpaceFromLocal, *TH );
375  glEnableVertexAttribArray( shader->VertexTexCoord );
376  gl_vboActivateAttribOffset( gl_squareVBO, shader->VertexTexCoord,
377  0, 2, GL_FLOAT, 0 );
378  }
379 
380  /* Set the texture(s). */
381  glBindTexture( GL_TEXTURE_2D, t->texture );
382  glUniform1i( shader->MainTex, 0 );
383  for (int i=0; i<array_size(shader->tex); i++) {
384  LuaTexture_t *lt = &shader->tex[i];
385  glActiveTexture( lt->active );
386  glBindTexture( GL_TEXTURE_2D, lt->texid );
387  glUniform1i( lt->uniform, lt->value );
388  }
389  glActiveTexture( GL_TEXTURE0 );
390 
391  /* Set shader uniforms. */
392  gl_uniformColor( shader->ConstantColor, col );
393  gl_Matrix4_Uniform( shader->ClipSpaceFromLocal, *H );
394 
395  /* Draw. */
396  glDrawArrays( GL_TRIANGLE_STRIP, 0, 4 );
397 
398  /* Clear state. */
399  glDisableVertexAttribArray( shader->VertexPosition );
400  if (shader->VertexTexCoord >= 0)
401  glDisableVertexAttribArray( shader->VertexTexCoord );
402 
403  /* anything failed? */
404  gl_checkErr();
405 
406  glUseProgram(0);
407 
408  return 0;
409 }

617 static int gfxL_printfWrap( lua_State *L )
618 {
619  const char *s;
620  int width, maxw;
621  glFont *font;
622  glPrintLineIterator iter;
623  int linenum;
624 
625  /* Parse parameters. */
626  font = luaL_checkfont(L,1);
627  s = luaL_checkstring(L,2);
628  width = luaL_checkinteger(L,3);
629  if (width < 0)
630  NLUA_ERROR(L,_("width has to be a positive value."));
631 
632  /* Process output into table. */
633  lua_newtable(L);
634  gl_printLineIteratorInit(&iter, font, s, width);
635  linenum = 1;
636  maxw = 0;
637  while (gl_printLineIteratorNext(&iter)) {
638  maxw = MAX(maxw, iter.l_width);
639 
640  /* Create entry of form { string, width } in the table. */
641  lua_newtable(L); /* t, t */
642  lua_pushlstring(L, &s[iter.l_begin], iter.l_end - iter.l_begin); /* t, t, s */
643  lua_rawseti(L,-2,1); /* t, t */
644  lua_pushinteger(L,iter.l_width);/* t, t, n */
645  lua_rawseti(L,-2,2); /* t, t */
646  lua_rawseti(L,-2,linenum++); /* t */
647  }
648 
649  /* Push max width. */
650  lua_pushinteger(L, maxw);
651 
652  return 2;
653 }

854 static int gfxL_setBlendMode( lua_State *L )
855 {
856  const char *mode, *alphamode;
857  GLenum func, srcRGB, srcA, dstRGB, dstA;
858 
859  /* Parse parameters. */
860  mode = luaL_checkstring(L,1);
861  alphamode= luaL_optstring(L,2,"alphamultiply");
862 
863  /* Translate to OpenGL enums. */
864  func = GL_FUNC_ADD;
865  srcRGB = GL_ONE;
866  srcA = GL_ONE;
867  dstRGB = GL_ZERO;
868  dstA = GL_ZERO;
869 
870  if (!strcmp( alphamode, "alphamultiply" ))
871  srcRGB = srcA = GL_SRC_ALPHA;
872  else if (!strcmp( alphamode, "premultiplied" )) {
873  if (!strcmp( mode, "lighten" ) || !strcmp( mode, "darken" ) || !strcmp( mode, "multiply" ))
874  NLUA_INVALID_PARAMETER(L);
875  }
876  else
877  NLUA_INVALID_PARAMETER(L);
878 
879  if (!strcmp( mode, "alpha" ))
880  dstRGB = dstA = GL_ONE_MINUS_SRC_ALPHA;
881  else if (!strcmp( mode, "multiply" ))
882  srcRGB = srcA = GL_DST_COLOR;
883  else if (!strcmp( mode, "subtract" ))
884  func = GL_FUNC_REVERSE_SUBTRACT;
885  else if (!strcmp( mode, "add" ) ) {
886  func = GL_FUNC_REVERSE_SUBTRACT;
887  srcA = GL_ZERO;
888  dstRGB = dstA = GL_ONE;
889  }
890  else if (!strcmp( mode, "lighten" ))
891  func = GL_MAX;
892  else if (!strcmp( mode, "darken" ))
893  func = GL_MIN;
894  else if (!strcmp( mode, "screen" ))
895  dstRGB = dstA = GL_ONE_MINUS_SRC_COLOR;
896  else if (strcmp( mode, "replace" ))
897  NLUA_INVALID_PARAMETER(L);
898 
899  glBlendEquation(func);
900  glBlendFuncSeparate(srcRGB, dstRGB, srcA, dstA);
901  gl_checkErr();
902 
903  return 0;
