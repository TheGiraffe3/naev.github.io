214 void gl_renderTexture( const glTexture* texture,
215  const double x, const double y,
216  const double w, const double h,
217  const double tx, const double ty,
218  const double tw, const double th, const glColour *c, const double angle )
219 {
220  // Half width and height
221  double hw, hh;
222  gl_Matrix4 projection, tex_mat;
223 
224  glUseProgram(shaders.texture.program);
225 
226  /* Bind the texture. */
227  glBindTexture( GL_TEXTURE_2D, texture->texture);
228 
229  /* Must have colour for now. */
230  if (c == NULL)
231  c = &cWhite;
232 
233  hw = w/2.;
234  hh = h/2.;
235 
236  /* Set the vertex. */
237  projection = gl_view_matrix;
238  if (angle==0.){
239  projection = gl_Matrix4_Translate(projection, x, y, 0);
240  projection = gl_Matrix4_Scale(projection, w, h, 1);
241  } else {
242  projection = gl_Matrix4_Translate(projection, x+hw, y+hh, 0);
243  projection = gl_Matrix4_Rotate2d(projection, angle);
244  projection = gl_Matrix4_Translate(projection, -hw, -hh, 0);
245  projection = gl_Matrix4_Scale(projection, w, h, 1);
246  }
247  glEnableVertexAttribArray( shaders.texture.vertex );
248  gl_vboActivateAttribOffset( gl_squareVBO, shaders.texture.vertex,
249  0, 2, GL_FLOAT, 0 );
250 
251  /* Set the texture. */
252  tex_mat = (texture->flags & OPENGL_TEX_VFLIP) ? gl_Matrix4_Ortho(-1, 1, 2, 0, 1, -1) : gl_Matrix4_Identity();
253  tex_mat = gl_Matrix4_Translate(tex_mat, tx, ty, 0);
254  tex_mat = gl_Matrix4_Scale(tex_mat, tw, th, 1);
255 
256  /* Set shader uniforms. */
257  gl_uniformColor(shaders.texture.color, c);
258  gl_Matrix4_Uniform(shaders.texture.projection, projection);
259  gl_Matrix4_Uniform(shaders.texture.tex_mat, tex_mat);
260 
261  /* Draw. */
262  glDrawArrays( GL_TRIANGLE_STRIP, 0, 4 );
263 
264  /* Clear state. */
265  glDisableVertexAttribArray( shaders.texture.vertex );
266 
267  /* anything failed? */
268  gl_checkErr();
269 
270  glUseProgram(0);
271 }

325 
326  /* Must have colour for now. */
327  if (c == NULL)
328  c = &cWhite;
329

331  projection = gl_view_matrix;
332  projection = gl_Matrix4_Translate(projection, x, y, 0);
333  projection = gl_Matrix4_Scale(projection, w, h, 1);

337  /* Set the texture. */
