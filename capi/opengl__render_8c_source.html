196 void gl_renderTexture( const glTexture* texture,
197  const double x, const double y,
198  const double w, const double h,
199  const double tx, const double ty,
200  const double tw, const double th, const glColour *c, const double angle )
201 {
202  // Half width and height
203  double hw, hh;
204  gl_Matrix4 projection, tex_mat;
205 
206  glUseProgram(shaders.texture.program);
207 
208  /* Bind the texture. */
209  glBindTexture( GL_TEXTURE_2D, texture->texture);
210 
211  /* Must have colour for now. */
212  if (c == NULL)
213  c = &cWhite;
214 
215  hw = w/2.;
216  hh = h/2.;
217 
218  /* Set the vertex. */
219  projection = gl_view_matrix;
220  if (angle==0.){
221  projection = gl_Matrix4_Translate(projection, x, y, 0);
222  projection = gl_Matrix4_Scale(projection, w, h, 1);
223  } else {
224  projection = gl_Matrix4_Translate(projection, x+hw, y+hh, 0);
225  projection = gl_Matrix4_Rotate2d(projection, angle);
226  projection = gl_Matrix4_Translate(projection, -hw, -hh, 0);
227  projection = gl_Matrix4_Scale(projection, w, h, 1);
228  }
229  glEnableVertexAttribArray( shaders.texture.vertex );
230  gl_vboActivateAttribOffset( gl_squareVBO, shaders.texture.vertex,
231  0, 2, GL_FLOAT, 0 );
232 
233  /* Set the texture. */
234  tex_mat = (texture->flags & OPENGL_TEX_VFLIP) ? gl_Matrix4_Ortho(-1, 1, 2, 0, 1, -1) : gl_Matrix4_Identity();
235  tex_mat = gl_Matrix4_Translate(tex_mat, tx, ty, 0);
236  tex_mat = gl_Matrix4_Scale(tex_mat, tw, th, 1);
237 
238  /* Set shader uniforms. */
239  gl_uniformColor(shaders.texture.color, c);
240  gl_Matrix4_Uniform(shaders.texture.projection, projection);
241  gl_Matrix4_Uniform(shaders.texture.tex_mat, tex_mat);
242 
243  /* Draw. */
244  glDrawArrays( GL_TRIANGLE_STRIP, 0, 4 );
245 
246  /* Clear state. */
247  glDisableVertexAttribArray( shaders.texture.vertex );
248 
249  /* anything failed? */
250  gl_checkErr();
251 
252  glUseProgram(0);
253 }

309  c = &cWhite;

312  projection = gl_view_matrix;

314  projection = gl_Matrix4_Scale(projection, w, h, 1);
