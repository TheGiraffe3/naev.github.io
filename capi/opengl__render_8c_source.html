250 void gl_blitTexture( const glTexture* texture,
251  const double x, const double y,
252  const double w, const double h,
253  const double tx, const double ty,
254  const double tw, const double th, const glColour *c, const double angle )
255 {
256  // Half width and height
257  double hw, hh;
258  gl_Matrix4 projection, tex_mat;
259 
260  glUseProgram(shaders.texture.program);
261 
262  /* Bind the texture. */
263  glBindTexture( GL_TEXTURE_2D, texture->texture);
264 
265  /* Must have colour for now. */
266  if (c == NULL)
267  c = &cWhite;
268 
269  hw = w/2.;
270  hh = h/2.;
271 
272  /* Set the vertex. */
273  projection = gl_view_matrix;
274  if (angle==0.){
275  projection = gl_Matrix4_Translate(projection, x, y, 0);
276  projection = gl_Matrix4_Scale(projection, w, h, 1);
277  } else {
278  projection = gl_Matrix4_Translate(projection, x+hw, y+hh, 0);
279  projection = gl_Matrix4_Rotate2d(projection, angle);
280  projection = gl_Matrix4_Translate(projection, -hw, -hh, 0);
281  projection = gl_Matrix4_Scale(projection, w, h, 1);
282  }
283  glEnableVertexAttribArray( shaders.texture.vertex );
284  gl_vboActivateAttribOffset( gl_squareVBO, shaders.texture.vertex,
285  0, 2, GL_FLOAT, 0 );
286 
287  /* Set the texture. */
288  tex_mat = (texture->flags & OPENGL_TEX_VFLIP) ? gl_Matrix4_Ortho(-1, 1, 2, 0, 1, -1) : gl_Matrix4_Identity();
289  tex_mat = gl_Matrix4_Translate(tex_mat, tx, ty, 0);
290  tex_mat = gl_Matrix4_Scale(tex_mat, tw, th, 1);
291 
292  /* Set shader uniforms. */
293  gl_uniformColor(shaders.texture.color, c);
294  gl_Matrix4_Uniform(shaders.texture.projection, projection);
295  gl_Matrix4_Uniform(shaders.texture.tex_mat, tex_mat);
296 
297  /* Draw. */
298  glDrawArrays( GL_TRIANGLE_STRIP, 0, 4 );
299 
300  /* Clear state. */
301  glDisableVertexAttribArray( shaders.texture.vertex );
302 
303  /* anything failed? */
304  gl_checkErr();
305 
306  glUseProgram(0);
307 }
