370 static int guiL_menuInfo( lua_State *L )
371 {
372  const char *str;
373  int window;
374 
375  NLUA_CHECKRW(L);
376 
377  if (menu_open)
378  return 0;
379 
380  if (lua_gettop(L) > 0)
381  str = luaL_checkstring(L,1);
382  else {
383  /* No parameter. */
384  menu_info( INFO_MAIN );
385  return 0;
386  }
387 
388  /* Parse string. */
389  if (strcasecmp( str, "main" )==0)
390  window = INFO_MAIN;
391  else if (strcasecmp( str, "ship" )==0)
392  window = INFO_SHIP;
393  else if (strcasecmp( str, "weapons" )==0)
394  window = INFO_WEAPONS;
395  else if (strcasecmp( str, "cargo" )==0)
396  window = INFO_CARGO;
397  else if (strcasecmp( str, "missions" )==0)
398  window = INFO_MISSIONS;
399  else if (strcasecmp( str, "standings" )==0)
400  window = INFO_STANDINGS;
401  else {
402  NLUA_ERROR(L,_("Invalid window info name '%s'."), str);
403  return 0;
404  }
405 
406  /* Open window. */
407  menu_info( window );
408 
409  return 0;
410 }
