341 static int guiL_menuInfo( lua_State *L )
342 {
343  const char *str;
344  int window;
345 
346  NLUA_CHECKRW(L);
347 
348  if (menu_open)
349  return 0;
350 
351  if (lua_gettop(L) > 0)
352  str = luaL_checkstring(L,1);
353  else {
354  /* No parameter. */
355  menu_info( INFO_MAIN );
356  return 0;
357  }
358 
359  /* Parse string. */
360  if (strcasecmp( str, "main" )==0)
361  window = INFO_MAIN;
362  else if (strcasecmp( str, "ship" )==0)
363  window = INFO_SHIP;
364  else if (strcasecmp( str, "weapons" )==0)
365  window = INFO_WEAPONS;
366  else if (strcasecmp( str, "cargo" )==0)
367  window = INFO_CARGO;
368  else if (strcasecmp( str, "missions" )==0)
369  window = INFO_MISSIONS;
370  else if (strcasecmp( str, "standings" )==0)
371  window = INFO_STANDINGS;
372  else {
373  NLUA_ERROR(L,_("Invalid window info name '%s'."), str);
374  return 0;
375  }
376 
377  /* Open window. */
378  menu_info( window );
379 
380  return 0;
381 }
