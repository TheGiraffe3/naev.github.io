180 static int nxml_persistDataNode( lua_State *L, xmlTextWriterPtr writer, int intable )
181 {
182  int ret, b;
183  Ship *sh;
184  ntime_t t;
185  LuaJump *lj;
186  Commodity *com;
187  Outfit *o;
188  Planet *pnt;
189  StarSystem *ss, *dest;
190  char buf[ PATH_MAX ];
191  const char *name, *str, *data;
192  int keynum;
193  size_t len, name_len;
194 
195  /* Default values. */
196  ret = 0;
197 
198  /* We receive data in the format of: key, value */
199 
200  /* key, value */
201  /* Handle different types of keys, we must not touch the stack after this operation. */
202  switch (lua_type(L, -2)) {
203  case LUA_TSTRING:
204  /* Can just tostring directly. */
205  name = lua_tolstring( L, -2, &name_len );
206  /* Isn't a number key. */
207  keynum = 0;
208  break;
209  case LUA_TNUMBER:
210  /* Can't tostring directly. */
211  lua_pushvalue(L,-2);
212  name = lua_tolstring( L, -1, &name_len );
213  lua_pop(L,1); /* Pop the new value. */
214  /* Is a number key. */
215  keynum = 1;
216  break;
217 
218  /* We only handle string or number keys, so ignore the rest. */
219  default:
220  lua_pop(L,1); /* key. */
221  return 0;
222  }
223  /* key, value */
224 
225  /* Now handle the value. */
226  switch (lua_type(L, -1)) {
227  /* Recursive for tables. */
228  case LUA_TTABLE:
229  /* Check if should save -- only if not in table.. */
230  if (!intable) {
231  lua_getfield(L, -1, "__save"); /* key, value, field */
232  b = lua_toboolean(L,-1);
233  lua_pop(L,1); /* key, value */
234  if (!b) /* No need to save. */
235  break;
236  }
237  /* Start the table. */
238  xmlw_startElem(writer,"data");
239  xmlw_attr(writer,"type","table");
240  nxml_saveNameAttribute( writer, name, name_len );
241  if (keynum)
242  xmlw_attr(writer,"keynum","1");
243  lua_pushnil(L); /* key, value, nil */
244  while (lua_next(L, -2) != 0) {
245  /* key, value, key, value */
246  ret |= nxml_persistDataNode( L, writer, 1 ); /* pops the value. */
247  /* key, value, key */
248  }
249  /* key, value */
250  xmlw_endElem(writer); /* "table" */
251  break;
252 
253  /* Normal number. */
254  case LUA_TNUMBER:
255  nxml_saveData( writer, "number", name, name_len, lua_tostring( L, -1 ), keynum );
256  /* key, value */
257  break;
258 
259  /* Boolean is either 1 or 0. */
260  case LUA_TBOOLEAN:
261  /* lua_tostring doesn't work on booleans. */
262  if (lua_toboolean(L,-1)) buf[0] = '1';
263  else buf[0] = '0';
264  buf[1] = '\0';
265  nxml_saveData( writer, "bool", name, name_len, buf, keynum );
266  /* key, value */
267  break;
268 
269  /* String is saved normally. */
270  case LUA_TSTRING:
271  data = lua_tolstring( L, -1, &len );
272  if ( nxml_canWriteString( data, len ) )
273  nxml_saveData( writer, "string", name, name_len, lua_tostring( L, -1 ), keynum );
274  else {
275  char *encoded = base64_encode_to_cstr( data, len );
276  nxml_saveData( writer, "string_base64", name, name_len, encoded, keynum );
277  free( encoded );
278  }
279  /* key, value */
280  break;
281 
282  /* User data must be handled here. */
283  case LUA_TUSERDATA:
284  if (lua_isplanet(L,-1)) {
285  pnt = planet_getIndex( *lua_toplanet(L,-1) );
286  if (pnt != NULL)
287  nxml_saveData( writer, PLANET_METATABLE, name, name_len, pnt->name, keynum );
288  else
289  WARN(_("Failed to save invalid planet."));
290  /* key, value */
291  break;
292  }
293  else if (lua_issystem(L,-1)) {
294  ss = system_getIndex( lua_tosystem(L,-1) );
295  if (ss != NULL)
296  nxml_saveData( writer, SYSTEM_METATABLE, name, name_len, ss->name, keynum );
297  else
298  WARN(_("Failed to save invalid system."));
299  /* key, value */
300  break;
301  }
302  else if (lua_isfaction(L,-1)) {
303  str = faction_name( lua_tofaction(L,-1) );
304  if (str == NULL)
305  break;
306  nxml_saveData( writer, FACTION_METATABLE, name, name_len, str, keynum );
307  /* key, value */
308  break;
309  }
310  else if (lua_isship(L,-1)) {
311  sh = lua_toship(L,-1);
312  str = sh->name;
313  if (str == NULL)
314  break;
315  nxml_saveData( writer, SHIP_METATABLE, name, name_len, str, keynum );
316  /* key, value */
317  break;
318  }
319  else if (lua_istime(L,-1)) {
320  t = *lua_totime(L,-1);
321  snprintf( buf, sizeof(buf), "%"PRId64, t );
322  nxml_saveData( writer, TIME_METATABLE, name, name_len, buf, keynum );
323  /* key, value */
324  break;
325  }
326  else if (lua_isjump(L,-1)) {
327  lj = lua_tojump(L,-1);
328  ss = system_getIndex( lj->srcid );
329  dest = system_getIndex( lj->destid );
330  if ((ss == NULL) || (dest == NULL))
331  WARN(_("Failed to save invalid jump."));
332  else
333  nxml_saveJump( writer, name, name_len, ss->name, dest->name );
334  }
335  else if (lua_iscommodity(L,-1)) {
336  com = lua_tocommodity(L,-1);
337  if( nxml_saveCommodity( writer, name, name_len, com ) != 0)
338  WARN( _("Failed to save invalid commodity.") );
339  /* key, value */
340  break;
341  }
342  else if (lua_isoutfit(L,-1)) {
343  o = lua_tooutfit(L,-1);
344  str = o->name;
345  if (str == NULL)
346  break;
347  nxml_saveData( writer, OUTFIT_METATABLE, name, name_len, str, keynum );
348  /* key, value */
349  break;
350  }
351  /* Purpose fallthrough. */
352 
353  /* Rest gets ignored, like functions, etc... */
354  default:
355  /* key, value */
356  break;
357  }
358  lua_pop(L,1); /* key */
359 
360  /* We must pop the value and leave only the key so it can continue iterating. */
361 
362  return ret;
363 }

402 static int nxml_unpersistDataNode( lua_State *L, xmlNodePtr parent )
403 {
404  LuaJump lj;
405  Planet *pnt;
406  StarSystem *ss, *dest;
407  xmlNodePtr node;
408  char *name, *type, *buf, *num, *data;
409  size_t len;
410 
411  node = parent->xmlChildrenNode;
412  do {
413  if (xml_isNode(node,"data")) {
414  /* Get general info. */
415  xmlr_attr_strd(node,"name",name);
416  xmlr_attr_strd(node,"type",type);
417  /* Check to see if key is a number. */
418  xmlr_attr_strd(node,"keynum",num);
419  if (num != NULL) {
420  lua_pushnumber(L, atof(name));
421  free(num);
422  }
423  else if ( name != NULL )
424  lua_pushstring(L, name);
425  else {
426  xmlr_attr_strd( node, "name_base64", name );
427  data = base64_decode_cstr( &len, name );
428  lua_pushlstring( L, data, len );
429  free( data );
430  }
431 
432  /* handle data types */
433  /* Recursive tables. */
434  if (strcmp(type,"table")==0) {
435  xmlr_attr_strd(node,"name",buf);
436  /* Create new table. */
437  lua_newtable(L);
438  /* Save data. */
439  nxml_unpersistDataNode(L,node);
440  /* Set table. */
441  free(buf);
442  }
443  else if (strcmp(type,"number")==0)
444  lua_pushnumber(L,xml_getFloat(node));
445  else if (strcmp(type,"bool")==0)
446  lua_pushboolean(L,xml_getInt(node));
447  else if (strcmp(type,"string")==0)
448  lua_pushstring(L,xml_get(node));
449  else if ( strcmp( type, "string_base64" ) == 0 ) {
450  data = base64_decode_cstr( &len, xml_get( node ) );
451  lua_pushlstring( L, data, len );
452  free( data );
453  }
454  else if (strcmp(type,PLANET_METATABLE)==0) {
455  pnt = planet_get(xml_get(node));
456  if (pnt != NULL) {
457  lua_pushplanet(L,planet_index(pnt));
458  }
459  else
460  WARN(_("Failed to load nonexistent planet '%s'"), xml_get(node));
461  }
462  else if (strcmp(type,SYSTEM_METATABLE)==0) {
463  ss = system_get(xml_get(node));
464  if (ss != NULL)
465  lua_pushsystem(L,system_index( ss ));
466  else
467  WARN(_("Failed to load nonexistent system '%s'"), xml_get(node));
468  }
469  else if (strcmp(type,FACTION_METATABLE)==0) {
470  lua_pushfaction(L,faction_get(xml_get(node)));
471  }
472  else if (strcmp(type,SHIP_METATABLE)==0)
473  lua_pushship(L,ship_get(xml_get(node)));
474  else if (strcmp(type,TIME_METATABLE)==0) {
475  lua_pushtime(L,xml_getLong(node));
476  }
477  else if (strcmp(type,JUMP_METATABLE)==0) {
478  ss = system_get(xml_get(node));
479  xmlr_attr_strd(node,"dest",buf);
480  dest = system_get( buf );
481  if ((ss != NULL) && (dest != NULL)) {
482  lj.srcid = ss->id;
483  lj.destid = dest->id;
484  lua_pushjump(L,lj);
485  }
486  else
487  WARN(_("Failed to load nonexistent jump from '%s' to '%s'"), xml_get(node), buf);
488  free(buf);
489  }
490  else if (strcmp(type,COMMODITY_METATABLE)==0)
491  lua_pushcommodity(L, nxml_loadCommodity( node ) );
492  else if (strcmp(type,OUTFIT_METATABLE)==0)
493  lua_pushoutfit(L,outfit_get(xml_get(node)));
494  else {
495  /* There are a few types knowingly left out above.
496  * Meaningless (?): PILOT_METATABLE.
497  * Seems doable, use case unclear: ARTICLE_METATABLE, VEC_METATABLE.
498  * Seems doable, but probably GUI-only: COL_METATABLE, TEX_METATABLE.
499  * */
500  WARN(_("Unknown Lua data type!"));
501  lua_pop(L,1);
502  return -1;
503  }
504 
505  /* Set field. */
506  lua_settable(L, -3);
507 
508  /* cleanup */
509  free(type);
510  free(name);
511  }
512  } while (xml_nextNode(node));
513 
514  return 0;
515 }
