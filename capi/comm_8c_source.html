83 int comm_openPilot( unsigned int pilot )
84 {
85  const char *msg;
86  char c;
87  unsigned int wid;
88  Pilot *p;
89  HookParam hparam[2];
90 
91  /* Get the pilot. */
92  p = pilot_get( pilot );
93  c = pilot_getFactionColourChar( p );
94 
95  /* Make sure pilot exists. */
96  if (p == NULL)
97  return -1;
98 
99  /* Make sure pilot in range. */
100  if (!pilot_isFlag(p, PILOT_HAILING) &&
101  pilot_inRangePilot( player.p, p, NULL ) <= 0) {
102  player_message(_("#rTarget is out of communications range"));
103  return -1;
104  }
105 
106  /* Destroy the window if it's already present. */
107  wid = window_get( "wdwComm" );
108  if (wid > 0) {
109  window_destroy( wid );
110  return 0;
111  }
112 
113  /* Must not be jumping. */
114  if (pilot_isFlag(p, PILOT_HYPERSPACE)) {
115  player_message(_("#%c%s#r is jumping and can't respond"), c, p->name);
116  return 0;
117  }
118 
119  /* Must not be disabled. */
120  if (pilot_isFlag(p, PILOT_DISABLED)) {
121  player_message(_("#%c%s#r does not respond"), c, p->name);
122  return 0;
123  }
124 
125  /* Check for player faction (escorts). */
126  if (p->faction == FACTION_PLAYER) {
127  escort_playerCommand( p );
128  return 0;
129  }
130 
131  /* Check to see if pilot wants to communicate. */
132  msg = comm_getString( p, "comm_no" );
133  if (msg != NULL) {
134  player_messageRaw( msg );
135  return 0;
136  }
137 
138  /* Set up for the comm_get* functions. */
139  ai_setPilot( p );
140 
141  /* Have pilot stop hailing. */
142  pilot_rmFlag( p, PILOT_HAILING );
143 
144  /* Don't close automatically. */
145  comm_commClose = 0;
146 
147  /* Run generic hail hooks. */
148  hparam[0].type = HOOK_PARAM_PILOT;
149  hparam[0].u.lp = p->id;
150  hparam[1].type = HOOK_PARAM_SENTINEL;
151  hooks_runParam( "hail", hparam );
152  pilot_runHook( p, PILOT_HOOK_HAIL );
153 
154  /* Close window if necessary. */
155  if (comm_commClose) {
156  p = NULL;
157  comm_planet = NULL;
158  comm_commClose = 0;
159  return 0;
160  }
161 
162  /* Set up environment first time. */
163  if (comm_env == LUA_NOREF) {
164  comm_env = nlua_newEnv(1);
165  nlua_loadStandard( comm_env );
166 
167  size_t bufsize;
168  char *buf = ndata_read( COMM_PATH, &bufsize );
169  if (nlua_dobufenv(comm_env, buf, bufsize, COMM_PATH) != 0) {
170  WARN( _("Error loading file: %s\n"
171  "%s\n"
172  "Most likely Lua file has improper syntax, please check"),
173  COMM_PATH, lua_tostring(naevL,-1));
174  free(buf);
175  return -1;
176  }
177  free(buf);
178  }
179 
180  /* Run Lua. */
181  nlua_getenv(comm_env,"comm");
182  lua_pushpilot(naevL, p->id);
183  if (nlua_pcall(comm_env, 1, 0)) { /* error has occurred */
184  WARN( _("Comm: '%s'"), lua_tostring(naevL,-1));
185  lua_pop(naevL,1);
186  }
187 
188  return 0;
189 }

381 static void comm_bribePlanet( unsigned int wid, char *unused )
382 {
383  (void) unused;
384  int answer;
385  credits_t price;
386 
387  /* Get price. */
388  price = comm_planet->bribe_price;
389 
390  /* No bribing. */
391  if (comm_planet->bribe_price <= 0.) {
392  dialogue_msgRaw( _("Bribe Starport"), comm_planet->bribe_msg );
393  return;
394  }
395 
396  /* Yes/No input. */
397  answer = dialogue_YesNoRaw( _("Bribe Starport"), comm_planet->bribe_msg );
398 
399  /* Said no. */
400  if (answer == 0) {
401  dialogue_msg(_("Bribe Starport"), _("You decide not to pay."));
402  return;
403  }
404 
405  /* Check if has the money. */
406  if (!player_hasCredits( price )) {
407  dialogue_msg(_("Bribe Starport"), _("You don't have enough credits for the bribery."));
408  return;
409  }
410 
411  /* Pay the money. */
412  player_modCredits( -price );
413  dialogue_msg(_("Bribe Starport"), _("You have permission to dock."));
414 
415  /* Mark as bribed and don't allow bribing again. */
416  comm_planet->bribed = 1;
417 
418  /* Reopen window. */
419  window_destroy( wid );
420  comm_openPlanet( comm_planet );
421 }
