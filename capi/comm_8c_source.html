403 static void comm_bribePilot( unsigned int wid, char *unused )
404 {
405  (void) unused;
406  int answer;
407  double d;
408  credits_t price;
409  char pricestr[ECON_CRED_STRLEN];
410  const char *str;
411 
412  /* Unbribable. */
413  str = comm_getString( "bribe_no" );
414  if (str != NULL) {
415  dialogue_msg(_("Bribe Pilot"), "%s", str );
416  return;
417  }
418 
419  /* Get amount pilot wants. */
420  if (comm_getNumber( &d, "bribe" )) {
421  WARN(_("Pilot '%s' accepts bribes but doesn't give price!"), comm_pilot->name );
422  d = 0.;
423  }
424  price = (credits_t) d;
425 
426  /* Check to see if already bribed. */
427  if (price == 0) {
428  dialogue_msg(_("Bribe Pilot"), _("\"Money won't save your hide now!\""));
429  return;
430  }
431 
432  /* Bribe message. */
433  price2str( pricestr, price, player.p->credits, -1 );
434  str = comm_getString( "bribe_prompt" );
435  if (str == NULL)
436  answer = dialogue_YesNo( _("Bribe Pilot"), _("\"I'm gonna need at least %s to not leave you as a hunk of floating debris.\"\n\nPay %s?"), pricestr, pricestr );
437  else
438  answer = dialogue_YesNo( _("Bribe Pilot"), _("%s\n\nPay %s?"), str, pricestr );
439 
440  /* Said no. */
441  if (answer == 0) {
442  dialogue_msg(_("Bribe Pilot"), _("You decide not to pay."));
443  return;
444  }
445 
446  /* Check if has the money. */
447  if (!player_hasCredits( price )) {
448  dialogue_msg(_("Bribe Pilot"), _("You don't have enough credits for the bribery."));
449  return;
450  }
451 
452  player_modCredits( -price );
453  str = comm_getString( "bribe_paid" );
454  if (str == NULL)
455  dialogue_msg(_("Bribe Pilot"), _("\"Pleasure to do business with you.\""));
456  else
457  dialogue_msg(_("Bribe Pilot"), "%s", str);
458 
459  /* Mark as bribed and don't allow bribing again. */
460  pilot_setFlag( comm_pilot, PILOT_BRIBED );
461  pilot_rmHostile( comm_pilot );
462 
463  /* Stop hyperspace if necessary. */
464  pilot_rmFlag( comm_pilot, PILOT_HYP_PREP );
465  pilot_rmFlag( comm_pilot, PILOT_HYP_BRAKE );
466  pilot_rmFlag( comm_pilot, PILOT_HYP_BEGIN );
467 
468  /* Don't allow rebribe. */
469  if (comm_pilot->ai != NULL) {
470  nlua_getenv(comm_pilot->ai->env, "mem");
471  lua_pushnumber(naevL, 0);
472  lua_setfield(naevL, -2, "bribe");
473  lua_pop(naevL,1);
474  }
475 
476  /* Reopen window. */
477  window_destroy( wid );
478  comm_openPilot( comm_pilot->id );
479 }

537 static void comm_requestFuel( unsigned int wid, char *unused )
538 {
539  (void) wid;
540  (void) unused;
541  double val;
542  const char *msg;
543  int ret;
544  credits_t price;
545  char creditstr[ECON_CRED_STRLEN];
546 
547  /* Check to see if ship has a no refuel message. */
548  msg = comm_getString( "refuel_no" );
549  if (msg != NULL) {
550  dialogue_msgRaw( _("Request Fuel"), msg );
551  return;
552  }
553 
554  /* Must need refueling. */
555  if (player.p->fuel >= player.p->fuel_max) {
556  dialogue_msg( _("Request Fuel"), _("Your fuel deposits are already full.") );
557  return;
558  }
559 
560  /* See if pilot has enough fuel. */
561  if (comm_pilot->fuel < 200.) {
562  dialogue_msg( _("Request Fuel"),
563  _("\"Sorry, I don't have enough fuel to spare at the moment.\"") );
564  return;
565  }
566 
567  /* See if player can get refueled. */
568  val = 0.;
569  ret = comm_getNumber( &val, "refuel" );
570  msg = comm_getString( "refuel_msg" );
571  if ((ret != 0) || (msg == NULL) || pilot_isFlag(comm_pilot, PILOT_MANUAL_CONTROL)) {
572  dialogue_msg( _("Request Fuel"), _("\"Sorry, I'm busy now.\"") );
573  return;
574  }
575  price = (credits_t) val;
576 
577  /* Check to see if is already refueling. */
578  if (pilot_isFlag(comm_pilot, PILOT_REFUELING)) {
579  dialogue_msg( _("Request Fuel"), _("Pilot is already refueling you.") );
580  return;
581  }
582 
583  /* See if player really wants to pay. */
584  if (price > 0) {
585  price2str( creditstr, price, player.p->credits, -1 );
586  ret = dialogue_YesNo( _("Request Fuel"), _("%s\n\nPay %s?"), msg, creditstr );
587  if (ret == 0) {
588  dialogue_msg( _("Request Fuel"), _("You decide not to pay.") );
589  return;
590  }
591  }
592  else
593  dialogue_msg( _("Request Fuel"), "%s", msg );
594 
595  /* Check if he has the money. */
596  if (!player_hasCredits( price )) {
597  credits2str( creditstr, price - player.p->credits, -1 );
598  dialogue_msg( _("Request Fuel"), _("You need %s more!"), creditstr );
599  return;
600  }
601 
602  /* Take money. */
603  player_modCredits( -price );
604  pilot_modCredits( comm_pilot, price );
605 
606  /* Start refueling. */
607  pilot_rmFlag( comm_pilot, PILOT_HYP_PREP);
608  pilot_rmFlag( comm_pilot, PILOT_HYP_BRAKE );
609  pilot_rmFlag( comm_pilot, PILOT_HYP_BEGIN);
610  pilot_setFlag( comm_pilot, PILOT_REFUELING);
611  ai_refuel( comm_pilot, player.p->id );
612 
613  /* Last message. */
614  if (price > 0)
615  dialogue_msg( _("Request Fuel"), _("\"On my way.\"") );
616 }
