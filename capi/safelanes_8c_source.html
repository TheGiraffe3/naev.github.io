240  tmp_planet_indices = array_create( int );
241  tmp_jump_edges = array_create( Edge );

246  Vertex v = {.system = system, .type = VERTEX_PLANET, .index = i};
247  array_push_back( &tmp_planet_indices, array_size(vertex_stack) );

255  Vertex v = {.system = system, .type = VERTEX_JUMP, .index = i};

275 static void safelanes_initStacks_edge (void)
276 {
277  const StarSystem *systems_stack;
278  int system, i, j, k;
279 
280  systems_stack = system_getAll();
281 
282  edge_stack = array_create( Edge );
283  lane_fmask = array_create( FactionMask );
284  tmp_edge_conduct = array_create( double );
285  for (system=0; system<array_size(systems_stack); system++) {
286  for (i = sys_to_first_vertex[system]; i < sys_to_first_vertex[1+system]; i++) {
287  const Vector2d *pi = vertex_pos( i );
288  for (j = sys_to_first_vertex[system]; j < i; j++) {
289  const Vector2d *pj = vertex_pos( j );
290  double lij = vect_dist( pi, pj );
291  int has_approx_midpoint = 0;
292  for (k = sys_to_first_vertex[system]; k < sys_to_first_vertex[1+system]; k++)
293  if (k!=i && k!=j && safelanes_triangleTooFlat( pi, pj, vertex_pos( k ), lij )) {
294  has_approx_midpoint = 1;
295  break;
296  }
297  if (has_approx_midpoint)
298  continue; /* The edge from i to j is disallowed in favor of a path through k. */
299  array_push_back_edge( &edge_stack, j, i );
300  array_push_back( &lane_fmask, MASK_COMPROMISE( vertex_faction( i ), vertex_faction( j ) ) );
301  array_push_back( &tmp_edge_conduct, 1/lij );
302  }
303  }
304  }
305  array_shrink( &edge_stack );
306 
307  lane_faction = array_create_size( int, array_size(edge_stack) );
308  memset( lane_faction, 0, array_size(lane_faction)*sizeof(lane_faction[0]) );
309 }

396 static void safelanes_initStiff (void)
397 {
398  int n, v, i;
399  int *pi, *pj;
400  double *pv;
401  double max_conductivity;
402 
403  cholmod_free_triplet( &stiff, &C );
404  v = array_size(vertex_stack);
405  n = 3*(array_size(edge_stack)+array_size(tmp_jump_edges)) + array_size(tmp_anchor_vertices);
406  stiff = cholmod_allocate_triplet( v, v, n, STORAGE_MODE_UPPER_TRIANGULAR_PART, CHOLMOD_REAL, &C );
407  /* Populate triplets: internal edges (ii ij jj), implicit jump connections (ditto), anchor conditions. */
408  pi = stiff->i; pj = stiff->j; pv = stiff->x;
409  for (i=0; i<array_size(edge_stack); i++) {
410  *pi++ = edge_stack[i][0]; *pj++ = edge_stack[i][0]; *pv++ = +tmp_edge_conduct[i];
411  *pi++ = edge_stack[i][0]; *pj++ = edge_stack[i][1]; *pv++ = -tmp_edge_conduct[i];
412  *pi++ = edge_stack[i][1]; *pj++ = edge_stack[i][1]; *pv++ = +tmp_edge_conduct[i];
413  }
414  for (i=0; i<array_size(tmp_jump_edges); i++) {
415  *pi++ = tmp_jump_edges[i][0]; *pj++ = tmp_jump_edges[i][0]; *pv++ = +JUMP_CONDUCTIVITY;
416  *pi++ = tmp_jump_edges[i][0]; *pj++ = tmp_jump_edges[i][1]; *pv++ = -JUMP_CONDUCTIVITY;
417  *pi++ = tmp_jump_edges[i][1]; *pj++ = tmp_jump_edges[i][1]; *pv++ = +JUMP_CONDUCTIVITY;
418  }
419  /* Add a Robin boundary condition, using the max conductivity (after activation) for spectral reasons. */
420  max_conductivity = JUMP_CONDUCTIVITY/(ALPHA+1);
421  for (i=0; i<array_size(edge_stack); i++)
422  max_conductivity = MAX( max_conductivity, tmp_edge_conduct[i] );
423  max_conductivity = MAX( JUMP_CONDUCTIVITY, (1+ALPHA)*max_conductivity ); /* Activation scales entries by 1+ALPHA later. */
424  for (i=0; i<array_size(tmp_anchor_vertices); i++) {
425  *pi++ = tmp_anchor_vertices[i]; *pj++ = tmp_anchor_vertices[i]; *pv++ = max_conductivity;
426  }
427 #if DEBUGGING
428  assert( cholmod_check_triplet( stiff, &C) );
429 #endif
430 }

509 static int safelanes_triangleTooFlat( const Vector2d* m, const Vector2d* n, const Vector2d* p, double lmn )
510 {
511  const double MAX_COSINE = cos(MIN_ANGLE);
512  double lnp = vect_dist( n, p );
513  double lmp = vect_dist( m, p );
514  double dpn = ((n->x-m->x)*(n->x-p->x) + (n->y-m->y)*(n->y-p->y)) / ( lmn * lnp );
515  double dpm = ((m->x-n->x)*(m->x-p->x) + (m->y-n->y)*(m->y-p->y)) / ( lmn * lmp );
516  return (dpn > MAX_COSINE && lnp < lmn) || (dpm > MAX_COSINE && lmp < lmn);
517 }
