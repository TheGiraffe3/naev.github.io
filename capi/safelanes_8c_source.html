184 SafeLane* safelanes_get( int faction, int standing, const StarSystem* system )
185 {
186  int i, j;
187  SafeLane *out, *l;
188  const Vertex *v[2];
189  int lf, fe, fa, skip;
190 
191  out = array_create( SafeLane );
192 
193  for (i=sys_to_first_edge[system->id]; i<sys_to_first_edge[1+system->id]; i++) {
194  lf = lane_faction[i];
195 
196  /* No lane on edge. */
197  if (lf <= 0)
198  continue;
199 
200  /* Filter by standing. */
201  if (faction >= 0) {
202  if (standing==0) {
203  /* Only match exact faction. */
204  if (lf != faction)
205  continue;
206  }
207  else {
208  /* Try to do more advanced matching. */
209  fe = areEnemies(faction,lf);
210  fa = areAllies(faction,lf);
211  skip = 1;
212  if ((standing & SAFELANES_FRIENDLY) && fa)
213  skip = 0;
214  if ((standing & SAFELANES_NEUTRAL) && !fe)
215  skip = 0;
216  if ((standing & SAFELANES_HOSTILE) && fe)
217  skip = 0;
218  if (skip)
219  continue;
220  }
221  }
222 
223  for (j=0; j<2; j++)
224  v[j] = &vertex_stack[edge_stack[i][j]];
225 
226  l = &array_grow( &out );
227  l->faction = lane_faction[i];
228  for (j=0; j<2; j++) {
229  switch (v[j]->type) {
230  case VERTEX_PLANET:
231  l->point_type[j] = SAFELANE_LOC_PLANET;
232  l->point_id[j] = system->planets[v[j]->index]->id;
233  break;
234  case VERTEX_JUMP:
235  l->point_type[j] = SAFELANE_LOC_DEST_SYS;
236  l->point_id[j] = system->jumps[v[j]->index].targetid;
237  break;
238  default:
239  ERR( _("Safe-lane vertex type is invalid.") );
240  }
241  }
242  }
243  return out;
244 }

298 static int safelanes_buildOneTurn (void)
299 {
300  cholmod_sparse *stiff_s;
301  cholmod_factor *stiff_f;
302  cholmod_dense *_QtQutilde, *Lambda_tilde, *Y_workspace, *E_workspace;
303  int turns_next_time;
304  double zero[] = {0, 0}, neg_1[] = {-1, 0};
305 
306  Y_workspace = E_workspace = Lambda_tilde = NULL;
307  stiff_s = cholmod_triplet_to_sparse( stiff, 0, &C );
308  stiff_f = cholmod_analyze( stiff_s, &C );
309  cholmod_factorize( stiff_s, stiff_f, &C );
310  cholmod_solve2( CHOLMOD_A, stiff_f, ftilde, NULL, &utilde, NULL, &Y_workspace, &E_workspace, &C );
311  _QtQutilde = cholmod_zeros( utilde->nrow, utilde->ncol, CHOLMOD_REAL, &C );
312  cholmod_sdmult( QtQ, 0, neg_1, zero, utilde, _QtQutilde, &C );
313  cholmod_solve2( CHOLMOD_A, stiff_f, _QtQutilde, NULL, &Lambda_tilde, NULL, &Y_workspace, &E_workspace, &C );
314  cholmod_free_dense( &_QtQutilde, &C );
315  cholmod_free_dense( &Y_workspace, &C );
316  cholmod_free_dense( &E_workspace, &C );
317  cholmod_free_factor( &stiff_f, &C );
318  cholmod_free_sparse( &stiff_s, &C );
319  turns_next_time = safelanes_activateByGradient( Lambda_tilde );
320  cholmod_free_dense( &Lambda_tilde, &C );
321 
322  return turns_next_time;
323 }

389 static void safelanes_initStacks_edge (void)
390 {
391  const StarSystem *systems_stack;
392  int system, i, j, k;
393 
394  systems_stack = system_getAll();
395 
396  edge_stack = array_create( Edge );
397  sys_to_first_edge = array_create( int );
398  array_push_back( &sys_to_first_edge, 0 );
399  lane_fmask = array_create( FactionMask );
400  tmp_edge_conduct = array_create( double );
401  for (system=0; system<array_size(systems_stack); system++) {
402  for (i = sys_to_first_vertex[system]; i < sys_to_first_vertex[1+system]; i++) {
403  const Vector2d *pi = vertex_pos( i );
404  for (j = sys_to_first_vertex[system]; j < i; j++) {
405  const Vector2d *pj = vertex_pos( j );
406  double lij = vect_dist( pi, pj );
407  int has_approx_midpoint = 0;
408  for (k = sys_to_first_vertex[system]; k < sys_to_first_vertex[1+system]; k++)
409  if (k!=i && k!=j && safelanes_triangleTooFlat( pi, pj, vertex_pos( k ), lij )) {
410  has_approx_midpoint = 1;
411  break;
412  }
413  if (has_approx_midpoint)
414  continue; /* The edge from i to j is disallowed in favor of a path through k. */
415  array_push_back_edge( &edge_stack, j, i );
416  array_push_back( &lane_fmask, MASK_COMPROMISE( vertex_faction( i ), vertex_faction( j ) ) );
417  array_push_back( &tmp_edge_conduct, 1/lij );
418  }
419  }
420  array_push_back( &sys_to_first_edge, array_size(edge_stack) );
421  }
422  array_shrink( &edge_stack );
423  array_shrink( &sys_to_first_edge );
424 
425  lane_faction = array_create_size( int, array_size(edge_stack) );
426  array_resize( &lane_faction, array_size(edge_stack) );
427  memset( lane_faction, 0, array_size(lane_faction)*sizeof(lane_faction[0]) );
428 }

533 static void safelanes_initStiff (void)
534 {
535  int nnz, v, i;
536  double max_conductivity;
537 
538  cholmod_free_triplet( &stiff, &C );
539  v = array_size(vertex_stack);
540  nnz = 3*(array_size(edge_stack)+array_size(tmp_jump_edges)) + array_size(tmp_anchor_vertices);
541  stiff = cholmod_allocate_triplet( v, v, nnz, STORAGE_MODE_UPPER_TRIANGULAR_PART, CHOLMOD_REAL, &C );
542  /* Populate triplets: internal edges (ii ij jj), implicit jump connections (ditto), anchor conditions. */
543  for (i=0; i<array_size(edge_stack); i++) {
544  triplet_entry( stiff, edge_stack[i][0], edge_stack[i][0], +tmp_edge_conduct[i] );
545  triplet_entry( stiff, edge_stack[i][0], edge_stack[i][1], -tmp_edge_conduct[i] );
546  triplet_entry( stiff, edge_stack[i][1], edge_stack[i][1], +tmp_edge_conduct[i] );
547  }
548  for (i=0; i<array_size(tmp_jump_edges); i++) {
549  triplet_entry( stiff, tmp_jump_edges[i][0], tmp_jump_edges[i][0], +JUMP_CONDUCTIVITY );
550  triplet_entry( stiff, tmp_jump_edges[i][0], tmp_jump_edges[i][1], -JUMP_CONDUCTIVITY );
551  triplet_entry( stiff, tmp_jump_edges[i][1], tmp_jump_edges[i][1], +JUMP_CONDUCTIVITY );
552  }
553  /* Add a Robin boundary condition, using the max conductivity (after activation) for spectral reasons. */
554  max_conductivity = JUMP_CONDUCTIVITY/(1+ALPHA);
555  for (i=0; i<array_size(edge_stack); i++)
556  max_conductivity = MAX( max_conductivity, tmp_edge_conduct[i] );
557  max_conductivity = MAX( JUMP_CONDUCTIVITY, (1+ALPHA)*max_conductivity ); /* Activation scales entries by 1+ALPHA later. */
558  for (i=0; i<array_size(tmp_anchor_vertices); i++)
559  triplet_entry( stiff, tmp_anchor_vertices[i], tmp_anchor_vertices[i], max_conductivity );
560 #if DEBUGGING
561  assert( stiff->nnz == stiff->nzmax );
562  assert( cholmod_check_triplet( stiff, &C) );
563 #endif
564 }

651  /* Form P, the pair-vertex projection where (Dirac notation) P |pair(i,j)> = |i> - |j>. It has a +1 and -1 per column. */
652  /* At least, pretend we did. We want (PD)(PD)*, where D is a diagonal matrix whose pair(i,j) are these presence sums: */
653 
654  component = calloc( np, sizeof(int) );
655  for (i=0; i<np; i++)
656  component[i] = unionfind_find( &tmp_sys_uf, vertex_stack[tmp_planet_indices[i]].system );
657 
658  for (i=0; i<np; i++) {
659  sys = vertex_stack[tmp_planet_indices[i]].system;
660  pnt = system_getIndex( sys )->planets[vertex_stack[tmp_planet_indices[i]].index];
661  pres = pnt->presenceAmount;
662  fi = FACTION_ID_TO_INDEX( pnt->faction );
663  if (fi < 0)
664  continue;
665  Di = PPl[fi]->x;
666  for (j=0; j<i; j++)
667  if (component[i] == component[j])
668  Di[np*i+j] += pres;
669  for (j=i+1; j<np; j++)
670  if (component[i] == component[j])
671  Di[np*j+i] += pres;
672  }
673 
674  /* At this point, PPl[fi]->x[np*i+j] holds the pair(i,j) entry of D. */
675  for (fi=0; fi<array_size(faction_stack); fi++)
676  for (i=0; i<np; i++)
677  for (j=0; j<i; j++) {
678  d = ((double*)PPl[fi]->x)[np*i+j];
679  d *= d;
680  ((double*)PPl[fi]->x)[np*i+j] = -d;
681  ((double*)PPl[fi]->x)[np*j+i] = -d;
682  ((double*)PPl[fi]->x)[np*i+i] += d;
683  ((double*)PPl[fi]->x)[np*j+j] += d;
684  }
685 
686  free( component );
687 }

694 static int safelanes_activateByGradient( cholmod_dense* Lambda_tilde )
695 {
696  int ei, eii, ei_best, fi, fii, *facind_opts, *edgeind_opts, si, sis, sjs, turns_next_time;
697  double *facind_vals, score, score_best, Linv, cost, cost_best, cost_cheapest_other;
698  StarSystem *sys;
699  cholmod_dense **lal;
700  size_t *lal_bases, lal_base, sys_base;
702  lal = calloc( array_size(faction_stack), sizeof(cholmod_dense*) );
703  lal_bases = calloc( array_size(faction_stack), sizeof(size_t) );
704  edgeind_opts = array_create( int );
705  facind_opts = array_create_size( int, array_size(faction_stack) );
706  facind_vals = array_create_size( double, array_size(faction_stack) );
707  for (fi=0; fi<array_size(faction_stack); fi++) {
708  array_push_back( &facind_opts, fi );
709  array_push_back( &facind_vals, 0 );
710  }
711  turns_next_time = 0;
712 
713  for (si=0; si<array_size(sys_to_first_vertex)-1; si++)
714  {
715  /* Factions with most presence here choose first. */
716  sys = system_getIndex( si );
717  for (fi=0; fi<array_size(faction_stack); fi++)
718  facind_vals[fi] = -system_getPresence( sys, faction_stack[fi].id ); /* FIXME: Is this better, or presence_budget? */
719  cmp_key_ref = facind_vals;
720  qsort( facind_opts, array_size(faction_stack), sizeof(int), cmp_key );
721 
722  for (fii=0; fii<array_size(faction_stack); fii++) {
723  fi = facind_opts[fii];
724  sys_base = sys_to_first_vertex[si];
725 
726  /* Get the base index to use for this system. Save the value we expect to be the next iteration's base index.
727  * The current system's rows are in the lal[fi] matrix if there's presence at the time we slice it.
728  * What we know is whether there's presence *now*. We can use that as a proxy and fix lal_bases[fi] if we
729  * deplete this presence before constructing lal[fi]. This is tricky, so there are assertions below,
730  * which can warn us if we fuck this up. */
731  lal_base = lal_bases[fi];
732  if (presence_budget[fi][si] <= 0)
733  continue;
734  /* We "should" find these DoF's interesting if/when we slice, and will unless we deplete this presence first. */
735  lal_bases[fi] += sys_to_first_vertex[1+si] - sys_to_first_vertex[si];
736 
737  array_resize( &edgeind_opts, 0 );
738  for (ei=sys_to_first_edge[si]; ei<sys_to_first_edge[1+si]; ei++)
739  if (!lane_faction[ei]
740  && presence_budget[fi][si] >= 1 / safelanes_initialConductivity(ei) / faction_stack[fi].lane_length_per_presence
741  && (lane_fmask[ei] & (1<<fi)))
742  array_push_back( &edgeind_opts, ei );
743 
744  if (array_size(edgeind_opts) == 0) {
745  presence_budget[fi][si] = 0; /* Nothing to build here! Tell ourselves to stop trying. */
746  if (lal[fi] == NULL)
747  lal_bases[fi] -= sys_to_first_vertex[1+si] - sys_to_first_vertex[si];
748  continue;
749  }
750 
751  ei_best = edgeind_opts[0];
752  cost_best = 1 / safelanes_initialConductivity(ei_best) / faction_stack[fi].lane_length_per_presence;
753  cost_cheapest_other = +HUGE_VAL;
754  if (array_size(edgeind_opts) > 1) {
755  /* There's an actual choice. Search for the best option. Lower is better. */
756  score_best = +HUGE_VAL;
757  for (eii=0; eii<array_size(edgeind_opts); eii++) {
758  ei = edgeind_opts[eii];
759  sis = edge_stack[ei][0];
760  sjs = edge_stack[ei][1];
761 
762  if (lal[fi] == NULL) { /* Is it time to evaluate the lazily-calculated matrix? */
763  cholmod_dense *lamt = safelanes_sliceByPresence( Lambda_tilde, presence_budget[fi] );
764  lal[fi] = ncholmod_ddmult( lamt, 0, PPl[fi] );
765  cholmod_free_dense( &lamt, &C );
766  }
767 
768  score = 0;
769  /* Evaluate (LUTll[0,0] + LUTll[1,1] - LUTll[0,1] - LUTll[1,0]), */
770  /* where LUTll = np.dot( lal[[sis,sjs],:] , utilde[[sis,sjs],:].T ) */
771  score += safelanes_row_dot_row( utilde, lal[fi], sis, sis - sys_base + lal_base );
772  score += safelanes_row_dot_row( utilde, lal[fi], sjs, sjs - sys_base + lal_base );
773  score -= safelanes_row_dot_row( utilde, lal[fi], sjs, sis - sys_base + lal_base );
774  score -= safelanes_row_dot_row( utilde, lal[fi], sis, sjs - sys_base + lal_base );
775  Linv = safelanes_initialConductivity(ei);
776  score *= ALPHA * Linv * Linv;
777 
778  cost = 1 / safelanes_initialConductivity(ei) / faction_stack[fi].lane_length_per_presence;
779  if (score < score_best) {
780  ei_best = ei;
781  score_best = score;
782  cost_cheapest_other = MIN( cost_cheapest_other, cost_best );
783  cost_best = cost;
784  }
785  else
786  cost_cheapest_other = MIN( cost_cheapest_other, cost );
787  }
788  }
789 
790  presence_budget[fi][si] -= cost_best;
791  if (presence_budget[fi][si] >= cost_cheapest_other)
792  turns_next_time++;
793  else {
794  presence_budget[fi][si] = 0; /* Nothing more to do here; tell ourselves. */
795  if (lal[fi] == NULL)
796  lal_bases[fi] -= sys_to_first_vertex[1+si] - sys_to_first_vertex[si];
797  }
798  safelanes_updateConductivity( ei_best );
799  lane_faction[ ei_best ] = faction_stack[fi].id;
800  }
801  }
802 
803 #if DEBUGGING
804  for (fi=0; fi<array_size(faction_stack); fi++)
805  if ( lal[fi] != NULL)
806  assert( "Correctly tracked row offsets between the 'lal' and 'utilde' matrices" && lal[fi]->nrow == lal_bases[fi] );
807 #endif
808 
809  for (fi=0; fi<array_size(faction_stack); fi++)
810  cholmod_free_dense( &lal[fi], &C );
811  free( lal );
812  free( lal_bases );
813  array_free( edgeind_opts );
814  array_free( facind_vals );
815  array_free( facind_opts );
816 
817  return turns_next_time;
818 }

832 static int safelanes_triangleTooFlat( const Vector2d* m, const Vector2d* n, const Vector2d* p, double lmn )
833 {
834  const double MAX_COSINE = cos(MIN_ANGLE);
835  double lnp = vect_dist( n, p );
836  double lmp = vect_dist( m, p );
837  double dpn = ((n->x-m->x)*(n->x-p->x) + (n->y-m->y)*(n->y-p->y)) / ( lmn * lnp );
838  double dpm = ((m->x-n->x)*(m->x-p->x) + (m->y-n->y)*(m->y-p->y)) / ( lmn * lmp );
839  return (dpn > MAX_COSINE && lnp < lmn) || (dpm > MAX_COSINE && lmp < lmn);
