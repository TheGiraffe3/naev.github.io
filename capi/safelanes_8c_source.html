192 {
193  int i, j;
194  SafeLane *out, *l;
195  const Vertex *v[2];
196 
197  out = array_create( SafeLane );
198 
199  for (i=sys_to_first_edge[system->id]; i<sys_to_first_edge[1+system->id]; i++) {
200  for (j=0; j<2; j++)
201  v[j] = &vertex_stack[edge_stack[i][j]];
202  if (lane_faction[i] > 0 && (faction < 0 || faction == lane_faction[i])) {
203  l = &array_grow( &out );
204  l->faction = lane_faction[i];
205  for (j=0; j<2; j++) {
206  switch (v[j]->type) {
207  case VERTEX_PLANET:

218  }
219  }

277 static double safelanes_changeFromOptimizer (void)
278 {
279  cholmod_sparse *stiff_s;
280  cholmod_factor *stiff_f;
281  cholmod_dense *new_utilde, *_QtQutilde, *Lambda_tilde;
282  double rel_change = HUGE_VAL;
283  double zero[] = {0, 0}, neg_1[] = {-1, 0};
284 
285  stiff_s = cholmod_triplet_to_sparse( stiff, 0, &C );
286  stiff_f = cholmod_analyze( stiff_s, &C );
287  cholmod_factorize( stiff_s, stiff_f, &C );
288  new_utilde = cholmod_solve( CHOLMOD_A, stiff_f, ftilde, &C );
289  if (utilde != NULL) {
290  /* Do utilde -= new_utilde, and compute ||utilde||/||utilde_new|| */
291  for (size_t i = 0; i < utilde->nzmax; i++)
292  ((double*)utilde->x)[i] -= ((double*)new_utilde->x)[i];
293  rel_change = frobenius_norm( utilde ) / frobenius_norm( new_utilde );
294  cholmod_free_dense( &utilde, &C );
295  }
296  utilde = new_utilde;
297 
298  if (rel_change > CONVERGENCE_THRESHOLD) {
299  _QtQutilde = cholmod_zeros( utilde->nrow, utilde->ncol, CHOLMOD_REAL, &C );
300  cholmod_sdmult( QtQ, 0, neg_1, zero, utilde, _QtQutilde, &C );
301  Lambda_tilde = cholmod_solve( CHOLMOD_A, stiff_f, _QtQutilde, &C );
302  cholmod_free_dense( &_QtQutilde, &C );
303  safelanes_activateByGradient( Lambda_tilde );
304  cholmod_free_dense( &Lambda_tilde, &C );
305  }
306  cholmod_free_factor( &stiff_f, &C );
307  cholmod_free_sparse( &stiff_s, &C );
308  return rel_change;
309 }

376 static void safelanes_initStacks_edge (void)
377 {
378  const StarSystem *systems_stack;
379  int system, i, j, k;
380 
381  systems_stack = system_getAll();
382 
383  edge_stack = array_create( Edge );
384  sys_to_first_edge = array_create( int );
385  array_push_back( &sys_to_first_edge, 0 );
386  lane_fmask = array_create( FactionMask );
387  tmp_edge_conduct = array_create( double );
388  for (system=0; system<array_size(systems_stack); system++) {
389  for (i = sys_to_first_vertex[system]; i < sys_to_first_vertex[1+system]; i++) {
390  const Vector2d *pi = vertex_pos( i );
391  for (j = sys_to_first_vertex[system]; j < i; j++) {
392  const Vector2d *pj = vertex_pos( j );
393  double lij = vect_dist( pi, pj );
394  int has_approx_midpoint = 0;
395  for (k = sys_to_first_vertex[system]; k < sys_to_first_vertex[1+system]; k++)
396  if (k!=i && k!=j && safelanes_triangleTooFlat( pi, pj, vertex_pos( k ), lij )) {
397  has_approx_midpoint = 1;
398  break;
399  }
400  if (has_approx_midpoint)
401  continue; /* The edge from i to j is disallowed in favor of a path through k. */
402  array_push_back_edge( &edge_stack, j, i );
403  array_push_back( &lane_fmask, MASK_COMPROMISE( vertex_faction( i ), vertex_faction( j ) ) );
404  array_push_back( &tmp_edge_conduct, 1/lij );
405  }
406  }
407  array_push_back( &sys_to_first_edge, array_size(edge_stack) );
408  }
409  array_shrink( &edge_stack );
410  array_shrink( &sys_to_first_edge );
411 
412  lane_faction = array_create_size( int, array_size(edge_stack) );
413  memset( lane_faction, 0, array_size(lane_faction)*sizeof(lane_faction[0]) );
414 }

520 static void safelanes_initStiff (void)
521 {
522  int nnz, v, i;
523  int *pi, *pj;
524  double *pv;
525  double max_conductivity;
526 
527  cholmod_free_triplet( &stiff, &C );
528  v = array_size(vertex_stack);
529  nnz = 3*(array_size(edge_stack)+array_size(tmp_jump_edges)) + array_size(tmp_anchor_vertices);
530  stiff = cholmod_allocate_triplet( v, v, nnz, STORAGE_MODE_UPPER_TRIANGULAR_PART, CHOLMOD_REAL, &C );
531  /* Populate triplets: internal edges (ii ij jj), implicit jump connections (ditto), anchor conditions. */
532  pi = stiff->i; pj = stiff->j; pv = stiff->x;
533  for (i=0; i<array_size(edge_stack); i++) {
534  *pi++ = edge_stack[i][0]; *pj++ = edge_stack[i][0]; *pv++ = +tmp_edge_conduct[i];
535  *pi++ = edge_stack[i][0]; *pj++ = edge_stack[i][1]; *pv++ = -tmp_edge_conduct[i];
536  *pi++ = edge_stack[i][1]; *pj++ = edge_stack[i][1]; *pv++ = +tmp_edge_conduct[i];
537  }
538  for (i=0; i<array_size(tmp_jump_edges); i++) {
539  *pi++ = tmp_jump_edges[i][0]; *pj++ = tmp_jump_edges[i][0]; *pv++ = +JUMP_CONDUCTIVITY;
540  *pi++ = tmp_jump_edges[i][0]; *pj++ = tmp_jump_edges[i][1]; *pv++ = -JUMP_CONDUCTIVITY;
541  *pi++ = tmp_jump_edges[i][1]; *pj++ = tmp_jump_edges[i][1]; *pv++ = +JUMP_CONDUCTIVITY;
542  }
543  /* Add a Robin boundary condition, using the max conductivity (after activation) for spectral reasons. */
544  max_conductivity = JUMP_CONDUCTIVITY/(ALPHA+1);
545  for (i=0; i<array_size(edge_stack); i++)
546  max_conductivity = MAX( max_conductivity, tmp_edge_conduct[i] );
547  max_conductivity = MAX( JUMP_CONDUCTIVITY, (1+ALPHA)*max_conductivity ); /* Activation scales entries by 1+ALPHA later. */
548  for (i=0; i<array_size(tmp_anchor_vertices); i++) {
549  *pi++ = tmp_anchor_vertices[i]; *pj++ = tmp_anchor_vertices[i]; *pv++ = max_conductivity;
550  }
551  stiff->nnz = nnz;
552 #if DEBUGGING
553  assert( cholmod_check_triplet( stiff, &C) );
554 #endif
555 }

699 static void safelanes_activateByGradient( cholmod_dense* Lambda_tilde )
700 {
701  int ei, eii, ei_best, fi, fii, *facind_opts, *edgeind_opts, si, sis, sjs;
702  double *facind_vals, score, score_best, Linv;
703  StarSystem *sys;
704  MatWrap *lal;
705  int *lal_bases, lal_base, sys_base;
706  MatWrap UTt = {.order = CblasRowMajor, .nrow = utilde->ncol, .ncol = utilde->nrow, .x = utilde->x};
707 
708  lal = calloc( array_size(faction_stack), sizeof(MatWrap) );
709  lal_bases = calloc( array_size(faction_stack), sizeof(int) );
710  edgeind_opts = array_create( int );
711  facind_opts = array_create_size( int, array_size(faction_stack) );
712  facind_vals = array_create_size( double, array_size(faction_stack) );
713  for (fi=0; fi<array_size(faction_stack); fi++) {
714  array_push_back( &facind_opts, fi );
715  array_push_back( &facind_vals, 0 );
716  }
717 
718  /* HACK: Because Lambda_tilde will never be used again, and we need to row-slice it, conver to row-major now. */
719  cblas_dimatcopy( CblasColMajor, CblasTrans, Lambda_tilde->nrow, Lambda_tilde->ncol, 1, Lambda_tilde->x,
720  Lambda_tilde->nrow, Lambda_tilde->ncol );
721 
722  for (si=0; si<array_size(sys_to_first_vertex)-1; si++)
723  {
724  /* Factions with most presence here choose first. */
725  sys = system_getIndex( si );
726  for (fi=0; fi<array_size(faction_stack); fi++)
727  facind_vals[fi] = -system_getPresence( sys, faction_stack[fi].id ); /* FIXME: Is this better, or presence_budget? */
728  cmp_key_ref = facind_vals;
729  qsort( facind_opts, array_size(faction_stack), sizeof(int), cmp_key );
730 
731  for (fii=0; fii<array_size(faction_stack); fii++) {
732  fi = facind_opts[fii];
733  sys_base = sys_to_first_vertex[si];
734 
735  /* Get the base index to use for this system. Save the value we expect to be the next iteration's base index.
736  * The current system's rows is in the lal[fi] matrix if there's presence at the time we slice it.
737  * What we know is whether there's presence *now*. We can use that as a proxy and fix lal_bases[fi] if we
738  * deplete this presence before constructing lal[fi]. This is tricky, so there are assertions below,
739  * which can warn us if we fuck this up. */
740  lal_base = lal_bases[fi];
741  if (presence_budget[fi][si] <= 0)
742  continue;
743  /* We "should" find these DoF's interesting if/when we slice, and will unless we deplete this presence first. */
744  lal_bases[fi] += sys_to_first_vertex[1+si] - sys_to_first_vertex[si];
745 
746  array_resize( &edgeind_opts, 0 );
747  for (ei=sys_to_first_edge[si]; ei<sys_to_first_edge[1+si]; ei++)
748  if (!lane_faction[ei]
749  && presence_budget[fi][si] >= 1 / safelanes_initialConductivity(ei) / faction_stack[fi].lane_length_per_presence
750  && (lane_fmask[ei] & (1<<fi)))
751  array_push_back( &edgeind_opts, ei );
752 
753  if (array_size(edgeind_opts) == 0) {
754  presence_budget[fi][si] = -1; /* Nothing to build here! Tell ourselves to stop trying. */
755  if (lal[fi].x == NULL)
756  lal_bases[fi] -= sys_to_first_vertex[1+si] - sys_to_first_vertex[si];
757  continue;
758  }
759 
760  ei_best = edgeind_opts[0];
761  if (array_size(edgeind_opts) > 1) {
762  /* There's an actual choice. Search for the best option. */
763  score_best = -HUGE_VAL;
764  for (eii=0; eii<array_size(edgeind_opts); eii++) {
765  ei = edgeind_opts[eii];
766  sis = edge_stack[ei][0];
767  sjs = edge_stack[ei][1];
768 
769  if (lal[fi].x == NULL) { /* Is it time to evaluate the lazily-calculated matrix? */
770  MatWrap lamt;
771  MatWrap PP = {.order = CblasColMajor, .nrow = PPl[fi]->ncol, .ncol = PPl[fi]->nrow, .x = PPl[fi]->x};
772  matwrap_sliceByPresence( Lambda_tilde, presence_budget[fi], &lamt );
773  matwrap_mul( lamt, PP, &lal[fi] );
774  matwrap_free( lamt );
775  }
776 
777  score = 0;
778  /* Evaluate (LUTll[0,0] + LUTll[1,1] - LUTll[0,1] - LUTll[1,0]), */
779  /* where LUTll = np.dot( lal[[sis,sjs],:] , utilde[[sis,sjs],:].T ) */
780  score += matwrap_mul_elem( lal[fi], UTt, sis - sys_base + lal_base, sis );
781  score += matwrap_mul_elem( lal[fi], UTt, sjs - sys_base + lal_base, sjs );
782  score -= matwrap_mul_elem( lal[fi], UTt, sis - sys_base + lal_base, sjs );
783  score -= matwrap_mul_elem( lal[fi], UTt, sjs - sys_base + lal_base, sis );
784  Linv = safelanes_initialConductivity(ei);
785  score *= ALPHA * Linv * Linv;
786 
787  if (score > score_best) {
788  ei_best = ei;
789  score_best = score;
790  }
791  }
792  }
793 
794  presence_budget[fi][si] -= 1 / safelanes_initialConductivity(ei_best) / faction_stack[fi].lane_length_per_presence;
795  if (presence_budget[fi][si] <= 0 && lal[fi].x == NULL)
796  lal_bases[fi] -= sys_to_first_vertex[1+si] - sys_to_first_vertex[si];
797  safelanes_updateConductivity( ei_best );
798  lane_faction[ ei_best ] = faction_stack[fi].id;
799  }
800  }
801 
802 #if DEBUGGING
803  for (fi=0; fi<array_size(faction_stack); fi++)
804  if ( lal[fi].x != NULL)
805  assert( "We correctly tracked the row offsets between the 'lal' and 'utilde' matrices" && lal[fi].nrow == lal_bases[fi] );
806 #endif
807 
808  for (fi=0; fi<array_size(faction_stack); fi++)
809  matwrap_free( lal[fi] );
810  free( lal );
811  free( lal_bases );
812  array_free( edgeind_opts );
813  array_free( facind_vals );
814  array_free( facind_opts );
815 }

829 static int safelanes_triangleTooFlat( const Vector2d* m, const Vector2d* n, const Vector2d* p, double lmn )
830 {
831  const double MAX_COSINE = cos(MIN_ANGLE);
832  double lnp = vect_dist( n, p );
833  double lmp = vect_dist( m, p );
834  double dpn = ((n->x-m->x)*(n->x-p->x) + (n->y-m->y)*(n->y-p->y)) / ( lmn * lnp );
835  double dpm = ((m->x-n->x)*(m->x-p->x) + (m->y-n->y)*(m->y-p->y)) / ( lmn * lmp );
836  return (dpn > MAX_COSINE && lnp < lmn) || (dpm > MAX_COSINE && lmp < lmn);
837 }
