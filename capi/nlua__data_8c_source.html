262 {

274

325 static int dataL_convolve2d( lua_State *L )
326 {
327  LuaData_t *lI = luaL_checkdata(L,1);
328  long iw = luaL_checklong(L,2);
329  long ih = luaL_checklong(L,3);
330  LuaData_t *lK = luaL_checkdata(L,4);
331  long kw = luaL_checklong(L,5);
332  long kh = luaL_checklong(L,6);
333  LuaData_t out;
334  int p, u,v, ku,kv, bu,bv;
335  int kw2,kh2, bw,bh, ow,oh;
336  float *I = (float*)lI->data;
337  float *K = (float*)lK->data;
338  float *B, *O;
339 
340  /* Checks. */
341  if (iw*ih*4*lI->elem != lI->size)
342  NLUA_ERROR(L,_("size mismatch for data: got %dx%dx4x%d, expected %d"), iw, ih, lI->elem, lI->size);
343  if (kw*kh*4*lK->elem != lK->size)
344  NLUA_ERROR(L,_("size mismatch for data: got %dx%dx4x%d, expected %d"), kw, kh, lK->elem, lK->size);
345  if (lI->type != LUADATA_NUMBER || lK->type != LUADATA_NUMBER)
346  NLUA_ERROR(L, _("%s is only implemented for number types"), __func__);
347 
348  /* Set up. */
349  kw2 = (kw-1)/2;
350  kh2 = (kh-1)/2;
351 
352  /* Create new data. */
353  ow = iw+kw2;
354  oh = ih+kw2;
355  out.elem = lI->elem;
356  out.type = lI->type;
357  out.size = ow*oh*4*out.elem;
358  out.data = calloc( out.size, 1 );
359  O = (float*)out.data;
360 
361 #define POS(U,V,W) (4*((V)*(W)+(U)))
362  /* Create buffer. */
363  bw = ow+2*kw2;
364  bh = oh+2*kh2;
365  B = calloc( bw*bh*4, sizeof(float) );
366  for (v=0; v<ih; v++)
367  memcpy( &B[ POS(kw2, v+kh2, bw) ],
368  &I[ POS( 0, v, iw) ],
369  4*sizeof(float)*iw );
370 
371  /* Convolve. */
372  for (v=0; v<oh; v++) {
373  for (u=0; u<ow; u++) {
374  for (kv=0; kv<kh; kv++) {
375  for (ku=0; ku<kw; ku++) {
376  bu = u + ku;
377  bv = v + kv;
378  for (p=0; p<4; p++)
379  O[ POS( u, v, ow )+p ] +=
380  B[ POS( bu, bv, bw )+p ]
381  * K[ POS( ku, kv, kw )+p ];
382  }
383  }
384  }
385  }
386 #undef POS
387 
388  /* Cleanup. */
389  free(B);
390 
391  /* Return new data. */
392  lua_pushdata(L,out);
393  lua_pushinteger(L,ow);
394  lua_pushinteger(L,oh);
395  return 3;
