281 {
282  LuaData_t *dest = luaL_checkdata(L,1);
283  LuaData_t *source = luaL_checkdata(L,2);
284  long dx = luaL_checklong(L,3) * dest->elem;

287 
288  /* Check fits. */

291  else if (sx+sw > (long)source->size)
292  NLUA_ERROR(L, _("size mismatch: out of bound access of source: %d of %d elements"), sx+sw, source->size);
293 
294  /* Copy memory over. */
295  memcpy( &dest->data[dx], &source->data[sx], sw );
296 
297  /* Return destination. */
298  lua_pushvalue(L,1);

346 static int dataL_convolve2d( lua_State *L )
347 {
348  LuaData_t *lI = luaL_checkdata(L,1);
349  long iw = luaL_checklong(L,2);
350  long ih = luaL_checklong(L,3);
351  LuaData_t *lK = luaL_checkdata(L,4);
352  long kw = luaL_checklong(L,5);
353  long kh = luaL_checklong(L,6);
354  LuaData_t out;
355  int p, u,v, ku,kv, bu,bv;
356  int kw2,kh2, bw,bh, ow,oh;
357  float *I = (float*)lI->data;
358  float *K = (float*)lK->data;
359  float *B, *O;
360 
361  /* Checks. */
362  if (iw*ih*4*lI->elem != lI->size)
363  NLUA_ERROR(L,_("size mismatch for data: got %dx%dx4x%d, expected %d"), iw, ih, lI->elem, lI->size);
364  if (kw*kh*4*lK->elem != lK->size)
365  NLUA_ERROR(L,_("size mismatch for data: got %dx%dx4x%d, expected %d"), kw, kh, lK->elem, lK->size);
366  if (lI->type != LUADATA_NUMBER || lK->type != LUADATA_NUMBER)
367  NLUA_ERROR(L, _("%s is only implemented for number types"), __func__);
368 
369  /* Set up. */
370  kw2 = (kw-1)/2;
371  kh2 = (kh-1)/2;
372 
373  /* Create new data. */
374  ow = iw+kw2;
375  oh = ih+kw2;
376  out.elem = lI->elem;
377  out.type = lI->type;
378  out.size = ow*oh*4*out.elem;
379  out.data = calloc( out.size, 1 );
380  O = (float*)out.data;
381 
382 #define POS(U,V,W) (4*((V)*(W)+(U)))
383  /* Create buffer. */
384  bw = ow+2*kw2;
385  bh = oh+2*kh2;
386  B = calloc( bw*bh*4, sizeof(float) );
387  for (v=0; v<ih; v++)
388  memcpy( &B[ POS(kw2, v+kh2, bw) ],
389  &I[ POS( 0, v, iw) ],
390  4*sizeof(float)*iw );
391 
392  /* Convolve. */
393  for (v=0; v<oh; v++) {
394  for (u=0; u<ow; u++) {
395  for (kv=0; kv<kh; kv++) {
396  for (ku=0; ku<kw; ku++) {
397  bu = u + ku;
398  bv = v + kv;
399  for (p=0; p<4; p++)
400  O[ POS( u, v, ow )+p ] +=
401  B[ POS( bu, bv, bw )+p ]
402  * K[ POS( ku, kv, kw )+p ];
403  }
404  }
405  }
406  }
407 #undef POS
408 
409  /* Cleanup. */
410  free(B);
411 
412  /* Return new data. */
413  lua_pushdata(L,out);
414  lua_pushinteger(L,ow);
415  lua_pushinteger(L,oh);
416  return 3;
417 }
