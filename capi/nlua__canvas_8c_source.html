175 static int canvasL_new( lua_State *L )
176 {
177  LuaCanvas_t lc;
178  int w, h;
179  GLenum status;
180  char *name;
181 
182  w = luaL_checkint(L,1);
183  h = luaL_checkint(L,2);
184 
185  memset( &lc, 0, sizeof(LuaCanvas_t) );
186 
187  /* Create the texture. */
188  asprintf( &name, "nlua_canvas_%03d", ++nlua_canvas_counter );
189  lc.tex = gl_loadImageData( NULL, w, h, 1, 1, name );
190  free( name );
191 
192  /* Create the frame buffer. */
193  glGenFramebuffers( 1, &lc.fbo );
194  glBindFramebuffer(GL_FRAMEBUFFER, lc.fbo);
195 
196  /* Attach the colour buffer. */
197  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, lc.tex->texture, 0);
198 
199  /* Check status. */
200  status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
201  if (status != GL_FRAMEBUFFER_COMPLETE)
202  NLUA_ERROR( L, _("Error setting up framebuffer!"));
203 
204  /* Clear the canvas. */
205  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
206 
207  /* Restore state. */
208  glBindFramebuffer(GL_FRAMEBUFFER, gl_screen.current_fbo);
209 
210  gl_checkErr();
211 
212  lua_pushcanvas( L, lc );
213  return 1;
214 }
