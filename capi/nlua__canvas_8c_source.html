159 {
160  GLenum status;
161  char *name;
162 
163  /* Create the texture. */
164  asprintf( &name, "nlua_canvas_%03d", ++nlua_canvas_counter );
165  lc->tex = gl_loadImageData( NULL, w, h, 1, 1, name );
166  free( name );
167 
168  /* Create the frame buffer. */
169  glGenFramebuffers( 1, &lc->fbo );
170  glBindFramebuffer(GL_FRAMEBUFFER, lc->fbo);
171 
172  /* Attach the colour buffer. */
173  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, lc->tex->texture, 0);
174 
175  /* Check status. */
176  status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
177  if (status != GL_FRAMEBUFFER_COMPLETE)
178  return -1;
179 
180  /* Clear the canvas. */
181  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
182 
183  /* Restore state. */
184  glBindFramebuffer(GL_FRAMEBUFFER, gl_screen.current_fbo);
185 
186  gl_checkErr();
187  return 0;
188 }
