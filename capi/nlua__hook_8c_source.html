150 {

153  /* ... Or do a no-op if caller passes nil. */
154  if (h < 0)

178 {

180 
181  /* If a table set __save, this won't work for tables of tables however. */
182  if (lua_istable(naevL, ind)) {

259 static unsigned int hookL_generic( lua_State *L, const char* stack, double sec, int pos, ntime_t date )
260 {
261  const char *func;
262  unsigned int h;
263  Event_t *running_event;
264  Mission *running_mission;
265 
266  /* Last parameter must be function to hook */
267  func = luaL_checkstring(L,pos);
268

270  running_event = event_getFromLua(L);
271  running_mission = misn_getFromLua(L);
272 
273  if (running_mission != NULL) {
274  int i;
275  /* make sure mission is a player mission */
276  for (i=0; i<MISSION_MAX; i++)
277  if (player_missions[i]->id == running_mission->id)
278  break;
279  if (i>=MISSION_MAX) {
280  WARN(_("Mission not in stack trying to hook, forgot to run misn.accept()?"));
281  return 0;
282  }
283 
284  if (stack != NULL)
285  h = hook_addMisn( running_mission->id, func, stack );
286  else if (date != 0)
287  h = hook_addDateMisn( running_mission->id, func, date );
288  else
289  h = hook_addTimerMisn( running_mission->id, func, sec );
290  }
291  else if (running_event != NULL) {
292  if (stack != NULL)
293  h = hook_addEvent( running_event->id, func, stack );
294  else if (date != 0)
295  h = hook_addDateEvt( running_event->id, func, date );
296  else
297  h = hook_addTimerEvt( running_event->id, func, sec );
298  }
299  else {
300  NLUA_ERROR(L,_("Attempting to set a hook outside of a mission or event."));
301  return 0;
302  }
303 
304  if (h == 0) {
305  NLUA_ERROR(L,_("No hook target was set."));
306  return 0;
307  }
308 
309  /* Check parameter. */
310  if (!lua_isnoneornil(L,pos+1))
311  hookL_setarg( h, pos+1 );
312 
313  return h;

886 static int hookL_trigger( lua_State *L )
887 {
888  int i, n;
889  HookParam hp[HOOK_MAX_PARAM];
890  const char *hookname = luaL_checkstring(L,1);
891  n = lua_gettop(L);
892 
893  /* Set up hooks. */
894  for (i=0; i< MIN(n, HOOK_MAX_PARAM-1); i++) {
895  HookParam *p = &hp[i];
896  switch (lua_type(L,i+1)) {
897  case LUA_TNIL:
898  p->type = HOOK_PARAM_NIL;
899  break;
900  case LUA_TNUMBER:
901  p->type = HOOK_PARAM_NUMBER;
902  p->u.num = lua_tonumber(L,i+1);
903  break;
904  case LUA_TBOOLEAN:
905  p->type = HOOK_PARAM_BOOL;
906  p->u.b = lua_toboolean(L,i+1);
907  break;
908  case LUA_TSTRING:
909  p->type = HOOK_PARAM_STRING;
910  p->u.str = lua_tostring(L,i+1);
911  break;
912  case LUA_TUSERDATA:
913  if (lua_ispilot(L,i+1)) {
914  p->type = HOOK_PARAM_PILOT;
915  p->u.lp = lua_topilot(L,i+1);
916  }
917  else if (lua_isfaction(L,i+1)) {
918  p->type = HOOK_PARAM_FACTION;
919  p->u.lf = lua_tofaction(L,i+1);
920  }
921  else if (lua_isplanet(L,i+1)) {
922  p->type = HOOK_PARAM_ASSET;
923  p->u.la = lua_toplanet(L,i+1);
924  }
925  else if (lua_isjump(L,i+1)) {
926  p->type = HOOK_PARAM_JUMP;
927  p->u.lj = *lua_tojump(L,i+1);
928  }
929  break;
930  default:
931  NLUA_ERROR(L, _("Unsupported Lua hook paramater type '%s'!"), lua_typename(L,i+1));
932  }
933  }
934  hp[i].type = HOOK_PARAM_SENTINEL;
935 
936  /* Run the deferred hooks. */
937  hooks_runParamDeferred( hookname, hp );
938  return 0;
939 }
