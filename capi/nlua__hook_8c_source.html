153 static int hookL_rm( lua_State *L )
154 {
155  long h;
156 
157  /* Remove the hook. */
158  h = luaL_optlong( L, 1, -1 );
159  /* ... Or do a no-op if caller passes nil. */
160  if (h < 0)
161  return 0;
162  hook_rm( (unsigned int) h );
163 
164  /* Clean up hook data. */
165  nlua_getenv(__NLUA_CURENV, "__hook_arg");
166  if (!lua_isnil(L,-1)) {
167  lua_pushnumber( L, h ); /* t, n */
168  lua_pushnil( L ); /* t, n, nil */
169  lua_settable( L, -3 ); /* t */
170  }
171  lua_pop( L, 1 ); /* */
172 
173  return 0;

184 static int hookL_setarg( unsigned int hook, int ind )
185 {
186  nlua_env env = hook_env(hook);
187 
188  /* If a table set __save, this won't work for tables of tables however. */
189  if (lua_istable(naevL, ind)) {
190  lua_pushboolean( naevL, 1 );/* b */
191  lua_setfield( naevL, ind, "__save" ); /* v */
192  }
193 
194  /* Create if necessary the actual hook argument table. */
195  nlua_getenv(env, "__hook_arg"); /* t */
196  if (lua_isnil(naevL,-1)) { /* nil */
197  lua_pop( naevL, 1 ); /* */
198  lua_newtable( naevL ); /* t */
199  lua_pushvalue( naevL, -1 ); /* t, t */
200  nlua_setenv(env, "__hook_arg");/*t */
201  lua_pushboolean( naevL, 1 ); /* t, s */
202  lua_setfield( naevL, -2, "__save" );/* t */

269 static unsigned int hook_generic( lua_State *L, const char* stack, double sec, int pos, ntime_t date )
270 {
271  int i;
272  const char *func;
273  unsigned int h;
274  Event_t *running_event;
275  Mission *running_mission;
276 
277  /* Last parameter must be function to hook */
278  func = luaL_checkstring(L,pos);
279 
280  /* Get stuff. */
281  running_event = event_getFromLua(L);
282  running_mission = misn_getFromLua(L);
283 
284  if (running_mission != NULL) {
285  /* make sure mission is a player mission */
286  for (i=0; i<MISSION_MAX; i++)
287  if (player_missions[i]->id == running_mission->id)
288  break;
289  if (i>=MISSION_MAX) {
290  WARN(_("Mission not in stack trying to hook, forgot to run misn.accept()?"));
291  return 0;
292  }
293 
294  if (stack != NULL)
295  h = hook_addMisn( running_mission->id, func, stack );
296  else if (date != 0)
297  h = hook_addDateMisn( running_mission->id, func, date );
298  else
299  h = hook_addTimerMisn( running_mission->id, func, sec );
300  }
301  else if (running_event != NULL) {
302  if (stack != NULL)
303  h = hook_addEvent( running_event->id, func, stack );
304  else if (date != 0)
305  h = hook_addDateEvt( running_event->id, func, date );
306  else
307  h = hook_addTimerEvt( running_event->id, func, sec );
308  }
309  else {
310  NLUA_ERROR(L,_("Attempting to set a hook outside of a mission or event."));
311  return 0;
312  }
313 
314  if (h == 0) {
315  NLUA_ERROR(L,_("No hook target was set."));
316  return 0;
317  }
318 
319  /* Check parameter. */
320  if (!lua_isnoneornil(L,pos+1))
321  hookL_setarg( h, pos+1 );
322 
323  return h;
324 }

813 static int hook_trigger( lua_State *L )
814 {
815  int i, n;
816  HookParam hp[HOOK_MAX_PARAM], *p;
817  const char *hookname = luaL_checkstring(L,1);
818  n = lua_gettop(L);
819 
820  /* Set up hooks. */
821  for (i=0; i< MIN(n, HOOK_MAX_PARAM-1); i++) {
822  p = &hp[i];
823  switch (lua_type(L,i+1)) {
824  case LUA_TNIL:
825  p->type = HOOK_PARAM_NIL;
826  break;
827  case LUA_TNUMBER:
828  p->type = HOOK_PARAM_NUMBER;
829  p->u.num = lua_tonumber(L,i+1);
830  break;
831  case LUA_TBOOLEAN:
832  p->type = HOOK_PARAM_BOOL;
833  p->u.b = lua_toboolean(L,i+1);
834  break;
835  case LUA_TSTRING:
836  p->type = HOOK_PARAM_STRING;
837  p->u.str = lua_tostring(L,i+1);
838  break;
839  case LUA_TUSERDATA:
840  if (lua_ispilot(L,i+1)) {
841  p->type = HOOK_PARAM_PILOT;
842  p->u.lp = lua_topilot(L,i+1);
843  }
844  else if (lua_isfaction(L,i+1)) {
845  p->type = HOOK_PARAM_FACTION;
846  p->u.lf = lua_tofaction(L,i+1);
847  }
848  else if (lua_isplanet(L,i+1)) {
849  p->type = HOOK_PARAM_ASSET;
850  p->u.la = *lua_toplanet(L,i+1);
851  }
852  else if (lua_isjump(L,i+1)) {
853  p->type = HOOK_PARAM_JUMP;
854  p->u.lj = *lua_tojump(L,i+1);
855  }
856  break;
857  default:
858  NLUA_ERROR(L, _("Unsupported Lua hook paramater type '%s'!"), lua_typename(L,i+1));
859  }
860  }
861  hp[i].type = HOOK_PARAM_SENTINEL;
862 
863  /* Run the deferred hooks. */
864  hooks_runParamDeferred( hookname, hp );
865  return 0;
866 }
