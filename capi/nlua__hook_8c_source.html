149 static int hookL_rm( lua_State *L )
150 {
151  long h;
152 
153  /* Remove the hook. */
154  h = luaL_optlong( L, 1, -1 );
155  /* ... Or do a no-op if caller passes nil. */
156  if (h < 0)
157  return 0;
158  hook_rm( (unsigned int) h );
159 
160  /* Clean up hook data. */
161  nlua_getenv(__NLUA_CURENV, "__hook_arg");
162  if (!lua_isnil(L,-1)) {
163  lua_pushnumber( L, h ); /* t, n */
164  lua_pushnil( L ); /* t, n, nil */
165  lua_settable( L, -3 ); /* t */
166  }
167  lua_pop( L, 1 ); /* */
168 
169  return 0;
170 }

180 static int hookL_setarg( unsigned int hook, int ind )
181 {
182  nlua_env env = hook_env(hook);
183 
184  lua_pushvalue( naevL, ind ); /* v */
185  /* If a table set __save, this won't work for tables of tables however. */
186  if (lua_istable(naevL, -1)) {
187  lua_pushboolean( naevL, 1 );/* v, b */
188  lua_setfield( naevL, -2, "__save" ); /* v */
189  }
190  /* Create if necessary the actual hook argument table. */
191  nlua_getenv(env, "__hook_arg"); /* v, t */
192  if (lua_isnil(naevL,-1)) { /* v, nil */
193  lua_pop( naevL, 1 ); /* v */
194  lua_newtable( naevL ); /* v, t */
195  lua_pushvalue( naevL, -1 ); /* v, t, t */
196  nlua_setenv(env, "__hook_arg"); /* v, t */
197  lua_pushboolean( naevL, 1 ); /* v, t, s */
198  lua_setfield( naevL, -2, "__save" ); /* v, t */
199  }
200  lua_pushnumber( naevL, hook ); /* v, t, k */
201  lua_pushvalue( naevL, -3 ); /* v, t, k, v */
202  lua_settable( naevL, -3 ); /* v, t */
203  lua_pop( naevL, 2 ); /* */
204  return 0;

265 static unsigned int hook_generic( lua_State *L, const char* stack, double ms, int pos, ntime_t date )
266 {
267  int i;
268  const char *func;
269  unsigned int h;
270  Event_t *running_event;
271  Mission *running_mission;
272 
273  /* Last parameter must be function to hook */
274  func = luaL_checkstring(L,pos);
275 
276  /* Get stuff. */
277  running_event = event_getFromLua(L);
278  running_mission = misn_getFromLua(L);
279 
280  if (running_mission != NULL) {
281  /* make sure mission is a player mission */
282  for (i=0; i<MISSION_MAX; i++)
283  if (player_missions[i]->id == running_mission->id)
284  break;
285  if (i>=MISSION_MAX) {
286  WARN(_("Mission not in stack trying to hook, forgot to run misn.accept()?"));
287  return 0;
288  }
289 
290  if (stack != NULL)
291  h = hook_addMisn( running_mission->id, func, stack );
292  else if (date != 0)
293  h = hook_addDateMisn( running_mission->id, func, date );
294  else
295  h = hook_addTimerMisn( running_mission->id, func, ms );
296  }
297  else if (running_event != NULL) {
298  if (stack != NULL)
299  h = hook_addEvent( running_event->id, func, stack );
300  else if (date != 0)
301  h = hook_addDateEvt( running_event->id, func, date );
302  else
303  h = hook_addTimerEvt( running_event->id, func, ms );
304  }
305  else {
306  NLUA_ERROR(L,_("Attempting to set a hook outside of a mission or event."));
307  return 0;
308  }
309 
310  if (h == 0) {
311  NLUA_ERROR(L,_("No hook target was set."));
312  return 0;
313  }
314 
315  /* Check parameter. */
316  if (!lua_isnil(L,pos+1))
317  hookL_setarg( h, pos+1 );
318 
319  return h;
320 }

866 static int hook_pilot( lua_State *L )
867 {
868  unsigned int h;
869  LuaPilot p;
870  int type;
871  const char *hook_type;
872  char buf[ PATH_MAX ];
873 
874  /* Parameters. */
875  if (lua_ispilot(L,1))
876  p = luaL_checkpilot(L,1);
877  else if (lua_isnil(L,1))
878  p = 0;
879  else {
880  NLUA_ERROR(L, _("Invalid parameter #1 for hook.pilot, expecting pilot or nil."));
881  return 0;
882  }
883  hook_type = luaL_checkstring(L,2);
884 
885  /* Check to see if hook_type is valid */
886  if (strcmp(hook_type,"death")==0) type = PILOT_HOOK_DEATH;
887  else if (strcmp(hook_type,"exploded")==0) type = PILOT_HOOK_EXPLODED;
888  else if (strcmp(hook_type,"boarding")==0) type = PILOT_HOOK_BOARDING;
889  else if (strcmp(hook_type,"board")==0) type = PILOT_HOOK_BOARD;
890  else if (strcmp(hook_type,"disable")==0) type = PILOT_HOOK_DISABLE;
891  else if (strcmp(hook_type,"undisable")==0) type = PILOT_HOOK_UNDISABLE;
892  else if (strcmp(hook_type,"jump")==0) type = PILOT_HOOK_JUMP;
893  else if (strcmp(hook_type,"hail")==0) type = PILOT_HOOK_HAIL;
894  else if (strcmp(hook_type,"land")==0) type = PILOT_HOOK_LAND;
895  else if (strcmp(hook_type,"attacked")==0) type = PILOT_HOOK_ATTACKED;
896  else if (strcmp(hook_type,"idle")==0) type = PILOT_HOOK_IDLE;
897  else if (strcmp(hook_type,"lockon")==0) type = PILOT_HOOK_LOCKON;
898  else { /* hook_type not valid */
899  NLUA_ERROR(L, _("Invalid pilot hook type: '%s'"), hook_type);
900  return 0;
901  }
902 
903  /* actually add the hook */
904  snprintf( buf, sizeof(buf), "p_%s", hook_type );
905  h = hook_generic( L, buf, 0., 3, 0 );
906  if (p==0)
907  pilots_addGlobalHook( type, h );
908  else
909  pilot_addHook( pilot_get(p), type, h );
910 
911  lua_pushnumber( L, h );
912  return 1;
913 }
