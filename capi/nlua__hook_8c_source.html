153 static int hookL_rm( lua_State *L )
154 {
155  long h;
156 
157  /* Remove the hook. */
158  h = luaL_optlong( L, 1, -1 );
159  /* ... Or do a no-op if caller passes nil. */
160  if (h < 0)
161  return 0;
162  hook_rm( (unsigned int) h );
163 
164  /* Clean up hook data. */
165  nlua_getenv(__NLUA_CURENV, "__hook_arg");

168  lua_pushnil( L ); /* t, n, nil */

170  }

184 static int hookL_setarg( unsigned int hook, int ind )
185 {
186  nlua_env env = hook_env(hook);
187 
188  lua_pushvalue( naevL, ind ); /* v */
189  /* If a table set __save, this won't work for tables of tables however. */
190  if (lua_istable(naevL, -1)) {
191  lua_pushboolean( naevL, 1 );/* v, b */
192  lua_setfield( naevL, -2, "__save" ); /* v */
193  }
194  /* Create if necessary the actual hook argument table. */
195  nlua_getenv(env, "__hook_arg"); /* v, t */
196  if (lua_isnil(naevL,-1)) { /* v, nil */
197  lua_pop( naevL, 1 ); /* v */
198  lua_newtable( naevL ); /* v, t */
199  lua_pushvalue( naevL, -1 ); /* v, t, t */
200  nlua_setenv(env, "__hook_arg"); /* v, t */
201  lua_pushboolean( naevL, 1 ); /* v, t, s */
202  lua_setfield( naevL, -2, "__save" ); /* v, t */
203  }

205  lua_pushvalue( naevL, -3 ); /* v, t, k, v */
206  lua_settable( naevL, -3 ); /* v, t */
207  lua_pop( naevL, 2 ); /* */

269 static unsigned int hook_generic( lua_State *L, const char* stack, double ms, int pos, ntime_t date )
270 {
271  int i;
272  const char *func;
273  unsigned int h;
274  Event_t *running_event;
275  Mission *running_mission;
276 
277  /* Last parameter must be function to hook */
278  func = luaL_checkstring(L,pos);
279 
280  /* Get stuff. */
281  running_event = event_getFromLua(L);
282  running_mission = misn_getFromLua(L);
283 
284  if (running_mission != NULL) {
285  /* make sure mission is a player mission */
286  for (i=0; i<MISSION_MAX; i++)
287  if (player_missions[i]->id == running_mission->id)
288  break;
289  if (i>=MISSION_MAX) {
290  WARN(_("Mission not in stack trying to hook, forgot to run misn.accept()?"));
291  return 0;
292  }
293 
294  if (stack != NULL)
295  h = hook_addMisn( running_mission->id, func, stack );
296  else if (date != 0)
297  h = hook_addDateMisn( running_mission->id, func, date );
298  else
299  h = hook_addTimerMisn( running_mission->id, func, ms );
300  }
301  else if (running_event != NULL) {
302  if (stack != NULL)
303  h = hook_addEvent( running_event->id, func, stack );
304  else if (date != 0)
305  h = hook_addDateEvt( running_event->id, func, date );
306  else
307  h = hook_addTimerEvt( running_event->id, func, ms );
308  }
309  else {
310  NLUA_ERROR(L,_("Attempting to set a hook outside of a mission or event."));
311  return 0;
312  }
313 
314  if (h == 0) {
315  NLUA_ERROR(L,_("No hook target was set."));
316  return 0;
317  }
318 
319  /* Check parameter. */
320  if (!lua_isnil(L,pos+1))
321  hookL_setarg( h, pos+1 );
322 
323  return h;
324 }

841 static int hook_trigger( lua_State *L )
842 {
843  int i, n;
844  HookParam hp[HOOK_MAX_PARAM], *p;
845  const char *hookname = luaL_checkstring(L,1);
846  n = lua_gettop(L);
847 
848  /* Set up hooks. */
849  for (i=0; i< MIN(n, HOOK_MAX_PARAM-1); i++) {
850  p = &hp[i];
851  switch (lua_type(L,i+1)) {
852  case LUA_TNIL:
853  p->type = HOOK_PARAM_NIL;
854  break;
855  case LUA_TNUMBER:
856  p->type = HOOK_PARAM_NUMBER;
857  p->u.num = lua_tonumber(L,i+1);
858  break;
859  case LUA_TBOOLEAN:
860  p->type = HOOK_PARAM_BOOL;
861  p->u.b = lua_toboolean(L,i+1);
862  break;
863  case LUA_TSTRING:
864  p->type = HOOK_PARAM_STRING;
865  p->u.str = lua_tostring(L,i+1);
866  break;
867  case LUA_TUSERDATA:
868  if (lua_ispilot(L,i+1)) {
869  p->type = HOOK_PARAM_PILOT;
870  p->u.lp = lua_topilot(L,i+1);
871  }
872  else if (lua_isfaction(L,i+1)) {
873  p->type = HOOK_PARAM_FACTION;
874  p->u.lf = lua_tofaction(L,i+1);
875  }
876  else if (lua_isplanet(L,i+1)) {
877  p->type = HOOK_PARAM_ASSET;
878  p->u.la = *lua_toplanet(L,i+1);
879  }
880  else if (lua_isjump(L,i+1)) {
881  p->type = HOOK_PARAM_JUMP;
882  p->u.lj = *lua_tojump(L,i+1);
883  }
884  break;
885  default:
886  NLUA_ERROR(L, _("Unsupported Lua hook paramater type '%s'!"), lua_typename(L,i+1));
887  }
888  }
889  hp[i].type = HOOK_PARAM_SENTINEL;
890 
891  /* Run the deferred hooks. */
892  hooks_runParamDeferred( hookname, hp );
893  return 0;
894 }
