143 static int hookL_rm( lua_State *L )
144 {
145  long h;
146 
147  /* Remove the hook. */
148  h = luaL_optlong( L, 1, -1 );
149  /* ... Or do a no-op if caller passes nil. */
150  if (h < 0)
151  return 0;
152  hook_rm( (unsigned int) h );
153 
154  /* Clean up hook data. */
155  nlua_getenv(__NLUA_CURENV, "__hook_arg");
156  if (!lua_isnil(L,-1)) {
157  lua_pushnumber( L, h ); /* t, n */
158  lua_pushnil( L ); /* t, n, nil */
159  lua_settable( L, -3 ); /* t */
160  }
161  lua_pop( L, 1 ); /* */
162 
163  return 0;
164 }

174 static int hookL_setarg( unsigned int hook, int ind )
175 {
176  nlua_env env = hook_env(hook);
177 
178  lua_pushvalue( naevL, ind ); /* v */
179  /* If a table set __save, this won't work for tables of tables however. */
180  if (lua_istable(naevL, -1)) {
181  lua_pushboolean( naevL, 1 );/* v, b */
182  lua_setfield( naevL, -2, "__save" ); /* v */
183  }
184  /* Create if necessary the actual hook argument table. */
185  nlua_getenv(env, "__hook_arg"); /* v, t */
186  if (lua_isnil(naevL,-1)) { /* v, nil */
187  lua_pop( naevL, 1 ); /* v */
188  lua_newtable( naevL ); /* v, t */
189  lua_pushvalue( naevL, -1 ); /* v, t, t */
190  nlua_setenv(env, "__hook_arg"); /* v, t */
191  lua_pushboolean( naevL, 1 ); /* v, t, s */
192  lua_setfield( naevL, -2, "__save" ); /* v, t */
193  }
194  lua_pushnumber( naevL, hook ); /* v, t, k */
195  lua_pushvalue( naevL, -3 ); /* v, t, k, v */
196  lua_settable( naevL, -3 ); /* v, t */
197  lua_pop( naevL, 2 ); /* */
198  return 0;

259 static unsigned int hook_generic( lua_State *L, const char* stack, double ms, int pos, ntime_t date )
260 {
261  int i;
262  const char *func;
263  unsigned int h;
264  Event_t *running_event;
265  Mission *running_mission;
266 
267  /* Last parameter must be function to hook */
268  func = luaL_checkstring(L,pos);
269 
270  /* Get stuff. */
271  running_event = event_getFromLua(L);
272  running_mission = misn_getFromLua(L);
273 
274  if (running_mission != NULL) {
275  /* make sure mission is a player mission */
276  for (i=0; i<MISSION_MAX; i++)
277  if (player_missions[i]->id == running_mission->id)
278  break;
279  if (i>=MISSION_MAX) {
280  WARN(_("Mission not in stack trying to hook, forgot to run misn.accept()?"));
281  return 0;
282  }
283 
284  if (stack != NULL)
285  h = hook_addMisn( running_mission->id, func, stack );
286  else if (date != 0)
287  h = hook_addDateMisn( running_mission->id, func, date );
288  else
289  h = hook_addTimerMisn( running_mission->id, func, ms );
290  }
291  else if (running_event != NULL) {
292  if (stack != NULL)
293  h = hook_addEvent( running_event->id, func, stack );
294  else if (date != 0)
295  h = hook_addDateEvt( running_event->id, func, date );
296  else
297  h = hook_addTimerEvt( running_event->id, func, ms );
298  }
299  else {
300  NLUA_ERROR(L,_("Attempting to set a hook outside of a mission or event."));
301  return 0;
302  }
303 
304  if (h == 0) {
305  NLUA_ERROR(L,_("No hook target was set."));
306  return 0;
307  }
308 
309  /* Check parameter. */
310  if (!lua_isnil(L,pos+1))
311  hookL_setarg( h, pos+1 );
312 
313  return h;
314 }

810 static int hook_pilot( lua_State *L )
811 {
812  unsigned int h;
813  LuaPilot p;
814  int type;
815  const char *hook_type;
816  char buf[ PATH_MAX ];
817 
818  /* Parameters. */
819  if (lua_ispilot(L,1))
820  p = luaL_checkpilot(L,1);
821  else if (lua_isnil(L,1))
822  p = 0;
823  else {
824  NLUA_ERROR(L, _("Invalid parameter #1 for hook.pilot, expecting pilot or nil."));
825  return 0;
826  }
827  hook_type = luaL_checkstring(L,2);
828 
829  /* Check to see if hook_type is valid */
830  if (strcmp(hook_type,"death")==0) type = PILOT_HOOK_DEATH;
831  else if (strcmp(hook_type,"exploded")==0) type = PILOT_HOOK_EXPLODED;
832  else if (strcmp(hook_type,"boarditf")==0) type = PILOT_HOOK_BOARDING;
833  else if (strcmp(hook_type,"board")==0) type = PILOT_HOOK_BOARD;
834  else if (strcmp(hook_type,"disable")==0) type = PILOT_HOOK_DISABLE;
835  else if (strcmp(hook_type,"undisable")==0) type = PILOT_HOOK_UNDISABLE;
836  else if (strcmp(hook_type,"jump")==0) type = PILOT_HOOK_JUMP;
837  else if (strcmp(hook_type,"hail")==0) type = PILOT_HOOK_HAIL;
838  else if (strcmp(hook_type,"land")==0) type = PILOT_HOOK_LAND;
839  else if (strcmp(hook_type,"attacked")==0) type = PILOT_HOOK_ATTACKED;
840  else if (strcmp(hook_type,"idle")==0) type = PILOT_HOOK_IDLE;
841  else if (strcmp(hook_type,"lockon")==0) type = PILOT_HOOK_LOCKON;
842  else { /* hook_type not valid */
843  NLUA_ERROR(L, _("Invalid pilot hook type: '%s'"), hook_type);
844  return 0;
845  }
846 
847  /* actually add the hook */
848  snprintf( buf, sizeof(buf), "p_%s", hook_type );
849  h = hook_generic( L, buf, 0., 3, 0 );
850  if (p==0)
851  pilots_addGlobalHook( type, h );
852  else
853  pilot_addHook( pilot_get(p), type, h );
854 
855  lua_pushnumber( L, h );
856  return 1;
