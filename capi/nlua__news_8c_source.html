345 int newsL_get( lua_State *L )
346 {
347  LuaArticle Larticle;
348  ntime_t date;
349  news_t *article_ptr;
350  char *characteristic;
351  int k, print_all;
352 
353  date = -1;
354  article_ptr = news_list;
355  characteristic = NULL;
356  print_all = 0;
357 
358  if (lua_isnil(L, 1) || lua_gettop(L) == 0) /* Case no argument */
359  print_all = 1;
360  else if (lua_isnumber(L, 1))
361  date = (ntime_t)lua_tonumber(L, 1);
362  else if (lua_isstring(L, 1))
363  characteristic = strdup(lua_tostring(L, 1));
364  else
365  NLUA_INVALID_PARAMETER(L); /* Bad Parameter */
366 
367  /* Now put all the matching articles in a table. */
368  lua_newtable(L);
369  k = 1;
370  do {
371 
372  if ( (article_ptr->title == NULL) || (article_ptr->desc == NULL)
373  || (article_ptr->faction == NULL) )
374  continue;
375 
376  if ( print_all || date == article_ptr->date
377  || ( characteristic
378  && ( strcmp( article_ptr->title, characteristic ) == 0
379  || strcmp( article_ptr->desc, characteristic ) == 0
380  || strcmp( article_ptr->faction, characteristic ) == 0
381  || ( article_ptr->tag != NULL && strcmp( article_ptr->tag, characteristic ) == 0 ) ) ) ) {
382  Larticle = article_ptr->id;
383  lua_pusharticle(L, Larticle); /* value */
384  lua_rawseti(L, -2, k++);
385  }
386 
387  } while ((article_ptr = article_ptr->next) != NULL);
388 
389  free(characteristic);
390 
391  return 1;
392 }
