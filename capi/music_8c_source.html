75 void music_update( double dt )
76 {
77  char *buf;
78 
79  if (music_disabled)
80  return;
81 
82  if (music_temp_disabled)
83  return;
84 
85  /* Timer stuff. */
86  if (music_timer > 0.) {
87  music_timer -= dt;
88  if (music_timer <= 0.)
89  music_runchoose = 1;
90  }
91 
92  /* Lock music and see if needs to update. */
93  SDL_mutexP(music_lock);
94  if (music_runchoose == 0) {
95  SDL_mutexV(music_lock);
96  return;
97  }
98  music_runchoose = 0;
99  buf = strdup( music_situation );
100  SDL_mutexV(music_lock);
101  music_runLua( buf );
102  free( buf );
103 
104  /* Make sure music is playing. */
105  if (!music_isPlaying())
106  music_choose("idle");
107 }

139 
140  return 0;
141 }

235  suflen = strlen(MUSIC_SUFFIX);
236  for (size_t i=0; files[i]!=NULL; i++) {
237  int flen = strlen(files[i]);
238  if ((flen > suflen) &&
239  strncmp( &files[i][flen - suflen], MUSIC_SUFFIX, suflen)==0) {
240 
241  /* grow the selection size */
242  nmusic++;
243  }
244  }
245 
246  DEBUG( n_("Loaded %d Song", "Loaded %d Songs", nmusic ), nmusic );
247 
248  /* More clean up. */
249  PHYSFS_freeList(files);
250 
251  return 0;
