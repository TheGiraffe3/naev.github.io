84 void music_update( double dt )
85 {
86  char *buf;
87 
88  if (music_disabled)
89  return;
90 
91  if (music_temp_disabled)
92  return;
93 
94  /* Timer stuff. */
95  if (music_timer > 0.) {
96  music_timer -= dt;
97  if (music_timer <= 0.)
98  music_runchoose = 1;
99  }
100 
101  /* Lock music and see if needs to update. */
102  SDL_mutexP(music_lock);
103  if (music_runchoose == 0) {
104  SDL_mutexV(music_lock);
105  return;
106  }
107  music_runchoose = 0;
108  buf = strdup( music_situation );
109  SDL_mutexV(music_lock);
110  music_runLua( buf );
111  free( buf );
112 
113  /* Make sure music is playing. */
114  if (!music_isPlaying())
115  music_choose("idle");
116 }

235 static int music_find (void)
236 {
237  char** files;
238  size_t i;
239  int suflen, flen;
240  int nmusic;
241 
242  if (music_disabled)
243  return 0;

246  files = PHYSFS_enumerateFiles( MUSIC_PATH );
247 
248  /* load the profiles */
249  nmusic = 0;
250  suflen = strlen(MUSIC_SUFFIX);
251  for (i=0; files[i]!=NULL; i++) {
252  flen = strlen(files[i]);
253  if ((flen > suflen) &&
254  strncmp( &files[i][flen - suflen], MUSIC_SUFFIX, suflen)==0) {
255 
256  /* grow the selection size */
257  nmusic++;
258  }
259  }
260 
261  DEBUG( n_("Loaded %d Song", "Loaded %d Songs", nmusic ), nmusic );
262 
263  /* More clean up. */
264  PHYSFS_freeList(files);
265 
266  return 0;
