526 static void gui_borderIntersection( double *cx, double *cy, double rx, double ry, double hw, double hh )
527 {
528  double a;
529  double w, h;
530 
531  /* Get angle. */
532  a = atan2( ry, rx );
533  if (a < 0.)
534  a += 2.*M_PI;
535 
536  /* Helpers. */
537  w = hw-7.;
538  h = hh-7.;
539 
540  /* Handle by quadrant. */
541  if ((a > gui_tr) && (a < gui_tl)) { /* Top. */
542  *cx = h * (rx/ry);
543  *cy = h;
544  }
545  else if ((a > gui_tl) && (a < gui_bl)) { /* Left. */
546  *cx = -w;
547  *cy = -w * (ry/rx);
548  }
549  else if ((a > gui_bl) && (a < gui_br)) { /* Bottom. */
550  *cx = -h * (rx/ry);
551  *cy = -h;
552  }
553  else { /* Right. */
554  *cx = w;
555  *cy = w * (ry/rx);
556  }
557 
558  /* Translate. */
559  *cx += hw;
560  *cy += hh;
561 }
