598 static void gui_borderIntersection( double *cx, double *cy, double rx, double ry, double hw, double hh )
599 {
600  double a;
601  double w, h;
602 
603  /* Get angle. */
604  a = atan2( ry, rx );
605  if (a < 0.)
606  a += 2.*M_PI;
607 
608  /* Helpers. */
609  w = hw-7.;
610  h = hh-7.;
611 
612  /* Handle by quadrant. */
613  if ((a > gui_tr) && (a < gui_tl)) { /* Top. */
614  *cx = h * (rx/ry);
615  *cy = h;
616  }
617  else if ((a > gui_tl) && (a < gui_bl)) { /* Left. */
618  *cx = -w;
619  *cy = -w * (ry/rx);
620  }
621  else if ((a > gui_bl) && (a < gui_br)) { /* Bottom. */
622  *cx = -h * (rx/ry);
623  *cy = -h;
624  }
625  else { /* Right. */
626  *cx = w;
627  *cy = w * (ry/rx);
628  }
629 
630  /* Translate. */
631  *cx += hw;
632  *cy += hh;
