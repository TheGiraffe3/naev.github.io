530  case OUTFIT_TYPE_TURRET_BOLT:
531  w->timer -= dt;
532  if (w->timer < 0.) {
533  spfx = -1;
534  /* See if we need armour death sprite. */

586  }
587

589  if (i >= array_size(wlayer))
590  break;
591

593  if (w == wlayer[i]) {
594  weapon_update(w,dt,layer);
595  if ((i < array_size(wlayer)) && (w == wlayer[i]))

756 static int weapon_checkCanHit( Weapon* w, Pilot *p )
757 {
758  Pilot *parent;
759 
760  /* Can't hit invincible stuff. */
761  if (pilot_isFlag(p, PILOT_INVINCIBLE))
762  return 0;
763 
764  /* Can't hit hidden stuff. */
765  if (pilot_isFlag(p, PILOT_HIDE))
766  return 0;
767 
768  /* Can never hit same faction. */
769  if (p->faction == w->faction)
770  return 0;
771 
772  /* Must not be landing nor taking off. */
773  if (pilot_isFlag(p, PILOT_LANDING) ||
774  pilot_isFlag(p, PILOT_TAKEOFF))
775  return 0;
776 
777  /* Go "through" dead pilots. */
778  if (pilot_isFlag(p, PILOT_DEAD))
779  return 0;
780 
781  /* Player can not hit special pilots. */
782  if ((w->faction == FACTION_PLAYER) &&
783  pilot_isFlag(p, PILOT_INVINC_PLAYER))
784  return 0;
785 
786  /* Always hit target. */
787  if (w->target == p->id)
788  return 1;
789 
790  /* Player behaves differently. */
791  if (w->faction == FACTION_PLAYER) {
792 
793  /* Always hit hostiles. */
794  if (pilot_isHostile(p))
795  return 1;
796 
797  /* Miss rest - can be neutral/ally. */
798  else
799  return 0;
800  }
801 
802  /* Let hostiles hit player. */
803  if (p->faction == FACTION_PLAYER) {
804  parent = pilot_get(w->parent);
805  if (parent != NULL) {
806  if (pilot_isHostile(parent))
807  return 1;
808  }
809  }
810 
811  /* Hit non-allies. */
812  if (areEnemies(w->faction, p->faction))
813  return 1;
814 
815  return 0;

1316 static double weapon_aimTurret( const Outfit *outfit, const Pilot *parent,
1317  const Pilot *pilot_target, const Vector2d *pos, const Vector2d *vel, double dir,
1318  double swivel, double time )

1328 
1329  if (pilot_target != NULL) {
1330  target_pos = pilot_target->solid->pos;
1331  target_vel = pilot_target->solid->vel;
1332  }
1333  else {
1334  if (parent->nav_asteroid < 0)
1335  return dir;

1338  ast = &field->asteroids[parent->nav_asteroid];

1350  t = 0.;

1367  /* the target is moving too fast for the turret to keep up */
1368  if (ANGLE(x, y) < VANGLE(relative_location))

1378  if (off > 0.)
1379  rdir = dir - swivel;
1380  else
1381  rdir = dir + swivel;
1382  }
1383 
1384  return rdir;
1385 }
