118 static int naevL_versionTest( lua_State *L)
119 {
120  const char *s1, *s2;
121  semver_t sv1, sv2;
122  int res;
123  /* Parse inputs. */
124  s1 = luaL_checkstring(L,1);
125  s2 = luaL_checkstring(L,2);
126  if (semver_parse( s1, &sv1 ))
127  WARN( _("Failed to parse version string '%s'!"), s1 );
128  if (semver_parse( s2, &sv2 ))
129  WARN( _("Failed to parse version string '%s'!"), s2 );
130 
131  /* Check version. */
132  res = semver_compare( sv1, sv2 );
133 
134  /* Cleanup. */
135  semver_free( &sv1 );
136  semver_free( &sv2 );
137 
138  lua_pushinteger(L,res);
139  return 1;
140 }
