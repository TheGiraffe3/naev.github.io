476 static int pilotL_addFleetFrom( lua_State *L, int from_ship )
477 {
478  Fleet *flt;
479  Ship *ship;
480  const char *fltname, *fltai;
481  int i, first, i_parameters;
482  unsigned int p;
483  double a, r;
484  Vector2d vv, vp, vn;
485  FleetPilot *plt;
486  LuaFaction lf;
487  StarSystem *ss;
488  Planet *planet;
489  JumpPoint *jump;
490  PilotFlags flags;
491  int ignore_rules;
492 
493  /* Default values. */
494  pilot_clearFlagsRaw( flags );
495  vectnull(&vn); /* Need to determine angle. */
496  jump = NULL;
497  a = 0.;
498 
499  /* Parse first argument - Fleet Name */
500  fltname = luaL_checkstring(L,1);
501 
502  /* pull the fleet */
503  ship = NULL;
504  flt = NULL;
505  if (from_ship) {
506  ship = ship_get( fltname );
507  if (ship == NULL) {
508  NLUA_ERROR(L,_("Ship '%s' not found!"), fltname);
509  return 0;
510  }
511  /* Get pilotname argument if provided. */
512  fltname = luaL_optstring( L, 4, fltname );
513  /* Get faction from string or number. */
514  lf = luaL_validfaction(L,2);
515  }
516  else {
517  flt = fleet_get( fltname );
518  if (flt == NULL) {
519  NLUA_ERROR(L,_("Fleet '%s' doesn't exist."), fltname);
520  return 0;
521  }
522  lf = flt->faction;
523  }
524 
525  /* Handle position/origin argument. */
526  if (lua_isvector(L,2+from_ship)) {
527  vp = *lua_tovector(L,2+from_ship);
528  a = RNGF() * 2.*M_PI;
529  vectnull( &vv );
530  }
531  else if (lua_issystem(L,2+from_ship)) {
532  ss = system_getIndex( lua_tosystem(L,2+from_ship) );
533  for (i=0; i<array_size(cur_system->jumps); i++) {
534  if ((cur_system->jumps[i].target == ss)
535  && !jp_isFlag( cur_system->jumps[i].returnJump, JP_EXITONLY )) {
536  jump = cur_system->jumps[i].returnJump;
537  break;
538  }
539  }
540  if (jump == NULL) {
541  if (array_size(cur_system->jumps) > 0) {
542  WARN(_("Fleet '%s' jumping in from non-adjacent system '%s' to '%s'."),
543  fltname, ss->name, cur_system->name );
544  jump = cur_system->jumps[RNG_BASE(0, array_size(cur_system->jumps)-1)].returnJump;
545  }
546  else
547  WARN(_("Fleet '%s' attempting to jump in from '%s', but '%s' has no jump points."),
548  fltname, ss->name, cur_system->name );
549  }
550  }
551  else if (lua_isplanet(L,2+from_ship)) {
552  planet = luaL_validplanet(L,2+from_ship);
553  pilot_setFlagRaw( flags, PILOT_TAKEOFF );
554  a = RNGF() * 2. * M_PI;
555  r = RNGF() * planet->radius;
556  vect_cset( &vp,
557  planet->pos.x + r * cos(a),
558  planet->pos.y + r * sin(a) );
559  a = RNGF() * 2.*M_PI;
560  vectnull( &vv );
561  }
562  /* Random. */
563  else {
564  /* Check if we should ignore the strict rules. */
565  ignore_rules = 0;
566  if (lua_isboolean(L,2+from_ship) && lua_toboolean(L,2+from_ship))
567  ignore_rules = 1;
568 
569  /* Choose the spawn point and act in consequence.*/
570  pilot_choosePoint( &vp, &planet, &jump, lf, ignore_rules, 0 );
571 
572  if (planet != NULL) {
573  pilot_setFlagRaw( flags, PILOT_TAKEOFF );
574  a = RNGF() * 2. * M_PI;
575  r = RNGF() * planet->radius;
576  vect_cset( &vp,
577  planet->pos.x + r * cos(a),
578  planet->pos.y + r * sin(a) );
579  a = RNGF() * 2.*M_PI;
580  vectnull( &vv );
581  }
582  else {
583  a = RNGF() * 2.*M_PI;
584  vectnull( &vv );
585  }
586  }
587 
588  /* Parse final argument - table of optional parameters */
589  i_parameters = 3+2*from_ship;
590  fltai = NULL;
591  if (lua_gettop( L ) >= i_parameters && !lua_isnil( L, i_parameters )) {
592  if (!lua_istable( L, i_parameters )) {
593  NLUA_ERROR( L, _("'parameters' should be a table of options or omitted!") );
594  return 0;
595  }
596  lua_getfield( L, i_parameters, "ai" );
597  fltai = luaL_optstring( L, -1, NULL );
598  lua_pop( L, 1 );
599 
600  lua_getfield( L, i_parameters, "naked" );
601  if (lua_toboolean(L, -1))
602  pilot_setFlagRaw( flags, PILOT_NO_OUTFITS );
603  lua_pop( L, 1 );
604  }
605 
606  /* Set up velocities and such. */
607  if (jump != NULL) {
608  space_calcJumpInPos( cur_system, jump->from, &vp, &vv, &a, NULL );
609  pilot_setFlagRaw( flags, PILOT_HYP_END );
610  }
611 
612  /* Make sure angle is valid. */
613  a = fmod( a, 2.*M_PI );
614  if (a < 0.)
615  a += 2.*M_PI;
616 
617  if (from_ship) {
618  /* Create the pilot. */
619  p = pilot_create( ship, fltname, lf, fltai, a, &vp, &vv, flags, 0, 0 );
620  lua_pushpilot(L,p);
621  }
622  else {
623  /* now we start adding pilots and toss ids into the table we return */
624  lua_newtable(L);
625  first = 1;
626  for (i=0; i<flt->npilots; i++) {
627  plt = &flt->pilots[i];
628 
629  /* Fleet displacement - first ship is exact. */
630  if (!first)
631  vect_cadd(&vp, RNG(75,150) * (RNG(0,1) ? 1 : -1),
632  RNG(75,150) * (RNG(0,1) ? 1 : -1));
633  first = 0;
634 
635  /* Create the pilot. */
636  p = fleet_createPilot( flt, plt, a, &vp, &vv, fltai, flags );
637 
638  /* we push each pilot created into a table and return it */
639  lua_pushnumber(L,i+1); /* index, starts with 1 */
640  lua_pushpilot(L,p); /* value = LuaPilot */
641  lua_rawset(L,-3); /* store the value in the table */
642  }
643  }
644  return 1;
645 }
