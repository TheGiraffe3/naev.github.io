236 int misn_runFunc( Mission *misn, const char *func, int nargs )
237 {
238  int i, ret;
239  const char* err;
240  int misn_delete;
241  Mission *cur_mission;
242  nlua_env env;
243  int isaccepted;
244 
245  /* Check to see if it is accepted first. */
246  isaccepted = misn->accepted;
247 
248  /* Set up and run function. */
249  env = misn->env;
250  ret = nlua_pcall(env, nargs, 0);
251 
252  /* The mission can change if accepted. */
253  nlua_getenv(env, "__misn");
254  cur_mission = *(Mission**) lua_touserdata(naevL, -1);
255  lua_pop(naevL, 1);
256 
257  if (ret != 0) { /* error has occurred */
258  err = (lua_isstring(naevL,-1)) ? lua_tostring(naevL,-1) : NULL;
259  if ((err==NULL) || (strcmp(err,NLUA_DONE)!=0)) {
260  WARN(_("Mission '%s' -> '%s': %s"),
261  cur_mission->data->name, func, (err) ? err : _("unknown error"));
262  ret = -1;
263  }
264  else
265  ret = 1;
266  lua_pop(naevL,1);
267  }
268 
269  /* Get delete. */
270  nlua_getenv(env, "__misn_delete");
271  misn_delete = lua_toboolean(naevL,-1);
272  lua_pop(naevL,1);
273 
274  /* Mission is finished */
275  if (misn_delete) {
276  ret = 2;
277  mission_cleanup( cur_mission );
278  for (i=0; i<MISSION_MAX; i++) {
279  if (cur_mission != player_missions[i])
280  continue;
281 
282  mission_shift(i);
283  break;
284  }
285  }
286  /* Mission became accepted. */
287  else if (!isaccepted && cur_mission->accepted)
288  ret = 3;
289 
290  return ret;
291 }

832
