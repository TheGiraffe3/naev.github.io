367 static int mapedit_mouse( unsigned int wid, SDL_Event* event, double mx, double my,
368  double w, double h, double xr, double yr, void *data )
369 {
370  (void) wid;
371  (void) data;
372  int found;
373  double x,y, t;
374  StarSystem *sys;
375 
376  t = 15.*15.; /* threshold */
377 
378  switch (event->type) {
379  case SDL_MOUSEWHEEL:
380  /* Must be in bounds. */
381  if ((mx < 0.) || (mx > w) || (my < 0.) || (my > h))
382  return 0;
383  if (event->wheel.y > 0)
384  mapedit_buttonZoom( 0, "btnZoomIn" );
385  else if (event->wheel.y < 0)
386  mapedit_buttonZoom( 0, "btnZoomOut" );
387  return 1;
388 
389  case SDL_MOUSEBUTTONDOWN:
390  /* Must be in bounds. */
391  if ((mx < 0.) || (mx > w) || (my < 0.) || (my > h))
392  return 0;
393 
394  /* Zooming */
395  if (event->button.button == SDL_BUTTON_X1) {
396  mapedit_buttonZoom( 0, "btnZoomIn" );
397  return 1;
398  }
399  else if (event->button.button == SDL_BUTTON_X2) {
400  mapedit_buttonZoom( 0, "btnZoomOut" );
401  return 1;
402  }
403 
404  /* selecting star system */
405  else {
406  mx -= w/2 - mapedit_xpos;
407  my -= h/2 - mapedit_ypos;
408 
409  for (int i=0; i<array_size(systems_stack); i++) {
410  sys = system_getIndex( i );
411 
412  /* get position */
413  x = sys->pos.x * mapedit_zoom;
414  y = sys->pos.y * mapedit_zoom;
415 
416  if ((pow2(mx-x)+pow2(my-y)) < t) {
417  /* Set last clicked system */
418  mapedit_iLastClickedSystem = i;
419 
420  /* Try to find in selected systems. */
421  found = 0;
422  for (int j=0; j<mapedit_nsys; j++) {
423  /* Must match. */
424  if (mapedit_sys[j] == sys) {
425  found = 1;
426  break;
427  } else
428  continue;
429  }
430 
431  /* Toggle system selection. */
432  if (found)
433  mapedit_selectRm( sys );
434  else
435  mapedit_selectAdd( sys );
436  return 1;
437  }
438  }
439 
440  /* Start dragging the viewport. */
441  mapedit_drag = 1;
442  mapedit_dragTime = SDL_GetTicks();
443  mapedit_moved = 0;
444  }
445  break;
446 
447  case SDL_MOUSEBUTTONUP:
448  /* Handles dragging viewport around. */
449  if (mapedit_drag) {
450  mapedit_drag = 0;
451  }
452  break;
453 
454  case SDL_MOUSEMOTION:
455  /* Update mouse positions. */
456  mapedit_mx = mx;
457  mapedit_my = my;
458 
459  /* Handle dragging. */
460  if (mapedit_drag) {
461  /* axis is inverted */
462  mapedit_xpos -= xr;
463  mapedit_ypos += yr;
464 
465  /* Update mouse movement. */
466  mapedit_moved += ABS(xr) + ABS(yr);
467  }
468  break;
469  }
470 
471  return 0;
472 }

596 
597  /* Transform coords back. */
598  mapedit_xpos *= mapedit_zoom;

879  char *description = NULL;

883  asprintf( &file, "%s%s", MAP_DATA_PATH, map_files[i] );

894  free(file);
895  xmlFreeDoc(doc);

897  }
898 
899  if (!xml_isNode(node,"outfit")) {
900  free(file);
901  xmlFreeDoc(doc);
902  return -1;
903  }
904 
905  /* Get "name" property from the "outfit" node */
906  xmlr_attr_strd( node, "name", name );
907 
908  /* Loop on the nodes to find <specific> node */

925  }

927  /* Get the "type" property from "specific" node */
928  xmlr_attr_strd( node, "type", outfitType );

931 
932  /* Break out of the loop, either with a map or not */
933  break;
934  } while (xml_nextNode(node));

1001 static int mapedit_saveMap( StarSystem **uniedit_sys, mapOutfitsList_t* ns )
1002 {
1003  xmlDocPtr doc;
1004  xmlTextWriterPtr writer;
1005  char *file;
1006 
1007  /* Create the writer. */
1008  writer = xmlNewTextWriterDoc(&doc, 0);
1009  if (writer == NULL) {
1010  WARN(_("testXmlwriterDoc: Error creating the xml writer"));
1011  return -1;
1012  }
1013 
1014  /* Set the writer parameters. */
1015  xmlw_setParams( writer );
1016 
1017  /* Start writer. */
1018  xmlw_start(writer);
1019  xmlw_startElem( writer, "outfit" );
1020 
1021  /* Attributes. */
1022  xmlw_attr( writer, "name", "%s", ns->mapName );
1023 
1024  /* General. */
1025  xmlw_startElem( writer, "general" );
1026  xmlw_elem( writer, "rarity", "%d", ns->rarity );
1027  xmlw_elem( writer, "mass", "%d", 0 );
1028  xmlw_elem( writer, "price", "%"CREDITS_PRI, ns->price );
1029  xmlw_elem( writer, "description", "%s", ns->description );
1030  xmlw_elem( writer, "gfx_store", "%s", "map.webp" );
1031  xmlw_endElem( writer ); /* "general" */
1032 
1033  xmlw_startElem( writer, "specific" );
1034  xmlw_attr( writer, "type", "map" );
1035 
1036  /* Iterate over all selected systems. Save said systems and any NORMAL jumps they might share. */
1037  for (int i=0; i < ns->numSystems; i++) {
1038  StarSystem *s = uniedit_sys[i];
1039  xmlw_startElem( writer, "sys" );
1040  xmlw_attr( writer, "name", "%s", s->name );
1041 
1042  /* Iterate jumps and see if they lead to any other systems in our array. */
1043  for (int j=0; j < array_size(s->jumps); j++) {
1044  /* Ignore hidden and exit-only jumps. */
1045  if (jp_isFlag(&s->jumps[j], JP_EXITONLY ))
1046  continue;
1047  if (jp_isFlag(&s->jumps[j], JP_HIDDEN))
1048  continue;
1049  /* This is a normal jump. */
1050  for (int k=0; k < ns->numSystems; k++) {
1051  if (s->jumps[j].target == uniedit_sys[k]) {
1052  xmlw_elem( writer, "jump", "%s", uniedit_sys[k]->name );
1053  break;
1054  }
1055  }
1056  }
1057 
1058  /* Iterate assets and add them */
1059  for (int j=0; j < array_size(s->planets); j++)
1060  xmlw_elem( writer, "asset", "%s", s->planets[j]->name );
1061 
1062  xmlw_endElem( writer ); /* "sys" */
1063  }
1064 
1065  xmlw_endElem( writer ); /* "specific" */
1066  xmlw_endElem( writer ); /* "outfit" */
1067  xmlw_done(writer);
1068 
1069  /* No need for writer anymore. */
1070  xmlFreeTextWriter(writer);
1071 
1072  /* Actually write data */
1073  asprintf( &file, "dat/outfits/maps/%s", ns->fileName );
1074  if (xmlSaveFileEnc( file, doc, "UTF-8" ) < 0)
1075  WARN("Failed writing '%s'!", file);
1076  free( file );
1077 
1078  /* Clean up. */
1079  xmlFreeDoc(doc);
1080 
1081  return 0;
1082 }
