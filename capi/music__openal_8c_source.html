128 static int music_thread( void* unused )
129 {
130  (void)unused;
131 
132  int ret;
133  int active = 0; /* active buffer */
134  ALint state;
135  ALuint removed[2];
136  ALenum value;
137  music_state_t cur_state;
138  ALfloat gain;
139  int fadein_start = 0;
140  uint32_t fade, fade_timer = 0;
141 
142  while (1) {
143  /* Handle states. */
144  musicLock();
145 
146  /* Handle new command. */
147  switch (music_command) {
148  case MUSIC_CMD_KILL:
149  if (music_state != MUSIC_STATE_IDLE)
150  music_state = MUSIC_STATE_STOPPING;
151  else
152  music_state = MUSIC_STATE_DEAD;
153 
154  /* Does not clear command. */
155  break;
156 
157  case MUSIC_CMD_STOP:
158  /* Notify of stopped. */
159  if (music_state == MUSIC_STATE_IDLE)
160  SDL_CondBroadcast( music_state_cond );
161  else
162  music_state = MUSIC_STATE_STOPPING;
163  break;
164 
165  case MUSIC_CMD_PLAY:
166  /* Set appropriate state. */
167  if (music_state == MUSIC_STATE_PAUSING)
168  music_state = MUSIC_STATE_RESUMING;
169  else if (music_state == MUSIC_STATE_FADEIN)
170  fade_timer = SDL_GetTicks() - MUSIC_FADEIN_DELAY;
171  else
172  music_state = MUSIC_STATE_LOADING;
173  /* Disable fadein. */
174  fadein_start = 0;
175  /* Clear command. */
176  music_command = MUSIC_CMD_NONE;
177  SDL_CondBroadcast( music_state_cond );
178  break;
179 
180  case MUSIC_CMD_FADEOUT:
181  /* Notify of stopped. */
182  if (music_state != MUSIC_STATE_IDLE) {
183  music_state = MUSIC_STATE_FADEOUT;
184  /* Set timer. */
185  fade_timer = SDL_GetTicks();
186  }
187  /* Clear command. */
188  music_command = MUSIC_CMD_NONE;
189  SDL_CondBroadcast( music_state_cond );
190  break;
191 
192  case MUSIC_CMD_FADEIN:
193  if ((music_state == MUSIC_STATE_FADEIN) ||
194  (music_state == MUSIC_STATE_PLAYING))
195  SDL_CondBroadcast( music_state_cond );
196  else {
197  music_state = MUSIC_STATE_LOADING;
198  /* Set timer. */
199  fade_timer = SDL_GetTicks();
200  fadein_start = 1;
201  }
202  /* Clear command. */
203  music_command = MUSIC_CMD_NONE;
204  break;
205 
206  case MUSIC_CMD_PAUSE:
207  if (music_state == MUSIC_STATE_PAUSED)
208  SDL_CondBroadcast( music_state_cond );
209  else if ((music_state == MUSIC_STATE_PLAYING) ||
210  (music_state == MUSIC_STATE_FADEIN))
211  music_state = MUSIC_STATE_PAUSING;
212  music_command = MUSIC_CMD_NONE;
213  break;
214 
215  case MUSIC_CMD_NONE:
216  break;
217  }
218  cur_state = music_state;
219  musicUnlock();
220 
221  /* Main processing loop. */
222  switch (cur_state) {
223  /* Basically send a message that thread is up and running. */
224  case MUSIC_STATE_STARTUP:
225  musicLock();
226  music_state = MUSIC_STATE_IDLE;
227  SDL_CondBroadcast( music_state_cond );
228  musicUnlock();
229  break;
230 
231  /* We died. */
232  case MUSIC_STATE_DEAD:
233  musicLock();
234  music_state = MUSIC_STATE_DEAD;
235  SDL_CondBroadcast( music_state_cond );
236  musicUnlock();
237  return 0;
238  break;
239 
240  /* Delays at the end. */
241  case MUSIC_STATE_PAUSED:
242  case MUSIC_STATE_IDLE:
243  break;
244 
245  /* Resumes the paused song. */
246  case MUSIC_STATE_RESUMING:
247  soundLock();
248  alSourcePlay( music_source );
249  alSourcef( music_source, AL_GAIN, music_vol );
250  /* Check for errors. */
251  al_checkErr();
252  soundUnlock();
253 
254  musicLock();
255  music_state = MUSIC_STATE_PLAYING;
256  SDL_CondBroadcast( music_state_cond );
257  musicUnlock();
258  break;
259 
260  /* Pause the song. */
261  case MUSIC_STATE_PAUSING:
262  soundLock();
263  alSourcePause( music_source );
264  /* Check for errors. */
265  al_checkErr();
266  soundUnlock();
267 
268  musicLock();
269  music_state = MUSIC_STATE_PAUSED;
270  SDL_CondBroadcast( music_state_cond );
271  musicUnlock();
272  break;
273 
274  /* Stop song setting to IDLE. */
275  case MUSIC_STATE_STOPPING:
276  soundLock();
277 
278  /* Stop and remove buffers. */
279  alSourceStop( music_source );
280  alGetSourcei( music_source, AL_BUFFERS_PROCESSED, &value );
281  if (value > 0)
282  alSourceUnqueueBuffers( music_source, value, removed );
283  /* Clear timer. */
284  fade_timer = 0;
285 
286  /* Reset volume. */
287  alSourcef( music_source, AL_GAIN, music_vol );
288 
289  soundUnlock();
290 
291  musicLock();
292  music_state = MUSIC_STATE_IDLE;
293  SDL_CondBroadcast( music_state_cond );
294  if (!music_forced)
295  music_rechoose();
296  musicUnlock();
297  break;
298 
299  /* Load the song. */
300  case MUSIC_STATE_LOADING:
301  /* Load buffer and start playing. */
302  active = 0; /* load first buffer */
303  ret = stream_loadBuffer( music_buffer[active] );
304  soundLock();
305  alSourceQueueBuffers( music_source, 1, &music_buffer[active] );
306 
307  /* Special case NULL file or error. */
308  if (ret < 0) {
309  soundUnlock();
310  /* Force state to stopped. */
311  musicLock();
312  music_state = MUSIC_STATE_IDLE;
313  SDL_CondBroadcast( music_state_cond );
314  if (!music_forced)
315  music_rechoose();
316  musicUnlock();
317  break;
318  }
319  /* Force volume level. */
320  alSourcef( music_source, AL_GAIN, (fadein_start) ? 0. : music_vol );
321 
322  /* Start playing. */
323  alSourcePlay( music_source );
324 
325  /* Check for errors. */
326  al_checkErr();
327 
328  soundUnlock();
329  /* Special case of a very short song. */
330  if (ret > 1) {
331  active = -1;
332 
333  musicLock();
334  if (fadein_start)
335  music_state = MUSIC_STATE_FADEIN;
336  else
337  music_state = MUSIC_STATE_PLAYING;
338  SDL_CondBroadcast( music_state_cond );
339  musicUnlock();
340  break;
341  }
342 
343  /* Load second buffer. */
344  active = 1;
345  ret = stream_loadBuffer( music_buffer[active] );
346  if (ret < 0)
347  active = -1;
348  else {
349  soundLock();
350  alSourceQueueBuffers( music_source, 1, &music_buffer[active] );
351  /* Check for errors. */
352  al_checkErr();
353  soundUnlock();
354  active = 1 - active;
355  }
356 
357  musicLock();
358  if (fadein_start)
359  music_state = MUSIC_STATE_FADEIN;
360  else
361  music_state = MUSIC_STATE_PLAYING;
362  SDL_CondBroadcast( music_state_cond );
363  musicUnlock();
364  break;
365 
366  /* Fades in the music. */
367  case MUSIC_STATE_FADEOUT:
368  case MUSIC_STATE_FADEIN:
369  /* See if must still fade. */
370  fade = SDL_GetTicks() - fade_timer;
371  if (cur_state == MUSIC_STATE_FADEIN) {
372  if (fade < MUSIC_FADEIN_DELAY) {
373  gain = (ALfloat)fade / (ALfloat)MUSIC_FADEIN_DELAY;
374  soundLock();
375  alSourcef( music_source, AL_GAIN, gain*music_vol );
376  /* Check for errors. */
377  al_checkErr();
378  soundUnlock();
379  }
380  /* No need to fade anymore. */
381  else {
382  /* Set volume to normal level. */
383  soundLock();
384  alSourcef( music_source, AL_GAIN, music_vol );
385  /* Check for errors. */
386  al_checkErr();
387  soundUnlock();
388 
389  /* Change state to playing. */
390  musicLock();
391  music_state = MUSIC_STATE_PLAYING;
392  musicUnlock();
393  }
394  }
395  else if (cur_state == MUSIC_STATE_FADEOUT) {
396  if (fade < MUSIC_FADEOUT_DELAY) {
397  gain = 1. - (ALfloat)fade / (ALfloat)MUSIC_FADEOUT_DELAY;
398  soundLock();
399  alSourcef( music_source, AL_GAIN, gain*music_vol );
400  /* Check for errors. */
401  al_checkErr();
402  soundUnlock();
403  }
404  else {
405  /* Music should stop. */
406  musicLock();
407  music_state = MUSIC_STATE_STOPPING;
408  musicUnlock();
409  break;
410  }
411  }
412  /* fallthrough */
413 
414  /* Play the song if needed. */
415  case MUSIC_STATE_PLAYING:
416  /* Special case where file has ended. */
417  if (active < 0) {
418  soundLock();
419  alGetSourcei( music_source, AL_SOURCE_STATE, &state );
420 
421  if (state == AL_STOPPED) {
422  alGetSourcei( music_source, AL_BUFFERS_PROCESSED, &value );
423  if (value > 0)
424  alSourceUnqueueBuffers( music_source, value, removed );
425  soundUnlock();
426 
427  musicLock();
428  music_state = MUSIC_STATE_IDLE;
429  if (!music_forced)
430  music_rechoose();
431  musicUnlock();
432  break;
433  }
434 
435  soundUnlock();
436 
437  break;
438  }
439 
440  soundLock();
441 
442  /* See if needs another buffer set. */
443  alGetSourcei( music_source, AL_BUFFERS_PROCESSED, &state );
444  if (state > 0) {
445  /* refill active buffer */
446  alSourceUnqueueBuffers( music_source, 1, removed );
447  ret = stream_loadBuffer( music_buffer[active] );
448  if (ret < 0)
449  active = -1;
450  else {
451  alSourceQueueBuffers( music_source, 1, &music_buffer[active] );
452  active = 1 - active;
453  }
454  }
455 
456  /* Check for errors. */
457  al_checkErr();
458 
459  soundUnlock();
460  }
461 
462  /* Global thread delay. */
463  SDL_Delay(0);
464 
465  }
466 
467  return 0;
468 }

479 static void rg_filter( float **pcm, long channels, long samples, void *filter_param )
480 {
481  int i, j;
482  float cur_sample;
483  alMusic *param = filter_param;
484  float scale_factor = param->rg_scale_factor;
485  float max_scale = param->rg_max_scale;
486 
487  /* Apply the gain, and any limiting necessary */
488  if (scale_factor > max_scale) {
489  for (i = 0; i < channels; i++)
490  for (j = 0; j < samples; j++) {
491  cur_sample = pcm[i][j] * scale_factor;
492  /*
493  * This is essentially the scaled hard-limiting algorithm
494  * It looks like the soft-knee to me
495  * I haven't found a better limiting algorithm yet...
496  */
497  if (cur_sample < -0.5)
498  cur_sample = tanh((cur_sample + 0.5) / (1-0.5)) * (1-0.5) - 0.5;
499  else if (cur_sample > 0.5)
500  cur_sample = tanh((cur_sample - 0.5) / (1-0.5)) * (1-0.5) + 0.5;
501  pcm[i][j] = cur_sample;
502  }
503  }
504  else if (scale_factor > 0.0)
505  for (i = 0; i < channels; i++)
506  for (j = 0; j < samples; j++)
507  pcm[i][j] *= scale_factor;
508 }

516 static int stream_loadBuffer( ALuint buffer )
517 {
518  int ret, size, section, result;
519 
520  musicVorbisLock();
521 
522  /* Make sure music is valid. */
523  if (music_vorbis.rw == NULL) {
524  musicVorbisUnlock();
525  return -1;
526  }
527 
528  ret = 0;
529  size = 0;
530  while (size < music_bufSize) { /* file up the entire data buffer */
531 
532  result = ov_read_filter(
533  &music_vorbis.stream, /* stream */
534  &music_buf[size], /* data */
535  music_bufSize - size, /* amount to read */
536  (SDL_BYTEORDER == SDL_BIG_ENDIAN),
537  2, /* 16 bit */
538  1, /* signed */
539  &section, /* current bitstream */
540  rg_filter, /* filter function */
541  &music_vorbis ); /* filter parameter */
542 
543  /* End of file. */
544  if (result == 0) {
545  if (size == 0) {
546  musicVorbisUnlock();
547  return -2;
548  }
549  ret = 1;
550  break;
551  }
552  /* Hole error. */
553  else if (result == OV_HOLE) {
554  musicVorbisUnlock();
555  WARN(_("OGG: Vorbis hole detected in music!"));
556  return 0;
557  }
558  /* Bad link error. */
559  else if (result == OV_EBADLINK) {
560  musicVorbisUnlock();
561  WARN(_("OGG: Invalid stream section or corrupt link in music!"));
562  return -1;
563  }
564 
565  size += result;
566  }
567 
568  musicVorbisUnlock();
569 
570  /* load the buffer up */
571  soundLock();
572  alBufferData( buffer, music_vorbis.format,
573  music_buf, size, music_vorbis.info->rate );
574  soundUnlock();
575 
576  return ret;
577 }

583 int music_al_init (void)
584 {
585  ALfloat v[] = { 0., 0., 0. };
586 
587  /* Create threading mechanisms. */
588  music_state_cond = SDL_CreateCond();
589  music_state_lock = SDL_CreateMutex();
590  music_vorbis_lock = SDL_CreateMutex();
591  music_vorbis.rw = NULL; /* indication it's not loaded */
592 
593  /* Create the buffer. */
594  music_bufSize = conf.al_bufsize * 1024;
595  music_buf = malloc( music_bufSize );
596 
597  soundLock();
598 
599  /* Allocate source for music. */
600  alGenSources( 1, &music_source );
601 
602  /* Generate buffers and sources. */
603  alGenBuffers( 2, music_buffer );
604 
605  /* Set up OpenAL properties. */
606  alSourcef( music_source, AL_GAIN, music_vol );
607  alSourcei( music_source, AL_SOURCE_RELATIVE, AL_TRUE );
608  alSourcefv( music_source, AL_POSITION, v );
609  alSourcefv( music_source, AL_VELOCITY, v );
610 
611  /* Check for errors. */
612  al_checkErr();
613 
614  /* Set state to none. */
615  music_state = 0;
616 
617  soundUnlock();
618 
619  /*
620  * Start up thread and have it inform us when it already reaches the main loop.
621  */
622  musicLock();
623  music_state = MUSIC_STATE_STARTUP;
624  music_player = SDL_CreateThread( music_thread,
625  "music_thread",
626  NULL );
627  SDL_CondWait( music_state_cond, music_state_lock );
628  musicUnlock();
629 
630  return 0;
631 }

665 int music_al_load( const char* name, SDL_RWops *rw )
666 {
667  int rg;
668  ALfloat track_gain_db, track_peak;
669  vorbis_comment *vc;
670  char *tag = NULL;
671 
672  musicVorbisLock();
673 
674  /* Load new ogg. */
675  music_vorbis.rw = rw;
676  if (ov_open_callbacks( music_vorbis.rw, &music_vorbis.stream,
677  NULL, 0, sound_al_ovcall ) < 0) {
678  WARN(_("Song '%s' does not appear to be a Vorbis bitstream."), name);
679  musicUnlock();
680  return -1;
681  }
682  music_vorbis.info = ov_info( &music_vorbis.stream, -1 );
683 
684  /* Get Replaygain information. */
685  vc = ov_comment( &music_vorbis.stream, -1 );
686  track_gain_db = 0.;
687  track_peak = 1.;
688  rg = 0;
689  if ((tag = vorbis_comment_query(vc, "replaygain_track_gain", 0))) {
690  track_gain_db = atof(tag);
691  rg = 1;
692  }
693  if ((tag = vorbis_comment_query(vc, "replaygain_track_peak", 0))) {
694  track_peak = atof(tag);
695  rg = 1;
696  }
697  music_vorbis.rg_scale_factor = pow(10.0, (track_gain_db + RG_PREAMP_DB)/20);
698  music_vorbis.rg_max_scale = 1.0 / track_peak;
699  if (!rg)
700  DEBUG(_("Song '%s' has no replaygain information."), name );
701 
702  /* Set the format */
703  if (music_vorbis.info->channels == 1)
704  music_vorbis.format = AL_FORMAT_MONO16;
705  else
706  music_vorbis.format = AL_FORMAT_STEREO16;
707 
708  musicVorbisUnlock();
709 
710  return 0;
711 }

762 
763  /* Check for errors. */
