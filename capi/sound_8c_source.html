99

101  if (sound_disabled && music_disabled)
102  return 0;
103 
104  /* Initialize sound backend. */
105  ret = sound_al_init();
106  if (ret != 0) {
107  sound_disabled = 1;
108  music_disabled = 1;
109  WARN(_("Sound disabled."));
110  return ret;
111  }
112 
113  /* Create voice lock. */
114  voice_mutex = SDL_CreateMutex();
115  if (voice_mutex == NULL)
116  WARN(_("Unable to create voice mutex."));
117 
118  /* Load available sounds. */
119  ret = sound_makeList();
120  if (ret != 0)
121  return ret;
122 
123  /* Initialize music. */
124  ret = music_init();
125  if (ret != 0) {
126  music_disabled = 1;
127  WARN(_("Music disabled."));
128  }
129

131  if ((conf.sound > 1.) || (conf.sound < 0.)) {
132  WARN(_("Sound has invalid value, clamping to [0:1]."));
133  conf.sound = CLAMP( 0., 1., conf.sound );

371

393  tv = v->prev;
394  if (tv == NULL) {

396  if (voice_active != NULL)

398  }

401  if (tv->next != NULL)
402  tv->next->prev = tv;
403  }
404

406  v->next = voice_pool;
407  v->prev = NULL;

409  if (v->next != NULL)
410  v->next->prev = v;
411 
412  /* Avoid loop blockage. */
413  v = (tv != NULL) ? tv->next : voice_active;

526 void sound_setSpeed( double s )
527 {
528  double v;
529  int playing;
530 
531  if (sound_disabled)
532  return;
533 
534  /* We implement the brown noise here. */
535  playing = (snd_compression_gain > 0.);
536  if (player.tc_max > 2.)
537  v = CLAMP( 0, 1., MAX( (s-2)/10., (s-2) / (player.tc_max-2) ) );
538  else
539  v = CLAMP( 0, 1., (s-2)/10. );
540 
541  if (v > 0.) {
542  if (snd_compression >= 0) {
543  if (!playing)
544  sound_playGroup( snd_compressionG, snd_compression, 0 ); /* Start playing only if it's not playing. */
545  sound_volumeGroup( snd_compressionG, v );
546  }
547  sound_al_setSpeedVolume( 1.-v );
548  }
549  else if (playing) {
550  if (snd_compression >= 0)
551  sound_stopGroup( snd_compressionG ); /* Stop compression sound. */
552  sound_al_setSpeedVolume( 1. ); /* Restore volume. */
553  }
554  snd_compression_gain = v;
555 
556  return sound_al_setSpeed( s );
557 }

592  continue;
593

595  snprintf( path, sizeof(path), SOUND_PATH"%s", files[i] );
596  rw = PHYSFSRWOPS_openRead( path );
597 
598  /* remove the suffix */
599  len = flen - suflen;
600  files[i][len] = '\0';
601 
602  source_newRW( rw, files[i], 0 );
603  SDL_RWclose( rw );
604  }
605 
606  DEBUG( n_("Loaded %d Sound", "Loaded %d Sounds", array_size(sound_list)), array_size(sound_list) );
607 
608  /* Clean up. */
609  PHYSFS_freeList( files );
610 
611  return 0;
