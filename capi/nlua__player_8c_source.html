1346  NLUA_ERROR(L,_("Can not teleport the player while they are boarded!"));

1352  sys = luaL_validsystem(L,1);
1353  name = system_getIndex(sys->id)->name;
1354  }

1357  pnt = luaL_validplanet(L,1);
1358  name = planet_getSystem( pnt->name );
1359  if (name == NULL) {
1360  NLUA_ERROR( L, _("Planet '%s' does not belong to a system."), pnt->name );
1361  return 0;
1362  }

1366  name = lua_tostring(L,1);

1369  /* No system found, assume destination string is the name of a planet. */

1379  NLUA_ERROR( L, _("Planet '%s' does not belong to a system."), pntname );
1380  return 0;
1381  }
1382  }
1383  else
1384  sys = system_get( sysname );
1385  }

1388
