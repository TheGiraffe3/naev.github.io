1192 static int playerL_teleport( lua_State *L )

1195  StarSystem *sys;
1196  const char *name, *pntname;
1197 
1198  NLUA_CHECKRW(L);
1199 
1200  /* Must not be landed. */
1201  if (landed)
1202  NLUA_ERROR(L,_("Can not teleport the player while landed!"));
1203  if (comm_isOpen())
1204  NLUA_ERROR(L,_("Can not teleport the player while the comm is open!"));
1205  if (player_isBoarded())
1206  NLUA_ERROR(L,_("Can not teleport the player while they are boarded!"));
1207 
1208  pnt = NULL;
1209 
1210  /* Get a system. */
1211  if (lua_issystem(L,1)) {
1212  sys = luaL_validsystem(L,1);
1213  name = system_getIndex(sys->id)->name;
1214  }
1215  /* Get a planet. */
1216  else if (lua_isplanet(L,1)) {
1217  pnt = luaL_validplanet(L,1);
1218  name = planet_getSystem( pnt->name );
1219  if (name == NULL) {
1220  NLUA_ERROR( L, _("Planet '%s' does not belong to a system."), pnt->name );
1221  return 0;
1222  }
1223  }
1224  /* Get destination from string. */
1225  else if (lua_isstring(L,1)) {
1226  name = lua_tostring(L,1);
1227  sys = system_get( name );
1228  if (sys == NULL) {
1229  /* No system found, assume destination string is the name of a planet. */
1230  pntname = name;
1231  name = planet_getSystem( pntname );
1232  pnt = planet_get( pntname );
1233  if (pnt == NULL) {
1234  NLUA_ERROR( L, _("'%s' is not a valid teleportation target."), name );
1235  return 0;
1236  }
1237 
1238  if (name == NULL) {
1239  NLUA_ERROR( L, _("Planet '%s' does not belong to a system."), pntname );
1240  return 0;
1241  }
1242  }
1243  }
1244  else
1245  NLUA_INVALID_PARAMETER(L);
1246 
1247  /* Check if system exists. */
1248  if (system_get( name ) == NULL) {
1249  NLUA_ERROR( L, _("System '%s' does not exist."), name );
1250  return 0;
1251  }
1252 
1253  /* Jump out hook is run first. */
1254  hooks_run( "jumpout" );
1255 
1256  /* Just in case remove hyperspace flags. */
1257  pilot_rmFlag( player.p, PILOT_HYPERSPACE );
1258  pilot_rmFlag( player.p, PILOT_HYP_BEGIN );
1259  pilot_rmFlag( player.p, PILOT_HYP_BRAKE );
1260  pilot_rmFlag( player.p, PILOT_HYP_PREP );
1261

1263  space_gfxUnload( cur_system );
1264

1266  * Both of these functions invoke gui_setNav(), which updates jump and
1267  * planet targets simultaneously. Thus, invalid reads may arise and the
1268  * target reset must be done prior to calling space_init and destroying
1269  * the old system.
1270  */
1271  player_targetHyperspaceSet( -1 );
1272  player_targetPlanetSet( -1 );
1273 
1274  /* Go to the new system. */
1275  space_init( name );
1276 
1277  /* Map gets deformed when jumping this way. */
1278  map_clear();
1279 
1280  /* Run hooks - order is important. */
1281  hooks_run( "jumpin" );
1282  hooks_run( "enter" );
1283  events_trigger( EVENT_TRIGGER_ENTER );
1284  missions_run( MIS_AVAIL_SPACE, -1, NULL, NULL );
1285 
1286  /* Move to planet. */
1287  if (pnt != NULL)

1289
