32 void pilot_heatCalc( Pilot *p )
33 {
34  double mass_kg;
35  mass_kg = 1000. * p->base_mass;
36  p->heat_emis = 0.8;
37  p->heat_cond = STEEL_HEAT_CONDUCTIVITY;
38  p->heat_C = STEEL_HEAT_CAPACITY * mass_kg;
39 
40  /* We'll approximate area for a sphere.
41  *
42  * Sphere:
43  * V = 4/3*pi*r^3
44  * A = 4*pi*r^2
45  *
46  * Ship:
47  * V = mass/density
48  *
49  * We then equal the ship V and sphere V to obtain r:
50  * r = (3*mass)/(4*pi*density))^(1/3)
51  *
52  * Substituting r in A we get:
53  * A = 4*pi*((3*mass)/(4*pi*density))^(2/3)
54  * */
55  p->heat_area = 4.*M_PI*pow( 3./4.*mass_kg/STEEL_DENSITY/M_PI, 2./3. ) * p->stats.heat_dissipation;
56 }

120 void pilot_heatAddSlot( Pilot *p, PilotOutfitSlot *o )
121 {
122  double hmod;
123  /* We consider that only 1% of the energy is lost in the form of heat,
124  * this keeps numbers safe. */
125  if (o->outfit->type == OUTFIT_TYPE_BOLT)
126  hmod = p->stats.fwd_heat;
127  else if (o->outfit->type == OUTFIT_TYPE_TURRET_BOLT)
128  hmod = p->stats.tur_heat;
129  else
130  hmod = 1.;
131  o->heat_T += hmod * outfit_heat(o->outfit) / o->heat_C;
132 
133  /* Enforce a minimum value as a safety measure. */
134  o->heat_T = MAX( o->heat_T, CONST_SPACE_STAR_TEMP );
135 }

177 double pilot_heatUpdateSlot( Pilot *p, PilotOutfitSlot *o, double dt )
178 {
179  double Q;
180 
181  /* Calculate energy leaving/entering ship chassis. */
182  Q = -p->heat_cond * (o->heat_T - p->heat_T) * o->heat_area * dt;

246 void pilot_heatUpdateCooldown( Pilot *p )
247 {
248  double t;
249  int i, ammo_threshold;
250  PilotOutfitSlot *o;
251  Outfit *ammo;
252 
253  t = pow2( 1. - p->ctimer / p->cdelay );
254  p->heat_T = p->heat_start - CONST_SPACE_STAR_TEMP - (p->heat_start -
255  CONST_SPACE_STAR_TEMP) * t + CONST_SPACE_STAR_TEMP;
256 
257  for (i=0; i<array_size(p->outfits); i++) {
258  o = p->outfits[i];
259  o->heat_T = o->heat_start - CONST_SPACE_STAR_TEMP - (o->heat_start -
260  CONST_SPACE_STAR_TEMP) * t + CONST_SPACE_STAR_TEMP;
261 
262  /* Refill ammo too (also part of Active Cooldown) */
263  /* Must be valid outfit. */
264  if (o->outfit == NULL)
265  continue;
266 
267  /* Add ammo if able to. */
268  ammo = outfit_ammo( o->outfit );
269  if (ammo == NULL)
270  continue;
271 
272  /* Initial (raw) ammo threshold */
273  ammo_threshold = round(t * pilot_maxAmmoO(p,o->outfit));
274 
275  /* Adjust for deployed fighters if needed */
276  if ( outfit_isFighterBay( o->outfit ) )
277  ammo_threshold -= o->u.ammo.deployed;
278 
279  if ( o->u.ammo.quantity < ammo_threshold )
280  pilot_addAmmo( p, p->outfits[i], ammo, ammo_threshold - o->u.ammo.quantity );
281  }
