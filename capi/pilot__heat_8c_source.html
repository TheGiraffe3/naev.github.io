38 void pilot_heatCalc( Pilot *p )
39 {
40  double mass_kg;
41  mass_kg = 1000. * p->base_mass;
42  p->heat_emis = 0.8;
43  p->heat_cond = STEEL_HEAT_CONDUCTIVITY;
44  p->heat_C = STEEL_HEAT_CAPACITY * mass_kg;
45 
46  /* We'll approximate area for a sphere.
47  *
48  * Sphere:
49  * V = 4/3*pi*r^3
50  * A = 4*pi*r^2
51  *
52  * Ship:
53  * V = mass/density
54  *
55  * We then equal the ship V and sphere V to obtain r:
56  * r = (3*mass)/(4*pi*density))^(1/3)
57  *
58  * Substituting r in A we get:
59  * A = 4*pi*((3*mass)/(4*pi*density))^(2/3)
60  * */
61  p->heat_area = 4.*M_PI*pow( 3./4.*mass_kg/STEEL_DENSITY/M_PI, 2./3. ) * p->stats.heat_dissipation;
62 }

146 void pilot_heatAddSlot( const Pilot *p, PilotOutfitSlot *o )
147 {
148  /* We consider that only 1% of the energy is lost in the form of heat,
149  * this keeps numbers safe. */
150  double hmod = pilot_heatOutfitMod( p, o->outfit );

196 {

264 void pilot_heatUpdateCooldown( Pilot *p )
265 {

270 
271  t = pow2( 1. - p->ctimer / p->cdelay );
272  p->heat_T = p->heat_start - CONST_SPACE_STAR_TEMP - (p->heat_start -
273  CONST_SPACE_STAR_TEMP) * t + CONST_SPACE_STAR_TEMP;
274 
275  for (i=0; i<array_size(p->outfits); i++) {
276  o = p->outfits[i];
277  o->heat_T = o->heat_start - CONST_SPACE_STAR_TEMP - (o->heat_start -
278  CONST_SPACE_STAR_TEMP) * t + CONST_SPACE_STAR_TEMP;
279 
280  /* Refill ammo too (also part of Active Cooldown) */
281  /* Must be valid outfit. */
282  if (o->outfit == NULL)
283  continue;

286  ammo = outfit_ammo( o->outfit );
287  if (ammo == NULL)
